140     final String source_template =
141                 "import java.lang.invoke.*;\n" +
142                 "import java.lang.invoke.MethodHandles.*;\n" +
143                 "class Test {\n" +
144                 "    static final #{TYPE} f = #{VALUE};\n" +
145 
146                 "    static #{TYPE} bsm(MethodHandles.Lookup lookup, String name, Class<?> type) {\n" +
147                 "        return f;\n" +
148                 "    }\n" +
149 
150                 "    static void test() {\n" +
151                 "        #{TYPE} i = f;\n" +
152                 "    }\n" +
153                 "}";
154 
155     @Override
156     public void doWork() throws IOException {
157         newCompilationTask()
158                 .withOption("-g")
159                 .withSourceFromTemplate(source_template)
160                 .withListenerFactory(context -> {
161                         Symtab syms = Symtab.instance(context);
162                         Names names = Names.instance(context);
163                         Types types = Types.instance(context);
164                         return new Condifier(syms, names, types);
165                     })
166                 .generate(this::verifyBytecode);
167     }

169     void verifyBytecode(Result<Iterable<? extends JavaFileObject>> res) {
170         if (res.hasErrors()) {
171             fail("Diags found when compiling instance: " + res.compilationInfo());
172             return;
173         }
174         try (InputStream is = res.get().iterator().next().openInputStream()){
175             ClassFile cf = ClassFile.read(is);
176             Method testMethod = null;
177             for (Method m : cf.methods) {
178                 if (m.getName(cf.constant_pool).equals("test")) {
179                     testMethod = m;
180                     break;
181                 }
182             }
183             if (testMethod == null) {
184                 fail("Test method not found");
185                 return;
186             }
187             Code_attribute ea =
188                     (Code_attribute)testMethod.attributes.get(Attribute.Code);
189             if (testMethod == null) {
190                 fail("Code attribute for test() method not found");
191                 return;
192             }
193 
194             int bsmIdx = -1;
195 
196             for (Instruction i : ea.getInstructions()) {
197                 if (i.getMnemonic().equals("ldc")) {
198                     CONSTANT_Dynamic_info condyInfo = (CONSTANT_Dynamic_info)cf.constant_pool.get(i.getByte(1));
199                     bsmIdx = condyInfo.bootstrap_method_attr_index;
200                     System.out.println("condyInfo.getNameAndTypeInfo().getType() " + condyInfo.getNameAndTypeInfo().getType());
201                     if (!condyInfo.getNameAndTypeInfo().getType().equals(type.bytecodeTypeStr)) {
202                         fail("type mismatch for CONSTANT_Dynamic_info");
203                         return;
204                     }
205                 }
206             }
207 
208 
209             if (bsmIdx == -1) {
210                 fail("Missing constantdynamic in generated code");
211                 return;
212             }
213 
214             BootstrapMethods_attribute bsm_attr =
215                     (BootstrapMethods_attribute)cf
216                     .getAttribute(Attribute.BootstrapMethods);
217             if (bsm_attr.bootstrap_method_specifiers.length != 1) {
218                 fail("Bad number of method specifiers " +
219                         "in BootstrapMethods attribute");
220                 return;
221             }
222             BootstrapMethods_attribute.BootstrapMethodSpecifier bsm_spec =
223                     bsm_attr.bootstrap_method_specifiers[0];
224 
225             CONSTANT_MethodHandle_info bsm_handle =
226                     (CONSTANT_MethodHandle_info)cf.constant_pool
227                     .get(bsm_spec.bootstrap_method_ref);
228 
229             if (bsm_handle.reference_kind != RefKind.REF_invokeStatic) {
230                 fail("Bad kind on boostrap method handle");
231                 return;
232             }
233 
234             CONSTANT_Methodref_info bsm_ref =
235                     (CONSTANT_Methodref_info)cf.constant_pool
236                     .get(bsm_handle.reference_index);
237 
238             if (!bsm_ref.getClassInfo().getName().equals("Test")) {
239                 fail("Bad owner of boostrap method");
240                 return;
241             }
242 
243             if (!bsm_ref.getNameAndTypeInfo().getName().equals("bsm")) {
244                 fail("Bad boostrap method name");
245                 return;
246             }
247 
248             if (!bsm_ref.getNameAndTypeInfo()
249                     .getType().equals(asBSMSignatureString())) {
250                 fail("Bad boostrap method type" +
251                         bsm_ref.getNameAndTypeInfo().getType() + " " +
252                         asBSMSignatureString());
253                 return;
254             }
255 
256             LineNumberTable_attribute lnt =
257                     (LineNumberTable_attribute)ea.attributes.get(Attribute.LineNumberTable);
258 
259             if (lnt == null) {
260                 fail("No LineNumberTable attribute");
261                 return;
262             }
263             if (lnt.line_number_table_length != 2) {
264                 fail("Wrong number of entries in LineNumberTable");
265                 return;
266             }
267         } catch (Exception e) {
268             e.printStackTrace();
269             fail("error reading classfile: " + res.compilationInfo());
270             return;
271         }
272     }

274     String asBSMSignatureString() {
275         StringBuilder buf = new StringBuilder();
276         buf.append("(Ljava/lang/invoke/MethodHandles$Lookup;Ljava/lang/String;Ljava/lang/Class;");
277         buf.append(")" + type.bytecodeTypeStr);
278         return buf.toString();
279     }

281     class Condifier extends TreeScanner<Void, Void> implements TaskListener {
282 
283         MethodHandleSymbol bsm;
284         Symtab syms;
285         Names names;
286         Types types;
287 
288         Condifier(Symtab syms, Names names, Types types) {
289             this.syms = syms;
290             this.names = names;
291             this.types = types;
292         }
293 
294         @Override
295         public void started(TaskEvent e) {
296             //do nothing
297         }
298 
299         @Override
300         public void finished(TaskEvent e) {
301             if (e.getKind() == TaskEvent.Kind.ANALYZE) {
302                 scan(e.getCompilationUnit(), null);
303             }
304         }
305 
306         @Override
307         public Void visitVariable(VariableTree node, Void p) {
308             super.visitVariable(node, p);
309             JCVariableDecl tree = (JCVariableDecl)node;
310             VarSymbol v = tree.sym;
311             if (tree.init != null && v.name.toString().equals("i")) {
312                 List<Type> bsm_staticArgs = List.of(syms.methodHandleLookupType,
313                         syms.stringType,
314                         syms.classType);
315                 Name bsmName = names.fromString("bsm");
316                 Symbol.DynamicVarSymbol dynSym = new Symbol.DynamicVarSymbol(bsmName,
317                         syms.noSymbol,
318                         bsm,
319                         v.type,
320                         new LoadableConstant[0]);
321                 ((JCIdent)tree.init).sym = dynSym;
322                 ((JCIdent)tree.init).name = bsmName;
323             }
324             return null;
325         }
326 
327         @Override
328         public Void visitMethod(MethodTree node, Void p) {
329             super.visitMethod(node, p);
330             if (node.getName().toString().equals("bsm")) {
331                 bsm = ((JCMethodDecl)node).sym.asHandle();
332             }
333             return null;
334         }
335     }
