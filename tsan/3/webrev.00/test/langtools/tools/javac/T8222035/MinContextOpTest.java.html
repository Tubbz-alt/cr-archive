24 /*
25  * @test
26  * @bug 8222035
27  * @summary minimal inference context optimization is forcing resolution with incomplete constraints
28  * @compile/fail/ref=MinContextOpTest.out -XDrawDiagnostics MinContextOpTest.java
29  */
30 
31 import java.util.Map;
32 import java.util.function.Function;
33 import java.util.stream.Collector;
34 import java.util.stream.Stream;
35 
36 public class MinContextOpTest {
37     abstract class A {
38         abstract static class T<K> {
39             abstract String f();
40         }
41 
42         abstract <E> Function<E, E> id();
43 
44         abstract static class ImmutableMap<K, V> implements Map<K, V> {}
45 
46         abstract <T, K, V> Collector<T, ?, ImmutableMap<K, V>> toImmutableMap(
47                 Function<? super T, ? extends K> k, Function<? super T, ? extends V> v);
48 
49         ImmutableMap<String, T<?>> test(Stream<T> stream) {
50             return stream.collect(toImmutableMap(T::f, id()));
51         }
52     }
53 }
