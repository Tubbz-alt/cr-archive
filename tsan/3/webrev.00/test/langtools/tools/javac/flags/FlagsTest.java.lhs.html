20 /**
21  * @test
22  * @bug 8211138
23  * @summary Missing Flag enum constants
24  * @library /tools/javac/lib
25  * @modules jdk.compiler/com.sun.tools.javac.code
26  * @run main FlagsTest
27  */
28 import com.sun.tools.javac.code.Flags;
29 import java.lang.reflect.Field;
30 import java.lang.reflect.Modifier;
31 
32 public class FlagsTest {
33     public static void main(String[] args) throws IllegalAccessException {
34         for (Field f : Flags.class.getFields()) {
35             if (!Modifier.isStatic(f.getModifiers())) {
36                 continue;
37             }
38             long flag = ((Number) f.get(null)).longValue();
39             try {
40                 Flags.asFlagSet(flag);
41             } catch (AssertionError e) {
42                 throw new AssertionError("missing Flags enum constant for: " + f.getName(), e);
43             }
44         }
45     }
46 }
