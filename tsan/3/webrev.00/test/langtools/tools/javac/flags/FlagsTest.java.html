24 /**
25  * @test
26  * @bug 8211138
27  * @summary Missing Flag enum constants
28  * @library /tools/javac/lib
29  * @modules jdk.compiler/com.sun.tools.javac.code
30  * @run main FlagsTest
31  */
32 import com.sun.tools.javac.code.Flags;
33 import java.lang.reflect.Field;
34 import java.lang.reflect.Modifier;
35 
36 public class FlagsTest {
37     public static void main(String[] args) throws IllegalAccessException {
38         for (Field f : Flags.class.getFields()) {
39             if (!Modifier.isStatic(f.getModifiers())) {
40                 continue;
41             }
42             long flag = ((Number) f.get(null)).longValue();
43             try {
44                 Flags.asFlagSet(flag);
45             } catch (AssertionError e) {
46                 throw new AssertionError("missing Flags enum constant for: " + f.getName(), e);
47             }
48         }
49     }
50 }
