24 /*
25  * @test
26  * @bug 8204610
27  * @summary Compiler confused by parenthesized "this" in final fields assignments
28  * @library /tools/javac/lib
29  * @modules jdk.compiler/com.sun.tools.javac.api
30  *          jdk.compiler/com.sun.tools.javac.code
31  *          jdk.compiler/com.sun.tools.javac.comp
32  *          jdk.compiler/com.sun.tools.javac.main
33  *          jdk.compiler/com.sun.tools.javac.tree
34  *          jdk.compiler/com.sun.tools.javac.util
35  * @build combo.ComboTestHelper
36 
37  * @run main T8204610
38  */
39 
40 import combo.ComboInstance;
41 import combo.ComboParameter;
42 import combo.ComboTask.Result;
43 import combo.ComboTestHelper;
44 
45 public class T8204610 extends ComboInstance<T8204610> {
46 
47     enum ParenKind implements ComboParameter {
48         NONE(""),
49         ONE("#P"),
50         TWO("#P#P"),
51         THREE("#P#P#P");
52 
53         String parensTemplate;
54 
55         ParenKind(String parensTemplate) {
56             this.parensTemplate = parensTemplate;
57         }
58 
59         @Override
60         public String expand(String optParameter) {
61             return parensTemplate.replaceAll("#P", optParameter.equals("OPEN") ? "(" : ")");
62         }
63     }
64 
65     public static void main(String... args) {
66         new ComboTestHelper<T8204610>()
67                 .withArrayDimension("PAREN", (x, pk, idx) -> x.parenKinds[idx] = pk, 3, ParenKind.values())
68                 .run(T8204610::new);
69     }
70 
71     ParenKind[] parenKinds = new ParenKind[3];
72 
73     @Override
74     public void doWork() {
75         newCompilationTask()
76                 .withSourceFromTemplate(bodyTemplate)
77                 .analyze(this::check);
78     }
79 
80     String bodyTemplate = "class Test {\n" +
81                           "   final int x;\n" +
82                           "   Test() {\n" +
83                           "      #{PAREN[0].OPEN} #{PAREN[1].OPEN} this #{PAREN[1].CLOSE} . #{PAREN[2].OPEN} x #{PAREN[2].CLOSE} #{PAREN[0].CLOSE} = 1;\n" +
84                           "   } }";
85 
86     void check(Result<?> res) {
87         boolean expectedFail = parenKinds[2] != ParenKind.NONE;
88         if (expectedFail != res.hasErrors()) {
89             fail("unexpected compilation result for source:\n" +
90                 res.compilationInfo());
91         }
92     }
93 }
