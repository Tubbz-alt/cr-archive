30 import java.io.IOException;
31 import java.net.URI;
32 import java.util.List;
33 import javax.lang.model.element.RecordComponentElement;
34 import javax.lang.model.element.TypeElement;
35 import javax.tools.JavaCompiler;
36 import javax.tools.SimpleJavaFileObject;
37 import javax.tools.ToolProvider;
38 
39 import com.sun.source.util.JavacTask;
40 import java.util.Objects;
41 import javax.lang.model.element.ExecutableElement;
42 import javax.lang.model.util.ElementFilter;
43 
44 import static javax.tools.JavaFileObject.Kind.SOURCE;
45 
46 public class MapAccessorToComponent {
47     public static void main(String... args) throws IOException {
48         new MapAccessorToComponent().run();
49     }
50 
51     public void run() throws IOException {
52         String code = "record R(int val1, int val2) {}";
53         JavaCompiler c = ToolProvider.getSystemJavaCompiler();
54         JavacTask t = (JavacTask) c.getTask(null, null, null,
55                 List.of("--enable-preview", "-source", Integer.toString(Runtime.version().feature())), null,
56                 List.of(new MyFileObject(code)));
57         TypeElement record = (TypeElement) t.analyze().iterator().next();
58         for (RecordComponentElement rce : ElementFilter.recordComponentsIn(record.getEnclosedElements())) {
59             ExecutableElement accessor = rce.getAccessor();
60             if (!Objects.equals(t.getElements().recordComponentFor(accessor), rce)) {
61                 throw new AssertionError("Did not find the correct record component!");
62             }
63         }
64     }
65 
66     class MyFileObject extends SimpleJavaFileObject {
67         private final String code;
68 
69         MyFileObject(String code) {
70             super(URI.create("myfo:///Test.java"), SOURCE);
71             this.code = code;
72         }
73         @Override
74         public String getCharContent(boolean ignoreEncodingErrors) {
75             return code;
76         }
77     }
78 }
