24 /*
25  * @test
26  * @summary check that members of abtract datum has protected access
27  * @ignore currently the compiler is generating a readResolve method but this test is failing now
28  * as the compiler will error if there is an explicit declaration of readResolve.
29  * @modules jdk.jdeps/com.sun.tools.classfile
30  *          jdk.compiler/com.sun.tools.javac.util
31  * @compile CheckReadResolveMethodTest.java
32  * @run main CheckReadResolveMethodTest
33  */
34 
35 import java.io.*;
36 
37 import com.sun.tools.classfile.*;
38 
39 public class CheckReadResolveMethodTest {
40 
41     // readResolve should be generated as a constructor is provided
42     record Point1(int i, int j) implements Serializable {
43         public Point1(int i, int j) {
44             this.i = i;
45             this.j = j;
46         }
47     }
48 
49     // readResolve should be generated the record implements Serializable
50     record Point2(int i, int j) implements Serializable {}
51 
52     // no readResolve should be generated as the record doesnt implement Serializable
53     record Point3(int i, int j) {}
54 
55     // no readResolve should be generated as an implementation is already provided
56     record Point4(int i, int j) {
57         private Object readResolve() {
58             return new Point4(i, j);
59         }
60     }
61 
62     // no readResolve should be generated as an implementation of readObject is already provided
63     record Point5(int i, int j) {
64         private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {}
65     }
66 
67     public static void main(String args[]) throws Throwable {
68         new CheckReadResolveMethodTest().run();
69     }
70 
71     void run() throws Throwable {
72         checkReadResolveMethod("Point1", true);
73         checkReadResolveMethod("Point2", true);
74         checkReadResolveMethod("Point3", false);
75         checkReadResolveMethod("Point4", true);
76         checkReadResolveMethod("Point5", false);
77     }
78 
79     void checkReadResolveMethod(String className, boolean shouldBeThere) throws Throwable {
80         File testClasses = new File(System.getProperty("test.classes"));
81         File file = new File(testClasses,
82                 CheckReadResolveMethodTest.class.getName() + "$" + className +".class");
83         ClassFile classFile = ClassFile.read(file);
84         boolean found = false;
85         for (Method f : classFile.methods) {
86             if (f.getName(classFile.constant_pool).equals("readResolve")) {
87                 found = true;
88             }
89         }
90         if (found != shouldBeThere) {
91             throw new AssertionError("incorrect generation of readResolve method at class " + className);
92         }
93     }
94 }
