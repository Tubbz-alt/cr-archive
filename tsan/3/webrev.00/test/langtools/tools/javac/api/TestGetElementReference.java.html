 24 /*
 25  * @test
 26  * @bug 8012929
 27  * @summary Trees.getElement should work not only for declaration trees, but also for use-trees
 28  * @modules jdk.compiler
 29  * @build TestGetElementReference
 30  * @run main TestGetElementReference
 31  */

 33 import com.sun.source.tree.CompilationUnitTree;
 34 import com.sun.source.tree.Tree;
 35 import com.sun.source.util.*;
 36 import java.io.File;
 37 import java.io.IOException;
 38 import java.net.URI;
 39 import java.util.ArrayList;
 40 import java.util.Arrays;
 41 import java.util.List;
 42 import java.util.regex.Matcher;
 43 import java.util.regex.Pattern;
 44 import javax.lang.model.element.Element;
 45 import javax.tools.Diagnostic;
 46 import javax.tools.DiagnosticCollector;
 47 import javax.tools.JavaFileObject;
 48 import javax.tools.SimpleJavaFileObject;
 49 import javax.tools.StandardJavaFileManager;
 50 import javax.tools.ToolProvider;
 51 
 52 public class TestGetElementReference {
 53 
 54     private static final String JDK_VERSION =
 55             Integer.toString(Runtime.getRuntime().version().feature());
 56 
 57     public static void main(String... args) throws IOException {
 58         analyze("TestGetElementReferenceData.java");
 59         analyze("mod/module-info.java", "mod/api/pkg/Api.java");
 60     }
 61 
 62     private static void analyze(String... fileNames) throws IOException {
 63         try (StandardJavaFileManager fm = ToolProvider.getSystemJavaCompiler().getStandardFileManager(null, null, null)) {
 64             List<JavaFileObject> files = new ArrayList<>();
 65             for (String fileName : fileNames) {
 66                 File source = new File(System.getProperty("test.src", "."), fileName.replace('/', File.separatorChar)).getAbsoluteFile();
 67                 for (JavaFileObject f : fm.getJavaFileObjects(source)) {
 68                     files.add(f);
 69                 }
 70             }
 71             DiagnosticCollector<JavaFileObject> diagnostics = new DiagnosticCollector<>();
 72             List<String> options = List.of("-Xjcov",
 73                                            "--enable-preview",
 74                                            "-source", JDK_VERSION);
 75             JavacTask ct = (JavacTask) ToolProvider.getSystemJavaCompiler().getTask(null, null, diagnostics, options, null, files);
 76             Trees trees = Trees.instance(ct);
 77             CompilationUnitTree cut = ct.parse().iterator().next();
 78 
 79             ct.analyze();
 80 
 81             for (Diagnostic<? extends JavaFileObject> d : diagnostics.getDiagnostics()) {
 82                 if (d.getKind() == Diagnostic.Kind.ERROR) {
 83                     throw new IllegalStateException("Should have been attributed without errors: " + diagnostics.getDiagnostics());
 84                 }
 85             }
 86 
 87             Pattern p = Pattern.compile("/\\*getElement:(.*?)\\*/");
 88             Matcher m = p.matcher(cut.getSourceFile().getCharContent(false));
 89 
 90             while (m.find()) {
 91                 TreePath tp = pathFor(trees, cut, m.start() - 1);
 92                 String expected = m.group(1);
 93                 if (expected.startsWith("getParentPath:")) {
 94                     tp = tp.getParentPath();
 95                     expected = expected.substring("getParentPath:".length());
 96                 }
 97                 Element found = trees.getElement(tp);
 98                 String actual = found != null ? found.getKind() + ":" + symbolToString(found) : "<null>";
 99 
100                 if (!expected.equals(actual)) {
101                     throw new IllegalStateException("expected=" + expected + "; actual=" + actual + "; tree: " + tp.getLeaf());
102                 }
103             }
104         }
105     }
106 
107     private static TreePath pathFor(final Trees trees, final CompilationUnitTree cut, final int pos) {
108         final TreePath[] result = new TreePath[1];
109 
110         new TreePathScanner<Void, Void>() {
111             @Override public Void scan(Tree node, Void p) {
112                 if (   node != null
113                     && trees.getSourcePositions().getStartPosition(cut, node) <= pos
114                     && pos <= trees.getSourcePositions().getEndPosition(cut, node)) {
115                     result[0] = new TreePath(getCurrentPath(), node);
116                     return super.scan(node, p);
117                 }
118                 return null;
119             }
120         }.scan(cut, null);
121 
122         return result[0];
123     }
124 
125     private static String symbolToString(Element el) {
126         switch (el.getKind()) {
127             case METHOD: return symbolToString(el.getEnclosingElement()) + "." + el.toString();
128             case CONSTRUCTOR: return symbolToString(el.getEnclosingElement().getEnclosingElement()) + "." + el.toString();
129             default:
130                 return el.toString();
131         }
132     }
133 
134     static class TestFileObject extends SimpleJavaFileObject {
135         private final String text;
136         public TestFileObject(String text) {
137             super(URI.create("myfo:/Test.java"), JavaFileObject.Kind.SOURCE);
138             this.text = text;
139         }
140         @Override public CharSequence getCharContent(boolean ignoreEncodingErrors) {
141             return text;
142         }
143     }
