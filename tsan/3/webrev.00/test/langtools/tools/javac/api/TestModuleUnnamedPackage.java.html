24 /*
25  * @test
26  * @bug     8234025
27  * @summary Elements.getPackageElement(ModuleElement,CharSequence) returns null for unnamed package
28  * @modules jdk.compiler/com.sun.tools.javac.api
29  *          jdk.compiler/com.sun.tools.javac.main
30  * @library /tools/lib /tools/javac/lib
31  * @build toolbox.ModuleBuilder toolbox.ToolBox
32  * @run main TestModuleUnnamedPackage
33  */
34 
35 import java.io.IOException;
36 import java.nio.file.Path;
37 import java.util.List;
38 import java.util.Set;
39 
40 import javax.annotation.processing.RoundEnvironment;
41 import javax.lang.model.element.ModuleElement;
42 import javax.lang.model.element.PackageElement;
43 import javax.lang.model.element.TypeElement;
44 import javax.lang.model.util.Elements;
45 import javax.tools.JavaCompiler;
46 import javax.tools.JavaFileObject;
47 import javax.tools.StandardJavaFileManager;
48 import javax.tools.StandardLocation;
49 import javax.tools.ToolProvider;
50 
51 import com.sun.source.util.JavacTask;
52 import toolbox.Assert;
53 import toolbox.ToolBox;
54 
55 public class TestModuleUnnamedPackage extends JavacTestingAbstractProcessor {
56     public static void main(String... args) throws IOException {
57         TestModuleUnnamedPackage t = new TestModuleUnnamedPackage();
58             t.run();
59     }
60 
61     void run() throws IOException {
62         ToolBox tb = new ToolBox();
63         Path src = Path.of("src");
64 
65         tb.writeJavaFiles(src, "module m { exports p; }",
66                 "package p; public class C { }");
67 
68         JavaCompiler c = ToolProvider.getSystemJavaCompiler();
69         StandardJavaFileManager fm = c.getStandardFileManager(null, null, null);
70         fm.setLocationFromPaths(StandardLocation.SOURCE_PATH, List.of(src));
71         List<String> options = List.of("-proc:only");
72         Iterable<? extends JavaFileObject> files = fm.getJavaFileObjects(tb.findJavaFiles(src));
73         JavacTask t = (JavacTask) c.getTask(null, fm, null, null, null, files);
74         t.setProcessors(List.of(this));
75         t.call();
76     }
77 
78     public boolean process(Set<? extends TypeElement> elems, RoundEnvironment rEnv) {
79         if (rEnv.processingOver()) {
80             Elements elements = processingEnv.getElementUtils();
81             ModuleElement m = elements.getModuleElement("m");
82             PackageElement unnamed = elements.getPackageElement(m, "");
83             System.out.println("module m: " + m);
84             System.out.println("module m, unnamed package: " + unnamed);
85             Assert.checkNonNull(unnamed, "unnamed package in module m");
86         }
87         return true;
88     }
89 }
