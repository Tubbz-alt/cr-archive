 24 /*
 25  * @test
 26  * @bug 8223443
 27  * @summary Verify errors are not reported when computing Scopes.
 28  * @modules jdk.compiler
 29  */
 30 
 31 import java.io.IOException;
 32 import java.net.URI;
 33 import java.util.ArrayList;
 34 import java.util.Arrays;
 35 import java.util.List;
 36 import java.util.concurrent.atomic.AtomicBoolean;
 37 
 38 import javax.tools.Diagnostic;
 39 import javax.tools.DiagnosticListener;
 40 import javax.tools.JavaCompiler;
 41 import javax.tools.JavaFileObject;
 42 import javax.tools.SimpleJavaFileObject;
 43 import javax.tools.ToolProvider;
 44 
 45 import static javax.tools.JavaFileObject.Kind.SOURCE;
 46 
 47 import com.sun.source.tree.CompilationUnitTree;
 48 import com.sun.source.tree.Tree;
 49 import com.sun.source.util.JavacTask;
 50 import com.sun.source.util.TaskEvent;
 51 import com.sun.source.util.TaskListener;
 52 import com.sun.source.util.TreePath;
 53 import com.sun.source.util.TreePathScanner;
 54 import com.sun.source.util.Trees;
 55 
 56 public class TestGetScopeErrors {
 57     public static void main(String... args) throws IOException {
 58         new TestGetScopeErrors().run();
 59     }
 60 
 61     void run() throws IOException {
 62         JavaCompiler c = ToolProvider.getSystemJavaCompiler();
 63         String code =
 64                 "public class Test {" +
 65                 "    private Object obj = new Object() {" +
 66                 "        private Unresolvable u;" +
 67                 "    };" +
 68                 "    void test() {" +
 69                 "        new Object() {" +
 70                 "            private Unresolvable u;" +
 71                 "        };" +
 72                 "    }" +
 73                 "}";
 74         class MyFileObject extends SimpleJavaFileObject {
 75             MyFileObject() {
 76                 super(URI.create("myfo:///Test.java"), SOURCE);
 77             }
 78             @Override
 79             public String getCharContent(boolean ignoreEncodingErrors) {
 80                 return code;
 81             }
 82         }
 83         AtomicBoolean enterDone = new AtomicBoolean();
 84         List<String> errors = new ArrayList<>();
 85         DiagnosticListener<JavaFileObject> noErrors = d -> {
 86             if (!enterDone.get() && d.getKind() == Diagnostic.Kind.ERROR) {
 87                 throw new AssertionError(d.toString());
 88             }
 89             errors.add(d.getSource().getName() + ":" +
 90                        d.getPosition() + ":" +
 91                        d.getCode());
 92         };
 93         JavacTask t =
 94                 (JavacTask) c.getTask(null, null, noErrors,
 95                                       Arrays.asList("-XDrawDiagnostics"),
 96                                       null, List.of(new MyFileObject()));
 97         CompilationUnitTree cut = t.parse().iterator().next();
 98         Trees trees = Trees.instance(t);
 99         t.addTaskListener(new TaskListener() {
100             @Override
101             public void finished(TaskEvent e) {
102                 if (e.getKind() == TaskEvent.Kind.ENTER) {
103                     new TreePathScanner<Void, Void>() {
104                         @Override
105                         public Void scan(Tree tree, Void p) {
106                             if (tree != null) {
107                                 TreePath path =
108                                         new TreePath(getCurrentPath(), tree);
109                                 trees.getScope(path);
110                             }
111                             return super.scan(tree, p);
112                         }
113                     }.scan(cut, null);
114                     enterDone.set(true);
115                 }
116             }
117         });
118 
119         t.analyze();
120 
121         List<String> expectedErrors = List.of(
122             "/Test.java:74:compiler.err.cant.resolve",
123             "/Test.java:154:compiler.err.cant.resolve"
124         );
125 
126         if (!expectedErrors.equals(errors)) {
127             throw new IllegalStateException("Unexpected errors: " + errors);
128         }
129     }
130 
131 }
