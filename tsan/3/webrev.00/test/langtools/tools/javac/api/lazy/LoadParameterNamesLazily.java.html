 24 /*
 25  * @test
 26  * @bug 8217047
 27  * @summary Verify the parameter names can be injected using ParameterNameProvider.
 28  * @library /tools/lib
 29  * @modules jdk.compiler/com.sun.tools.javac.api
 30  *          jdk.compiler/com.sun.tools.javac.main
 31  * @build toolbox.JavacTask toolbox.TestRunner toolbox.ToolBox
 32  * @run main LoadParameterNamesLazily
 33  */
 34 
 35 import java.io.IOException;
 36 import java.nio.file.Files;
 37 import java.nio.file.Path;
 38 import java.nio.file.Paths;
 39 
 40 import javax.lang.model.element.ExecutableElement;
 41 import javax.lang.model.element.Name;
 42 import javax.lang.model.element.TypeElement;
 43 import javax.lang.model.util.ElementFilter;
 44 
 45 import com.sun.source.util.*;
 46 
 47 import toolbox.JavacTask;
 48 import toolbox.Task;
 49 import toolbox.TestRunner;
 50 import toolbox.ToolBox;
 51 
 52 public class LoadParameterNamesLazily extends TestRunner {
 53 
 54     public static void main(String... args) throws Exception {
 55         LoadParameterNamesLazily t = new LoadParameterNamesLazily();
 56         t.runTests();
 57     }
 58 
 59     private static final String libClass =
 60             "package lib;" +
 61             "/**Lib javadoc.*/" +
 62             "public class Lib {" +
 63             "    /**Lib method javadoc.*/" +
 64             "    public static void m(int param, int other) {}" +
 65             "}";
 66     private final ToolBox tb = new ToolBox();
 67 
 68     LoadParameterNamesLazily() throws IOException {
 69         super(System.err);
 70     }
 71 
 72     @Test
 73     public void testLoadTreesLazily() throws IOException {
 74         Path libSrc = Paths.get("lib-src");
 75         tb.writeJavaFiles(libSrc, libClass);
 76         Path libClasses = Paths.get("lib-classes");
 77         Files.createDirectories(libClasses);
 78 
 79         new JavacTask(tb)
 80               .outdir(libClasses)
 81               .options()
 82               .files(tb.findJavaFiles(libSrc))
 83               .run(Task.Expect.SUCCESS)
 84               .writeAll();
 85 
 86         Path src = Paths.get("src");
 87         tb.writeJavaFiles(src,
 88                           "class Use {" +
 89                           " lib.Lib lib;" +
 90                           "}");
 91         Path classes = Paths.get("classes");
 92         Files.createDirectories(classes);
 93 
 94         new JavacTask(tb)
 95               .outdir(classes)
 96               .options("-classpath", libClasses.toString())
 97               .files(tb.findJavaFiles(src))
 98               .callback(task -> {
 99                   task.setParameterNameProvider(parameter -> {
100                       ExecutableElement method = (ExecutableElement) parameter.getEnclosingElement();
101                       TypeElement clazz =
102                               (TypeElement) method.getEnclosingElement();
103                       if (clazz.getQualifiedName().contentEquals("lib.Lib")) {
104                           if (method.getParameters().indexOf(parameter) == 0) {
105                               return "testName";
106                           } else {
107                               return null;
108                           }
109                       }
110                       return null;
111                   });
112                   task.addTaskListener(new TaskListener() {
113                       @Override
114                       public void finished(TaskEvent e) {
115                           if (e.getKind() == TaskEvent.Kind.ANALYZE) {
116                               TypeElement lib = task.getElements().getTypeElement("lib.Lib");
117                               lib.getClass(); //not null
118                               ExecutableElement method =
119                                       ElementFilter.methodsIn(lib.getEnclosedElements()).get(0);
120                               Name paramName0 = method.getParameters().get(0).getSimpleName();
121                               if (!paramName0.contentEquals("testName")) {
122                                   throw new IllegalStateException("Unexpected parameter name: " +
123                                                                   paramName0);
124                               }
125                               Name paramName1 = method.getParameters().get(1).getSimpleName();
126                               if (!paramName1.contentEquals("arg1")) {
127                                   throw new IllegalStateException("Unexpected parameter name: " +
128                                                                   paramName1);
129                               }
130                           }
131                       }
132                   });
133               })
134               .run(Task.Expect.SUCCESS)
135               .writeAll();
136     }
137 
138 }
