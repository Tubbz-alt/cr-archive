 35 import java.io.File;
 36 import java.io.IOException;
 37 import java.nio.file.Path;
 38 import java.util.Collections;
 39 import java.util.List;
 40 import javax.tools.JavaFileObject;
 41 import javax.tools.StandardJavaFileManager;
 42 import javax.tools.StandardLocation;
 43 
 44 /**
 45  * For those paths supported by a file manager, verify that the paths
 46  * can be encapsulated by file objects, such that the file objects can
 47  * be used by a tool such as javac.
 48  */
 49 public class SJFM_GetFileObjects extends SJFM_TestBase {
 50     public static void main(String... args) throws Exception {
 51         new SJFM_GetFileObjects().run();
 52     }
 53 
 54     @Test
 55     void test_getJavaFileObjects(StandardJavaFileManager fm) throws IOException {
 56         test_getJavaFileObjects(fm, getTestFilePaths());
 57         test_getJavaFileObjects(fm, getTestZipPaths());
 58     }
 59 
 60     /**
 61      * Tests the getJavaFileObjects method for a specific file manager
 62      * and a series of paths.
 63      *
 64      * Note: instances of MyStandardJavaFileManager only support
 65      * encapsulating paths for files in the default file system.
 66      *
 67      * @param fm  the file manager to be tested
 68      * @param paths  the paths to be tested
 69      * @throws IOException
 70      */
 71     void test_getJavaFileObjects(StandardJavaFileManager fm, List<Path> paths) throws IOException {
 72         boolean expectException = !isGetFileObjectsSupported(fm, paths);
 73         try {
 74             compile(fm.getJavaFileObjects(paths.toArray(new Path[paths.size()])));
 75             if (expectException)
 76                 error("expected exception not thrown");
 77         } catch (RuntimeException e) {
 78             if (expectException && e instanceof IllegalArgumentException)
 79                 return;
 80             error("unexpected exception thrown: " + e);
 81         }
 82     }

154     /**
155      * Compiles a set of files.
156      *
157      * @param files the files to be compiled.
158      * @throws IOException
159      */
160     void compile(Iterable<? extends JavaFileObject> files) throws IOException {
161         String name = "compile" + (compileCount++);
162         try (StandardJavaFileManager fm = comp.getStandardFileManager(null, null, null)) {
163             File f = new File(name);
164             f.mkdirs();
165             // use setLocation(Iterable<File>) to avoid relying on setLocationFromPaths
166             fm.setLocation(StandardLocation.CLASS_OUTPUT, Collections.singleton(f));
167             boolean ok = comp.getTask(null, fm, null, null, null, files).call();
168             if (!ok)
169                 error(name + ": compilation failed");
170         }
171     }
172 
173     int compileCount;
