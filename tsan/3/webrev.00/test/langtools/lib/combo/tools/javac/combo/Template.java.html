24 package tools.javac.combo;
25 
26 import java.util.Map;
27 import java.util.function.Function;
28 import java.util.regex.Matcher;
29 import java.util.regex.Pattern;
30 
31 /**
32  * A template into which tags of the form {@code #\{KEY\}} or
33  * {@code #\{KEY.SUBKEY\}} can be expanded.
34  */
35 public interface Template {
36     static final Pattern KEY_PATTERN = Pattern.compile("#\\{([A-Z_][A-Z0-9_]*(?:\\[\\d+\\])?)(?:\\.([A-Z0-9_]*))?\\}");
37 
38     String expand(String selector);
39 
40         /* Looks for expandable keys.  An expandable key can take the form:
41          *   #{MAJOR}
42          *   #{MAJOR.}
43          *   #{MAJOR.MINOR}
44          * where MAJOR can be IDENTIFIER or IDENTIFIER[NUMERIC_INDEX]
45          * and MINOR can be an identifier.
46          *
47          * The ability to have an empty minor is provided on the
48          * assumption that some tests that can be written with this
49          * will find it useful to make a distinction akin to
50          * distinguishing F from F(), where F is a function pointer,
51          * and also cases of #{FOO.#{BAR}}, where BAR expands to an
52          * empty string.
53          *
54          * However, this being a general-purpose framework, the exact
55          * use is left up to the test writers.
56          */
57     public static String expandTemplate(String template,
58                                         Map<String, Template> vars) {
59         return expandTemplate(template, vars::get);
60         }
61 
62     private static String expandTemplate(String template, Function<String, Template> resolver) {
63             CharSequence in = template;
64             StringBuffer out = new StringBuffer();
65             while (true) {
66                 boolean more = false;
67             Matcher m = KEY_PATTERN.matcher(in);
68                 while (m.find()) {
69                     String major = m.group(1);
70                     String minor = m.group(2);
71                 Template key = resolver.apply(major);
72                     if (key == null)
73                         throw new IllegalStateException("Unknown major key " + major);
74 
75                     String replacement = key.expand(minor == null ? "" : minor);
76                 more |= KEY_PATTERN.matcher(replacement).find();
77                     m.appendReplacement(out, replacement);
78                 }
79                 m.appendTail(out);
80                 if (!more)
81                     return out.toString();
82                 else {
83                     in = out;
84                     out = new StringBuffer();
85                 }
86             }
87     }
88 }
