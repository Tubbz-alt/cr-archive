24 /*
25  * @test
26  * @bug 8235474 8236715
27  * @summary Tests for evalution of records
28  * @modules jdk.jshell

30  * @run testng RecordsTest

32 
33 import org.testng.annotations.Test;
34 
35 import javax.lang.model.SourceVersion;
36 import jdk.jshell.Snippet.Status;
37 import jdk.jshell.UnresolvedReferenceException;
38 import static org.testng.Assert.assertEquals;
39 import org.testng.annotations.BeforeMethod;
40 
41 @Test
42 public class RecordsTest extends KullaTesting {
43 
44     public void testRecordClass() {
45         assertEval("record R(String s, int i) { }");
46         assertEquals(varKey(assertEval("R r = new R(\"r\", 42);")).name(), "r");
47         assertEval("r.s()", "\"r\"");
48         assertEval("r.i()", "42");
49     }
50 
51     public void testRecordCorralling() {
52         //simple record with a mistake that can be fixed by corralling:
53         assertEval("record R1(int i) { int g() { return j; } }", ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_DEFINED, true, null));
54         assertEval("R1 r1 = new R1(1);", null, UnresolvedReferenceException.class, DiagCheck.DIAG_OK, DiagCheck.DIAG_OK, added(Status.VALID));
55         //record with a concise constructor and a mistake take can be fixed by corralling:
56         assertEval("record R2(int i) { public R2 {} int g() { return j; } }", ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_DEFINED, true, null));
57         assertEval("R2 r2 = new R2(1);", null, UnresolvedReferenceException.class, DiagCheck.DIAG_OK, DiagCheck.DIAG_OK, added(Status.VALID));
58         //record with a full constructor and a mistake take can be fixed by corralling:
59         assertEval("record R3(int i) { public R3(int i) {this.i = i;} int g() { return j; } }", ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_DEFINED, true, null));
60         assertEval("R3 r3 = new R3(1);", null, UnresolvedReferenceException.class, DiagCheck.DIAG_OK, DiagCheck.DIAG_OK, added(Status.VALID));
61         //record with an accessor and a mistake take can be fixed by corralling:
62         assertEval("record R4(int i) { public int i() { return i; } int g() { return j; } }", ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_DEFINED, true, null));
63         assertEval("R4 r4 = new R4(1);", null, UnresolvedReferenceException.class, DiagCheck.DIAG_OK, DiagCheck.DIAG_OK, added(Status.VALID));
64         //record with an accessor with a mistake take can be fixed by corralling:
65         assertEval("record R5(int i) { public int i() { return j; } }", ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_DEFINED, true, null));
66         assertEval("R5 r5 = new R5(1);", null, UnresolvedReferenceException.class, DiagCheck.DIAG_OK, DiagCheck.DIAG_OK, added(Status.VALID));
67     }
68 
69     public void testRecordField() {
70         assertEquals(varKey(assertEval("String record = \"\";")).name(), "record");
71         assertEval("record.length()", "0");
72     }
73 
74     public void testRecordMethod() {
75         assertEquals(methodKey(assertEval("String record(String record) { return record + record; }")).name(), "record");
76         assertEval("record(\"r\")", "\"rr\"");
77         assertEval("record(\"r\").length()", "2");
78     }
79 
80     @BeforeMethod
81     public void setUp() {
82         setUp(b -> b.compilerOptions("--enable-preview", "-source", String.valueOf(SourceVersion.latest().ordinal()))
83                     .remoteVMOptions("--enable-preview"));
84     }
85 }
