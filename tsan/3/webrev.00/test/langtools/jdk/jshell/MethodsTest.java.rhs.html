 48     public void testSignature1() {

 57     public void testSignature2() {

308     public void methodSignatureUnresolved() {
309         MethodSnippet key = (MethodSnippet) methodKey(assertEval("und m() { return new und(); }", added(RECOVERABLE_NOT_DEFINED)));
310         assertMethodDeclSnippet(key, "m", "()und", RECOVERABLE_NOT_DEFINED, 1, 0);

341     public void lambdas() {
342         assertEval("class Inner1 implements Runnable {" +
343                 "public Runnable lambda1 = () -> {};" +
344                 "public void function() {}" +
345                 "public void run() {}" +
346                 "}");
347 
348         assertEval("class Inner2 {" +
349                 "private Runnable lambda1 = () -> {};" +
350                 "private static void staticFunction() {}" +
351                 "}");
352 
353         // the following method references and lambda functions
354         // generate synthetic methods
355         assertEval("Runnable run = () -> {};");
356         assertEval("Inner1 inner = new Inner1();");
357         assertEval("Runnable l1 = inner::function;");
358         assertEval("Runnable l2 = Inner1::new;");
359         assertEval("inner.lambda1 = inner::function;");
360         assertEval("java.util.stream.IntStream.of(2).mapToObj(int[]::new);");
361         assertNumberOfActiveMethods(0);
362         assertActiveKeys();
