24 /*
25  * @test
26  * @bug      4924383 8149402 8196202
27  * @summary  Test to make sure the -group option works correctly
28  *           with the given pattern usages.

29  * @library  ../../lib
30  * @modules  jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestGroupOption
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestGroupOption extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestGroupOption tester = new TestGroupOption();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test1() {
46         // Make sure the warning is not printed when -group is used correctly.
47         javadoc("-d", "out-1",
48                 "-sourcepath", testSrc,
49                 "-group", "Package One", "pkg1",
50                 "-group", "Package Two", "pkg2",
51                 "-group", "Package Three", "pkg3",
52                 "pkg1", "pkg2", "pkg3");
53         checkExit(Exit.OK);
54 
55         checkOutput(Output.OUT, false,
56                 "-group");
57     }
58 
59      @Test
60     // Make sure the "Other packages" section is printed and the header for empty section is not.
61     // Make sure that the headers of group that is defined using patterns are printed.
62     public void test2() {
63         javadoc("-d", "out-2",

64                 "-sourcepath", testSrc,
65                 "-group", "Group pkg*", "pkg*",
66                 "-group", "Group abc*", "abc*",
67                 "-group", "Empty group", "qwerty*",
68                 "-group", "Group a*", "a*",
69                 "pkg1", "pkg2", "pkg3", "abc1",  "abc2", "abc3", "other", testSrc("InUnnamedPackage.java"));
70         checkExit(Exit.OK);
71 
72         checkOutput("index.html", true, "Group pkg*", "Group abc*", "Other Packages");
73         checkOutput("index.html", false, "Empty group", "Group a*");
74     }
75 
76     @Test
77     public void test3() {
78         // Make sure the warning is printed when -group is not used correctly.
79         javadoc("-d", "out-3",
80                 "-sourcepath", testSrc,
81                 "-group", "Package One", "pkg1",
82                 "-group", "Package One", "pkg2",
83                 "-group", "Package One", "pkg3",
84                 "pkg1", "pkg2", "pkg3");
85         checkExit(Exit.OK);
86 
87         checkOutput(Output.OUT, true,
88                 "-group");
89 
90     }
91 }
