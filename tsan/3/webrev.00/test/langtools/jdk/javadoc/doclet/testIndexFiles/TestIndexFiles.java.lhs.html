24 /*
25  * @test
26  * @bug      8170825 8196027
27  * @summary  Perform tests on index files generated by javadoc.
28  * @author   bpatel
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestIndexFiles
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestIndexFiles extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestIndexFiles tester = new TestIndexFiles();
41         tester.runTests();
42     }
43 
44     @Test
45     public void testIndexFiles() {
46         javadoc("-d", "out", "-splitindex", "-Xdoclint:none", "-sourcepath", testSrc,
47                 "-use", "pkg");
48         checkExit(Exit.OK);
49         checkIndexFiles(false);
50     }
51 
52     void checkIndexFiles(boolean found) {
53         checkOutput("index-files/index-1.html", found,
54                 "<li>Prev Letter</li>\n"
55                 + "<li><a href=\"index-2.html\">Next Letter</a></li>");
56         checkOutput("index-files/index-5.html", found,
57                 "<li><a href=\"index-4.html\">Prev Letter</a></li>\n"
58                 + "<li>Next Letter</li>");
59         checkOutput("index-files/index-1.html", found,
60                 "<li><a href=\"index-0.html\">Prev Letter</a></li>\n"
61                 + "<li><a href=\"index-1.html\">Next Letter</a></li>");
62         checkOutput("index-files/index-5.html", found,
63                 "<li><a href=\"index-4.html\">Prev Letter</a></li>\n"
64                 + "<li><a href=\"index-5.html\">Next Letter</a></li>");
65     }
66 }
