24 /*
25  * @test
26  * @bug 4857717 8025633 8026567 8164407 8182765 8205593
27  * @summary Test to make sure that externally overriden and implemented methods
28  * are documented properly.  The method should still include "implements" or
29  * "overrides" documentation even though the method is external.
30  * @author jamieh
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.* TestExternalOverridenMethod
34  * @run main TestExternalOverridenMethod
35  */
36 import javadoc.tester.JavadocTester;
37 
38 public class TestExternalOverridenMethod extends JavadocTester {
39 
40     static final String uri = "http://java.sun.com/j2se/1.4.1/docs/api";
41 
42     public static void main(String... args) throws Exception {
43         TestExternalOverridenMethod tester = new TestExternalOverridenMethod();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         javadoc("-d", "out",
50                 "-source","8",
51                 "-sourcepath", testSrc,
52                 "-linkoffline", uri, testSrc,
53                 "pkg");
54         checkExit(Exit.OK);
55 
56         checkOutput("pkg/XReader.html", true,
57                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
58                 + "<dd><code><a href=\"" + uri + "/java/io/FilterReader.html?is-external=true#read()\" "
59                 + "title=\"class or interface in java.io\" class=\"externalLink\">read</a></code>&nbsp;in class&nbsp;<code>"
60                 + "<a href=\"" + uri + "/java/io/FilterReader.html?is-external=true\" "
61                 + "title=\"class or interface in java.io\" class=\"externalLink\">FilterReader</a></code></dd>",
62                 "<dt><span class=\"overrideSpecifyLabel\">Specified by:</span></dt>\n"
63                 + "<dd><code><a href=\"" + uri + "/java/io/DataInput.html?is-external=true#readInt()\" "
64                 + "title=\"class or interface in java.io\" class=\"externalLink\">readInt</a></code>&nbsp;in interface&nbsp;<code>"
65                 + "<a href=\"" + uri + "/java/io/DataInput.html?is-external=true\" "
66                 + "title=\"class or interface in java.io\" class=\"externalLink\">DataInput</a></code></dd>"
67         );
68     }
69 }
