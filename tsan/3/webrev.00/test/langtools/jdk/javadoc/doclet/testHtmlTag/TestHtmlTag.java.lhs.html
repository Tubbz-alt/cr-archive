 24 /*
 25  * @test
 26  * @bug 6786682 4649116 8182765
 27  * @summary This test verifies the use of lang attribute by <HTML>.
 28  * @author Bhavesh Patel
 29  * @library ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build javadoc.tester.*
 32  * @run main TestHtmlTag
 33  */
 34 
 35 import java.util.Locale;
 36 
 37 import javadoc.tester.JavadocTester;
 38 
 39 public class TestHtmlTag extends JavadocTester {
 40     private static final String defaultLanguage = Locale.getDefault().getLanguage();
 41     public static void main(String... args) throws Exception {
 42         TestHtmlTag tester = new TestHtmlTag();
 43         tester.runTests();
 44     }
 45     @Test
 46     public void test_default() {
 47         javadoc("-locale", defaultLanguage,
 48                 "-d", "out-default",
 49                 "-sourcepath", testSrc,
 50                 "pkg1");
 51 
 52         checkExit(Exit.OK);
 53 
 54         checkOutput("pkg1/C1.html", true,
 55             "<html lang=\"" + defaultLanguage + "\">");
 56 
 57         checkOutput("pkg1/package-summary.html", true,
 58             "<html lang=\"" + defaultLanguage + "\">");
 59 
 60         checkOutput("pkg1/C1.html", false,
 61                 "<html>");
 62     }
 63 
 64     @Test
 65     public void test_ja() {
 66         // TODO: why does this test need/use pkg2; why can't it use pkg1
 67         // like the other two tests, so that we can share the check methods?
 68         javadoc("-locale", "ja",
 69                 "-d", "out-ja",
 70                 "-sourcepath", testSrc,
 71                 "pkg2");
 72         checkExit(Exit.OK);
 73 
 74         checkOutput("pkg2/C2.html", true,
 75                 "<html lang=\"ja\">");
 76 
 77         checkOutput("pkg2/package-summary.html", true,
 78                 "<html lang=\"ja\">");
 79 
 80         checkOutput("pkg2/C2.html", false,
 81                 "<html>");
 82     }
