24 /*
25  * @test
26  * @bug 4232882 8014636
27  * @summary Javadoc strips all of the leading spaces when the comment
28  *    does not begin with a star.  This RFE allows users to
29  *    begin their comment without a leading star without leading
30  *    spaces stripped
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.*
34  * @run main LeadingSpaces
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class LeadingSpaces extends JavadocTester {
40     /**
41      * The entry point of the test.
42      * @param args the array of command line arguments.
43      * @throws Exception if the test fails
44      */
45     public static void main(String... args) throws Exception {
46         LeadingSpaces tester = new LeadingSpaces();
47         tester.runTests();
48     }
49 
50     @Test
51     public void testLeadingSpaces() {
52         javadoc("-d", "out", "-sourcepath", testSrc,
53                 testSrc("LeadingSpaces.java"));
54         checkExit(Exit.OK);
55         checkOutput("LeadingSpaces.html", true,
56                   "        1\n"
57                 + "          2\n"
58                 + "            3\n"
59                 + "              4\n"
60                 + "                5\n"
61                 + "                  6\n"
62                 + "                    7");
63     }
64 
65     /**
66        This leading spaces in the &lt;pre&gt; block below should be
67        preserved.
68        <pre>
69         1
70           2
71             3
72               4
73                 5
74                   6
75                     7
76        </pre>
77      */
78     public void method(){}
79 
80 }
