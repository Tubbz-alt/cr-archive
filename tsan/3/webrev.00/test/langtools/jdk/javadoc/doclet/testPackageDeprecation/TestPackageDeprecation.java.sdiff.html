35 import javadoc.tester.JavadocTester;
36 
37 public class TestPackageDeprecation extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestPackageDeprecation tester = new TestPackageDeprecation();
41         tester.runTests();
42     }
43 
44     @Test
45     public void testDefault() {
46         javadoc("-d", "out-default",
47                 "-sourcepath", testSrc,
48                 "-use",
49                 "--frames",
50                 "pkg", "pkg1", testSrc("C2.java"), testSrc("FooDepr.java"));
51         checkExit(Exit.OK);
52 
53         checkOutput("pkg1/package-summary.html", true,
54                 "<div class=\"deprecationBlock\"><span class=\"deprecatedLabel\">Deprecated.</span>\n"
55                 + "<div class=\"deprecationComment\">This package is Deprecated.</div>"
56         );
57 
58         checkOutput("deprecated-list.html", true,
59             "<li><a href=\"#package\">Packages</a></li>"
60         );
61     }
62 
63     @Test
64     public void testNoDeprecated() {
65         javadoc("-d", "out-nodepr",
66                 "-sourcepath", testSrc,
67                 "-use",
68                 "-nodeprecated",
69                 "--frames",
70                 "pkg", "pkg1", testSrc("C2.java"), testSrc("FooDepr.java"));
71         checkExit(Exit.OK);
72 
73         checkOutput("overview-summary.html", false,
74                 "pkg1");
75         checkOutput("allclasses-frame.html", false,
76                 "FooDepr");
77         checkOutput("class-use/C2.ModalExclusionType.html", true,
78                 "<th class=\"colFirst\" scope=\"row\"><a href=\"#unnamed.package\">&lt;Unnamed&gt;</a></th>");
79 
80         checkFiles(false,
81                 "pkg1/package-summary.html",
82                 "FooDepr.html");
83     }
84 }
