 36 import javadoc.tester.JavadocTester;
 37 
 38 public class TestUseOption extends JavadocTester {
 39 
 40     public static void main(String... args) throws Exception {
 41         TestUseOption tester = new TestUseOption();
 42         tester.runTests();
 43     }
 44 
 45     @Test
 46     public void test1() {
 47         javadoc("-d", "out-1",
 48                 "-sourcepath", testSrc,
 49                 "-use",
 50                 "--frames",
 51                 "pkg1", "pkg2");
 52         checkExit(Exit.OK);
 53 
 54         // Eight tests for class use.
 55         for (int i = 1; i <= 8; i++) {
 56             checkOutput("pkg1/class-use/C1.html", true,
 57                     "Test " + i + " passes");
 58         }
 59 
 60         // Three more tests for package use.
 61         for (int i = 1; i <= 3; i++) {
 62             checkOutput("pkg1/package-use.html", true,
 63                     "Test " + i + " passes");
 64         }
 65 
 66         checkOrder("pkg1/class-use/UsedClass.html",
 67                 "Field in C1.",
 68                 "Field in C2.",
 69                 "Field in C4.",
 70                 "Field in C5.",
 71                 "Field in C6.",
 72                 "Field in C7.",
 73                 "Field in C8.",
 74                 "Method in C1.",
 75                 "Method in C2.",
 76                 "Method in C4.",
 77                 "Method in C5.",
 78                 "Method in C6.",
 79                 "Method in C7.",
 80                 "Method in C8."
 81         );
 82 
 83         checkOutput("pkg2/class-use/C3.html", true,
 84                 "<a href=\"../../index.html?pkg2/class-use/C3.html\" target=\"_top\">"
 85                 + "Frames</a></li>"
 86         );
 87         checkOutput("pkg1/class-use/UsedClass.html", true,
 88           "that return types with arguments of type"
 89         );
 90         checkOutput("pkg1/class-use/UsedClass.html", true,
 91           "<a href=\"../C1.html#methodInC1ReturningType()\">methodInC1ReturningType</a>"
 92         );
 93         checkOutput("pkg1/class-use/UsedInterface.html", true,
 94           "Classes in <a href=\"../package-summary.html\">pkg1</a> that implement " +
 95           "<a href=\"../UsedInterface.html\" title=\"interface in pkg1\">UsedInterface</a>"
 96         );
 97         checkOutput("pkg1/class-use/UsedInterfaceA.html", true,
 98           "Classes in <a href=\"../package-summary.html\">pkg1</a> that implement " +
 99           "<a href=\"../UsedInterfaceA.html\" title=\"interface in pkg1\">UsedInterfaceA</a>"
100         );
101         checkOutput("pkg1/class-use/UsedClass.html", false,
102            "methodInC1Protected"
103         );
104         checkOutput("pkg1/class-use/UsedInterface.html", true,
105            "<a href=\"../AnAbstract.html\" title=\"class in pkg1\">AnAbstract</a>"
106         );
107         checkOutput("pkg1/class-use/UsedInterface.html", true,
108             "../C10.html#withReturningTypeParameters()"
109         );
110         checkOutput("pkg1/class-use/UsedInterface.html", true,
111             "../C10.html#withTypeParametersOfType(java.lang.Class)"
112         );
113         checkOutput("pkg1/class-use/UsedInterface.html", true,
114             "\"../package-summary.html\">pkg1</a> that return " +
115             "<a href=\"../UsedInterface.html\" title=\"interface in pkg1\""
116         );
117         checkOutput("pkg1/class-use/UsedInterface.html", true,
118             "<a href=\"../C10.html#addAll(pkg1.UsedInterface...)\">addAll</a>"
119         );
120         checkOutput("pkg1/class-use/UsedInterface.html", true,
121             "<a href=\"../C10.html#create(pkg1.UsedInterfaceA,pkg1." +
122             "UsedInterface,java.lang.String)\">"
123         );
124         checkOutput("pkg1/class-use/UsedInterface.html", true,
125             "<a href=\"../C10.html#withTypeParametersOfType(java.lang.Class)\">" +
126             "withTypeParametersOfType</a>"
127         );
128         checkOutput("pkg1/class-use/UsedInterface.html", true,
129             "Subinterfaces of <a href=\"../UsedInterface.html\" title=\"interface in pkg1\">"
130             + "UsedInterface</a> in <a href=\"../package-summary.html\">pkg1",
131             "<td class=\"colFirst\"><code>interface&nbsp;</code></td>\n<th class=\"colSecond\" scope=\"row\">"
132             + "<code><span class=\"memberNameLink\"><a href=\"../SubInterface.html\" "
133             + "title=\"interface in pkg1\">SubInterface</a>&lt;T&gt;</span></code></th>"
134         );
135         checkOutput("pkg1/class-use/UsedThrowable.html", true,
136             "Methods in <a href=\"../package-summary.html\">pkg1</a> that throw "
137             + "<a href=\"../UsedThrowable.html\" title=\"class in pkg1\">UsedThrowable</a>",
138             "<td class=\"colFirst\"><code>void</code></td>\n<th class=\"colSecond\" scope=\"row\"><span class="
139             + "\"typeNameLabel\">C1.</span><code><span class=\"memberNameLink\">"
140             + "<a href=\"../C1.html#methodInC1ThrowsThrowable()\">methodInC1ThrowsThrowable"
141             + "</a></span>()</code></th>"
142         );
143     }

145     @Test
146     public void test2() {
147         javadoc("-d", "out-2",
148                 "-sourcepath", testSrc,
149                 "-use",
150                 testSrc("C.java"), testSrc("UsedInC.java"), "pkg3");
151         checkExit(Exit.OK);
152 
153         checkOutput("class-use/UsedInC.html", true,
154                 "Uses of <a href=\"../UsedInC.html\" title=\"class in &lt;Unnamed&gt;\">"
155                 + "UsedInC</a> in <a href=\"../package-summary.html\">&lt;Unnamed&gt;</a>"
156         );
157         checkOutput("class-use/UsedInC.html", true,
158                 "<li class=\"blockList\">\n"
159                 + "<section role=\"region\"><a id=\"unnamed.package\">"
160         );
161         checkOutput("package-use.html", true,
162                 "<th class=\"colFirst\" scope=\"row\">"
163                 + "<a href=\"class-use/UsedInC.html#unnamed.package\">UsedInC</a></th>",
164                 "<th class=\"colFirst\" scope=\"row\"><a href=\"#unnamed.package\">&lt;Unnamed&gt;</a></th>\n"
165                 + "<td class=\"colLast\">&nbsp;</td>"
166         );
167     }
