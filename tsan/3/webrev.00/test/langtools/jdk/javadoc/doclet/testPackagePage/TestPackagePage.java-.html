 24 /*
 25  * @test
 26  * @bug 4492643 4689286 8196201 8184205
 27  * @summary Test that a package page is properly generated when a .java file
 28  * passed to Javadoc.  Also test that the proper package links are generated
 29  * when single or multiple packages are documented.
 30  * @author jamieh
 31  * @library ../../lib
 32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 33  * @build javadoc.tester.*
 34  * @run main TestPackagePage
 35  */
 36 
 37 import javadoc.tester.JavadocTester;
 38 
 39 public class TestPackagePage extends JavadocTester {
 40 
 41     public static void main(String... args) throws Exception {
 42         TestPackagePage tester = new TestPackagePage();
 43         tester.runTests();
 44     }
 45 
 46     @Test
 47     public void testSinglePackage() {
 48         javadoc("-d", "out-1",
 49                 "-sourcepath", testSrc,
 50                 testSrc("com/pkg/C.java"));
 51         checkExit(Exit.OK);
 52 
 53         checkOutput("com/pkg/package-summary.html", true,
 54             "This is a package page.");
 55 
 56         // With just one package, all general pages link to the single package page.
 57         checkOutput("com/pkg/C.html", true,
 58             "<a href=\"package-summary.html\">Package</a>");
 59         checkOutput("com/pkg/package-tree.html", true,
 60             "<li><a href=\"package-summary.html\">Package</a></li>");
 61         checkOutput("deprecated-list.html", true,
 62             "<li><a href=\"com/pkg/package-summary.html\">Package</a></li>");
 63         checkOutput("index-all.html", true,
 64             "<li><a href=\"com/pkg/package-summary.html\">Package</a></li>");
 65         checkOutput("help-doc.html", true,
 66             "<li><a href=\"com/pkg/package-summary.html\">Package</a></li>");
 67     }
 68 
 69     private static final String[][] TEST1 = {
 70     };
 71 
 72 
 73     @Test
 74     public void testMultiplePackages() {
 75         javadoc("-d", "out-2",
 76                 "-sourcepath", testSrc,
 77                 "com.pkg", "pkg2");
 78         checkExit(Exit.OK);
 79 
 80         //With multiple packages, there is no package link in general pages.
 81         checkOutput("deprecated-list.html", true,
 82             "<li>Package</li>");
 83         checkOutput("index-all.html", true,
 84             "<li>Package</li>");
 85         checkOutput("help-doc.html", true,
 86             "<li>Package</li>");
 87         checkOutput("allclasses-index.html", true,
 88                 "<div class=\"typeSummary\">\n<table>\n"
 89                 + "<caption><span>Class Summary</span><span class=\"tabEnd\">&nbsp;</span></caption>\n"
 90                 + "<tr>\n"
 91                 + "<th class=\"colFirst\" scope=\"col\">Class</th>\n"
 92                 + "<th class=\"colLast\" scope=\"col\">Description</th>\n"
 93                 + "</tr>\n");
 94         checkOutput("allpackages-index.html", true,
 95                 "<div class=\"packagesSummary\">\n<table>\n"
 96                 + "<caption><span>Package Summary</span><span class=\"tabEnd\">&nbsp;</span></caption>\n"
 97                 + "<tr>\n"
 98                 + "<th class=\"colFirst\" scope=\"col\">Package</th>\n"
 99                 + "<th class=\"colLast\" scope=\"col\">Description</th>\n"
100                 + "</tr>\n");
101         checkOutput("type-search-index.js", true,
102                 "{\"l\":\"All Classes\",\"url\":\"allclasses-index.html\"}");
103         checkOutput("package-search-index.js", true,
104                 "{\"l\":\"All Packages\",\"url\":\"allpackages-index.html\"}");
105         checkOutput("index-all.html", true,
106                 "<br><a href=\"allclasses-index.html\">All&nbsp;Classes</a>&nbsp;"
107                 + "<a href=\"allpackages-index.html\">All&nbsp;Packages</a>");
108     }
109 }
