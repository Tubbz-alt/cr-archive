 24 /*
 25  * @test
 26  * @bug 6553182 8025416 8029504
 27  * @summary This test verifies the -Xdocrootparent option.

 28  * @library ../../lib
 29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 30  * @build javadoc.tester.*
 31  * @run main TestDocRootLink
 32  */
 33 import javadoc.tester.JavadocTester;
 34 
 35 public class TestDocRootLink extends JavadocTester {
 36 
 37     public static void main(String... args) throws Exception {
 38         TestDocRootLink tester = new TestDocRootLink();
 39 
 40         // The test files intentionally contain examples of links that should
 41         // or should not be affected by the -Xdocrootparent option, and the
 42         // results are checked explicitly; so, disable the automatic link
 43         // checker to prevent spurious "missing files" errors from some of
 44         // these links.
 45         tester.setAutomaticCheckLinks(false);
 46 
 47         tester.runTests();
 48     }
 49 
 50     @Test
 51     public void test1() {
 52         javadoc("-d", "out-1",
 53                 "-sourcepath", testSrc,
 54                 "pkg1", "pkg2");
 55         checkExit(Exit.OK);
 56 
 57         checkOutput("pkg1/C1.html", true,
 58             "Refer <a href=\"../../technotes/guides/index.html\">Here</a>",
 59             "This <a href=\"../pkg2/C2.html\">Here</a> should not be replaced\n" +
 60             " with an absolute link.",
 61             "Testing <a href=\"../technotes/guides/index.html\">Link 1</a> and\n" +
 62             " <a href=\"../pkg2/C2.html\">Link 2</a>.");
 63 
 64         checkOutput("pkg1/package-summary.html", true,
 65             "<a href=\"../../technotes/guides/index.html\">\n" +
 66             "            Test document 1</a>",
 67             "<a href=\"../pkg2/C2.html\">\n" +
 68             "            Another Test document 1</a>",
 69             "<a href=\"../technotes/guides/index.html\">\n" +
 70             "            Another Test document 2.</a>");
 71 
 72         // TODO: should this check *any* reference to http://download.oracle.com/
 73         checkOutput("pkg1/C1.html", false,
 74             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">",
 75             "<a href=\"http://download.oracle.com/javase/7/docs/pkg2/C2.html\">");
 76 
 77         checkOutput("pkg1/package-summary.html", false,
 78             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">",
 79             "<a href=\"http://download.oracle.com/javase/7/docs/pkg2/C2.html\">");
 80     }
 81 
 82     @Test
 83     public void test2() {
 84         javadoc("-d", "out-2",
 85                 "-Xdocrootparent", "http://download.oracle.com/javase/7/docs",
 86                 "-sourcepath", testSrc,
 87                 "pkg1", "pkg2");
 88         checkExit(Exit.OK);
 89 
 90         checkOutput("pkg2/C2.html", true,
 91             "Refer <a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">Here</a>",
 92             "This <a href=\"../pkg1/C1.html\">Here</a> should not be replaced\n" +
 93             " with an absolute link.",
 94             "Testing <a href=\"../technotes/guides/index.html\">Link 1</a> and\n" +
 95             " <a href=\"../pkg1/C1.html\">Link 2</a>.");
 96 
 97         checkOutput("pkg2/package-summary.html", true,
 98             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">\n" +
 99             "            Test document 1</a>",
100             "<a href=\"../pkg1/C1.html\">\n" +
101             "            Another Test document 1</a>",
102             "<a href=\"../technotes/guides/index.html\">\n" +
103             "            Another Test document 2.</a>");
104 
105         checkOutput("pkg2/C2.html", false,
106             "<a href=\"../../technotes/guides/index.html\">",
107             "<a href=\"http://download.oracle.com/javase/7/docs/pkg1/C1.html\">");
108 
109         checkOutput("pkg2/package-summary.html", false,
110             "<a href=\"../../technotes/guides/index.html\">",
111             "<a href=\"http://download.oracle.com/javase/7/docs/pkg1/C1.html\">");
112     }
113 }
