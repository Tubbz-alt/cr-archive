 39 public class TestHtmlTag extends JavadocTester {
 40     private static final String defaultLanguage = Locale.getDefault().getLanguage();

 42         TestHtmlTag tester = new TestHtmlTag();
 43         tester.runTests();
 44     }
 45     @Test
 46     public void test_default() {
 47         javadoc("-locale", defaultLanguage,
 48                 "-d", "out-default",

 95         checkOutput("pkg1/package-summary.html", true,
 96                 "<html lang=\"en\">");
 97 
 98         checkOutput("pkg1/C1.html", false,
 99                 "<html>");
100     }
