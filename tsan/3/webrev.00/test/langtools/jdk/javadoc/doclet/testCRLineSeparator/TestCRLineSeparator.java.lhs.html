25  * @test
26  * @bug      4979486
27  * @summary  Make sure tool parses CR line separators properly.
28  * @author   jamieh
29  * @library  ../../lib

32  * @run main TestCRLineSeparator

34 
35 import java.io.*;
36 import java.util.*;
37 
38 import javadoc.tester.JavadocTester;
39 
40 public class TestCRLineSeparator extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         TestCRLineSeparator tester = new TestCRLineSeparator();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() throws IOException {
49         initFiles(new File(testSrc), new File("src"), "pkg");
50         javadoc("-d", "out",
51                 "-sourcepath", "src",
52                 "pkg");
53         checkExit(Exit.OK);
54 
55         checkOutput("pkg/MyClass.html", true,
56                 "Line 1\n"
57                 + " Line 2");
58     }
59 
60     // recursively copy files from fromDir to toDir, replacing newlines
61     // with \r
62     void initFiles(File fromDir, File toDir, String f) throws IOException {
63         File from_f = new File(fromDir, f);
64         File to_f = new File(toDir, f);
65         if (from_f.isDirectory()) {
66             to_f.mkdirs();
67             for (String child: from_f.list()) {
68                 initFiles(from_f, to_f, child);
69             }
70         } else {
71             List<String> lines = new ArrayList<>();
72             try (BufferedReader in = new BufferedReader(new FileReader(from_f))) {
73                 String line;
74                 while ((line = in.readLine()) != null)
75                     lines.add(line);
76             }
77             try (BufferedWriter out = new BufferedWriter(new FileWriter(to_f))) {
78                 for (String line: lines) {
79                     out.write(line);
80                     out.write("\r");
81                 }
82             }
83         }
84     }
85 }
