 24 /*
 25  * @test
 26  * @bug      4460354 8014636 8043186 8195805 8182765 8196202
 27  * @summary  Test to make sure that relative paths are redirected in the
 28  *           output so that they are not broken.
 29  * @author   jamieh
 30  * @library  ../../lib
 31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build    javadoc.tester.*
 33  * @run main TestRelativeLinks
 34  */
 35 
 36 import java.io.File;
 37 import java.io.FileOutputStream;
 38 import java.io.IOException;
 39 
 40 import javadoc.tester.JavadocTester;
 41 
 42 public class TestRelativeLinks extends JavadocTester {
 43 
 44     public static void main(String... args) throws Exception {
 45         TestRelativeLinks tester = new TestRelativeLinks();
 46         tester.runTests();
 47     }
 48 
 49     @Test
 50     public void test() {
 51         javadoc("-d", "out",
 52                 "-use",
 53                 "--frames",
 54                 "-sourcepath", testSrc,
 55                 "pkg", "pkg2");
 56         checkExit(Exit.ERROR);
 57 
 58         checkOutput(Output.OUT, true,
 59                 "attribute not supported in HTML5: name");
 60 
 61         // These relative paths should stay relative because they appear
 62         // in the right places.
 63         checkOutput("pkg/C.html", true,
 64             "<a href=\"relative-class-link.html\">relative class link</a>",
 65             "<a href=\"relative-field-link.html\">relative field link</a>",
 66             "<a href=\"relative-method-link.html\">relative method link</a>",
 67             " <a href=\"relative-multi-line-link.html\">relative-multi-line-link</a>.");
 68         checkOutput("pkg/package-summary.html", true,
 69             "<a href=\"relative-package-link.html\">relative package link</a>");
 70 
 71         // These relative paths should be redirected because they are in different
 72         // places.
 73 
 74         // INDEX PAGE
 75         checkOutput("index-all.html", true,
 76             "<a href=\"./pkg/relative-class-link.html\">relative class link</a>",
 77             "<a href=\"./pkg/relative-field-link.html\">relative field link</a>",
 78             "<a href=\"./pkg/relative-method-link.html\">relative method link</a>",
 79             "<a href=\"./pkg/relative-package-link.html\">relative package link</a>",
 80             " <a href=\"./pkg/relative-multi-line-link.html\">relative-multi-line-link</a>.");
 81 
 82         // This is not a relative path and should not be redirected.
 83         checkOutput("index-all.html", true,
 84             "<div class=\"block\"><a name=\"masters\"></a>");
 85         checkOutput("index-all.html", false,
 86             "<div class=\"block\"><a name=\"./pkg/masters\"></a>");
 87 
 88         // PACKAGE USE
 89         checkOutput("pkg/package-use.html", true,
 90             "<a href=\"../pkg/relative-package-link.html\">relative package link</a>.",
 91             "<a href=\"../pkg/relative-class-link.html\">relative class link</a>");
 92 
 93         // CLASS_USE
 94         checkOutput("pkg/class-use/C.html", true,
 95             "<a href=\"../../pkg/relative-field-link.html\">relative field link</a>",
 96             "<a href=\"../../pkg/relative-method-link.html\">relative method link</a>",
 97             "<a href=\"../../pkg/relative-package-link.html\">relative package link</a>",
 98             " <a href=\"../../pkg/relative-multi-line-link.html\">relative-multi-line-link</a>.");
 99 
100         // PACKAGE OVERVIEW
101         checkOutput("overview-summary.html", true,
102             "<a href=\"./pkg/relative-package-link.html\">relative package link</a>");
103     }
104 
105     private void touch(String file) {
106         File f = new File(outputDir, file);
107         out.println("touch " + f);
108         try (FileOutputStream fos = new FileOutputStream(f)) {
109         } catch (IOException e) {
110             checking("Touch file");
111             failed("Error creating file: " + e);
112         }
113     }
114 }
