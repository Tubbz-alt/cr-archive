27  * @summary Javadoc wrongly inserts </DD> tags when using multiple @author tags
28  * @author dkramer

30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main AuthorDD

34 
35 /**
36  * Runs javadoc and runs regression tests on the resulting HTML.
37  */
38 import javadoc.tester.JavadocTester;
39 
40 public class AuthorDD extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         AuthorDD tester = new AuthorDD();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         // Test for all cases except the split index page
50         javadoc("-d", "out",
51                 "-author",
52                 "-version",
53                 "-sourcepath", testSrc,
54                 "p1");
55         checkExit(Exit.OK);
56 
57         checkOutput("p1/C1.html", true,
58                 // Test single @since tag:
59                 "<dt><span class=\"simpleTagLabel\">Since:</span></dt>\n"
60                 + "<dd>JDK 1.0</dd>",
61                 // Test multiple @author tags:
62                 "<dt><span class=\"simpleTagLabel\">Author:</span></dt>\n"
63                 + "<dd>Doug Kramer, Jamie, Neal</dd>");
64     }
65 }
