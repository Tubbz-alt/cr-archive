24 package examples;
25 
26 import java.io.Serializable;
27 
28 /**
29  * A class with a record for a serialization proxy.
30  */
31 public class SerializableProxy implements Serializable {
32     private int p;
33     private int q;
34 
35     /**
36      * The proxy.
37      *
38      * @param x the first item in the serialized form
39      * @param y the second item in the serialized form
40      *
41      * @serial include
42      */
43     private record Proxy(int x, int y) {
44         /**
45          * Returns a deserialized object.
46          *
47          * @returns the deserialized object
48          */
49         public SerializableProxy readResolve() {
50             return new SerializableProxy(x, y);
51         }
52     }
53 
54     /**
55      * Returns the proxy object.
56      *
57      * @return the proxy object
58      */
59     public Object writeReplace() {
60         return new Proxy(p, q);
61     }
62 }
