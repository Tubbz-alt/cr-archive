24 /*
25  * @test
26  * @bug      8170825 8196027
27  * @summary  Perform tests on index files generated by javadoc.

28  * @library  ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestIndexFiles
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestIndexFiles extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestIndexFiles tester = new TestIndexFiles();
40         tester.runTests();
41     }
42 
43     @Test
44     public void testIndexFiles() {
45         javadoc("-d", "out", "-splitindex", "-Xdoclint:none", "-sourcepath", testSrc,
46                 "-use", "pkg");
47         checkExit(Exit.OK);
48         checkIndexFiles(false);
49     }
50 
51     void checkIndexFiles(boolean found) {
52         checkOutput("index-files/index-1.html", found,
53                 "<li>Prev Letter</li>\n"
54                 + "<li><a href=\"index-2.html\">Next Letter</a></li>");
55         checkOutput("index-files/index-5.html", found,
56                 "<li><a href=\"index-4.html\">Prev Letter</a></li>\n"
57                 + "<li>Next Letter</li>");
58         checkOutput("index-files/index-1.html", found,
59                 "<li><a href=\"index-0.html\">Prev Letter</a></li>\n"
60                 + "<li><a href=\"index-1.html\">Next Letter</a></li>");
61         checkOutput("index-files/index-5.html", found,
62                 "<li><a href=\"index-4.html\">Prev Letter</a></li>\n"
63                 + "<li><a href=\"index-5.html\">Next Letter</a></li>");
64     }
65 }
