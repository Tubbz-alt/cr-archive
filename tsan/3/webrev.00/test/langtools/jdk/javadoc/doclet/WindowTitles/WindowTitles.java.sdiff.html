35 /**
36  * Runs javadoc and runs regression tests on the resulting HTML.
37  */
38 import javadoc.tester.JavadocTester;
39 
40 public class WindowTitles extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         WindowTitles tester = new WindowTitles();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         // Test for all cases except the split index page
50         javadoc("-d", "out-1",
51                 "-use",
52                 "--frames",
53                 "-sourcepath", testSrc,
54                 "p1", "p2");
55         checkExit(Exit.OK);
56         checkFiles(false, "allclasses-noframe.html");
57 
58         checkTitle("overview-summary.html",     "Overview");
59         checkTitle("overview-tree.html",        "Class Hierarchy");
60         checkTitle("overview-frame.html",       "Overview List");
61         checkTitle("p1/package-summary.html",   "p1");
62         checkTitle("p1/package-frame.html",     "p1");
63         checkTitle("p1/package-tree.html",      "p1 Class Hierarchy");
64         checkTitle("p1/package-use.html",       "Uses of Package p1");
65         checkTitle("p1/C1.html",                "C1");
66         checkTitle("allclasses-frame.html",     "All Classes");
67         checkTitle("constant-values.html",      "Constant Field Values");
68         checkTitle("deprecated-list.html",      "Deprecated List");
69         checkTitle("serialized-form.html",      "Serialized Form");
70         checkTitle("help-doc.html",             "API Help");
71         checkTitle("index-all.html",            "Index");
72         checkTitle("p1/class-use/C1.html",      "Uses of Class p1.C1");
73     }
74 
75     @Test
76     public void test2() {
77         // Test only for the split-index case (and run on only one package)
78         javadoc("-d", "out-2",
79                 "-splitindex",
80                 "-sourcepath", testSrc,
81                 "p1");

83
