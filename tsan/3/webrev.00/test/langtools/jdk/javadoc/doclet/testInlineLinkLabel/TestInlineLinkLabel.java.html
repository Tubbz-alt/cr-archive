24 /*
25  * @test
26  * @bug 4524136
27  * @summary Test to make sure label is used for inline links.
28  * @library ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.*
31  * @run main TestInlineLinkLabel
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestInlineLinkLabel extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestInlineLinkLabel tester = new TestInlineLinkLabel();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test() {
45         javadoc("-d", "out",
46                 "-sourcepath", testSrc,
47                 "pkg");
48         checkExit(Exit.OK);
49 
50         checkOutput("pkg/C1.html", true,
51                 //Search for the label to the package link.
52                 "<a href=\"package-summary.html\"><code>Here is a link to a package</code></a>",
53                 //Search for the label to the class link
54                 "<a href=\"C2.html\" title=\"class in pkg\"><code>Here is a link to a class</code></a>");
55     }
56 }
