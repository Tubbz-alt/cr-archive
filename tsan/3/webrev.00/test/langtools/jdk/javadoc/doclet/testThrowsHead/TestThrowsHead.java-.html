24 /*
25  * @test
26  * @bug 4530727 8026567
27  * @summary When an exception is declared in the method signature but
28  * not documented with a throws tag, we generate a link to it in the
29  * throws section.  Make sure that the link is below a Throws heading.
30  * @author jamieh
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.*
34  * @run main TestThrowsHead
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestThrowsHead extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestThrowsHead tester = new TestThrowsHead();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         javadoc("-d", "out",
49                 testSrc("C.java"));
50         checkExit(Exit.OK);
51 
52         checkOutput("C.html", true,
53                 "<dt><span class=\"throwsLabel\">Throws:</span>");
54     }
55 }
