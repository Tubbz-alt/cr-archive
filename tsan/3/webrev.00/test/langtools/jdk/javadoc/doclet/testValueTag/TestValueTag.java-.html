 36 import java.io.IOException;
 37 import java.nio.file.Files;
 38 import java.nio.file.Path;
 39 import java.nio.file.Paths;
 40 import java.util.List;
 41 
 42 import javadoc.tester.JavadocTester;
 43 
 44 public class TestValueTag extends JavadocTester {
 45 
 46     public static void main(String... args) throws Exception {
 47         TestValueTag tester = new TestValueTag();
 48         tester.runTests();
 49     }
 50 
 51     @Test
 52     public void test1() {
 53         javadoc("-d", "out1",
 54                 "-sourcepath", testSrc,
 55                 "-tag", "todo",
 56                 "pkg1", "pkg2");
 57         checkExit(Exit.ERROR);
 58 
 59         checkOutput("pkg1/Class1.html", true,
 60                 // Base case:  using @value on a constant.
 61                 "Result:  \"Test 1 passes\"",
 62                 // Retrieve value of constant in same class.
 63                 "Result:  <a href=\"#TEST_2_PASSES\">\"Test 2 passes\"</a>",
 64                 "Result:  <a href=\"#TEST_3_PASSES\">\"Test 3 passes\"</a>",
 65                 "Result:  <a href=\"#TEST_4_PASSES\">\"Test 4 passes\"</a>",
 66                 "Result:  <a href=\"#TEST_5_PASSES\">\"Test 5 passes\"</a>",
 67                 "Result:  <a href=\"#TEST_6_PASSES\">\"Test 6 passes\"</a>");
 68 
 69         checkOutput("pkg1/Class2.html", true,
 70                 // Retrieve value of constant in different class.
 71                 "Result:  <a href=\"Class1.html#TEST_7_PASSES\">\"Test 7 passes\"</a>",
 72                 "Result:  <a href=\"Class1.html#TEST_8_PASSES\">\"Test 8 passes\"</a>",
 73                 "Result:  <a href=\"Class1.html#TEST_9_PASSES\">\"Test 9 passes\"</a>",
 74                 "Result:  <a href=\"Class1.html#TEST_10_PASSES\">\"Test 10 passes\"</a>",
 75                 "Result:  <a href=\"Class1.html#TEST_11_PASSES\">\"Test 11 passes\"</a>",
 76                 // Retrieve value of constant in different package
 77                 "Result:  <a href=\"../pkg2/Class3.html#TEST_12_PASSES\">\"Test 12 passes\"</a>",
 78                 "Result:  <a href=\"../pkg2/Class3.html#TEST_13_PASSES\">\"Test 13 passes\"</a>",
 79                 "Result:  <a href=\"../pkg2/Class3.html#TEST_14_PASSES\">\"Test 14 passes\"</a>",
 80                 "Result:  <a href=\"../pkg2/Class3.html#TEST_15_PASSES\">\"Test 15 passes\"</a>",
 81                 "Result:  <a href=\"../pkg2/Class3.html#TEST_16_PASSES\">\"Test 16 passes\"</a>");
 82 
 83         checkOutput("pkg2/package-summary.html", true,
 84                 // Retrieve value of constant from a package page
 85                 "Result: <a href=\"Class3.html#TEST_17_PASSES\">\"Test 17 passes\"</a>");
 86 
 87         checkOutput("pkg1/CustomTagUsage.html", true,
 88                 // Test @value tag used with custom tag.
 89                 "<dt><span class=\"simpleTagLabel\">Todo:</span></dt>\n" +
 90                 "<dd>the value of this constant is 55.</dd>");
 91 
 92         checkOutput(Output.OUT, true,
 93                 // Test @value errors printed due to invalid use or when used with
 94                 // non-constant or with bad references.
 95                 "error: value does not refer to a constant\n"
 96                 + "     * Result:  {@value TEST_12_ERROR}",
 97                 "error: {@value} not allowed here\n"
 98                 + "     * Result:  {@value}",
 99                 "error: value does not refer to a constant\n"
100                 + "     * Result:  {@value NULL}",
101                 "error: {@value} not allowed here\n"
102                 + "     * Invalid (null): {@value}",
103                 "error: {@value} not allowed here\n"
104                 + "     * Invalid (non-constant field): {@value}",
105                 "error: value does not refer to a constant\n"
106                 + "     * Here is a bad value reference: {@value UnknownClass#unknownConstant}",
107                 "error: reference not found\n"
108                 + "     * Here is a bad value reference: {@value UnknownClass#unknownConstant}",
109                 "error: {@value} not allowed here\n"
110                 + "     * @todo the value of this constant is {@value}"
111         );
112 
113         checkOutput("pkg1/Class1.html", false,
114                 // Base case:  using @value on a constant.
115                 "Result:  <a href=\"#TEST_12_ERROR\">\"Test 12 "
116                 + "generates an error message\"</a>");
117 
118         checkForException();
119     }

121     @Test
122     public void test2() {
123         javadoc("-Xdoclint:none",
124                 "-d", "out2",
125                 "-sourcepath", testSrc,
126                 "-tag", "todo",
127                 "pkg1", "pkg2");
128         checkExit(Exit.OK);
129         checkOutput(Output.OUT, true,
130                 // Test @value warning printed when used with non-constant.
131                 "warning - @value tag (which references nonConstant) "
132                 + "can only be used in constants.",
133                 "warning - @value tag (which references NULL) "
134                 + "can only be used in constants.",
135                 "warning - @value tag (which references TEST_12_ERROR) "
136                 + "can only be used in constants.",
137                 // Test warning printed for bad reference.
138                 "warning - {@value UnknownClass#unknownConstant}"
139                 + " (referenced by @value tag) is an unknown reference."
140         );
141         checkForException();
142     }

144     @Test
145     public void test3() {
146         javadoc("-d", "out3",
147                 "-sourcepath", testSrc,
148                 "pkg2", "pkg3");
149         checkExit(Exit.OK);
150 
151         checkOrder("pkg3/RT.html",
152                 "The value is <a href=\"#CONSTANT\">\"constant\"</a>.",
153                 "The value1 is <a href=\"#CONSTANT\">\"constant\"</a>.",
154                 "The value2 is <a href=\"#CONSTANT\">\"constant\"</a>.",
155                 "The value3 is <a href=\"../pkg2/Class3.html#TEST_12_PASSES\">"
156                 + "\"Test 12 passes\"</a>.");
157         checkForException();
158     }
159 
160     @Test
161     public void test4() throws IOException {
162         Path base = Paths.get("test4");
163         Path src = base.resolve("src");
164         Files.createDirectories(src.resolve("p"));
165         Files.write(src.resolve("p").resolve("C.java"), List.of(
166                 "package p;",
167                 "/** This class defines specialChars: {@value C#specialChars}. */",
168                 "public class C {",
169                 "    /** The value is {@value}. */",
170                 "    public static final String specialChars = \"abc < def & ghi > jkl\";",
171                 "}"));
172 
173         javadoc("-d", base.resolve("out").toString(),
174                 "-sourcepath", src.toString(),
175                 "p");
176         checkExit(Exit.OK);
177         checkOutput("p/C.html", false,
178                 "The value is \"abc < def & ghi > jkl\".");
179         checkOutput("p/C.html", true,
180                 "The value is \"abc &lt; def &amp; ghi &gt; jkl\".");
181 
182         checkForException();
183     }
184 
185 
186     void checkForException() {
187         checkOutput(Output.STDERR, false, "DocletAbortException");
188         checkOutput(Output.STDERR, false, "IllegalArgumentException");
189     }
