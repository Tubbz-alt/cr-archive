24 /*
25  * @test
26  * @summary Determine if proper warning messages are printed.
27  * @author jamieh
28  * @library ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.*
31  * @build TestTagMisuse
32  * @run main TestTagMisuse
33  */
34 import javadoc.tester.JavadocTester;
35 
36 public class TestTagMisuse extends JavadocTester {
37 
38     /**
39      * The entry point of the test.
40      * @param args the array of command line arguments.
41      * @throws Exception if the test fails
42      */
43     public static void main(String... args) throws Exception {
44         TestTagMisuse tester = new TestTagMisuse();
45         tester.runTests();
46     }
47 
48     @Test
49     public void test() {
50         javadoc("-Xdoclint:none",
51                 "-d", "out",
52                 testSrc("TestTagMisuse.java"));
53         checkExit(Exit.OK);
54 
55         checkOutput(Output.OUT, true,
56                 "warning - Tag @param cannot be used in field documentation.",
57                 "warning - Tag @throws cannot be used in field documentation.",
58                 "warning - Tag @return cannot be used in constructor documentation."
59                 /* DCerroneous, "warning - Tag @throws cannot be used in inline documentation."*/);
60         checkOutput(Output.OUT, false, "DocletAbortException");
61     }
62 
63     /**
64      * {@throws blah}
65      * Here is a bad field tag:
66      * @throws foo
67      * @param foo.
68      */
69     public int field;
70 
71     /**
72      * Here is a bad constructor tag:
73      * @return blah
74      */
75     public TestTagMisuse(){}
76 
77     /**
78      * for @see and {@link}), and ThrowsTag (for @throws).
79      */
80     public void thisShouldNotCrash() {}
81 }
