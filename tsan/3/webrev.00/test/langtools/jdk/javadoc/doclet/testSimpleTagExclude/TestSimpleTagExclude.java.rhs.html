24 /*
25  * @test
26  * @bug 4628181
27  * @summary Test the parsing of the -tag option.  The user should be able to
28  * exclude a simple tag by using -tag tagname:X

29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestSimpleTagExclude
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestSimpleTagExclude extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestSimpleTagExclude tester = new TestSimpleTagExclude();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "-tag", "todo:X",
49                 testSrc("DummyClass.java"));
50         checkExit(Exit.OK);
51 
52         checkOutput("DummyClass.html", false,
53                 "todo");
54     }
55 }
