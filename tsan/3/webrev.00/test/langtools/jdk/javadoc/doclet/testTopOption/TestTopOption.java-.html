24 /*
25  * @test
26  * @bug      6227616 8043186 8196202
27  * @summary  Test the new -top option.
28  * @author   jamieh
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestTopOption
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestTopOption extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestTopOption tester = new TestTopOption();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-overview", testSrc("overview.html"),
47                 "-use",
48                 "-top", "TOP TEXT",
49                 "-d", "out-1",
50                 "--frames",
51                 "-sourcepath", testSrc,
52                 "pkg");
53         checkExit(Exit.OK);
54 
55         checkTopText(
56                 "pkg/AnnotationType.html",
57                 "pkg/class-use/AnnotationType.html",
58                 "pkg/Cl.html",
59                 "pkg/class-use/Cl.html",
60                 "pkg/package-summary.html",
61                 "pkg/package-use.html",
62                 "overview-summary.html",
63                 "overview-tree.html",
64                 "constant-values.html",
65                 "help-doc.html");
66     }
67 
68     @Test
69     public void testDocRootRewrite() {
70         javadoc("-overview", testSrc("overview.html"),
71                 "-use",
72                 "-top", "\u0130{@docroot}TOP TEXT",
73                 "-d", "out-2",
74                 "--frames",
75                 "-sourcepath", testSrc,
76                 "pkg");
77         checkExit(Exit.OK);
78 
79         checkTopText(
80                 "pkg/AnnotationType.html",
81                 "pkg/class-use/AnnotationType.html",
82                 "pkg/Cl.html",
83                 "pkg/class-use/Cl.html",
84                 "pkg/package-summary.html",
85                 "pkg/package-use.html",
86                 "overview-summary.html",
87                 "overview-tree.html",
88                 "constant-values.html",
89                 "help-doc.html");
90     }
91 
92     void checkTopText(String... files) {
93         for (String file : files) {
94             checkOutput(file, true, "TOP TEXT");
95         }
96     }
97 }
