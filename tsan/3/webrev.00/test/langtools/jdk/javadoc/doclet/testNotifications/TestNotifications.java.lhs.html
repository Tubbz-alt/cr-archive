24 /*
25  * @test
26  * @bug      4657239 4775743
27  * @summary  Make sure a notification is printed when an output directory must
28  *           be created.
29  *           Make sure classname is not include in javadoc usage message.
30  * @author   jamieh
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build    javadoc.tester.*
34  * @run main TestNotifications
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestNotifications extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestNotifications tester = new TestNotifications();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test1() {
48         String outDir = "out";
49 
50         // Notify that the destination directory must be created.
51         javadoc("-d", outDir, "-sourcepath", testSrc, "pkg");
52         checkExit(Exit.OK);
53         checkOutput(Output.OUT, true,
54                 "Creating destination directory: \"" + outDir);
55 
56         // No need to notify that the destination must be created because
57         // it already exists.
58         setOutputDirectoryCheck(DirectoryCheck.NONE);
59         javadoc("-d", outDir, "-sourcepath", testSrc, "pkg");
60         checkExit(Exit.OK);
61         checkOutput(Output.OUT, false,
62                 "Creating destination directory: \"" + outDir);
63     }
64 
65     @Test
66     public void test() {
67         //Make sure classname is not include in javadoc usage message.
68         setOutputDirectoryCheck(DirectoryCheck.NO_HTML_FILES);
69         javadoc("-help");
70         checkOutput(Output.OUT, false,
71                 "[classnames]");
72     }
73 }
