24 /*
25  * @test
26  * @bug      4490068
27  * @summary  Warn when a return tag is used on a method without a return type.
28  * @author   jamieh
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestReturnTag
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestReturnTag extends JavadocTester {
38 
39     /**
40      * Trigger warning message when return tag is used on a void method.
41      *
42      * @return I really don't return anything.
43      */
44     public void method() {}
45 
46     public static void main(String... args) throws Exception {
47         TestReturnTag tester = new TestReturnTag();
48         tester.runTests();
49     }
50 
51     @Test
52     public void tests() {
53         javadoc("-Xdoclint:none",
54                 "-d", "out",
55                 "-sourcepath", testSrc,
56                 testSrc("TestReturnTag.java"));
57         checkExit(Exit.OK);
58 
59         checkOutput(Output.OUT, true,
60             "warning - @return tag cannot be used in method with void return type.");
61     }
62 }
