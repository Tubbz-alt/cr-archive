24 /*
25  * @test
26  * @bug 4638723 8015882 8176131 8176331
27  * @summary Test to ensure that the refactored version of the standard
28  * doclet still works with Taglets that implement the 1.4.0 interface.

29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.* ToDoTaglet UnderlineTaglet Check
32  * @run main TestLegacyTaglet
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestLegacyTaglet extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestLegacyTaglet tester = new TestLegacyTaglet();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "-tagletpath", System.getProperty("test.classes", "."),
49                 "-taglet", "ToDoTaglet",
50                 "-taglet", "Check",
51                 "-taglet", "UnderlineTaglet",
52                 testSrc("C.java"));
53         checkExit(Exit.OK);
54         checkOutput("C.html", true,
55                 "This is an <u>underline</u>",
56                 "<DT><B>To Do:</B><DD><table summary=\"Summary\" cellpadding=2 cellspacing=0><tr>" +
57                 "<td bgcolor=\"yellow\">Finish this class.</td></tr></table></DD>",
58                 "<DT><B>To Do:</B><DD><table summary=\"Summary\" cellpadding=2 cellspacing=0><tr>" +
59                 "<td bgcolor=\"yellow\">Tag in Method.</td></tr></table></DD>");
60         checkOutput(Output.STDERR, false,
61                 "NullPointerException");
62     }
63 }
