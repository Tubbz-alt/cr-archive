24 /*
25  * @test
26  * @bug      4665566 4855876 7025314 8012375 8015997 8016328 8024756 8148985 8151921 8151743 8196202 8223378
27  * @summary  Verify that the output has the right javascript.
28  * @library  ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestJavascript
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestJavascript extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestJavascript tester = new TestJavascript();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test() {
45         javadoc("-d", "out",
46                 "-sourcepath", testSrc,
47                 "pkg", testSrc("TestJavascript.java"));
48         checkExit(Exit.OK);
49 
50         checkOutput("pkg/C.html", false,
51                 "<script type=\"text/javascript\"><!--\n"
52                 + "$('.navPadding').css('padding-top', $('.fixedNav').css(\"height\"));\n"
53                 + "//-->\n"
54                 + "</script>");
55 
56         checkOutput("index.html", false,
57                 "<script type=\"text/javascript\"><!--\n"
58                 + "$('.navPadding').css('padding-top', $('.fixedNav').css(\"height\"));\n"
59                 + "//-->\n");
60 
61         checkOutput("script.js", false,
62                 "$(window).resize(function() {\n"
63                 + "        $('.navPadding').css('padding-top', $('.fixedNav').css(\"height\"));\n"
64                 + "    });");
65     }
66 }
