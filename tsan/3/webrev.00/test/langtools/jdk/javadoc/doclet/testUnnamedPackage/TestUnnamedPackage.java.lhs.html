 24 /*
 25  * @test
 26  * @bug      4904075 4774450 5015144 8043698 8196201 8203791 8184205
 27  * @summary  Reference unnamed package as "Unnamed", not empty string.
 28  *           Generate a package summary for the unnamed package.
 29  * @author   jamieh
 30  * @library  ../../lib
 31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build    javadoc.tester.*
 33  * @run main TestUnnamedPackage
 34  */
 35 
 36 import javadoc.tester.JavadocTester;
 37 
 38 public class TestUnnamedPackage extends JavadocTester {
 39 
 40     public static void main(String... args) throws Exception {
 41         TestUnnamedPackage tester = new TestUnnamedPackage();
 42         tester.runTests();
 43     }
 44 
 45     @Test
 46     public void test() {
 47         javadoc("-d", "out",
 48                 "-sourcepath", testSrc,
 49                 testSrc("C.java"));
 50         checkExit(Exit.OK);
 51 
 52         checkOutput("package-summary.html", true,
 53                 "<h1 title=\"Package\" class=\"title\">Package&nbsp;&lt;Unnamed&gt;</h1>",
 54                 "This is a package comment for the unnamed package.",
 55                 "This is a class in the unnamed package.");
 56 
 57         checkOutput("package-summary.html", true,
 58                 "<title>&lt;Unnamed&gt;</title>");
 59 
 60         checkOutput("package-tree.html", true,
 61                 "<h1 class=\"title\">Hierarchy For Package &lt;Unnamed&gt;</h1>");
 62 
 63         checkOutput("index-all.html", true,
 64                 "title=\"class in &lt;Unnamed&gt;\"");
 65 
 66         checkOutput("C.html", true,
 67                 "<a href=\"package-summary.html\">");
 68 
 69         checkOutput("allclasses-index.html", true,
 70                 "<div class=\"typeSummary\">\n<table>\n"
 71                 + "<caption><span>Class Summary</span><span class=\"tabEnd\">&nbsp;</span></caption>\n"

 72                 + "<tr>\n"
 73                 + "<th class=\"colFirst\" scope=\"col\">Class</th>\n"
 74                 + "<th class=\"colLast\" scope=\"col\">Description</th>\n"
 75                 + "</tr>\n"

 76                 + "<tbody>\n"
 77                 + "<tr class=\"altColor\" id=\"i0\">\n"
 78                 + "<td class=\"colFirst\"><a href=\"C.html\" title=\"class in &lt;Unnamed&gt;\">C</a></td>\n"
 79                 + "<th class=\"colLast\" scope=\"row\">\n"
 80                 + "<div class=\"block\">This is a class in the unnamed package.</div>\n"
 81                 + "</th>\n"
 82                 + "</tr>\n"
 83                 + "</tbody>\n"
 84                 + "</table>");
 85 
 86         checkOutput("allpackages-index.html", true,
 87                 "<div class=\"packagesSummary\">\n<table>\n"
 88                 + "<caption><span>Package Summary</span><span class=\"tabEnd\">&nbsp;</span></caption>\n"

 89                 + "<tr>\n"
 90                 + "<th class=\"colFirst\" scope=\"col\">Package</th>\n"
 91                 + "<th class=\"colLast\" scope=\"col\">Description</th>\n"
 92                 + "</tr>\n"

 93                 + "<tbody>\n"
 94                 + "<tr class=\"altColor\">\n"
 95                 + "<th class=\"colFirst\" scope=\"row\"><a href=\"package-summary.html\">&lt;Unnamed&gt;</a></th>\n"
 96                 + "<td class=\"colLast\">\n"
 97                 + "<div class=\"block\">This is a package comment for the unnamed package.</div>\n"
 98                 + "</td>\n"
 99                 + "</tr>\n"
100                 + "</tbody>\n"
101                 + "</table>");
102 
103         checkOutput("type-search-index.js", true,
104                 "{\"l\":\"All Classes\",\"url\":\"allclasses-index.html\"}");
105 
106         checkOutput("package-search-index.js", true,
107                 "{\"l\":\"All Packages\",\"url\":\"allpackages-index.html\"}");
108 
109         checkOutput("index-all.html", true,
110                 "<br><a href=\"allclasses-index.html\">All&nbsp;Classes</a>&nbsp;"

111                 + "<a href=\"allpackages-index.html\">All&nbsp;Packages</a>");
112 
113         checkOutput(Output.OUT, false,
114                 "BadSource");
115     }
116 }
