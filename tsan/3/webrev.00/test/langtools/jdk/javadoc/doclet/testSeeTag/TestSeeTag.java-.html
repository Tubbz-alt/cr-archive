24 /*
25  * @test
26  * @bug      8017191 8182765 8200432
27  * @summary  Javadoc is confused by at-link to imported classes outside of the set of generated packages
28  * @author   jjg
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestSeeTag
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestSeeTag extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestSeeTag tester = new TestSeeTag();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "pkg");
49         checkExit(Exit.OK);
50 
51         checkOutput("pkg/Test.html", true,
52             "<code>List</code>",
53             "<dl>\n"
54             + "<dt><span class=\"seeLabel\">See Also:</span></dt>\n"
55             + "<dd><a href=\"Test.InnerOne.html#foo()\"><code>Test.InnerOne.foo()</code></a>, \n"
56             + "<a href=\"Test.InnerOne.html#bar(java.lang.Object)\"><code>Test.InnerOne.bar(Object)</code></a>, \n"
57             + "<a href=\"http://docs.oracle.com/javase/7/docs/technotes/tools/windows/javadoc.html#see\">Javadoc</a>, \n"
58             + "<a href=\"Test.InnerOne.html#baz(float)\"><code>something</code></a></dd>\n"
59             + "</dl>");
60 
61         checkOutput("pkg/Test.html", false,
62           "&lt;code&gt;List&lt;/code&gt;");
63 
64         checkOutput("pkg/Test2.html", true,
65            "<code>Serializable</code>");
66 
67         checkOutput("pkg/Test2.html", false,
68            ">Serialized Form<");
69     }
70 
71     @Test
72     public void testBadReference() {
73         javadoc("-d", "out-badref",
74                 "-sourcepath", testSrc,
75                 "badref");
76         checkExit(Exit.ERROR);
77 
78         checkOutput("badref/Test.html", true,
79                 "<dl>\n"
80                 + "<dt><span class=\"seeLabel\">See Also:</span></dt>\n"
81                 + "<dd><code>Object</code>, \n"
82                 + "<code>Foo<String></code></dd>\n"
83                 + "</dl>");
84     }
85 }
