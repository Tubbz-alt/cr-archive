24 /*
25  * @test
26  * @bug      4924383 8149402 8196202
27  * @summary  Test to make sure the -group option works correctly
28  *           with the given pattern usages.
29  * @author   jamieh
30  * @library  ../../lib
31  * @modules  jdk.javadoc/jdk.javadoc.internal.tool
32  * @build    javadoc.tester.*
33  * @run main TestGroupOption
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestGroupOption extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestGroupOption tester = new TestGroupOption();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test1() {
47         // Make sure the warning is not printed when -group is used correctly.
48         javadoc("-d", "out-1",
49                 "-sourcepath", testSrc,
50                 "-group", "Package One", "pkg1",
51                 "-group", "Package Two", "pkg2",
52                 "-group", "Package Three", "pkg3",
53                 "pkg1", "pkg2", "pkg3");
54         checkExit(Exit.OK);
55 
56         checkOutput(Output.OUT, false,
57                 "-group");
58     }
59 
60      @Test
61     // Make sure the "Other packages" section is printed and the header for empty section is not.
62     // Make sure that the headers of group that is defined using patterns are printed.
63     public void test2() {
64         javadoc("-d", "out-2",
65                 "--frames",
66                 "-sourcepath", testSrc,
67                 "-group", "Group pkg*", "pkg*",
68                 "-group", "Group abc*", "abc*",
69                 "-group", "Empty group", "qwerty*",
70                 "-group", "Group a*", "a*",
71                 "pkg1", "pkg2", "pkg3", "abc1",  "abc2", "abc3", "other", testSrc("InUnnamedPackage.java"));
72         checkExit(Exit.OK);
73 
74         checkOutput("overview-summary.html", true, "Group pkg*", "Group abc*", "Other Packages");
75         checkOutput("overview-summary.html", false, "Empty group", "Group a*");
76     }
77 
78     @Test
79     public void test3() {
80         // Make sure the warning is printed when -group is not used correctly.
81         javadoc("-d", "out-3",
82                 "-sourcepath", testSrc,
83                 "-group", "Package One", "pkg1",
84                 "-group", "Package One", "pkg2",
85                 "-group", "Package One", "pkg3",
86                 "pkg1", "pkg2", "pkg3");
87         checkExit(Exit.OK);
88 
89         checkOutput(Output.OUT, true,
90                 "-group");
91 
92     }
93 }
