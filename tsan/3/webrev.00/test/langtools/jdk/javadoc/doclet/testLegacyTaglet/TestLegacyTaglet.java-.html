24 /*
25  * @test
26  * @bug 4638723 8015882 8176131 8176331
27  * @summary Test to ensure that the refactored version of the standard
28  * doclet still works with Taglets that implement the 1.4.0 interface.
29  * @author jamieh
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.* ToDoTaglet UnderlineTaglet Check
33  * @run main TestLegacyTaglet
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestLegacyTaglet extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestLegacyTaglet tester = new TestLegacyTaglet();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-sourcepath", testSrc,
49                 "-tagletpath", System.getProperty("test.classes", "."),
50                 "-taglet", "ToDoTaglet",
51                 "-taglet", "Check",
52                 "-taglet", "UnderlineTaglet",
53                 testSrc("C.java"));
54         checkExit(Exit.OK);
55         checkOutput("C.html", true,
56                 "This is an <u>underline</u>",
57                 "<DT><B>To Do:</B><DD><table summary=\"Summary\" cellpadding=2 cellspacing=0><tr>" +
58                 "<td bgcolor=\"yellow\">Finish this class.</td></tr></table></DD>",
59                 "<DT><B>To Do:</B><DD><table summary=\"Summary\" cellpadding=2 cellspacing=0><tr>" +
60                 "<td bgcolor=\"yellow\">Tag in Method.</td></tr></table></DD>");
61         checkOutput(Output.STDERR, false,
62                 "NullPointerException");
63     }
64 }
