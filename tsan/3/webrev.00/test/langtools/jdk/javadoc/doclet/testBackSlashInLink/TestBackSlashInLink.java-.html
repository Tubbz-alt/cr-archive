24 /*
25  * @test
26  * @bug 4511110
27  * @summary Test to make sure that the link to source documentation
28  * has a forward slash.  It would be wrong to use a back slash.
29  * @author jamieh
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main TestBackSlashInLink
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestBackSlashInLink extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestBackSlashInLink tester = new TestBackSlashInLink();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-sourcepath", testSrc,
49                 "-linksource",
50                 testSrc("C.java"));
51         checkExit(Exit.OK);
52 
53         checkOutput("C.html", true,
54                 "src-html/C.html#line.7");
55     }
56 }
