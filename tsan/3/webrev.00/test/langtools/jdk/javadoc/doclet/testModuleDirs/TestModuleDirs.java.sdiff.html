 38 import java.io.IOException;
 39 import java.nio.file.Path;
 40 import java.nio.file.Paths;

 42 import javadoc.tester.JavadocTester;
 43 import toolbox.ModuleBuilder;
 44 import toolbox.ToolBox;

 46 public class TestModuleDirs extends JavadocTester {

 59     public void testNoModules(Path base) throws IOException {
 60         Path src = base.resolve("src");
 61         tb.writeJavaFiles(src, "package p; public class C { }");
 62 
 63         javadoc("-d", base.resolve("api").toString(),
 64                 "-sourcepath", src.toString(),
 65                 "-quiet",
 66                 "p");
 67 
 68         checkExit(Exit.OK);
 69         checkFiles(true, "p/package-summary.html");
 70     }

 72     @Test
 73     public void testNoModuleDirs(Path base) throws IOException {
 74         Path src = base.resolve("src");
 75         new ModuleBuilder(tb, "ma")
 76                 .classes("package pa; public class A {}")
 77                 .exports("pa")
 78                 .write(src);
 79         new ModuleBuilder(tb, "mb")
 80                 .classes("package pb; public class B {}")
 81                 .exports("pb")
 82                 .write(src);
 83 
 84         javadoc("-d", base.resolve("api").toString(),
 85                 "-quiet",
 86                 "--frames",
 87                 "--module-source-path", src.toString(),
 88                 "--no-module-directories",
 89                 "--module", "ma,mb");
 90 
 91         checkExit(Exit.OK);
 92         checkFiles(true,
 93                 "ma-frame.html",
 94                 "ma-summary.html",
 95                 "pa/package-summary.html");
 96         checkFiles(false,
 97                 "ma/module-frame.html",
 98                 "ma/module-summary.html",
 99                 "ma/pa/package-summary.html");
100         checkOutput("ma-frame.html", true,
101                 "<ul>\n"
102                 + "<li><a href=\"allclasses-frame.html\" target=\"packageFrame\">All&nbsp;Classes</a></li>\n"
103                 + "<li><a href=\"overview-frame.html\" target=\"packageListFrame\">All&nbsp;Packages</a></li>\n"
104                 + "<li><a href=\"module-overview-frame.html\" target=\"packageListFrame\">All&nbsp;Modules</a></li>\n"
105                 + "</ul>\n");
106         checkOutput("ma-summary.html", false,
107                 "<ul class=\"navList\" id=\"allclasses_navbar_top\">\n"
108                 + "<li><a href=\"allclasses-noframe.html\">All&nbsp;Classes</a></li>\n"
109                 + "</ul>\n");
110         checkOutput("pa/package-summary.html", true,
111                 "<li><a href=\"../deprecated-list.html\">Deprecated</a></li>\n"
112                 + "<li><a href=\"../index-all.html\">Index</a></li>");
113 
114     }
