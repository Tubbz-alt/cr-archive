24 /*
25  * @test
26  * @bug      4802275 4967243 8026567
27  * @summary  Make sure param tags are still printed even though they do not
28  *           match up with a real parameters.
29  *           Make sure inheritDoc cannot be used in an invalid param tag.
30  * @author   jamieh
31  * @library  ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build    javadoc.tester.*
34  * @run main TestParamTaglet
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestParamTaglet extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestParamTaglet tester = new TestParamTaglet();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         javadoc("-d", "out",
49                 "-sourcepath", testSrc,
50                 "pkg");
51         checkExit(Exit.ERROR);
52 
53         checkOutput("pkg/C.html", true,
54                 //Regular param tags.
55                 "<span class=\"paramLabel\">Parameters:</span></dt>\n"
56                 + "<dd><code>param1</code> - testing 1 2 3.</dd>\n"
57                 + "<dd><code>param2</code> - testing 1 2 3.",
58                 //Param tags that don't match with any real parameters.
59                 "<span class=\"paramLabel\">Parameters:</span></dt>\n"
60                 + "<dd><code>p1</code> - testing 1 2 3.</dd>\n"
61                 + "<dd><code>p2</code> - testing 1 2 3.",
62                 //{@inherit} doc misuse does not cause doclet to throw exception.
63                 // Param is printed with nothing inherited.
64                 //XXX: in the future when Configuration is available during doc inheritence,
65                 //print a warning for this mistake.
66                 "<code>inheritBug</code> -");
67     }
68 }
