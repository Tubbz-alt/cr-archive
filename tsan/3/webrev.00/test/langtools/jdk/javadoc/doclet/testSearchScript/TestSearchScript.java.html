 56     private Invocable getEngine() throws ScriptException, IOException, NoSuchMethodException {
 57         ScriptEngineManager engineManager = new ScriptEngineManager();
 58         ScriptEngine engine = engineManager.getEngineByName("nashorn");
 59         engine.eval(new BufferedReader(new FileReader(new File(testSrc, "javadoc-search.js"))));
 60         Invocable inv = (Invocable) engine;
 61         inv.invokeFunction("loadIndexFiles", outputDir.getAbsolutePath());
 62         return inv;
 63     }

291     void checkSearch(Invocable inv, String query, List<String> results) throws ScriptException, NoSuchMethodException {
292         checkList((List) inv.invokeFunction("search", query), results);
293     }
294 
295     void checkList(List<?> result, List<?> expected) {
296         checking("Checking list: " + result);
297         if (!expected.equals(result)) {
298             failed("Expected: " + expected + ", got: " + result);
299         } else {
300             passed("List matches expected result");
301         }
302     }
303 
304     void checkNullSearch(Invocable inv, String query) throws ScriptException, NoSuchMethodException {
305         Object result = inv.invokeFunction("search", query);
306         checking("Checking null result");
307         if (result == null) {
308             passed("Result is null as expected");
309         } else {
310             failed("Expected: null, got: " + result);
311         }
312     }
