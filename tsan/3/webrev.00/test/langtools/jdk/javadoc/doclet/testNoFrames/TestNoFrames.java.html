24 /*
25  * @test
26  * @bug 8215599
27  * @summary Remove support for javadoc "frames" mode
28  * @library ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.*
31  * @run main TestNoFrames
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestNoFrames extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestNoFrames tester = new TestNoFrames();
40         tester.runTests();
41     }
42 
43     @Test
44     public void testFrames() {
45         javadoc("-d", "out-1",
46                 "--frames",
47                 "-sourcepath",testSrc,
48                 testSrc("TestNoFrames.java"));
49         checkExit(Exit.ERROR);
50     }
51 
52     @Test
53     public void testDefault() {
54         javadoc("-d", "out-2",
55                 "--no-frames",
56                 "-sourcepath",testSrc,
57                 testSrc("TestNoFrames.java"));
58         checkExit(Exit.OK);
59 
60         checkOutput(Output.OUT, true,
61                 "javadoc: warning - The --no-frames option is no longer required and may be removed\n"
62                 + "in a future release.");
63     }
64 }
