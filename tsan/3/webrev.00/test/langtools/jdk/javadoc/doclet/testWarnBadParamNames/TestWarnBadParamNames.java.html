24 /*
25  * @test
26  * @bug 4693440
27  * @summary Test to make sure that warning is printed when bad parameter
28  * name is used with param.
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestWarnBadParamNames
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestWarnBadParamNames extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestWarnBadParamNames tester = new TestWarnBadParamNames();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-Xdoclint:none",
47                 "-d", "out",
48                 testSrc("C.java"));
49         checkExit(Exit.OK);
50 
51         checkOutput(Output.OUT, true,
52                 "warning - @param argument \"int\" is not a parameter name.",
53                 "warning - @param argument \"IDontExist\" is not a parameter name.",
54                 "warning - Parameter \"arg\" is documented more than once.");
55     }
56 }
