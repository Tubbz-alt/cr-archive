164     /**
165      * The following test is somewhat simplistic, but it is useful
166      * in conjunction with the W3C Validation Service at https://validator.w3.org/nu/#file
167      * @param base A working directory for this method, in which some UTF-8 source files
168      *      will be generated
169      * @throws IOException if there is a problem generating the source files
170      */
171     @Test
172     public void testNonAscii(Path base) throws IOException {
173         Path src = base.resolve("src");
174         tb.writeJavaFiles(src,
175                 "package p; public class Def {\n"
176                 + "    public int \u00e0\u00e9;\n"              // a`e'
177                 + "    public void \u00c0\u00c9() { }\n"        // A`E'
178                 + "    public int \u03b1\u03b2\u03b3;\n"        // alpha beta gamma
179                 + "    public void \u0391\u0392\u0393() { }\n"  // ALPHA BETA GAMMA
180                 + "}",
181                 "package p; \n"
182                 + "/**\n"
183                 + " * {@link Def#\u00e0\u00e9 &agrave;&eacute;}<br>\n"
184                 + " * {@link Def#\u00c0\u00c9() &Agrave;&Eacute;}<br>\n"
185                 + " * {@link Def#\u03b1\u03b2\u03b3 &alpha;&beta;&gamma;}<br>\n"
186                 + " * {@link Def#\u0391\u0392\u0393() &Alpha;&Beta;&Gamma;}<br>\n"
187                 + " */\n"
188                 + "public class Ref { }");
189 
190         javadoc("-d", "out-nonAscii",
191                 "-sourcepath", src.toString(),
192                 "-html5",
193                 "-encoding", "utf-8",
194                 "p");
195         checkExit(Exit.OK);
196 
197         checkOutput("p/Def.html", true,
198                 "<a id=\"\u00e0\u00e9\">",
199                 "<a id=\"\u00c0\u00c9()\">",
200                 "<a id=\"\u03b1\u03b2\u03b3\">",
201                 "<a id=\"\u0391\u0392\u0393()\">");
202 
203         checkOutput("p/Ref.html", true,
204                 "<a href=\"Def.html#%C3%A0%C3%A9\"><code>&agrave;&eacute;</code></a>",
205                 "<a href=\"Def.html#%C3%80%C3%89()\"><code>&Agrave;&Eacute;</code></a>",
206                 "<a href=\"Def.html#%CE%B1%CE%B2%CE%B3\"><code>&alpha;&beta;&gamma;</code></a>",
207                 "<a href=\"Def.html#%CE%91%CE%92%CE%93()\"><code>&Alpha;&Beta;&Gamma;</code></a>");
208 
209     }
