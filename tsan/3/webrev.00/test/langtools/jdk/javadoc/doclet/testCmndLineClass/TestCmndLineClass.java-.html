24 /*
25  * @test
26  * @bug 4506980
27  * @summary Test to make sure that there is no difference in the output
28  * when specifying packages on the command line and specifying individual
29  * classes.
30  * @author jamieh
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.*
34  * @run main TestCmndLineClass
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestCmndLineClass extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestCmndLineClass tester = new TestCmndLineClass();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         String outdir1 = "out-1";
49         String outdir2 = "out-2";
50 
51         javadoc("-d", outdir1,
52                 "-sourcepath", testSrc,
53                 "-notimestamp",
54                 testSrc("C5.java"), "pkg1", "pkg2");
55         checkExit(Exit.OK);
56 
57         javadoc("-d", outdir2,
58                 "-sourcepath", testSrc,
59                 "-notimestamp",
60                 testSrc("C5.java"),
61                 testSrc("pkg1/C1.java"),
62                 testSrc("pkg1/C2.java"),
63                 testSrc("pkg2/C3.java"),
64                 testSrc("pkg2/C4.java"));
65         checkExit(Exit.OK);
66 
67         diff(outdir1, outdir2,
68                 "C5.html",
69                 "pkg1/C1.html",
70                 "pkg1/C2.html",
71                 "pkg2/C3.html",
72                 "pkg2/C4.html");
73     }
74 }
