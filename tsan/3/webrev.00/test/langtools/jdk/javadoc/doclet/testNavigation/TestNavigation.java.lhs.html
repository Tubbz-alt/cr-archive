181     @Test
182     public void test3(Path base) throws IOException {
183         Path src = base.resolve("src");
184         tb.writeJavaFiles(src,
185                 "package pkg1; public class A {\n"
186                 + "    /**\n"
187                 + "     * Class with members.\n"
188                 + "     */\n"
189                 + "    public static class X {\n"
190                 + "        /**\n"
191                 + "         * A ctor\n"
192                 + "         */\n"
193                 + "        public X() {\n"
194                 + "        }\n"
195                 + "        /**\n"
196                 + "         * A field\n"
197                 + "         */\n"
198                 + "        public int field;\n"
199                 + "        /**\n"
200                 + "         * A method\n"
201                 + "         */\n"
202                 + "        public void method() {\n"
203                 + "        }\n"
204                 + "        /**\n"
205                 + "         * An inner class\n"
206                 + "         */\n"
207                 + "        public static class IC {\n"
208                 + "        }\n"
209                 + "    }\n"
210                 + "    /**\n"
211                 + "     * Class with all inherited members.\n"
212                 + "     */\n"
213                 + "    public static class Y extends X {\n"
214                 + "    }\n"
215                 + "}");
216 
217         tb.writeJavaFiles(src,
218                 "package pkg1; public class C {\n"
219                 + "}");
220 
221         tb.writeJavaFiles(src,
222                 "package pkg1; public interface InterfaceWithNoMembers {\n"
223                 + "}");
224 
225         javadoc("-d", "out-3",
226                 "-sourcepath", src.toString(),
227                 "pkg1");
228         checkExit(Exit.OK);
229 
230         checkOrder("pkg1/A.X.html",
231                 "Summary",
232                 "<li><a href=\"#nested.class.summary\">Nested</a>&nbsp;|&nbsp;</li>",
233                 "<li><a href=\"#field.summary\">Field</a>&nbsp;|&nbsp;</li>",
234                 "<li><a href=\"#constructor.summary\">Constr</a>&nbsp;|&nbsp;</li>",
235                 "<li><a href=\"#method.summary\">Method</a></li>");
236 
237         checkOrder("pkg1/A.Y.html",
238                 "Summary",
239                 "<li><a href=\"#nested.class.summary\">Nested</a>&nbsp;|&nbsp;</li>",
240                 "<li><a href=\"#field.summary\">Field</a>&nbsp;|&nbsp;</li>",
241                 "<li><a href=\"#constructor.summary\">Constr</a>&nbsp;|&nbsp;</li>",
242                 "<li><a href=\"#method.summary\">Method</a></li>");
243 
244         checkOrder("pkg1/A.X.IC.html",
245                 "Summary",
246                 "<li>Nested&nbsp;|&nbsp;</li>",
247                 "<li>Field&nbsp;|&nbsp;</li>",
248                 "<li><a href=\"#constructor.summary\">Constr</a>&nbsp;|&nbsp;</li>",
249                 "<li><a href=\"#method.summary\">Method</a></li>");
250 
251         checkOrder("pkg1/C.html",
252                 "Summary",
253                 "<li>Nested&nbsp;|&nbsp;</li>",
254                 "<li>Field&nbsp;|&nbsp;</li>",
255                 "<li><a href=\"#constructor.summary\">Constr</a>&nbsp;|&nbsp;</li>",
256                 "<li><a href=\"#method.summary\">Method</a></li>");
257 
258         checkOrder("pkg1/InterfaceWithNoMembers.html",
259                 "Summary",
260                 "<li>Nested&nbsp;|&nbsp;</li>",
261                 "<li>Field&nbsp;|&nbsp;</li>",
262                 "<li>Constr&nbsp;|&nbsp;</li>",
263                 "<li>Method</li>");
264     }
