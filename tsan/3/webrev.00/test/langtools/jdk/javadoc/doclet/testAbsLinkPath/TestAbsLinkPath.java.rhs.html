24 /*
25  * @test
26  * @bug 4640745
27  * @summary This test verifys that the -link option handles absolute paths.

28  * @library ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.*
31  * @run main TestAbsLinkPath
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestAbsLinkPath extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestAbsLinkPath tester = new TestAbsLinkPath();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test1() {
45         String out1 = "out1";
46         javadoc("-d", out1, "-sourcepath", testSrc, "pkg2");
47         checkExit(Exit.OK);
48 
49         javadoc("-d", "out2",
50                 "-sourcepath", testSrc,
51                 "-link", "../" + out1,
52                 "pkg1");
53         checkExit(Exit.OK);
54 
55         checkOutput("pkg1/C1.html", true,
56                 "C2.html");
57     }
58 }
