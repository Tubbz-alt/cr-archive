24 /*
25  * @test
26  * @bug 8008164 8169819 8183037 8182765 8196202 8184205
27  * @summary Test styles on HTML tables generated by javadoc.
28  * @author Bhavesh Patel
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestHtmlTableStyles
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestHtmlTableStyles extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestHtmlTableStyles tester = new TestHtmlTableStyles();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "-use",
49                 "--frames",
50                 "pkg1", "pkg2");
51         checkExit(Exit.ERROR);
52         checkOutput(Output.OUT, true,
53                 "attribute not supported in HTML5: summary",
54                 "attribute border for table only accepts \"\" or \"1\", use CSS instead: BORDER",
55                 "attribute not supported in HTML5: cellpadding",
56                 "attribute not supported in HTML5: cellspacing",
57                 "attribute not supported in HTML5: align");
58 
59         checkOutput("pkg1/TestTable.html", true,
60                 "<table summary=\"Summary\" border cellpadding=3 cellspacing=1>",
61                 "<div class=\"memberSummary\">\n<table>",
62                 "<div class=\"memberSummary\">\n<table>",
63                 "<div class=\"memberSummary\">\n<table>");
64 
65         checkOutput("pkg1/package-summary.html", true,
66                 "<div class=\"typeSummary\">\n<table>");
67 
68         checkOutput("pkg1/class-use/TestTable.html", true,
69                 "<div class=\"useSummary\">\n<table>");
70 
71         checkOutput("overview-summary.html", true,
72                 "<div class=\"overviewSummary\">\n<table>");
73 
74         checkOutput("deprecated-list.html", true,
75             "<div class=\"deprecatedSummary\">\n<table>");
76 
77         checkOutput("constant-values.html", true,
78             "<div class=\"constantsSummary\">\n<table>");
79     }
80 }
