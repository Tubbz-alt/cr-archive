24 /*
25  * @test
26  * @bug      7180906 8026567
27  * @summary  Test to make sure that the since tag works correctly
28  * @author   Bhavesh Patel
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestSinceTag
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestSinceTag extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestSinceTag tester = new TestSinceTag();
41         tester.runTests();
42         tester.printSummary();
43     }
44 
45     @Test
46     public void testSince() {
47         javadoc("-d", "out-since",
48                 "-sourcepath", testSrc,
49                 "pkg1");
50         checkExit(Exit.OK);
51 
52         checkSince(true);
53     }
54 
55     @Test
56     public void testNoSince() {
57         javadoc("-d", "out-nosince",
58                 "-sourcepath", testSrc,
59                 "-nosince",
60                 "pkg1");
61         checkExit(Exit.OK);
62 
63         checkSince(false);
64     }
65 
66     void checkSince(boolean on) {
67         checkOutput("pkg1/C1.html", on,
68                 "<dl>\n"
69                 + "<dt><span class=\"simpleTagLabel\">Since:</span></dt>\n"
70                 + "<dd>JDK1.0</dd>");
71 
72         checkOutput("serialized-form.html", on,
73                 "<dl>\n"
74                 + "<dt><span class=\"simpleTagLabel\">Since:</span></dt>\n"
75                 + "<dd>1.4</dd>");
76     }
77 }
