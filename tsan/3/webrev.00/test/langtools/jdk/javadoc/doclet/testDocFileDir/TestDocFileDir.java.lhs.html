24 /*
25  * @test
26  * @bug 4258405 4973606 8024096
27  * @summary This test verifies that the doc-file directory does not
28  *          get overwritten when the sourcepath is equal to the destination
29  *          directory.
30  *          Also test that -docfilessubdirs and -excludedocfilessubdir both work.
31  * @author jamieh
32  * @library ../../lib
33  * @modules jdk.javadoc/jdk.javadoc.internal.tool
34  * @build javadoc.tester.*
35  * @run main TestDocFileDir
36  */
37 
38 import javadoc.tester.JavadocTester;
39 
40 public class TestDocFileDir extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         TestDocFileDir tester = new TestDocFileDir();
44         tester.runTests();
45     }
46 
47     // Output dir = "", Input dir = ""
48     @Test
49     public void test1() {
50         copyDir(testSrc("pkg"), ".");
51         setOutputDirectoryCheck(DirectoryCheck.NO_HTML_FILES);
52         javadoc("pkg/C.java");
53         checkExit(Exit.OK);
54         checkOutput("pkg/doc-files/testfile.txt", true,
55             "This doc file did not get trashed.");
56     }
57 
58     // Output dir = Input Dir
59     @Test
60     public void test2() {
61         String outdir = "out2";
62         copyDir(testSrc("pkg"), outdir);
63         setOutputDirectoryCheck(DirectoryCheck.NO_HTML_FILES);
64         javadoc("-d", outdir,
65             "-sourcepath", "blah" + PS + outdir + PS + "blah",
66             "pkg");
67         checkExit(Exit.OK);
68         checkOutput("pkg/doc-files/testfile.txt", true,
69             "This doc file did not get trashed.");
70     }
71 
72     // Exercising -docfilessubdirs and -excludedocfilessubdir
73     @Test
74     public void test3() {
75         String outdir = "out3";
76         setOutputDirectoryCheck(DirectoryCheck.NONE);
77         javadoc("-d", outdir,
78                 "-sourcepath", testSrc,
79                 "-docfilessubdirs",
80                 "-excludedocfilessubdir", "subdir-excluded1:subdir-excluded2",
81                 "pkg");
82         checkExit(Exit.OK);
83         checkFiles(true,
84                 "pkg/doc-files/subdir-used1/testfile.txt",
85                 "pkg/doc-files/subdir-used2/testfile.txt");
86         checkFiles(false,
87                 "pkg/doc-files/subdir-excluded1/testfile.txt",
88                 "pkg/doc-files/subdir-excluded2/testfile.txt");
89     }
90 }
