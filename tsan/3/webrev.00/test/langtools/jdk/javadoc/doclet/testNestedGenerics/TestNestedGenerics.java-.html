35 import javadoc.tester.JavadocTester;
36 
37 public class TestNestedGenerics extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestNestedGenerics tester = new TestNestedGenerics();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "pkg");
49         checkExit(Exit.OK);
50 
51         checkOutput("pkg/NestedGenerics.html", true,
52             "<div class=\"block\">Contains <a " +
53             "href=\"#foo(java.util.Map)\"><code>foo" +
54             "(java.util.Map&lt;A, java.util.Map&lt;A, A&gt;&gt;)</code></a></div>");
55     }
56 }
