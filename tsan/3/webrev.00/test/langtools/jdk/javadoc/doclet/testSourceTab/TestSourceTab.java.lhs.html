 24 /*
 25  * @test
 26  * @bug 4510979
 27  * @summary Test to make sure that the source documentation is indented properly
 28  * when -linksourcetab is used.
 29  * @author jamieh
 30  * @library ../../lib
 31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build javadoc.tester.*
 33  * @run main TestSourceTab
 34  */
 35 
 36 import java.io.*;
 37 
 38 import javadoc.tester.JavadocTester;
 39 
 40 public class TestSourceTab extends JavadocTester {
 41 
 42     public static void main(String... args) throws Exception {
 43         TestSourceTab tester = new TestSourceTab();
 44         tester.runTests();
 45     }
 46 
 47     @Test
 48     public void test() throws Exception {
 49         String tmpSrcDir = "tmpSrc";
 50         String outdir1 = "out-tabLengthEight";
 51         String outdir2 = "out-tabLengthFour";
 52         initTabs(new File(testSrc), new File(tmpSrcDir));
 53 
 54         // Run Javadoc on a source file with that is indented with a single tab per line
 55         javadoc("-d", outdir1,
 56                 "-sourcepath", tmpSrcDir,
 57                 "-notimestamp",
 58                 "-linksource",
 59                 tmpSrcDir + "/SingleTab/C.java");
 60         checkExit(Exit.OK);
 61 
 62         // Run Javadoc on a source file with that is indented with a two tab per line
 63         // If we double the tabs and decrease the tab length by a half, the output should
 64         // be the same as the one generated above.
 65         javadoc("-d", outdir2,
 66                 "-sourcepath", tmpSrcDir,
 67                 "-notimestamp",
 68                 "-sourcetab", "4",
 69                 tmpSrcDir + "/DoubleTab/C.java");
 70         checkExit(Exit.OK);
 71 
 72         diff(outdir1, outdir2,
 73                 "src-html/C.html",
 74                 "C.html");
 75     }
 76 
 77     void initTabs(File from, File to) throws IOException {
 78         for (File f: from.listFiles()) {
 79             File t = new File(to, f.getName());
 80             if (f.isDirectory()) {
 81                 initTabs(f, t);
 82             } else if (f.getName().endsWith(".java")) {
 83                 write(t, read(f).replace("\\t", "\t"));
 84             }
 85         }
 86     }
 87 
 88     String read(File f) throws IOException {
 89         StringBuilder sb = new StringBuilder();
 90         try (BufferedReader in = new BufferedReader(new FileReader(f))) {
 91             String line;
 92             while ((line = in.readLine()) != null) {
 93                 sb.append(line);
 94                 sb.append(NL);
 95             }
 96         }
 97         return sb.toString();
 98     }
 99 
100     void write(File f, String s) throws IOException {
101         f.getParentFile().mkdirs();
102         try (Writer out = new FileWriter(f)) {
103             out.write(s);
104         }
105     }
106 }
