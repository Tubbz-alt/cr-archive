24 /*
25  * @test
26  * @bug 4691095 6306394
27  * @summary Make sure that Javadoc emits a useful warning
28  *          when a bad package.html exists in a JAR archive.
29  * @author jamieh
30  * @library /tools/lib ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.* toolbox.ToolBox toolbox.JarTask
33  * @run main TestBadPackageFileInJar
34  */
35 
36 import toolbox.JarTask;
37 import toolbox.Task.Result;
38 import toolbox.ToolBox;
39 
40 import java.io.IOException;
41 import java.nio.file.Path;
42 import java.nio.file.Paths;
43 
44 import javadoc.tester.JavadocTester;
45 
46 public class TestBadPackageFileInJar extends JavadocTester {
47 
48     final ToolBox tb = new ToolBox();
49 
50     public static void main(String... args) throws Exception {
51         TestBadPackageFileInJar tester = new TestBadPackageFileInJar();
52         tester.runTests();
53     }
54 
55     @Test
56     public void test() throws IOException {
57         // create the file
58         Path pkgDir = Paths.get("pkg");
59         tb.createDirectories(pkgDir);
60         Path pkgfilePath = pkgDir.resolve("package.html");
61         tb.writeFile(pkgfilePath, "<html>\n\n</html>");
62 
63         // create the jar file
64         Path jarFile = Paths.get("badPackageFileInJar.jar");
65         JarTask jar = new JarTask(tb, "badPackageFileInJar.jar");
66         jar.files(pkgDir.toString()).run();
67 
68         // clean up to prevent accidental pick up
69         tb.cleanDirectory(pkgDir);
70         tb.deleteFiles(pkgDir.toString());
71 
72         javadoc("-d", "out",
73                 "-sourcepath", testSrc,
74                 "-classpath",  jarFile.toString(),
75                 "pkg");
76         checkExit(Exit.OK);
77 
78         checkOutput(Output.OUT, true,
79                 "badPackageFileInJar.jar(/pkg/package.html):1: warning: no comment");
80     }
81 }
