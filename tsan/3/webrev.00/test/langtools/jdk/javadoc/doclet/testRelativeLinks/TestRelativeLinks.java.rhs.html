 24 /*
 25  * @test
 26  * @bug      4460354 8014636 8043186 8195805 8182765 8196202
 27  * @summary  Test to make sure that relative paths are redirected in the
 28  *           output so that they are not broken.

 29  * @library  ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build    javadoc.tester.*
 32  * @run main TestRelativeLinks
 33  */
 34 
 35 import java.io.File;
 36 import java.io.FileOutputStream;
 37 import java.io.IOException;
 38 
 39 import javadoc.tester.JavadocTester;
 40 
 41 public class TestRelativeLinks extends JavadocTester {
 42 
 43     public static void main(String... args) throws Exception {
 44         TestRelativeLinks tester = new TestRelativeLinks();
 45         tester.runTests();
 46     }
 47 
 48     @Test
 49     public void test() {
 50         javadoc("-d", "out",
 51                 "-use",

 52                 "-sourcepath", testSrc,
 53                 "pkg", "pkg2");
 54         checkExit(Exit.ERROR);
 55 
 56         checkOutput(Output.OUT, true,
 57                 "attribute not supported in HTML5: name");
 58 
 59         // These relative paths should stay relative because they appear
 60         // in the right places.
 61         checkOutput("pkg/C.html", true,
 62             "<a href=\"relative-class-link.html\">relative class link</a>",
 63             "<a href=\"relative-field-link.html\">relative field link</a>",
 64             "<a href=\"relative-method-link.html\">relative method link</a>",
 65             " <a href=\"relative-multi-line-link.html\">relative-multi-line-link</a>.");
 66         checkOutput("pkg/package-summary.html", true,
 67             "<a href=\"relative-package-link.html\">relative package link</a>");
 68 
 69         // These relative paths should be redirected because they are in different
 70         // places.
 71 
 72         // INDEX PAGE
 73         checkOutput("index-all.html", true,
 74             "<a href=\"./pkg/relative-class-link.html\">relative class link</a>",
 75             "<a href=\"./pkg/relative-field-link.html\">relative field link</a>",
 76             "<a href=\"./pkg/relative-method-link.html\">relative method link</a>",
 77             "<a href=\"./pkg/relative-package-link.html\">relative package link</a>",
 78             " <a href=\"./pkg/relative-multi-line-link.html\">relative-multi-line-link</a>.");
 79 
 80         // This is not a relative path and should not be redirected.
 81         checkOutput("index-all.html", true,
 82             "<div class=\"block\"><a name=\"masters\"></a>");
 83         checkOutput("index-all.html", false,
 84             "<div class=\"block\"><a name=\"./pkg/masters\"></a>");
 85 
 86         // PACKAGE USE
 87         checkOutput("pkg/package-use.html", true,
 88             "<a href=\"../pkg/relative-package-link.html\">relative package link</a>.",
 89             "<a href=\"../pkg/relative-class-link.html\">relative class link</a>");
 90 
 91         // CLASS_USE
 92         checkOutput("pkg/class-use/C.html", true,
 93             "<a href=\"../../pkg/relative-field-link.html\">relative field link</a>",
 94             "<a href=\"../../pkg/relative-method-link.html\">relative method link</a>",
 95             "<a href=\"../../pkg/relative-package-link.html\">relative package link</a>",
 96             " <a href=\"../../pkg/relative-multi-line-link.html\">relative-multi-line-link</a>.");
 97 
 98         // PACKAGE OVERVIEW
 99         checkOutput("index.html", true,
100             "<a href=\"./pkg/relative-package-link.html\">relative package link</a>");
101     }
102 
103     private void touch(String file) {
104         File f = new File(outputDir, file);
105         out.println("touch " + f);
106         try (FileOutputStream fos = new FileOutputStream(f)) {
107         } catch (IOException e) {
108             checking("Touch file");
109             failed("Error creating file: " + e);
110         }
111     }
112 }
