 24 /*
 25  * @test
 26  * @bug 8219313
 27  * @summary Support module specific stylesheets
 28  * @library /tools/lib ../../lib
 29  * @modules jdk.compiler/com.sun.tools.javac.api
 30  *          jdk.compiler/com.sun.tools.javac.main
 31  *          jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build javadoc.tester.* toolbox.ToolBox
 33  * @run main TestModuleSpecificStylesheet
 34  */
 35 
 36 
 37 import java.nio.file.Files;
 38 import java.nio.file.Path;
 39 import java.nio.file.Paths;
 40 import java.util.List;
 41 
 42 import toolbox.ModuleBuilder;
 43 import toolbox.ToolBox;
 44 
 45 import javadoc.tester.JavadocTester;
 46 
 47 public class TestModuleSpecificStylesheet extends JavadocTester {
 48 
 49     final ToolBox tb;
 50 
 51     public static void main(String... args) throws Exception {
 52         TestModuleSpecificStylesheet tester = new TestModuleSpecificStylesheet();
 53         tester.runTests(m -> new Object[]{Paths.get(m.getName())});
 54     }
 55 
 56     TestModuleSpecificStylesheet() {
 57         tb = new ToolBox();
 58     }
 59 
 60     @Test
 61     public void test(Path base) throws Exception {
 62         Path srcDir = base.resolve("src");
 63         Path outDir = base.resolve("out");
 64 
 65         new ModuleBuilder(tb, "ma")
 66                 .classes("package pa; public class A{}")
 67                 .classes("package pa.pb; public class B{}")
 68                 .exports("pa")
 69                 .exports("pa.pb")
 70                 .write(srcDir);
 71 
 72         Path docFilesDir = Files.createDirectories(srcDir.resolve("ma").resolve("doc-files"));
 73         Path stylesheet = docFilesDir.resolve("spanstyle.css");
 74         Files.createFile(stylesheet);
 75         Files.write(stylesheet, List.of("span{ color:blue; }"));
 76 
 77         javadoc("-d", outDir.toString(),
 78                 "--module-source-path", srcDir.toString(),
 79                 "--module", "ma");
 80 
 81         checkExit(Exit.OK);
 82 
 83         checkOutput("ma/module-summary.html", true,
 84                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../ma/doc-files/spanstyle.css\"" +
 85                         " title=\"Style\">");
 86 
 87         checkOutput("ma/pa/package-summary.html", true,
 88                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../../ma/doc-files/spanstyle.css\"" +
 89                         " title=\"Style\">");
 90 
 91         checkOutput("ma/pa/A.html", true,
 92                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../../ma/doc-files/spanstyle.css\"" +
 93                         " title=\"Style\">");
 94 
 95         checkOutput("ma/pa/pb/B.html", true,
 96                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../../../ma/doc-files/spanstyle.css\"" +
 97                         " title=\"Style\">");
 98 
 99         checkOutput("ma/pa/pb/package-summary.html", true,
100                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../../../ma/doc-files/spanstyle.css\"" +
101                         " title=\"Style\">");
102     }
103 }
