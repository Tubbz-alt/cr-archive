26  * @bug 4275630 4749453 4625400 4753048 4415270 8074521 8182765 8196202
27  * @summary  Generated HTML is invalid with frames.
28  *           Displays unnecessary horizontal scroll bars.
29  *           Missing whitespace in DOCTYPE declaration
30  *           HTML table tags inserted in wrong place in pakcage use page
31  * @author dkramer
32  * @library ../../lib
33  * @modules jdk.javadoc/jdk.javadoc.internal.tool
34  * @build javadoc.tester.*
35  * @run main ValidHtml
36  */
37 
38 import javadoc.tester.JavadocTester;
39 
40 public class ValidHtml extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         ValidHtml tester = new ValidHtml();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         // Test for all cases except the split index page
50         javadoc("-d", "out",
51                     "-doctitle", "Document Title",
52                     "-windowtitle", "Window Title",
53                     "-use",
54                     "--frames",
55                     "-overview", testSrc("overview.html"),
56                     "-sourcepath", testSrc,
57                     "p1", "p2");
58         checkExit(Exit.OK);
59         // Test for IFRAME element:
60         checkOutput("index.html", true,
61                 "<iframe");
62         // Test the table elements are in the correct order:
63         checkOutput("p1/package-use.html", true,
64                 "</td>\n"
65                 + "</tr>");
66         String HTML5 = "<!DOCTYPE HTML>";
67         checkValidHTML(HTML5);
68     }
69 
70     void checkValidHTML(String doctype) {
71         // Test the proper DOCTYPE element are present:
72         checkOutput("index.html", true, doctype);
73         checkOutput("overview-summary.html", true, doctype);
74         checkOutput("p1/package-summary.html", true, doctype);
75         checkOutput("p1/C.html", true, doctype);
76         checkOutput("overview-frame.html", true, doctype);
77         checkOutput("allclasses-frame.html", true, doctype);
78         checkOutput("p1/package-frame.html", true, doctype);
79     }
80 }
