32  * @run main AccessH1

35 
36 import javadoc.tester.JavadocTester;
37 
38 public class AccessH1 extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         AccessH1 tester = new AccessH1();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-doctitle", "Document Title",
49                 "--frames",
50                 "-sourcepath", testSrc,
51                 "p1", "p2");
52         checkExit(Exit.OK);
53 
54         // Test the style sheet
55         checkOutput("stylesheet.css", true,
56                 "h1 {\n"
57                 + "    font-size:20px;\n"
58                 + "}");
59 
60         // Test the doc title in the overview page
61         checkOutput("overview-summary.html", true,
62                 "<h1 class=\"title\">Document Title</h1>");
63     }
64 }
