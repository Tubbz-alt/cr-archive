24 /*
25  * @test
26  * @bug 4634891 8025633 8026567 8182765
27  * @summary Determine if overridden methods are properly documented when
28  * -protected (default) visibility flag is used.
29  * @author jamieh
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main TestOverriddenPrivateMethodsWithPackageFlag
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestOverriddenPrivateMethodsWithPackageFlag extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestOverriddenPrivateMethodsWithPackageFlag tester = new TestOverriddenPrivateMethodsWithPackageFlag();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-sourcepath", testSrc,
49                 "-package",
50                 "pkg1", "pkg2");
51         checkExit(Exit.OK);
52 
53         // The public method should be overridden
54         checkOutput("pkg1/SubClass.html", true,
55                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
56                 + "<dd><code><a href=\"BaseClass.html#publicMethod()\">"
57                 + "publicMethod</a></code>&nbsp;in class&nbsp;<code>"
58                 + "<a href=\"BaseClass.html\" title=\"class in pkg1\">BaseClass</a></code></dd>");
59 
60         // The public method in different package should be overridden
61         checkOutput("pkg2/SubClass.html", true,
62                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
63                 + "<dd><code><a href=\"../pkg1/BaseClass.html#publicMethod()\">"
64                 + "publicMethod</a></code>&nbsp;in class&nbsp;<code>"
65                 + "<a href=\"../pkg1/BaseClass.html\" title=\"class in pkg1\">BaseClass</a></code></dd>");
66 
67         // The package private method should be overridden since the base and sub class are in the same
68         // package.
69         checkOutput("pkg1/SubClass.html", true,
70                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
71                 + "<dd><code><a href=\"BaseClass.html#packagePrivateMethod()\">"
72                 + "packagePrivateMethod</a></code>&nbsp;in class&nbsp;<code>"
73                 + "<a href=\"BaseClass.html\" title=\"class in pkg1\">BaseClass</a></code></dd>");
74 
75         // The private method in should not be overridden
76         checkOutput("pkg1/SubClass.html", false,
77                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
78                 + "<dd><code><a href=\"BaseClass.html#privateMethod--\">");
79 
80         // The private method in different package should not be overridden
81         checkOutput("pkg2/SubClass.html", false,
82                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
83                 + "<dd><code><a href=\"../pkg1/BaseClass.html#privateMethod--\">");
84 
85         // The package private method should not be overridden since the base and sub class are in
86         // different packages.
87         checkOutput("pkg2/SubClass.html", false,
88                 "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
89                 + "<dd><code><a href=\"../pkg1/BaseClass.html#packagePrivateMethod--\">");
90     }
91 }
