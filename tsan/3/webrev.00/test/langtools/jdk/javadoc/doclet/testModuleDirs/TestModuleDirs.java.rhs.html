 24 /*
 25  * @test
 26  * @bug 8195795 8201396 8196202 8215582
 27  * @summary test the use of module directories in output,
 28  *          and the --no-module-directories option
 29  * @modules jdk.javadoc/jdk.javadoc.internal.api
 30  *          jdk.javadoc/jdk.javadoc.internal.tool
 31  *          jdk.compiler/com.sun.tools.javac.api
 32  *          jdk.compiler/com.sun.tools.javac.main
 33  * @library ../../lib /tools/lib
 34  * @build toolbox.ToolBox toolbox.ModuleBuilder javadoc.tester.*
 35  * @run main TestModuleDirs
 36  */
 37 
 38 import java.io.IOException;
 39 import java.nio.file.Path;
 40 import java.nio.file.Paths;
 41 
 42 import javadoc.tester.JavadocTester;
 43 import toolbox.ModuleBuilder;
 44 import toolbox.ToolBox;
 45 
 46 public class TestModuleDirs extends JavadocTester {
 47 
 48     public final ToolBox tb;
 49     public static void main(String... args) throws Exception {
 50         TestModuleDirs tester = new TestModuleDirs();
 51         tester.runTests(m -> new Object[] { Paths.get(m.getName()) });
 52     }
 53 
 54     public TestModuleDirs() {
 55         tb = new ToolBox();
 56     }
 57 
 58     @Test
 59     public void testNoModules(Path base) throws IOException {
 60         Path src = base.resolve("src");
 61         tb.writeJavaFiles(src, "package p; public class C { }");
 62 
 63         javadoc("-d", base.resolve("api").toString(),
 64                 "-sourcepath", src.toString(),
 65                 "-quiet",
 66                 "p");
 67 
 68         checkExit(Exit.OK);
 69         checkFiles(true, "p/package-summary.html");
 70     }
 71 
 72     @Test
 73     public void testNoModuleDirs(Path base) throws IOException {
 74         Path src = base.resolve("src");
 75         new ModuleBuilder(tb, "ma")
 76                 .classes("package pa; public class A {}")
 77                 .exports("pa")
 78                 .write(src);




 79 
 80         javadoc("-d", base.resolve("api").toString(),


 81                 "--module-source-path", src.toString(),
 82                 "--no-module-directories",
 83                 "--module", "ma,mb");
 84         checkExit(Exit.ERROR);























 85     }
 86 
 87     @Test
 88     public void testModuleDirs(Path base) throws IOException {
 89         Path src = base.resolve("src");
 90         new ModuleBuilder(tb, "ma")
 91                 .classes("package pa; public class A {}")
 92                 .exports("pa")
 93                 .write(src);
 94         new ModuleBuilder(tb, "mb")
 95                 .classes("package pb; public class B {}")
 96                 .exports("pb")
 97                 .write(src);
 98 
 99         javadoc("-d", base.resolve("api").toString(),
100                 "-quiet",

101                 "--module-source-path", src.toString(),
102                 "--module", "ma,mb");
103 
104         checkExit(Exit.OK);
105         checkFiles(false,

106                 "ma-summary.html",
107                 "pa/package-summary.html");
108         checkFiles(true,

109                 "ma/module-summary.html",
110                 "ma/pa/package-summary.html");






111         checkOutput("ma/module-summary.html", false,
112                 "<ul class=\"navList\" id=\"allclasses_navbar_top\">\n"
113                 + "<li><a href=\"../allclasses-noframe.html\">All&nbsp;Classes</a></li>\n"
114                 + "</ul>\n");
115         checkOutput("ma/pa/package-summary.html", true,
116                 "<li><a href=\"../../deprecated-list.html\">Deprecated</a></li>\n"
117                 + "<li><a href=\"../../index-all.html\">Index</a></li>");
118     }
119 }
