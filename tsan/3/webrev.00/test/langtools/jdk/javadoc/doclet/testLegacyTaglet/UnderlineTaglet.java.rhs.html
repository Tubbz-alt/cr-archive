24 import java.util.EnumSet;
25 import java.util.List;
26 import java.util.Set;
27 import javax.lang.model.element.Element;
28 
29 import com.sun.source.doctree.DocTree;
30 import jdk.javadoc.doclet.Taglet;
31 import static jdk.javadoc.doclet.Taglet.Location.*;
32 
33 /**
34  * A sample Inline Taglet representing {@underline ...}.  The text
35  * is simple underlined.  For example, "@underline UNDERLINE ME" would
36  * be shown as: <u>UNDERLINE ME</u>.



37  */
38 
39 public class UnderlineTaglet implements Taglet {
40 
41     private final String NAME = "underline";
42 
43     /**
44      * Return the name of this custom tag.
45      */
46     @Override
47     public String getName() {
48         return NAME;
49     }
50     private final EnumSet<Location> allowedSet = EnumSet.of(CONSTRUCTOR);
51 
52     @Override
53     public Set<Taglet.Location> getAllowedLocations() {
54         return allowedSet;
55     }
56 
57     /**
58      * Will return true since this is an inline tag.
59      * @return true since this is an inline tag.
60      */
61 
62     @Override
63     public boolean isInlineTag() {
64         return true;
65     }
66 
67     /**
68      * Given the <code>DocTree</code> representation of this custom
69      * tag, return its string representation.
70      * @param tags the <code>DocTree</code> representation of this custom tag.
71      * @param element the declaration to which the enclosing comment belongs
72      */
73     @Override
74     public String toString(List<? extends DocTree> tags, Element element) {
75         return "<u>" + ToDoTaglet.getText(tags.get(0)) + "</u>";
76     }
77 }
