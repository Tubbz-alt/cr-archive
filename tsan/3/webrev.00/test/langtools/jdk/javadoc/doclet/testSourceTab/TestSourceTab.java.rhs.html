 24 /*
 25  * @test
 26  * @bug 4510979
 27  * @summary Test to make sure that the source documentation is indented properly
 28  * when -linksourcetab is used.

 29  * @library ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build javadoc.tester.*
 32  * @run main TestSourceTab
 33  */
 34 
 35 import java.io.*;
 36 
 37 import javadoc.tester.JavadocTester;
 38 
 39 public class TestSourceTab extends JavadocTester {
 40 
 41     public static void main(String... args) throws Exception {
 42         TestSourceTab tester = new TestSourceTab();
 43         tester.runTests();
 44     }
 45 
 46     @Test
 47     public void test() throws Exception {
 48         String tmpSrcDir = "tmpSrc";
 49         String outdir1 = "out-tabLengthEight";
 50         String outdir2 = "out-tabLengthFour";
 51         initTabs(new File(testSrc), new File(tmpSrcDir));
 52 
 53         // Run Javadoc on a source file with that is indented with a single tab per line
 54         javadoc("-d", outdir1,
 55                 "-sourcepath", tmpSrcDir,
 56                 "-notimestamp",
 57                 "-linksource",
 58                 tmpSrcDir + "/SingleTab/C.java");
 59         checkExit(Exit.OK);
 60 
 61         // Run Javadoc on a source file with that is indented with a two tab per line
 62         // If we double the tabs and decrease the tab length by a half, the output should
 63         // be the same as the one generated above.
 64         javadoc("-d", outdir2,
 65                 "-sourcepath", tmpSrcDir,
 66                 "-notimestamp",
 67                 "-sourcetab", "4",
 68                 tmpSrcDir + "/DoubleTab/C.java");
 69         checkExit(Exit.OK);
 70 
 71         diff(outdir1, outdir2,
 72                 "src-html/C.html",
 73                 "C.html");
 74     }
 75 
 76     void initTabs(File from, File to) throws IOException {
 77         for (File f: from.listFiles()) {
 78             File t = new File(to, f.getName());
 79             if (f.isDirectory()) {
 80                 initTabs(f, t);
 81             } else if (f.getName().endsWith(".java")) {
 82                 write(t, read(f).replace("\\t", "\t"));
 83             }
 84         }
 85     }
 86 
 87     String read(File f) throws IOException {
 88         StringBuilder sb = new StringBuilder();
 89         try (BufferedReader in = new BufferedReader(new FileReader(f))) {
 90             String line;
 91             while ((line = in.readLine()) != null) {
 92                 sb.append(line);
 93                 sb.append(NL);
 94             }
 95         }
 96         return sb.toString();
 97     }
 98 
 99     void write(File f, String s) throws IOException {
100         f.getParentFile().mkdirs();
101         try (Writer out = new FileWriter(f)) {
102             out.write(s);
103         }
104     }
105 }
