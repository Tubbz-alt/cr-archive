24 /*
25  * @test
26  * @bug      4515705 4804296 4702454 4697036 8025633 8182765
27  * @summary  Make sure that first sentence warning only appears once.
28  *           Make sure that only warnings/errors are printed when quiet is used.
29  *           Make sure that links to private/unincluded methods do not cause
30  *           a "link unresolved" warning.
31  *           Make sure error message starts with "error -".
32  * @author   jamieh
33  * @library  ../../lib
34  * @modules jdk.javadoc/jdk.javadoc.internal.tool
35  * @build    javadoc.tester.*
36  * @run main TestWarnings
37  */
38 
39 import javadoc.tester.JavadocTester;
40 
41 public class TestWarnings extends JavadocTester {
42     public static void main(String... args) throws Exception  {
43         TestWarnings tester = new TestWarnings();
44         tester.runTests();
45     }
46 
47     @Test
48     public void testDefault() {
49         javadoc("-d", "out-default",
50                 "-sourcepath", testSrc,
51                 "pkg");
52         checkExit(Exit.ERROR);
53 
54         checkOutput(Output.OUT, true,
55                 "X.java:23: error: self-closing element not allowed");
56 
57         checkOutput(Output.OUT, true,
58                 "X.java:24: error: self-closing element not allowed");
59 
60         checkOutput(Output.OUT, true,
61                 "X.java:25: error: self-closing element not allowed");
62 
63         checkOutput(Output.OUT, true,
64                 "X.java:26: error: self-closing element not allowed");
65 
66         checkOutput("pkg/X.html", false,
67                 "can't find m()");
68         checkOutput("pkg/X.html", false,
69                 "can't find X()");
70         checkOutput("pkg/X.html", false,
71                 "can't find f");
72     }
73 
74     @Test
75     public void testPrivate() {
76         javadoc("-d", "out-private",
77                 "-private",
78                 "-sourcepath", testSrc,
79                 "pkg");
80         checkExit(Exit.ERROR);
81 
82         checkOutput("pkg/X.html", true,
83             "<a href=\"#m()\"><code>m()</code></a><br/>",
84             "<a href=\"#%3Cinit%3E()\"><code>X()</code></a><br/>",
85             "<a href=\"#f\"><code>f</code></a><br/>");
86     }
87 }
