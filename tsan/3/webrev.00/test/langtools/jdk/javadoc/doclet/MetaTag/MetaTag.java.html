 24 /*
 25  * @test
 26  * @bug      4034096 4764726 6235799 8182765 8196202
 27  * @summary  Add support for HTML keywords via META tag for
 28  *           class and member names to improve API search
 29  * @library ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build    javadoc.tester.*
 32  * @run main MetaTag
 33  */
 34 
 35 import java.text.SimpleDateFormat;
 36 import java.util.Date;
 37 
 38 import javadoc.tester.JavadocTester;
 39 
 40 public class MetaTag extends JavadocTester {
 41 
 42     /**
 43      * The entry point of the test.
 44      * @param args the array of command line arguments
 45      * @throws Exception if the test fails
 46      */
 47     public static void main(String... args) throws Exception {
 48         MetaTag tester = new MetaTag();
 49         tester.runTests();
 50     }
 51 
 52     @Test
 53     public void testStandard() {
 54         javadoc("-d", "out-1",
 55                 "-sourcepath", testSrc,
 56                 "-keywords",
 57                 "-doctitle", "Sample Packages",
 58                 "p1", "p2");
 59 
 60         checkExit(Exit.OK);
 61 
 62         checkMeta("dc.created", true);
 63     }
 64 
 65     @Test
 66     public void testNoTimestamp() {
 67         javadoc("-d", "out-2",
 68                 "-sourcepath", testSrc,
 69                 "-notimestamp",
 70                 "-doctitle", "Sample Packages",
 71                 "p1", "p2");
 72         checkExit(Exit.OK);
 73 
 74         // No keywords when -keywords is not used.
 75         checkMeta("dc.created", false);
 76     }
 77 
 78     void checkMeta(String metaNameDate, boolean found) {
 79         checkOutput("p1/C1.html", found,
 80                 "<meta name=\"keywords\" content=\"p1.C1 class\">",
 81                 "<meta name=\"keywords\" content=\"field1\">",
 82                 "<meta name=\"keywords\" content=\"field2\">",
 83                 "<meta name=\"keywords\" content=\"method1()\">",
 84                 "<meta name=\"keywords\" content=\"method2()\">");
 85 
 86         checkOutput("p1/package-summary.html", found,
 87                 "<meta name=\"keywords\" content=\"p1 package\">");
 88 
 89         checkOutput("index.html", found,
 90                 "<meta name=\"keywords\" content=\"Overview, Sample Packages\">");
 91 
 92         // NOTE: Hopefully, this regression test is not run at midnight.  If the output
 93         // was generated yesterday and this test is run today, this check could fail ...
 94         // so make sure the date has not changed since the test started
 95         String date = date();
 96         if (date.equals(startDate)) {
 97             checkOutput("index.html", found,
 98                     "<meta name=\"" + metaNameDate + "\" content=\"" + date + "\">");
 99         }
100     }
101 
102     private static final SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
103     private static final String startDate = date();
104 
105     static String date() {
106         return dateFormat.format(new Date());
107     }
108 }
