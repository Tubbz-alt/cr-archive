24 /*
25  * @test
26  * @bug 8173302 8182765 8196202 8210047
27  * @summary make sure the overview-summary and module-summary pages don't
28  *          don't have the See link, and the overview is copied correctly.
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestOverview
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestOverview extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestOverview tester = new TestOverview();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test1() {
46         javadoc("-d", "out-1",
47                     "-doctitle", "Document Title",
48                     "-windowtitle", "Window Title",
49                     "-overview", testSrc("overview.html"),
50                     "--frames",
51                     "-sourcepath", testSrc("src"),
52                     "p1", "p2");
53         checkExit(Exit.OK);
54         checkOverview();
55     }
56 
57     @Test
58     public void test2() {
59         javadoc("-d", "out-2",
60                     "-doctitle", "Document Title",
61                     "-windowtitle", "Window Title",
62                     "-overview", testSrc("overview.html"),
63                     "--frames",
64                     "-sourcepath", testSrc("msrc"),
65                     "p1", "p2");
66         checkExit(Exit.OK);
67         checkOverview();
68     }
69 
70     void checkOverview() {
71         checkOutput("overview-summary.html", true,
72                 "<main role=\"main\">\n"
73                 + "<div class=\"header\">\n"
74                 + "<h1 class=\"title\">Document Title</h1>\n"
75                 + "</div>\n"
76                 + "<div class=\"contentContainer\">\n"
77                 + "<div class=\"block\">This is line1. This is line 2.</div>\n"
78                 + "</div>\n"
79                 + "<div class=\"contentContainer\">");
80     }
81 }
