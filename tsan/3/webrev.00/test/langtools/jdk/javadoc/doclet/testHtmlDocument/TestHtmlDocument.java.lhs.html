 24 /*
 25  * @test
 26  * @bug 6851834
 27  * @summary This test verifies the HTML document generation for javadoc output.
 28  * @library ../../lib
 29  * @modules jdk.javadoc/jdk.javadoc.internal.doclets.formats.html.markup
 30  *          jdk.javadoc/jdk.javadoc.internal.doclets.toolkit.util
 31  *          jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build javadoc.tester.*
 33  * @author Bhavesh Patel
 34  * @run main TestHtmlDocument
 35  */
 36 
 37 
 38 import jdk.javadoc.internal.doclets.formats.html.markup.*;
 39 
 40 import static jdk.javadoc.internal.doclets.toolkit.util.DocletConstants.NL;
 41 
 42 /**
 43  * The class reads each file, complete with newlines, into a string to easily
 44  * compare the existing markup with the generated markup.
 45  */
 46 import javadoc.tester.JavadocTester;
 47 
 48 public class TestHtmlDocument extends JavadocTester {
 49 
 50     // Entry point
 51     public static void main(String... args) throws Exception {
 52         TestHtmlDocument tester = new TestHtmlDocument();
 53         tester.runTests();
 54     }
 55 
 56     @Test
 57     public void test() {
 58         checking("markup");
 59         // Check whether the generated markup is same as the existing markup.
 60         String expected = readFile(testSrc, "testMarkup.html").replace("\n", NL);
 61         String actual = generateHtmlTree();
 62         if (actual.equals(expected)) {
 63             passed("");
 64         } else {
 65             failed("expected content in " + testSrc("testMarkup.html") + "\n"
 66                 + "Actual output:\n"
 67                 + actual);
 68         }
 69     }
 70 
 71     // Generate the HTML output using the HTML document generation within doclet.
 72     public static String generateHtmlTree() {
 73         // Document type for the HTML document
 74         HtmlTree html = new HtmlTree(HtmlTag.HTML);
 75         HtmlTree head = new HtmlTree(HtmlTag.HEAD);
 76         HtmlTree title = new HtmlTree(HtmlTag.TITLE);
 77         // String content within the document
 78         StringContent titleContent = new StringContent("Markup test");
 79         title.add(titleContent);
 80         head.add(title);
 81         // Test META tag
 82         HtmlTree meta = new HtmlTree(HtmlTag.META);
 83         meta.put(HtmlAttr.NAME, "keywords");
 84         meta.put(HtmlAttr.CONTENT, "testContent");
 85         head.add(meta);
 86         // Test invalid META tag
 87         HtmlTree invmeta = new HtmlTree(HtmlTag.META);
 88         head.add(invmeta);
 89         // Test LINK tag
 90         HtmlTree link = new HtmlTree(HtmlTag.LINK);
 91         link.put(HtmlAttr.REL, "testRel");
 92         link.put(HtmlAttr.HREF, "testLink.html");
 93         head.add(link);
 94         // Test invalid LINK tag
 95         HtmlTree invlink = new HtmlTree(HtmlTag.LINK);
 96         head.add(invlink);
 97         html.add(head);
 98         // Comment within the document
 99         Comment bodyMarker = new Comment("======== START OF BODY ========");
100         html.add(bodyMarker);
101         HtmlTree body = new HtmlTree(HtmlTag.BODY);
102         Comment pMarker = new Comment("======== START OF PARAGRAPH ========");
103         body.add(pMarker);
104         HtmlTree p = new HtmlTree(HtmlTag.P);
105         StringContent bodyContent = new StringContent(
106                 "This document is generated from sample source code and HTML " +
107                 "files with examples of a wide variety of Java language constructs: packages, " +
108                 "subclasses, subinterfaces, nested classes, nested interfaces," +
109                 "inheriting from other packages, constructors, fields," +
110                 "methods, and so forth. ");
111         p.add(bodyContent);
112         StringContent anchorContent = new StringContent("Click Here");
113         p.add(HtmlTree.A("testLink.html", anchorContent));
114         StringContent pContent = new StringContent(" to <test> out a link.");
115         p.add(pContent);
116         body.add(p);
117         HtmlTree p1 = new HtmlTree(HtmlTag.P);
118         // Test another version of A tag.
119         HtmlTree anchor = new HtmlTree(HtmlTag.A);
120         anchor.put(HtmlAttr.HREF, "testLink.html");
121         anchor.put(HtmlAttr.ID, "Another version of a tag");
122         p1.add(anchor);
123         body.add(p1);
124         // Test for empty tags.
125         HtmlTree dl = new HtmlTree(HtmlTag.DL);
126         html.add(dl);
127         // Test for empty nested tags.
128         HtmlTree dlTree = new HtmlTree(HtmlTag.DL);
129         dlTree.add(new HtmlTree(HtmlTag.DT));
130         dlTree.add(new HtmlTree (HtmlTag.DD));
131         html.add(dlTree);
132         HtmlTree dlDisplay = new HtmlTree(HtmlTag.DL);
133         dlDisplay.add(new HtmlTree(HtmlTag.DT));
134         HtmlTree dd = new HtmlTree (HtmlTag.DD);
135         StringContent ddContent = new StringContent("Test DD");
136         dd.add(ddContent);
137         dlDisplay.add(dd);
138         body.add(dlDisplay);
139         StringContent emptyString = new StringContent("");
140         body.add(emptyString);
141         Comment emptyComment = new Comment("");
142         body.add(emptyComment);
143         HtmlTree hr = new HtmlTree(HtmlTag.HR);
144         body.add(hr);
145         html.add(body);
146         HtmlDocument htmlDoc = new HtmlDocument(html);
147         return htmlDoc.toString();
148     }
149 }
