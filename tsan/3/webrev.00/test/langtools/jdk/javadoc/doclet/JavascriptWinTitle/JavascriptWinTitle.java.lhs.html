24 /*
25  * @test
26  * @bug 4645058 4747738 4855054 8024756 8141492 8196202 8205593
27  * @summary  Javascript IE load error when linked by -linkoffline
28  *           Window title shouldn't change when loading left frames (javascript)
29  * @author dkramer
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main JavascriptWinTitle
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class JavascriptWinTitle extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         JavascriptWinTitle tester = new JavascriptWinTitle();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-source", "8",
49                 "--frames",
50                 "-doctitle", "Document Title",
51                 "-windowtitle", "Window Title",
52                 "-overview", testSrc("overview.html"),
53                 "-linkoffline", "http://java.sun.com/j2se/1.4/docs/api", testSrc,
54                 "-sourcepath", testSrc,
55                 "p1", "p2");
56         checkExit(Exit.OK);
57         checkOutput("overview-summary.html", true,
58                 "<script type=\"text/javascript\">",
59                 "<body class=\"package-index\">");
60 
61         // Test that "onload" is not present in BODY tag:
62         checkOutput("p1/package-summary.html", true, "<body class=\"package-declaration\">");
63         checkOutput("overview-frame.html", true, "<body class=\"package-index-frame\">");
64         checkOutput("allclasses-frame.html", true, "<body class=\"all-classes-frame\">");
65         checkOutput("p1/package-frame.html", true, "<body class=\"package-frame\">");
66 
67         // Test that win title javascript is followed by NOSCRIPT code.
68         checkOutput("p1/C.html", true,
69                 "<script type=\"text/javascript\"><!--\n"
70                 + "    try {\n"
71                 + "        if (location.href.indexOf('is-external=true') == -1) {\n"
72                 + "            parent.document.title=\"C (Window Title)\";\n"
73                 + "        }\n"
74                 + "    }\n"
75                 + "    catch(err) {\n"
76                 + "    }\n"
77                 + "//-->\n");
78     }
79 }
