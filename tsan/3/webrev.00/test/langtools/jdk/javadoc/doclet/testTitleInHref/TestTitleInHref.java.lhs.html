24 /*
25  * @test
26  * @bug 4714257 8164407 8205593
27  * @summary Test to make sure that the title attribute shows up in links.
28  * @author jamieh
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestTitleInHref
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestTitleInHref extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestTitleInHref tester = new TestTitleInHref();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         String uri = "http://java.sun.com/j2se/1.4/docs/api";
47         javadoc("-d", "out",
48                 "-source", "8",
49                 "-sourcepath", testSrc,
50                 "-linkoffline", uri, testSrc,
51                 "pkg");
52         checkExit(Exit.OK);
53 
54         checkOutput("pkg/Links.html", true,
55                 //Test to make sure that the title shows up in a class link.
56                 "<a href=\"Class.html\" title=\"class in pkg\">",
57                 //Test to make sure that the title shows up in an interface link.
58                 "<a href=\"Interface.html\" title=\"interface in pkg\">",
59                 //Test to make sure that the title shows up in cross link shows up
60                 "<a href=\"" + uri + "/java/io/File.html?is-external=true\" "
61                 + "title=\"class or interface in java.io\" class=\"externalLink\">"
62                 + "<code>This is a cross link to class File</code></a>");
63     }
64 }
