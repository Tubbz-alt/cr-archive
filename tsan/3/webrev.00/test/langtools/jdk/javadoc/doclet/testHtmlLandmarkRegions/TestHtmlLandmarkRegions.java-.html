 38 import java.nio.file.Files;
 39 import java.nio.file.Path;
 40 import java.nio.file.Paths;
 41 import java.util.List;
 42 
 43 import builder.ClassBuilder;
 44 import toolbox.ModuleBuilder;
 45 import toolbox.ToolBox;
 46 
 47 import javadoc.tester.JavadocTester;
 48 
 49 public class TestHtmlLandmarkRegions extends JavadocTester {
 50 
 51     final ToolBox tb;
 52 
 53     public static void main(String... args) throws Exception {
 54         TestHtmlLandmarkRegions tester = new TestHtmlLandmarkRegions();
 55         tester.runTests(m -> new Object[]{Paths.get(m.getName())});
 56     }
 57 
 58     TestHtmlLandmarkRegions() {
 59         tb = new ToolBox();
 60         setAutomaticCheckLinks(false); // @ignore 8217013
 61     }
 62 
 63     @Test
 64     public void testModules(Path base) throws Exception {
 65         Path srcDir = base.resolve("src");
 66         createModules(srcDir);
 67 
 68         Path outDir = base.resolve("out");
 69         javadoc("-d", outDir.toString(),
 70                 "-doctitle", "Document Title",
 71                 "-header", "Test Header",
 72                 "--frames",
 73                 "--module-source-path", srcDir.toString(),
 74                 "--module", "m1,m2");
 75 
 76         checkExit(Exit.OK);
 77 
 78         checkOrder("module-overview-frame.html",
 79                 "<header role=\"banner\">\n"
 80                 + "<h1 title=\"Test Header\" class=\"bar\">Test Header</h1>\n"
 81                 + "<nav role=\"navigation\" class=\"indexNav\">",
 82                 "<main role=\"main\">\n"
 83                 + "<div class=\"indexContainer\">\n"
 84                 + "<h2 title=\"Modules\">Modules</h2>\n"
 85                 + "<ul title=\"Modules\">",
 86                 "<footer role=\"contentinfo\">");
 87 
 88         checkOrder("m1/module-frame.html",
 89                 "<header role=\"banner\">\n"
 90                 + "<h1 title=\"Test Header\" class=\"bar\">Test Header</h1>\n"
 91                 + "<nav role=\"navigation\" class=\"indexNav\">",
 92                 "<main role=\"main\">\n"
 93                 + "<div class=\"indexContainer\">\n"
 94                 + "<h2 title=\"m1\"><a href=\"module-summary.html\" target=\"classFrame\">m1</a>&nbsp;Packages</h2>",
 95                 "<footer role=\"contentinfo\">");
 96 
 97         checkOrder("overview-summary.html",
 98                 "<header role=\"banner\">\n"
 99                 + "<nav role=\"navigation\">",
100                 "<main role=\"main\">\n"
101                 + "<div class=\"header\">\n"
102                 + "<h1 class=\"title\">Document Title</h1>",
103                 "<footer role=\"contentinfo\">\n"
104                 + "<nav role=\"navigation\">");
105     }
106 
107     @Test
108     public void testPackages(Path base) throws Exception {
109         Path srcDir = base.resolve("src");
110         createPackages(srcDir);
111 
112         Path outDir = base.resolve("out");
113         javadoc("-d", outDir.toString(),
114                 "-doctitle", "Document Title",
115                 "-header", "Test Header",
116                 "--frames",
117                 "-sourcepath", srcDir.toString(),
118                 "pkg1", "pkg2");
119 
120         checkExit(Exit.OK);
121 
122         checkOrder("overview-summary.html",
123                 "<header role=\"banner\">\n"
124                 + "<nav role=\"navigation\">",
125                 "<main role=\"main\">\n"
126                 + "<div class=\"header\">\n"
127                 + "<h1 class=\"title\">Document Title</h1>",
128                 "<footer role=\"contentinfo\">\n" +
129                         "<nav role=\"navigation\">");
130 
131         checkOrder("overview-frame.html",
132                 "<header role=\"banner\">\n"
133                 + "<h1 title=\"Test Header\" class=\"bar\">Test Header</h1>\n"
134                 + "<nav role=\"navigation\" class=\"indexNav\">",
135                 "<main role=\"main\">\n"
136                 + "<div class=\"indexContainer\">\n"
137                 + "<h2 title=\"Packages\">Packages</h2>",
138                 "<footer role=\"contentinfo\">");
139     }
140 
141     @Test
142     public void testDocFiles(Path base) throws Exception {
143         Path srcDir = base.resolve("src");
144         createPackages(srcDir);
145         Path docFiles = Files.createDirectory(srcDir.resolve("pkg1").resolve("doc-files"));
146         Files.write(docFiles.resolve("s.html"), List.of(
147                 "<html>\n"
148                 + "  <head>\n"
149                 + "    <title>\"Hello World\"</title>\n"
150                 + "  </head>\n"
151                 + "  <body>\n"
152                 + "     A sample doc file.\n"
153                 + "  </body>\n"
154                 + "</html>"));
155 
156         Path outDir = base.resolve("out");
157         javadoc("-d", outDir.toString(),
158                 "-sourcepath", srcDir.toString(),
159                 "pkg1", "pkg2");
160 
161         checkExit(Exit.OK);
162 
163         checkOrder("pkg1/doc-files/s.html",
164                 "<header role=\"banner\">\n"
165                 + "<nav role=\"navigation\">\n",
166                 "<main role=\"main\">A sample doc file",
167                 "<footer role=\"contentinfo\">\n"
168                 + "<nav role=\"navigation\">"
169                 );
170     }
171 
172     void createModules(Path srcDir) throws Exception {
173         new ModuleBuilder(tb, "m1")
174                 .classes("package p1; public class a { }")
175                 .classes("package p2; public class b { }")
176                 .write(srcDir);
177         new ModuleBuilder(tb, "m2")
178                 .classes("package p3; public class c { }")
179                 .classes("package p4; public class d { }")
180                 .write(srcDir);
181     }
182 
183     void createPackages(Path srcDir) throws Exception {
184         new ClassBuilder(tb, "pkg1.A")
185                 .setModifiers("public", "class")
186                 .write(srcDir);
187         new ClassBuilder(tb, "pkg2.B")
188                 .setModifiers("public", "class")
189                 .write(srcDir);
190     }
