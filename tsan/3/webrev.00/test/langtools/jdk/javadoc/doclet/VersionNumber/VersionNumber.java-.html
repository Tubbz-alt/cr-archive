36 /**

40 
41 public class VersionNumber extends JavadocTester {
42 
43     public static void main(String... args) throws Exception {
44         VersionNumber tester = new VersionNumber();
45         tester.runTests();
46     }
47 
48     @Test
49     public void test() {
50         javadoc("-d", "out",
51                 "p1");
52         checkExit(Exit.OK);

54         // Test the proper DOCTYPE element is present:
55         checkOutput("p1/C.html", true,
56                 "<!-- Generated by javadoc (");
57     }
