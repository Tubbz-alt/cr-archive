24 /*
25  * @test
26  * @summary Determine if proper warning messages are printed.

27  * @library ../../lib
28  * @modules jdk.javadoc/jdk.javadoc.internal.tool
29  * @build javadoc.tester.*
30  * @build TestTagMisuse
31  * @run main TestTagMisuse
32  */
33 import javadoc.tester.JavadocTester;
34 
35 public class TestTagMisuse extends JavadocTester {
36 
37     /**
38      * The entry point of the test.
39      * @param args the array of command line arguments.
40      * @throws Exception if the test fails
41      */
42     public static void main(String... args) throws Exception {
43         TestTagMisuse tester = new TestTagMisuse();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         javadoc("-Xdoclint:none",
50                 "-d", "out",
51                 testSrc("TestTagMisuse.java"));
52         checkExit(Exit.OK);
53 
54         checkOutput(Output.OUT, true,
55                 "warning - Tag @param cannot be used in field documentation.",
56                 "warning - Tag @throws cannot be used in field documentation.",
57                 "warning - Tag @return cannot be used in constructor documentation."
58                 /* DCerroneous, "warning - Tag @throws cannot be used in inline documentation."*/);
59         checkOutput(Output.OUT, false, "DocletAbortException");
60     }
61 
62     /**
63      * {@throws blah}
64      * Here is a bad field tag:
65      * @throws foo
66      * @param foo.
67      */
68     public int field;
69 
70     /**
71      * Here is a bad constructor tag:
72      * @return blah
73      */
74     public TestTagMisuse(){}
75 
76     /**
77      * for @see and {@link}), and ThrowsTag (for @throws).
78      */
79     public void thisShouldNotCrash() {}
80 }
