 24 /*
 25  * @test
 26  * @bug      4034096 4764726 6235799 8182765 8196202
 27  * @summary  Add support for HTML keywords via META tag for
 28  *           class and member names to improve API search
 29  * @author   dkramer
 30  * @library ../../lib
 31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 32  * @build    javadoc.tester.*
 33  * @run main MetaTag
 34  */
 35 
 36 import java.text.SimpleDateFormat;
 37 import java.util.Date;
 38 
 39 import javadoc.tester.JavadocTester;
 40 
 41 public class MetaTag extends JavadocTester {
 42 
 43     /**
 44      * The entry point of the test.
 45      * @param args the array of command line arguments
 46      * @throws Exception if the test fails
 47      */
 48     public static void main(String... args) throws Exception {
 49         MetaTag tester = new MetaTag();
 50         tester.runTests();
 51     }
 52 
 53     @Test
 54     public void testStandard() {
 55         javadoc("-d", "out-1",
 56                 "-sourcepath", testSrc,
 57                 "-keywords",
 58                 "--frames",
 59                 "-doctitle", "Sample Packages",
 60                 "p1", "p2");
 61 
 62         checkExit(Exit.OK);
 63 
 64         checkMeta("dc.created", true);
 65     }
 66 
 67     @Test
 68     public void testNoTimestamp() {
 69         javadoc("-d", "out-2",
 70                 "-sourcepath", testSrc,
 71                 "-notimestamp",
 72                 "--frames",
 73                 "-doctitle", "Sample Packages",
 74                 "p1", "p2");
 75         checkExit(Exit.OK);
 76 
 77         // No keywords when -keywords is not used.
 78         checkMeta("dc.created", false);
 79     }
 80 
 81     void checkMeta(String metaNameDate, boolean found) {
 82         checkOutput("p1/C1.html", found,
 83                 "<meta name=\"keywords\" content=\"p1.C1 class\">",
 84                 "<meta name=\"keywords\" content=\"field1\">",
 85                 "<meta name=\"keywords\" content=\"field2\">",
 86                 "<meta name=\"keywords\" content=\"method1()\">",
 87                 "<meta name=\"keywords\" content=\"method2()\">");
 88 
 89         checkOutput("p1/package-summary.html", found,
 90                 "<meta name=\"keywords\" content=\"p1 package\">");
 91 
 92         checkOutput("overview-summary.html", found,
 93                 "<meta name=\"keywords\" content=\"Overview, Sample Packages\">");
 94 
 95         // NOTE: Hopefully, this regression test is not run at midnight.  If the output
 96         // was generated yesterday and this test is run today, the test will fail.
 97         checkOutput("overview-summary.html", found,
 98                 "<meta name=\"" + metaNameDate + "\" content=\"" + date() + "\">");
 99     }
100 
101     private static final SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd");
102 
103     String date() {
104         return dateFormat.format(new Date());
105     }
106 }
