24 /*
25  * @test
26  * @bug 4530730 8196202
27  * @summary stddoclet: With frames off, window titles have "()" appended

28  * @library ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.*
31  * @run main WindowTitles
32  */
33 
34 /**
35  * Runs javadoc and runs regression tests on the resulting HTML.
36  */
37 import javadoc.tester.JavadocTester;
38 
39 public class WindowTitles extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         WindowTitles tester = new WindowTitles();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         // Test for all cases except the split index page
49         javadoc("-d", "out-1",
50                 "-use",

51                 "-sourcepath", testSrc,
52                 "p1", "p2");
53         checkExit(Exit.OK);
54         checkFiles(false, "allclasses-noframe.html");
55 
56         checkTitle("index.html",                "Overview");
57         checkTitle("overview-tree.html",        "Class Hierarchy");

58         checkTitle("p1/package-summary.html",   "p1");

59         checkTitle("p1/package-tree.html",      "p1 Class Hierarchy");
60         checkTitle("p1/package-use.html",       "Uses of Package p1");
61         checkTitle("p1/C1.html",                "C1");

62         checkTitle("constant-values.html",      "Constant Field Values");
63         checkTitle("deprecated-list.html",      "Deprecated List");
64         checkTitle("serialized-form.html",      "Serialized Form");
65         checkTitle("help-doc.html",             "API Help");
66         checkTitle("index-all.html",            "Index");
67         checkTitle("p1/class-use/C1.html",      "Uses of Class p1.C1");
68     }
69 
70     @Test
71     public void test2() {
72         // Test only for the split-index case (and run on only one package)
73         javadoc("-d", "out-2",
74                 "-splitindex",
75                 "-sourcepath", testSrc,
76                 "p1");
77         checkExit(Exit.OK);
78 
79         checkTitle("index-files/index-1.html", "C-Index");
80     }
81 
82     void checkTitle(String file, String title) {
83         checkOutput(file, true, "<title>" + title + "</title>");
84     }
85 
86 }
