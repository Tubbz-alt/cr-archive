24 /*
25  * @test
26  * @bug 8026370 8026567 8183511 8074407
27  * @summary This test checks the generated tag output.
28  * @author Bhavesh Patel
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestTagOutput
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestTagOutput extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestTagOutput tester = new TestTagOutput();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "pkg1");
49         checkExit(Exit.OK);
50 
51         checkOutput("pkg1/DeprecatedTag.html", true,
52                 "<div class=\"deprecationBlock\"><span class=\"deprecatedLabel\">Deprecated.</span></div>",
53                 "<div class=\"deprecationBlock\"><span class=\"deprecatedLabel\">Deprecated.</span>\n"
54                 + "<div class=\"deprecationComment\">Do not use this.</div>\n"
55                 + "</div>");
56 
57         checkOutput("pkg1/DeprecatedTag.html", false,
58                 "<div class=\"deprecationBlock\"><span class=\"deprecatedLabel\">Deprecated.</span>\n"
59                 + "<div class=\"deprecationComment\"></div>\n"
60                 + "</div>");
61     }
62 }
