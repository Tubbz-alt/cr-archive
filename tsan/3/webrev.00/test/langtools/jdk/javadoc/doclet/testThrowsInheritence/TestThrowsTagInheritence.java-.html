47     @Test
48     public void test() {
49         javadoc("-d", "out",
50                 "-sourcepath", testSrc,

52                 testSrc("I.java"),
53                 testSrc("Foo.java"),
54                 testSrc("Iface.java"));

57         // The class should not inherit the tag from the interface.
58         checkOutput("Foo.html", true,

63                 "Test 1 fails.");
64     }
