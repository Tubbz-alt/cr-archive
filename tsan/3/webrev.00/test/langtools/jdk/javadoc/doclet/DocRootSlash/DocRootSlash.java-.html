 24 /*
 25  * @test
 26  * @bug 4524350 4662945 4633447 8196202
 27  * @summary stddoclet: {@docRoot} inserts an extra trailing "/"
 28  * @author dkramer

 31  * @build javadoc.tester.*
 32  * @run main DocRootSlash

 34 
 35 import java.util.regex.*;
 36 
 37 /**
 38  * Runs javadoc and runs regression tests on the resulting HTML.
 39  * It reads each file, complete with newlines, into a string to easily
 40  * find strings that contain newlines.
 41  */
 42 import javadoc.tester.JavadocTester;
 43 
 44 public class DocRootSlash extends JavadocTester {
 45 
 46     public static void main(String... args) throws Exception {
 47         DocRootSlash tester = new DocRootSlash();
 48         tester.runTests();
 49     }
 50 
 51     @Test
 52     public void test() {
 53         // Directory that contains source files that javadoc runs on
 54         String srcdir = System.getProperty("test.src", ".");
 55 
 56 
 57         javadoc("-d", "out",
 58                 "-Xdoclint:none",
 59                 "-overview", (srcdir + "/overview.html"),
 60                 "--frames",
 61                 "-header", "<A HREF=\"{@docroot}/element-list\">{&#064;docroot}</A> <A HREF=\"{@docRoot}/help-doc.html\">{&#064;docRoot}</A>",
 62                 "-sourcepath", srcdir,
 63                 "p1", "p2");
 64 
 65         checkFiles(
 66                 "p1/C1.html",
 67                 "p1/package-summary.html",
 68                 "overview-summary.html");
 69 
 70         // Bug 4633447: Special test for overview-frame.html
 71         // Find two strings in file "overview-frame.html"
 72         checkOutput("overview-frame.html", true,
 73                 "<A HREF=\"./element-list\">",
 74                 "<A HREF=\"./help-doc.html\">");
 75     }
 76 
 77     void checkFiles(String... filenameArray) {
 78         int count = 0;
 79 
 80         for (String f : filenameArray) {
 81             // Read contents of file into a string
 82             String fileString = readFile(f);
 83             System.out.println("\nSub-tests for file: " + f + " --------------");
 84             // Loop over all tests in a single file
 85             for ( int j = 0; j < 7; j++ ) {
 86 
 87                 // Compare actual to expected string for a single subtest
 88                 compareActualToExpected(++count, fileString);
 89             }
 90         }
 91     }
 92 
 93     /**
 94      * Regular expression pattern matching code
 95      *
 96      * Prefix Pattern:
 97      * flag   (?i)            (case insensitive, so "a href" == "A HREF" and all combinations)
 98      * group1 (
 99      *          <a or <A
100      *          \\s+          (one or more whitespace characters)
101      *          href or HREF
102      *          \"            (double quote)
103      *        )
104      * group2 ([^\"]*)        (link reference -- characters that don't include a quote)
105      * group3 (\".*?>)        (" target="frameName">)
106      * group4 (.*?)           (label - zero or more characters)
107      * group5 (</a>)          (end tag)
108      */
109     private static final String prefix = "(?i)(<a\\s+href=";    // <a href=     (start group1)
110     private static final String ref1   = "\")([^\"]*)(\".*?>)"; // doublequotes (end group1, group2, group3)
111 
112     /**
113      * Compares the actual string to the expected string in the specified string
114      * @param str   String to search through
115      */
116     void compareActualToExpected(int count, String str) {
117         checking("comparison for " + str);
118 
119         // Pattern must be compiled each run because numTestsRun is incremented
120         Pattern actualLinkPattern1 =
121             Pattern.compile("Sub-test " + count + " Actual: " + prefix + ref1, Pattern.DOTALL);
122         Pattern expectLinkPattern1 =
123             Pattern.compile("Sub-test " + count + " Expect: " + prefix + ref1, Pattern.DOTALL);
124         // Pattern linkPattern2 = Pattern.compile(prefix + ref2 + label + end, Pattern.DOTALL);
125 
126         Matcher actualLinkMatcher1 = actualLinkPattern1.matcher(str);
127         Matcher expectLinkMatcher1 = expectLinkPattern1.matcher(str);
128         if (expectLinkMatcher1.find() && actualLinkMatcher1.find()) {
129             String expectRef = expectLinkMatcher1.group(2);
130             String actualRef = actualLinkMatcher1.group(2);
131             if (actualRef.equals(expectRef)) {
132                 passed(expectRef);
133                 // System.out.println("pattern:   " + actualLinkPattern1.pattern());
134                 // System.out.println("actualRef: " + actualRef);
135                 // System.out.println("group0:    " + actualLinkMatcher1.group());
136                 // System.out.println("group1:    " + actualLinkMatcher1.group(1));
137                 // System.out.println("group2:    " + actualLinkMatcher1.group(2));
138                 // System.out.println("group3:    " + actualLinkMatcher1.group(3));
139                 // System.exit(0);
140             } else {
141                 failed("\n"
142                         + "Actual: \"" + actualRef + "\"\n"
143                         + "Expect: \"" + expectRef + "\"");
144             }
145         } else {
146             failed("Didn't find <A HREF> that fits the pattern: "
147                     + expectLinkPattern1.pattern());
148         }
149     }
150 }
