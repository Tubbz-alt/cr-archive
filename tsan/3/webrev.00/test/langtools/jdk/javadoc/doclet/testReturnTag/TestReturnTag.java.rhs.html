24 /*
25  * @test
26  * @bug      4490068
27  * @summary  Warn when a return tag is used on a method without a return type.

28  * @library  ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestReturnTag
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestReturnTag extends JavadocTester {
37 
38     /**
39      * Trigger warning message when return tag is used on a void method.
40      *
41      * @return I really don't return anything.
42      */
43     public void method() {}
44 
45     public static void main(String... args) throws Exception {
46         TestReturnTag tester = new TestReturnTag();
47         tester.runTests();
48     }
49 
50     @Test
51     public void tests() {
52         javadoc("-Xdoclint:none",
53                 "-d", "out",
54                 "-sourcepath", testSrc,
55                 testSrc("TestReturnTag.java"));
56         checkExit(Exit.OK);
57 
58         checkOutput(Output.OUT, true,
59             "warning - @return tag cannot be used in method with void return type.");
60     }
61 }
