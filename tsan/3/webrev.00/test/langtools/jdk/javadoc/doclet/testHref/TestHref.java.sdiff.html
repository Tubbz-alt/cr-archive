35 import javadoc.tester.JavadocTester;
36 
37 public class TestHref extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestHref tester = new TestHref();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-Xdoclint:none",
47                 "-d", "out",
48                 "-source", "8",
49                 "-sourcepath", testSrc,
50                 "-linkoffline", "http://java.sun.com/j2se/1.4/docs/api/", testSrc,
51                 "pkg");
52         checkExit(Exit.OK);
53 
54         checkOutput("pkg/C1.html", true,
55                 //External link.
56                 "href=\"http://java.sun.com/j2se/1.4/docs/api/java/lang/Object.html?is-external=true#wait(long,int)\"",
57                 //Member summary table link.
58                 "href=\"#method(int,int,java.util.ArrayList)\"",
59                 //Anchor test.
60                 "<a id=\"method(int,int,java.util.ArrayList)\">\n"
61                 + "<!--   -->\n"
62                 + "</a>",
63                 //Backward compatibility anchor test."pkg/C1.html",
64                 "<a id=\"method(int,int,java.util.ArrayList)\">\n"
65                 + "<!--   -->\n"
66                 + "</a>");
67 
68         checkOutput("pkg/C2.html", true,
69                 //{@link} test.
70                 "Link: <a href=\"C1.html#method(int,int,java.util.ArrayList)\">",
71                 //@see test.
72                 "See Also:</span></dt>\n"
73                 + "<dd><a href=\"C1.html#method(int,int,java.util.ArrayList)\">"
74         );
75 
76         checkOutput("pkg/C4.html", true,
77                 //Header does not link to the page itself.
78                 "Class C4&lt;E extends C4&lt;E&gt;&gt;</h1>",
79                 //Signature does not link to the page itself.
80                 "public abstract class <span class=\"typeNameLabel\">C4&lt;E extends C4&lt;E&gt;&gt;</span>"
81         );
82 
83         checkOutput(Output.OUT, false,
84                 "<a> tag is malformed");
85     }
