 24 /*
 25  * @test
 26  * @bug 5076751
 27  * @summary System properties documentation needed in javadocs
 28  * @library /tools/lib ../../lib
 29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 30  * @build javadoc.tester.* toolbox.ToolBox builder.ClassBuilder
 31  * @run main TestSystemPropertyTaglet
 32  */
 33 
 34 
 35 import java.nio.file.Path;
 36 import java.nio.file.Paths;
 37 
 38 import builder.ClassBuilder;
 39 import builder.ClassBuilder.MethodBuilder;
 40 import toolbox.ToolBox;
 41 
 42 import javadoc.tester.JavadocTester;
 43 
 44 public class TestSystemPropertyTaglet extends JavadocTester {
 45 
 46     final ToolBox tb;
 47 
 48     public static void main(String... args) throws Exception {
 49         TestSystemPropertyTaglet tester = new TestSystemPropertyTaglet();
 50         tester.runTests(m -> new Object[]{Paths.get(m.getName())});
 51     }
 52 
 53     TestSystemPropertyTaglet() {
 54         tb = new ToolBox();
 55     }
 56 
 57     @Test
 58     public void test(Path base) throws Exception {
 59         Path srcDir = base.resolve("src");
 60         Path outDir = base.resolve("out");
 61 
 62         MethodBuilder method = MethodBuilder
 63                 .parse("public void func(A a) {}")
 64                 .setComments("test with {@systemProperty java.version}");
 65 
 66         new ClassBuilder(tb, "pkg.A")
 67                 .setComments("test with {@systemProperty user.name}")
 68                 .setModifiers("public", "class")
 69                 .addMembers(method)
 70                 .write(srcDir);
 71 
 72         javadoc("-d", outDir.toString(),
 73                 "-sourcepath", srcDir.toString(),
 74                 "pkg");
 75 
 76         checkExit(Exit.OK);
 77 
 78         checkOrder("pkg/A.html",
 79                 "<h1 title=\"Class A\" class=\"title\">Class A</h1>",
 80                 "test with <code><a id=\"user.name\" class=\"searchTagResult\">user.name</a></code>",
 81                 "<h2>Method Detail</h2>",
 82                 "test with <code><a id=\"java.version\" class=\"searchTagResult\">java.version</a></code>");























































 83 
 84         checkOrder("index-all.html",
 85                 "<h2 class=\"title\">J</h2>",
 86                 "<dt><span class=\"searchTagLink\"><a href=\"pkg/A.html#java.version\">java.version</a>"
 87                 + "</span> - Search tag in pkg.A</dt>\n<dd>System Property</dd>",
 88                 "<h2 class=\"title\">U</h2>",
 89                 "<dt><span class=\"searchTagLink\"><a href=\"pkg/A.html#user.name\">user.name</a></span>"
 90                 + " - Search tag in pkg.A</dt>\n<dd>System Property</dd>");
 91 
 92         checkOutput("tag-search-index.js", true,
 93                 "{\"l\":\"java.version\",\"h\":\"pkg.A\",\"d\":\"System Property\","
 94                 + "\"u\":\"pkg/A.html#java.version\"}");


































 95 
 96         checkOutput("tag-search-index.js", true,
 97                 "{\"l\":\"user.name\",\"h\":\"pkg.A\",\"d\":\"System Property\","
 98                 + "\"u\":\"pkg/A.html#user.name\"}");









































 99     }
100 
101     @Test
102     public void testSystemProperytWithinATag(Path base) throws Exception {
103         Path srcDir = base.resolve("src");
104         Path outDir = base.resolve("out");
105 
106         new ClassBuilder(tb, "pkg2.A")
107                 .setModifiers("public", "class")
108                 .addMembers(MethodBuilder.parse("public void func(){}")
109                         .setComments("a within a : <a href='..'>{@systemProperty user.name}</a>"))
110                 .write(srcDir);
111 
112         javadoc("-d", outDir.toString(),
113                 "-sourcepath", srcDir.toString(),
114                 "pkg2");
115 
116         checkExit(Exit.OK);
117 
118         checkOutput(Output.OUT, true,
119                 "warning: {@systemProperty} tag, which expands to <a>, within <a>");
120     }
121 }
