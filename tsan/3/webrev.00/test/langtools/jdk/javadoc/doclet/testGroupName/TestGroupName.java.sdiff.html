 53     @Test
 54     public void testPackageGroups(Path base) throws IOException {
 55         Path src = base.resolve("src");
 56         tb.writeJavaFiles(src,
 57                 "package p1; public class C1 { }",
 58                 "package p2; public class C2 { }",
 59                 "package p3; public class C3 { }");
 60 
 61         javadoc("-d", base.resolve("out").toString(),
 62                 "--frames",
 63                 "-sourcepath", src.toString(),
 64                 "-group", "abc < & > def", "p1",
 65                 "p1", "p2", "p3");
 66         checkExit(Exit.OK);
 67 
 68         checkOutput("overview-summary.html", true,
 69                 "<button role=\"tab\" aria-selected=\"false\" aria-controls=\"overviewSummary_tabpanel\""
 70                 + " tabindex=\"-1\" onkeydown=\"switchTab(event)\" id=\"t1\" class=\"tableTab\""
 71                 + " onclick=\"show(1);\">abc &lt; &amp; &gt; def</button>",
 72                 ",\"abc < & > def\"],");
 73     }
