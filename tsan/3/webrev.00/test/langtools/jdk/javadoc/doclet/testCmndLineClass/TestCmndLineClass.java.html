24 /*
25  * @test
26  * @bug 4506980
27  * @summary Test to make sure that there is no difference in the output
28  * when specifying packages on the command line and specifying individual
29  * classes.
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main TestCmndLineClass
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestCmndLineClass extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestCmndLineClass tester = new TestCmndLineClass();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         String outdir1 = "out-1";
48         String outdir2 = "out-2";
49 
50         javadoc("-d", outdir1,
51                 "-sourcepath", testSrc,
52                 "-notimestamp",
53                 testSrc("C5.java"), "pkg1", "pkg2");
54         checkExit(Exit.OK);
55 
56         javadoc("-d", outdir2,
57                 "-sourcepath", testSrc,
58                 "-notimestamp",
59                 testSrc("C5.java"),
60                 testSrc("pkg1/C1.java"),
61                 testSrc("pkg1/C2.java"),
62                 testSrc("pkg2/C3.java"),
63                 testSrc("pkg2/C4.java"));
64         checkExit(Exit.OK);
65 
66         diff(outdir1, outdir2,
67                 "C5.html",
68                 "pkg1/C1.html",
69                 "pkg1/C2.html",
70                 "pkg2/C3.html",
71                 "pkg2/C4.html");
72     }
73 }
