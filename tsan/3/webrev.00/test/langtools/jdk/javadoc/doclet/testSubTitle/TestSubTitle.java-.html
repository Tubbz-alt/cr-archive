24 /*
25  * @test
26  * @bug 7010342 8150000 8174974
27  * @summary Test for correct sub title generation.
28  * @author Bhavesh Patel
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestSubTitle
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestSubTitle extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestSubTitle tester = new TestSubTitle();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "pkg");
49         checkExit(Exit.OK);
50 
51         checkOutput("pkg/package-summary.html", true,
52             "<div class=\"block\">This is the description of package pkg.</div>");
53 
54         checkOutput("pkg/C.html", true,
55                 "<div class=\"subTitle\"><span class=\"packageLabelInType\">" +
56                 "Package</span>&nbsp;<a href=\"package-summary.html\">pkg</a></div>");
57 
58         checkOutput("pkg/package-summary.html", false,
59             "<p class=\"subTitle\">\n" +
60             "<div class=\"block\">This is the " +
61             "description of package pkg.</div>\n" +
62             "</p>");
63 
64         checkOutput("pkg/C.html", false,
65             "<p class=\"subTitle\">pkg</p>");
66     }
67 }
