 24 /*
 25  * @test
 26  * @bug 8196027 8196202
 27  * @summary test navigation links
 28  * @modules jdk.compiler/com.sun.tools.javac.api
 29  *          jdk.compiler/com.sun.tools.javac.main
 30  *          jdk.javadoc/jdk.javadoc.internal.api
 31  *          jdk.javadoc/jdk.javadoc.internal.tool
 32  * @library ../../lib /tools/lib
 33  * @build toolbox.ToolBox toolbox.ModuleBuilder javadoc.tester.*
 34  * @run main TestModuleNavigation
 35  */
 36 
 37 import java.nio.file.Files;
 38 import java.nio.file.Path;
 39 import java.nio.file.Paths;
 40 
 41 import javadoc.tester.JavadocTester;
 42 import toolbox.ModuleBuilder;
 43 import toolbox.ToolBox;
 44 
 45 public class TestModuleNavigation extends JavadocTester {
 46 
 47     public final ToolBox tb;
 48     public static void main(String... args) throws Exception {
 49         TestModuleNavigation  tester = new TestModuleNavigation ();
 50         tester.runTests(m -> new Object[] { Paths.get(m.getName()) });
 51     }
 52 
 53     public TestModuleNavigation () {
 54         tb = new ToolBox();
 55     }
 56 
 57     @Test
 58     public void checkNavbar(Path base) throws Exception {
 59         Path src = Files.createDirectories(base.resolve("src"));
 60         ModuleBuilder mb = new ModuleBuilder(tb, "m")
 61                 .comment("module m.\n@uses p1.A")
 62                 .uses("p1.A")
 63                 .uses("p1.B")
 64                 .exports("p1")
 65                 .classes("package p1; public class A {}")
 66                 .classes("package p1; public class B {}");
 67         mb.write(src);
 68         ModuleBuilder mb1 = new ModuleBuilder(tb, "m2")
 69                 .comment("module m2.\n@uses m2p1.Am2")
 70                 .uses("m2p1.Am2")
 71                 .uses("m2p1.Bm2")
 72                 .exports("m2p1")
 73                 .classes("package m2p1; public class Am2 {}")
 74                 .classes("package m2p1; public class Bm2 {}");
 75         mb1.write(src);
 76 
 77         javadoc("-d", base.resolve("out").toString(), "-use",
 78                 "-quiet",

 79                 "--module-source-path", src.toString(),
 80                 "--module", "m,m2");
 81         checkExit(Exit.OK);
 82 
 83         checkOutput("index.html", false,
 84                 "Prev",
 85                 "Next",
 86                 "All&nbsp;Classes",
 87                 "<script type=\"text/javascript\"><!--\n"
 88                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
 89                 "<script type=\"text/javascript\"><!--\n"
 90                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
 91 
 92         checkOutput("m/module-summary.html", false,
 93                 "Prev&nbsp;Module",
 94                 "Next&nbsp;Module",
 95                 "All&nbsp;Classes",
 96                 "<script type=\"text/javascript\"><!--\n"
 97                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
 98                 "<script type=\"text/javascript\"><!--\n"
 99                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
100 
101         checkOutput("m2/m2p1/package-summary.html", false,
102                 "Prev&nbsp;Package",
103                 "Next&nbsp;Package",
104                 "All&nbsp;Classes",
105                 "<script type=\"text/javascript\"><!--\n"
106                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
107                 "<script type=\"text/javascript\"><!--\n"
108                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
109 
110         checkOutput("m2/m2p1/Am2.html", false,
111                 "Prev&nbsp;Class",
112                 "Next&nbsp;Class",
113                 "All&nbsp;Classes",
114                 "<script type=\"text/javascript\"><!--\n"
115                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
116                 "<script type=\"text/javascript\"><!--\n"
117                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
118 
119         checkOutput("m2/m2p1/class-use/Am2.html", false,
120                 "Prev",
121                 "Next",
122                 "All&nbsp;Classes",
123                 "<script type=\"text/javascript\"><!--\n"
124                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
125                 "<script type=\"text/javascript\"><!--\n"
126                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
127 
128         checkOutput("m2/m2p1/package-tree.html", false,
129                 "Prev",
130                 "Next",
131                 "All&nbsp;Classes",
132                 "<script type=\"text/javascript\"><!--\n"
133                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
134                 "<script type=\"text/javascript\"><!--\n"
135                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
136 
137         checkOutput("deprecated-list.html", false,
138                 "Prev",
139                 "Next",
140                 "All&nbsp;Classes",
141                 "<script type=\"text/javascript\"><!--\n"
142                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
143                 "<script type=\"text/javascript\"><!--\n"
144                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
145 
146         checkOutput("index-all.html", false,
147                 "Prev",
148                 "Next",
149                 "<script type=\"text/javascript\"><!--\n"
150                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
151                 "<script type=\"text/javascript\"><!--\n"
152                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
153     }
