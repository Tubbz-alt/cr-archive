 24 /*
 25  * @test
 26  * @bug 6553182 8025416 8029504
 27  * @summary This test verifies the -Xdocrootparent option.
 28  * @author Bhavesh Patel
 29  * @library ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build javadoc.tester.*
 32  * @run main TestDocRootLink
 33  */
 34 import javadoc.tester.JavadocTester;
 35 
 36 public class TestDocRootLink extends JavadocTester {
 37 
 38     public static void main(String... args) throws Exception {
 39         TestDocRootLink tester = new TestDocRootLink();
 40 
 41         // The test files intentionally contain examples of links that should
 42         // or should not be affected by the -Xdocrootparent option, and the
 43         // results are checked explicitly; so, disable the automatic link
 44         // checker to prevent spurious "missing files" errors from some of
 45         // these links.
 46         tester.setAutomaticCheckLinks(false);
 47 
 48         tester.runTests();
 49     }
 50 
 51     @Test
 52     public void test1() {
 53         javadoc("-d", "out-1",
 54                 "-sourcepath", testSrc,
 55                 "pkg1", "pkg2");
 56         checkExit(Exit.OK);
 57 
 58         checkOutput("pkg1/C1.html", true,
 59             "Refer <a href=\"../../technotes/guides/index.html\">Here</a>",
 60             "This <a href=\"../pkg2/C2.html\">Here</a> should not be replaced\n" +
 61             " with an absolute link.",
 62             "Testing <a href=\"../technotes/guides/index.html\">Link 1</a> and\n" +
 63             " <a href=\"../pkg2/C2.html\">Link 2</a>.");
 64 
 65         checkOutput("pkg1/package-summary.html", true,
 66             "<a href=\"../../technotes/guides/index.html\">\n" +
 67             "            Test document 1</a>",
 68             "<a href=\"../pkg2/C2.html\">\n" +
 69             "            Another Test document 1</a>",
 70             "<a href=\"../technotes/guides/index.html\">\n" +
 71             "            Another Test document 2.</a>");
 72 
 73         // TODO: should this check *any* reference to http://download.oracle.com/
 74         checkOutput("pkg1/C1.html", false,
 75             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">",
 76             "<a href=\"http://download.oracle.com/javase/7/docs/pkg2/C2.html\">");
 77 
 78         checkOutput("pkg1/package-summary.html", false,
 79             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">",
 80             "<a href=\"http://download.oracle.com/javase/7/docs/pkg2/C2.html\">");
 81     }
 82 
 83     @Test
 84     public void test2() {
 85         javadoc("-d", "out-2",
 86                 "-Xdocrootparent", "http://download.oracle.com/javase/7/docs",
 87                 "-sourcepath", testSrc,
 88                 "pkg1", "pkg2");
 89         checkExit(Exit.OK);
 90 
 91         checkOutput("pkg2/C2.html", true,
 92             "Refer <a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">Here</a>",
 93             "This <a href=\"../pkg1/C1.html\">Here</a> should not be replaced\n" +
 94             " with an absolute link.",
 95             "Testing <a href=\"../technotes/guides/index.html\">Link 1</a> and\n" +
 96             " <a href=\"../pkg1/C1.html\">Link 2</a>.");
 97 
 98         checkOutput("pkg2/package-summary.html", true,
 99             "<a href=\"http://download.oracle.com/javase/7/docs/technotes/guides/index.html\">\n" +
100             "            Test document 1</a>",
101             "<a href=\"../pkg1/C1.html\">\n" +
102             "            Another Test document 1</a>",
103             "<a href=\"../technotes/guides/index.html\">\n" +
104             "            Another Test document 2.</a>");
105 
106         checkOutput("pkg2/C2.html", false,
107             "<a href=\"../../technotes/guides/index.html\">",
108             "<a href=\"http://download.oracle.com/javase/7/docs/pkg1/C1.html\">");
109 
110         checkOutput("pkg2/package-summary.html", false,
111             "<a href=\"../../technotes/guides/index.html\">",
112             "<a href=\"http://download.oracle.com/javase/7/docs/pkg1/C1.html\">");
113     }
114 }
