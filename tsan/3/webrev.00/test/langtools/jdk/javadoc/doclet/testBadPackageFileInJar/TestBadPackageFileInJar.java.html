24 /*
25  * @test
26  * @bug 4691095 6306394
27  * @summary Make sure that Javadoc emits a useful warning
28  *          when a bad package.html exists in a JAR archive.
29  * @library /tools/lib ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.* toolbox.ToolBox toolbox.JarTask
32  * @run main TestBadPackageFileInJar
33  */
34 
35 import toolbox.JarTask;
36 import toolbox.Task.Result;
37 import toolbox.ToolBox;
38 
39 import java.io.IOException;
40 import java.nio.file.Path;
41 import java.nio.file.Paths;
42 
43 import javadoc.tester.JavadocTester;
44 
45 public class TestBadPackageFileInJar extends JavadocTester {
46 
47     final ToolBox tb = new ToolBox();
48 
49     public static void main(String... args) throws Exception {
50         TestBadPackageFileInJar tester = new TestBadPackageFileInJar();
51         tester.runTests();
52     }
53 
54     @Test
55     public void test() throws IOException {
56         // create the file
57         Path pkgDir = Paths.get("pkg");
58         tb.createDirectories(pkgDir);
59         Path pkgfilePath = pkgDir.resolve("package.html");
60         tb.writeFile(pkgfilePath, "<html>\n\n</html>");
61 
62         // create the jar file
63         Path jarFile = Paths.get("badPackageFileInJar.jar");
64         JarTask jar = new JarTask(tb, "badPackageFileInJar.jar");
65         jar.files(pkgDir.toString()).run();
66 
67         // clean up to prevent accidental pick up
68         tb.cleanDirectory(pkgDir);
69         tb.deleteFiles(pkgDir.toString());
70 
71         javadoc("-d", "out",
72                 "-sourcepath", testSrc,
73                 "-classpath",  jarFile.toString(),
74                 "pkg");
75         checkExit(Exit.OK);
76 
77         checkOutput(Output.OUT, true,
78                 "badPackageFileInJar.jar(/pkg/package.html):1: warning: no comment");
79     }
80 }
