24 /*
25  * @test
26  * @bug 6786028 8026567
27  * @summary This test verifies the use of <strong> HTML tag instead of <B> by Javadoc std doclet.
28  * @author Bhavesh Patel
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestHtmlStrongTag
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestHtmlStrongTag extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestHtmlStrongTag tester = new TestHtmlStrongTag();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test1() {
46         javadoc("-d", "out-1",
47                 "-sourcepath", testSrc,
48                 "pkg1");
49         checkExit(Exit.OK);
50 
51         checkOutput("pkg1/C1.html", true,
52             "<span class=\"seeLabel\">See Also:</span>");
53 
54         checkOutput("pkg1/C1.html", false,
55             "<STRONG>Method Summary</STRONG>",
56             "<B>");
57 
58         checkOutput("pkg1/package-summary.html", false,
59             "<STRONG>Class Summary</STRONG>");
60     }
61 
62     @Test
63     public void test2() {
64         javadoc("-d", "out-2",
65                 "-sourcepath", testSrc,
66                 "pkg2");
67         checkExit(Exit.OK);
68 
69         checkOutput("pkg2/C2.html", true,
70                 "<B>Comments:</B>");
71 
72         checkOutput("pkg2/C2.html", false,
73                 "<STRONG>Method Summary</STRONG>");
74     }
75 }
