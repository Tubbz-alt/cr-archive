 24 /*
 25  * @test
 26  * @bug 8073100 8182765 8196202
 27  * @summary ensure the hidden tag works as intended
 28  * @library ../../lib
 29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 30  * @build javadoc.tester.*
 31  * @run main TestHiddenTag
 32  */
 33 
 34 import javadoc.tester.JavadocTester;
 35 
 36 public class TestHiddenTag extends JavadocTester {
 37 
 38     public static void main(String... args) throws Exception {
 39         TestHiddenTag tester = new TestHiddenTag();
 40         tester.runTests();
 41     }
 42 
 43     /**
 44      * Perform tests on &#64;hidden tags
 45      */
 46     @Test
 47     public void test1() {
 48         javadoc("-d", "out1",
 49                 "--frames",
 50                 "-sourcepath", testSrc,
 51                 "-package",
 52                 "pkg1");
 53         checkExit(Exit.OK);
 54 
 55         checkOutput("pkg1/A.html", true,
 56                 "<a id=\"visibleField\">",
 57                 "<a id=\"visibleMethod()\">",
 58                 "<dt>Direct Known Subclasses:</dt>\n" +
 59                 "<dd><code><a href=\"A.VisibleInner.html\" title=\"class in pkg1\">" +
 60                 "A.VisibleInner</a></code>, <code><a href=\"A.VisibleInnerExtendsInvisibleInner.html\" " +
 61                 "title=\"class in pkg1\">A.VisibleInnerExtendsInvisibleInner</a></code></dd>");
 62 
 63         checkOutput("pkg1/A.html", false,
 64                 "<a id=\"inVisibleField\">",
 65                 "<a id=\"inVisibleMethod()\">");
 66 
 67         checkOutput("pkg1/A.VisibleInner.html", true,
 68                 "<code><a href=\"A.html#visibleField\">visibleField</a></code>",
 69                 "<code><a href=\"A.html#visibleMethod()\">visibleMethod</a></code>",
 70                 "<h2>Nested classes/interfaces inherited from class&nbsp;pkg1." +

 71                 "<a href=\"A.html\" title=\"class in pkg1\">A</a></h2>\n" +
 72                 "<code><a href=\"A.VisibleInner.html\" title=\"class in pkg1\">" +
 73                 "A.VisibleInner</a>, <a href=\"A.VisibleInnerExtendsInvisibleInner.html\" " +
 74                 "title=\"class in pkg1\">A.VisibleInnerExtendsInvisibleInner</a></code></li>\n" +
 75                 "</ul>");
 76 
 77         checkOutput("pkg1/A.VisibleInner.html", false,
 78                 "../pkg1/A.VisibleInner.html#VisibleInner()",
 79                 "<a id=\"inVisibleField\">",
 80                 "<a id=\"inVisibleMethod()\">");
 81 
 82         checkOutput("pkg1/A.VisibleInnerExtendsInvisibleInner.html", true,
 83                 "<pre>public static class <span class=\"typeNameLabel\">" +
 84                 "A.VisibleInnerExtendsInvisibleInner</span>\n" +
 85                 "extends <a href=\"A.html\" title=\"class in pkg1\">A</a></pre>",
 86                 "<code><a href=\"A.html#visibleField\">visibleField</a></code></li>",
 87                 "<code><a href=\"A.html#visibleMethod()\">visibleMethod</a></code>");
 88 
 89         checkOutput("pkg1/A.VisibleInnerExtendsInvisibleInner.html", false,
 90                 "invisibleField",
 91                 "invisibleMethod",
 92                 "A.InvisibleInner");
 93 
 94         checkOutput("pkg1/package-frame.html", false, "A.InvisibleInner");
 95 
 96         checkOutput("pkg1/package-summary.html", false, "A.InvisibleInner");
 97 
 98         checkOutput("pkg1/package-tree.html", false, "A.InvisibleInner");
 99 
100         checkFiles(false,
101                 "pkg1/A.InvisibleInner.html",
102                 "pkg1/A.InvisibleInnerExtendsVisibleInner.html");
103     }
104 }
