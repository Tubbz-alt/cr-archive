24 /*
25  * @test
26  * @bug 4369014 4851991 8164407 8205593
27  * @summary Determine if the docRoot inline tag works properly.
28  * If docRoot performs as documented, the test passes.
29  * Make sure that the docRoot tag works with the -bottom option.
30  * @author jamieh
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.*
34  * @run main TestDocRootInlineTag
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestDocRootInlineTag extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestDocRootInlineTag tester = new TestDocRootInlineTag();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         String uri = "http://www.java.sun.com/j2se/1.4/docs/api";
49 
50         javadoc("-bottom", "The value of @docRoot is \"{@docRoot}\"",
51                 "-d", "out",
52                 "-source", "8",
53                 "-sourcepath", testSrc,
54                 "-linkoffline", uri, testSrc,
55                 testSrc("TestDocRootTag.java"), "pkg");
56         checkExit(Exit.OK);
57 
58         checkOutput("TestDocRootTag.html", true,
59                 "<a href=\"" + uri + "/java/io/File.html?is-external=true\" "
60                 + "title=\"class or interface in java.io\" class=\"externalLink\"><code>File</code></a>",
61                 "<a href=\"./index-all.html\">index</a>",
62                 "<a href=\"" + uri + "/java/io/File.html?is-external=true\" "
63                 + "title=\"class or interface in java.io\" class=\"externalLink\"><code>Second File Link</code></a>",
64                 "The value of @docRoot is \"./\"");
65 
66         checkOutput("index-all.html", true,
67                 "My package page is <a href=\"./pkg/package-summary.html\">here</a>");
68     }
69 }
