24 /*
25  * @test
26  * @bug 4671694
27  * @summary Test to make sure link to superclass is generated for
28  * each class in serialized form page.

29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestSuperClassInSerialForm
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestSuperClassInSerialForm extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestSuperClassInSerialForm tester = new TestSuperClassInSerialForm();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 "pkg");
49         checkExit(Exit.OK);
50 
51         checkOutput("serialized-form.html", true,
52                 "<a href=\"pkg/SubClass.html\" title=\"class in pkg\">pkg.SubClass</a>"
53                 + " extends <a href=\"pkg/SuperClass.html\" title=\"class in pkg\">SuperClass</a>");
54     }
55 }
