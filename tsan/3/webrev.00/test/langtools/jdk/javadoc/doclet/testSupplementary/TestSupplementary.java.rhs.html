24 /*
25  * @test
26  * @bug 4914724
27  * @summary Test to make sure that "see" tag and "serialField" tag handle supplementary
28  *    characters correctly.  This test case needs to be run in en_US locale.

29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestSupplementary
33  */
34 
35 import java.util.Locale;
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestSupplementary extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         Locale saveLocale = Locale.getDefault();
43         try {
44             TestSupplementary tester = new TestSupplementary();
45             tester.runTests();
46         } finally {
47             Locale.setDefault(saveLocale);
48         }
49     }
50 
51     @Test
52     public void test() {
53         javadoc("-locale", "en_US",
54                 "-d", "out",
55                 testSrc("C.java"));
56         checkExit(Exit.ERROR);
57 
58         checkOutput(Output.OUT, true,
59             "C.java:36: error: unexpected text",
60             "C.java:41: error: unexpected text",
61             "C.java:42: error: identifier expected");
62     }
63 }
