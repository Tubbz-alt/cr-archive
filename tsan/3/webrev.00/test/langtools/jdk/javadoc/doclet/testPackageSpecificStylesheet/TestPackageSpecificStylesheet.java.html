24 /*
25  * @test
26  * @bug 8213354
27  * @summary Support package-specific stylesheets
28  * @library /tools/lib ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build javadoc.tester.* toolbox.ToolBox builder.ClassBuilder
31  * @run main TestPackageSpecificStylesheet
32  */
33 
34 
35 import java.nio.file.Files;
36 import java.nio.file.Path;
37 import java.nio.file.Paths;
38 import java.util.List;
39 
40 import builder.ClassBuilder;
41 import toolbox.ToolBox;
42 
43 import javadoc.tester.JavadocTester;
44 
45 public class TestPackageSpecificStylesheet extends JavadocTester {
46 
47     final ToolBox tb;
48 
49     public static void main(String... args) throws Exception {
50         TestPackageSpecificStylesheet tester = new TestPackageSpecificStylesheet();
51         tester.runTests(m -> new Object[]{Paths.get(m.getName())});
52     }
53 
54     TestPackageSpecificStylesheet() {
55         tb = new ToolBox();
56     }
57 
58     @Test
59     public void test(Path base) throws Exception {
60         Path srcDir = base.resolve("src");
61         Path outDir = base.resolve("out");
62 
63         new ClassBuilder(tb, "pkg.A")
64                 .setComments("<span>test span style</span>")
65                 .setModifiers("public", "class")
66                 .write(srcDir);
67 
68         Path docFilesDir = Files.createDirectories(srcDir.resolve("pkg").resolve("doc-files"));
69         Path stylesheet = docFilesDir.resolve("spanstyle.css");
70         Files.createFile(stylesheet);
71         Files.write(stylesheet, List.of("span{ color:blue; }"));
72 
73         new ClassBuilder(tb, "pkg2.B")
74                 .setModifiers("public", "class")
75                 .write(srcDir);
76 
77         javadoc("-d", outDir.toString(),
78                 "-sourcepath", srcDir.toString(),
79                 "pkg", "pkg2");
80 
81         checkExit(Exit.OK);
82 
83         checkOutput("pkg/A.html", true,
84                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../pkg/doc-files/spanstyle.css\"" +
85                         " title=\"Style\">");
86 
87         checkOutput("pkg/package-summary.html", true,
88                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../pkg/doc-files/spanstyle.css\"" +
89                         " title=\"Style\">");
90 
91         checkOutput("pkg2/B.html", false,
92                 "<link rel=\"stylesheet\" type=\"text/css\" href=\"../pkg2/doc-files/spanstyle.css\"" +
93                         " title=\"Style\">");
94 
95     }
96 }
