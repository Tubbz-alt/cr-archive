 36 import javadoc.tester.JavadocTester;
 37 
 38 public class TestHeadings extends JavadocTester {
 39 
 40     public static void main(String... args) throws Exception {
 41         TestHeadings tester = new TestHeadings();
 42         tester.runTests();
 43     }
 44 
 45     @Test
 46     public void test() {
 47         javadoc("-d", "out",
 48                 "-sourcepath", testSrc,
 49                 "-use",
 50                 "--frames",
 51                 "-header", "Test Files",
 52                 "pkg1", "pkg2");
 53         checkExit(Exit.OK);
 54 
 55         //Package summary
 56         checkOutput("pkg1/package-summary.html", true,
 57                 "<th class=\"colFirst\" scope=\"col\">"
 58                 + "Class</th>\n"
 59                 + "<th class=\"colLast\" scope=\"col\""
 60                 + ">Description</th>");
 61 
 62         // Class documentation
 63         checkOutput("pkg1/C1.html", true,
 64                 "<th class=\"colFirst\" scope=\"col\">Modifier and Type</th>\n"
 65                 + "<th class=\"colSecond\" scope=\"col\">Field</th>\n"
 66                 + "<th class=\"colLast\" scope=\"col\">Description</th>",
 67                 "<h3>Methods inherited from class&nbsp;java.lang.Object</h3>");

 68 
 69         // Class use documentation
 70         checkOutput("pkg1/class-use/C1.html", true,
 71                 "<th class=\"colFirst\" scope=\"col\">Package</th>\n"
 72                 + "<th class=\"colLast\" scope=\"col\">Description</th>",
 73                 "<th class=\"colFirst\" scope=\"col\">Modifier and Type</th>\n"
 74                 + "<th class=\"colSecond\" scope=\"col\">Field</th>\n"
 75                 + "<th class=\"colLast\" scope=\"col\">Description</th>");
 76 
 77         // Deprecated
 78         checkOutput("deprecated-list.html", true,
 79                 "<th class=\"colFirst\" scope=\"col\">Method</th>\n"
 80                 + "<th class=\"colLast\" scope=\"col\">Description</th>");
 81 
 82         // Constant values
 83         checkOutput("constant-values.html", true,
 84                 "<th class=\"colFirst\" scope=\"col\">"
 85                 + "Modifier and Type</th>\n"
 86                 + "<th class=\"colSecond\" scope=\"col\">Constant Field</th>\n"
 87                 + "<th class=\"colLast\" scope=\"col\">Value</th>");
 88 
 89         // Serialized Form
 90         checkOutput("serialized-form.html", true,
 91                 "<h2 title=\"Package\">Package&nbsp;pkg1</h2>",
 92                 "<h3>Class <a href=\"pkg1/C1.html\" title=\"class in pkg1\">"
 93                 + "pkg1.C1</a> extends java.lang.Object implements Serializable</h3>",
 94                 "<h4>Serialized Fields</h4>");
 95 
 96         // Overview Frame
 97         checkOutput("overview-frame.html", true,
 98                 "<h1 title=\"Test Files\" class=\"bar\">Test Files</h1>",
 99                 "<title>Overview List</title>");
100 
101         // Overview Summary
102         checkOutput("overview-summary.html", true,
103                 "<title>Overview</title>");
104     }
105 }
