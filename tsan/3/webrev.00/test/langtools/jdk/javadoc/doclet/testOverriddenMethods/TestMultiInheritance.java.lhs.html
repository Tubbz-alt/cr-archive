24 /*
25  * @test
26  * @bug      4933335
27  * @summary  Make sure that all inherited methods from multiple extended
28  *           interfaces are documented
29  * @author   jamieh
30  * @library  ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build    javadoc.tester.*
33  * @run main TestMultiInheritance
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestMultiInheritance extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestMultiInheritance tester = new TestMultiInheritance();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-sourcepath", testSrc,
49                 "pkg3");
50         checkExit(Exit.OK);
51 
52         // Method foo() is inherited from BOTH I2 and I3
53 
54         checkOutput("pkg3/I1.html", true,
55                 "Methods inherited from interface&nbsp;pkg3."
56                 + "<a href=\"I2.html\" title=\"interface in pkg3\">"
57                 + "I2</a>",
58                 "Methods inherited from interface&nbsp;pkg3."
59                 + "<a href=\"I3.html\" title=\"interface in pkg3\">"
60                 + "I3</a>");
61 
62         checkOutput("pkg3/I0.html", true,
63                 "Methods inherited from interface&nbsp;pkg3."
64                 + "<a href=\"I2.html\" title=\"interface in pkg3\">"
65                 + "I2</a>",
66                 "Methods inherited from interface&nbsp;pkg3."
67                 + "<a href=\"I3.html\" title=\"interface in pkg3\">"
68                 + "I3</a>");
69 
70         // Method foo() is NOT inherited from I4 because it is overriden by I3.
71 
72         checkOutput("pkg3/I1.html", false,
73                 "Methods inherited from interface&nbsp;pkg3."
74                 + "<a href=\"I4.html\" title=\"interface in pkg3\">"
75                 + "I4</a>");
76 
77         checkOutput("pkg3/I0.html", false,
78                 "Methods inherited from interface&nbsp;pkg3."
79                 + "<a href=\"I4.html\" title=\"interface in pkg3\">"
80                 + "I4</a>");
81     }
82 }
