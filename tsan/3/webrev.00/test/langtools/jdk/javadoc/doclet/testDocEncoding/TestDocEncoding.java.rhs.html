28 /*
29  * @test
30  * @bug      8000743
31  * @summary  Run tests on -docencoding to see if the value is
32              used for stylesheet as well.

33  * @library  ../../lib
34  * @modules jdk.javadoc/jdk.javadoc.internal.tool
35  * @build    javadoc.tester.*
36  * @run main TestDocEncoding
37  */
38 
39 import java.nio.charset.Charset;
40 
41 import javadoc.tester.JavadocTester;
42 
43 public class TestDocEncoding extends JavadocTester {
44 
45     public static void main(String... args) throws Exception {
46         TestDocEncoding tester = new TestDocEncoding();
47         tester.runTests();
48     }
49 
50     @Test
51     public void test() {
52         javadoc("-d", "out",
53                 "-docencoding", "Cp930",
54                 "-sourcepath", testSrc,
55                 "-notimestamp",
56                 "pkg");
57         checkExit(Exit.OK);
58 
59         checkOutput("stylesheet.css", true,
60                 "body {\n"
61                 + "    background-color:#ffffff;");
62 
63         // reset the charset, for a negative test, that the -docencoding
64         // was effective and that the output is not in UTF-8.
65         charset = Charset.forName("UTF-8");
66         checkOutput("stylesheet.css", false,
67                 "body {\n"
68                 + "    background-color:#ffffff;");
69     }
70 }
