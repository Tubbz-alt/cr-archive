24 /*
25  * @test
26  * @bug 4652655 4857717 8025633 8026567 8071982 8164407 8182765 8205593
27  * @summary This test verifies that class cross references work properly.
28  * @author jamieh
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @build TestClassCrossReferences
33  * @run main TestClassCrossReferences
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestClassCrossReferences extends JavadocTester {
39 
40     static final String uri = "http://docs.oracle.com/javase/8/docs/api/";
41 
42     public static void main(String... args) throws Exception {
43         TestClassCrossReferences tester = new TestClassCrossReferences();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49         javadoc("-d", "out",
50                 "-source", "8",
51                 "-Xdoclint:none",
52                 "-sourcepath", testSrc,
53                 "-linkoffline", uri, testSrc,
54                 testSrc("C.java"));
55         checkExit(Exit.OK);
56 
57         checkOutput("C.html", true,
58                 "<a href=\"" + uri + "java/math/package-summary.html?is-external=true\" class=\"externalLink\">"
59                 + "<code>Link to math package</code></a>",
60                 "<a href=\"" + uri + "javax/swing/text/AbstractDocument.AttributeContext.html?is-external=true\" "
61                 + "title=\"class or interface in javax.swing.text\" class=\"externalLink\"><code>Link to AttributeContext innerclass</code></a>",
62                 "<a href=\"" + uri + "java/math/BigDecimal.html?is-external=true\" "
63                 + "title=\"class or interface in java.math\" class=\"externalLink\"><code>Link to external class BigDecimal</code></a>",
64                 "<a href=\"" + uri + "java/math/BigInteger.html?is-external=true#gcd(java.math.BigInteger)\" "
65                 + "title=\"class or interface in java.math\" class=\"externalLink\"><code>Link to external member gcd</code></a>",
66                 "<a href=\"" + uri + "javax/tools/SimpleJavaFileObject.html?is-external=true#uri\" "
67                 + "title=\"class or interface in javax.tools\" class=\"externalLink\"><code>Link to external member URI</code></a>",
68                 "<dl>\n"
69                 + "<dt><span class=\"overrideSpecifyLabel\">Overrides:</span></dt>\n"
70                 + "<dd><code>toString</code>&nbsp;in class&nbsp;<code>java.lang.Object</code></dd>\n"
71                 + "</dl>");
72     }
73 
74     @Test
75     public void test_error() {
76         javadoc("-d", "out-error",
77                 "-Xdoclint:none",
78                 "-sourcepath", testSrc,
79                 "-linkoffline", uri, testSrc,
80                 testSrc("C.java"));
81         checkExit(Exit.ERROR);
82         checkOutput(Output.OUT, true,
83                 "The code being documented uses modules but the packages defined"
84                 + " in http://docs.oracle.com/javase/8/docs/api/ are in the unnamed module");
85     }
86 }
