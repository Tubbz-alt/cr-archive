 24 import java.util.EnumSet;
 25 import java.util.List;
 26 import java.util.Map;
 27 import java.util.Set;
 28 import javax.lang.model.element.Element;
 29 
 30 import com.sun.source.doctree.DocTree;
 31 import com.sun.source.doctree.TextTree;
 32 import com.sun.source.doctree.UnknownBlockTagTree;
 33 import com.sun.source.doctree.UnknownInlineTagTree;
 34 import com.sun.source.util.SimpleDocTreeVisitor;
 35 import jdk.javadoc.doclet.Taglet;
 36 import jdk.javadoc.doclet.Taglet.Location;
 37 import static jdk.javadoc.doclet.Taglet.Location.*;
 38 
 39 
 40 /**
 41  * A sample Taglet representing @todo. This tag can be used in any kind of
 42  * {@link javax.lang.model.Element}.  It is not an inline tag. The text is displayed
 43  * in yellow to remind the developer to perform a task.  For
 44  * example, "@todo Fix this!" would be shown as:
 45  * <DL>
 46  * <DT>
 47  * <B>To Do:</B>
 48  * <DD><table summary="Summary" cellpadding=2 cellspacing=0><tr><td bgcolor="yellow">Fix this!
 49  * </td></tr></table></DD>
 50  * </DL>
 51  */
 52 
 53 public class ToDoTaglet implements Taglet {
 54 
 55     private static final String NAME = "todo";
 56     private static final String HEADER = "To Do:";
 57 
 58     /**
 59      * Return the name of this custom tag.
 60      */
 61     public String getName() {
 62         return NAME;
 63     }
 64 
 65     private final EnumSet<Location> allowedSet = EnumSet.allOf(Location.class);
 66 
 67     @Override
 68     public Set<Taglet.Location> getAllowedLocations() {
 69         return allowedSet;
 70     }
 71 
 72     /**
 73      * Will return false since <code>@todo</code>
 74      * is not an inline tag.
 75      * @return false since <code>@todo</code>
 76      * is not an inline tag.
 77      */
 78 
 79     public boolean isInlineTag() {
 80         return false;
 81     }
 82 
 83     /**
 84      * Given an array of <code>Tag</code>s representing this custom
 85      * tag, return its string representation.
 86      * @param tags  the array of <code>DocTree</code>s representing this custom tag.
 87      * @param element the declaration to which the enclosing comment belongs
 88      */
 89     @Override
 90     public String toString(List<? extends DocTree> tags, Element element) {
 91         if (tags.isEmpty()) {
 92             return "";
 93         }
 94         String result = "\n<DT><B>" + HEADER + "</B><DD>";
 95         result += "<table summary=\"Summary\" cellpadding=2 cellspacing=0><tr><td bgcolor=\"yellow\">";
 96         for (int i = 0; i < tags.size(); i++) {
 97             if (i > 0) {
 98                 result += ", ";
 99             }
100             result += getText(tags.get(i));
101         }
102         return result + "</td></tr></table></DD>\n";
103     }
104 
105     static String getText(DocTree dt) {
106         return new SimpleDocTreeVisitor<String, Void>() {
107             @Override
108             public String visitUnknownBlockTag(UnknownBlockTagTree node, Void p) {
109                 for (DocTree dt : node.getContent()) {
110                     return dt.accept(this, null);
111                 }
112                 return "";
113             }
114 
115             @Override
116             public String visitUnknownInlineTag(UnknownInlineTagTree node, Void p) {
117                 for (DocTree dt : node.getContent()) {
118                     return dt.accept(this, null);
119                 }
120                 return "";
121             }
122 
123             @Override
124             public String visitText(TextTree node, Void p) {
125                 return node.getBody();
126             }
127 
128             @Override
129             protected String defaultAction(DocTree node, Void p) {
130                 return "";
131             }
132 
133         }.visit(dt, null);
134     }
135 }
