 24 /*
 25  * @test
 26  * @bug 6802694 8025633 8026567 8183511 8074407 8182765
 27  * @summary This test verifies deprecation info in serialized-form.html.
 28  * @author Bhavesh Patel
 29  * @library ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  * @build javadoc.tester.*
 32  * @run main TestSerializedFormDeprecationInfo
 33  */
 34 
 35 import javadoc.tester.JavadocTester;
 36 
 37 public class TestSerializedFormDeprecationInfo extends JavadocTester {
 38 
 39     public static void main(String... args) throws Exception {
 40         TestSerializedFormDeprecationInfo tester = new TestSerializedFormDeprecationInfo();
 41         tester.runTests();
 42     }
 43 
 44     @Test
 45     public void testDefault() {
 46         javadoc("-d", "out-default",
 47                 "-sourcepath", testSrc,
 48                 "pkg1");
 49         checkExit(Exit.OK);
 50 
 51         checkCommentDeprecated(true);
 52         checkNoComment(false);
 53     }
 54 
 55     @Test
 56     public void testNoComment() {
 57         javadoc("-d", "out-nocmnt",
 58                 "-nocomment",
 59                 "-sourcepath", testSrc,
 60                 "pkg1");
 61         checkExit(Exit.OK);
 62 
 63         checkNoComment(true);
 64         checkCommentDeprecated(false);
 65     }
 66 
 67     @Test
 68     public void testNoDeprecated() {
 69         javadoc("-d", "out-nodepr",
 70                 "-nodeprecated",
 71                 "-sourcepath", testSrc,
 72                 "pkg1");
 73         checkExit(Exit.OK);
 74 
 75         checkNoDeprecated(true);
 76         checkNoCommentNoDeprecated(false);
 77     }
 78 
 79     @Test
 80     public void testNoCommentNoDeprecated() {
 81         javadoc("-d", "out-nocmnt-nodepr",
 82                 "-nocomment",
 83                 "-nodeprecated",
 84                 "-sourcepath", testSrc,
 85                 "pkg1");
 86         checkExit(Exit.OK);
 87         checkNoCommentNoDeprecated(true);
 88         checkNoDeprecated(false);
 89     }
 90 
 91     // Test for normal run of javadoc. The serialized-form.html should
 92     // display the inline comments, tags and deprecation information if any.
 93     void checkCommentDeprecated(boolean expectFound) {
 94         checkOutput("serialized-form.html", expectFound,
 95                 "<dl>\n"
 96                 + "<dt><span class=\"throwsLabel\">Throws:</span></dt>\n"
 97                 + "<dd><code>"
 98                 + "java.io.IOException</code> - on error</dd>\n"
 99                 + "<dt><span class=\"seeLabel\">See Also:</span>"
100                 + "</dt>\n"
101                 + "<dd><a href=\"pkg1/C1.html#setUndecorated(boolean)\">"
102                 + "<code>C1.setUndecorated(boolean)</code></a></dd>\n"
103                 + "</dl>",
104                 "<span class=\"deprecatedLabel\">Deprecated.</span>\n"
105                 + "<div class=\"deprecationComment\">As of JDK version 1.5, replaced by\n"
106                 + " <a href=\"pkg1/C1.html#setUndecorated(boolean)\">"
107                 + "<code>setUndecorated(boolean)</code></a>.</div>\n"
108                 + "</div>\n"
109                 + "<div class=\"block\">This field indicates whether the C1 "
110                 + "is undecorated.</div>\n"
111                 + "&nbsp;\n"
112                 + "<dl>\n"
113                 + "<dt><span class=\"simpleTagLabel\">Since:</span></dt>\n"
114                 + "<dd>1.4</dd>\n"
115                 + "<dt><span class=\"seeLabel\">See Also:</span>"
116                 + "</dt>\n"
117                 + "<dd><a href=\"pkg1/C1.html#setUndecorated(boolean)\">"
118                 + "<code>C1.setUndecorated(boolean)</code></a></dd>\n"
119                 + "</dl>",
120                 "<span class=\"deprecatedLabel\">Deprecated.</span>\n"
121                 + "<div class=\"deprecationComment\">As of JDK version 1.5, replaced by\n"
122                 + " <a href=\"pkg1/C1.html#setUndecorated(boolean)\">"
123                 + "<code>setUndecorated(boolean)</code></a>.</div>\n"
124                 + "</div>\n"
125                 + "<div class=\"block\">Reads the object stream.</div>\n"
126                 + "<dl>\n"
127                 + "<dt><span class=\"throwsLabel\">Throws:</span></dt>\n"
128                 + "<dd><code>java.io.IOException</code> - on error</dd>\n"
129                 + "</dl>",
130                 "<span class=\"deprecatedLabel\">Deprecated.</span>"
131                 + "</div>\n"
132                 + "<div class=\"block\">"
133                 + "The name for this class.</div>");
134     }
135 
136     // Test with -nocomment option. The serialized-form.html should
137     // not display the inline comments and tags but should display deprecation
138     // information if any.
139     void checkNoComment(boolean expectFound) {
140         checkOutput("serialized-form.html", expectFound,
141                 "<pre>boolean undecorated</pre>\n"
142                 + "<div class=\"deprecationBlock\"><span class=\"deprecatedLabel\">Deprecated.</span>\n"
143                 + "<div class=\"deprecationComment\">"
144                 + "As of JDK version 1.5, replaced by\n"
145                 + " <a href=\"pkg1/C1.html#setUndecorated(boolean)\"><code>"
146                 + "setUndecorated(boolean)</code></a>.</div>\n"
147                 + "</div>\n"
148                 + "</li>",
149                 "<span class=\"deprecatedLabel\">"
150                 + "Deprecated.</span>\n"
151                 + "<div class=\"deprecationComment\">As of JDK version"
152                 + " 1.5, replaced by\n"
153                 + " <a href=\"pkg1/C1.html#setUndecorated(boolean)\">"
154                 + "<code>setUndecorated(boolean)</code></a>.</div>\n"
155                 + "</div>\n"
156                 + "</li>");
157     }
158 
159     // Test with -nodeprecated option. The serialized-form.html should
160     // ignore the -nodeprecated tag and display the deprecation info. This
161     // test is similar to the normal run of javadoc in which inline comment, tags
162     // and deprecation information will be displayed.
163     void checkNoDeprecated(boolean expectFound) {
164         checkCommentDeprecated(expectFound);
165     }
166 
167     // Test with -nodeprecated and -nocomment options. The serialized-form.html should
168     // ignore the -nodeprecated tag and display the deprecation info but should not
169     // display the inline comments and tags. This test is similar to the test with
170     // -nocomment option.
171     void checkNoCommentNoDeprecated(boolean expectFound) {
172         checkNoComment(expectFound);
173     }
174 }
