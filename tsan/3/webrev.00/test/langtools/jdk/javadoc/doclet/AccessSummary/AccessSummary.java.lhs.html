27  * @summary  Test the tables for summary attribute
28  * @author   dkramer

31  * @build    javadoc.tester.*
32  * @run main AccessSummary
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class AccessSummary extends JavadocTester {
38     /**
39      * The entry point of the test.
40      * @param args the array of command line arguments.
41      * @throws Exception if the test fails
42      */
43     public static void main(String... args) throws Exception {
44         AccessSummary tester = new AccessSummary();
45         tester.runTests();
46     }
47 
48     @Test
49     public void testAccessSummary() {
50         javadoc("-d", "out",
51                 "--frames",
52                 "-sourcepath", testSrc,
53                 "p1", "p2");
54         checkExit(Exit.OK);
55         checkSummary(false);
56     }
57 
58     void checkSummary(boolean found) {
59         checkOutput("overview-summary.html", found,
60                  "summary=\"Package Summary table, listing packages, and an explanation\"");
61 
62         // Test that the summary attribute appears or not
63         checkOutput("p1/C1.html", found,
64                  "summary=\"Constructor Summary table, listing constructors, and an explanation\"");
65 
66         // Test that the summary attribute appears or not
67         checkOutput("constant-values.html", found,
68                  "summary=\"Constant Field Values table, listing constant fields, and values\"");
69     }
70 }
