24 /*
25  * @test
26  * @bug      4904038
27  * @summary  The field detail comment should not show up in the output if there
28  *           are no fields to document.
29  * @author   jamieh
30  * @library  ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build    javadoc.tester.*
33  * @run main TestHtmlComments
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestHtmlComments extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestHtmlComments tester = new TestHtmlComments();
42         tester.runTests();
43     }
44 
45     @Test
46     public void run() {
47         javadoc("-d", "out",
48                 "-sourcepath", testSrc,
49                 testSrc("C.java"));
50         checkExit(Exit.OK);
51 
52         checkOutput("C.html", false,
53             "<!-- ============ FIELD DETAIL =========== -->");
54     }
55 }
