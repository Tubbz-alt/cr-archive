 24 /*
 25  * @test
 26  * @bug 8196027 8196202
 27  * @summary test navigation links
 28  * @modules jdk.compiler/com.sun.tools.javac.api
 29  *          jdk.compiler/com.sun.tools.javac.main
 30  *          jdk.javadoc/jdk.javadoc.internal.api
 31  *          jdk.javadoc/jdk.javadoc.internal.tool
 32  * @library ../../lib /tools/lib
 33  * @build toolbox.ToolBox toolbox.ModuleBuilder javadoc.tester.*
 34  * @run main TestModuleNavigation
 35  */
 36 
 37 import java.nio.file.Files;
 38 import java.nio.file.Path;
 39 import java.nio.file.Paths;
 40 
 41 import javadoc.tester.JavadocTester;
 42 import toolbox.ModuleBuilder;
 43 import toolbox.ToolBox;
 44 
 45 public class TestModuleNavigation extends JavadocTester {
 46 
 47     public final ToolBox tb;
 48     public static void main(String... args) throws Exception {
 49         TestModuleNavigation  tester = new TestModuleNavigation ();
 50         tester.runTests(m -> new Object[] { Paths.get(m.getName()) });
 51     }
 52 
 53     public TestModuleNavigation () {
 54         tb = new ToolBox();
 55     }
 56 
 57     @Test
 58     public void checkNavbar(Path base) throws Exception {
 59         Path src = Files.createDirectories(base.resolve("src"));
 60         ModuleBuilder mb = new ModuleBuilder(tb, "m")
 61                 .comment("module m.\n@uses p1.A")
 62                 .uses("p1.A")
 63                 .uses("p1.B")
 64                 .exports("p1")
 65                 .classes("package p1; public class A {}")
 66                 .classes("package p1; public class B {}");
 67         mb.write(src);
 68         ModuleBuilder mb1 = new ModuleBuilder(tb, "m2")
 69                 .comment("module m2.\n@uses m2p1.Am2")
 70                 .uses("m2p1.Am2")
 71                 .uses("m2p1.Bm2")
 72                 .exports("m2p1")
 73                 .classes("package m2p1; public class Am2 {}")
 74                 .classes("package m2p1; public class Bm2 {}");
 75         mb1.write(src);
 76 
 77         javadoc("-d", base.resolve("out").toString(), "-use",
 78                 "-quiet",
 79                 "--frames",
 80                 "--module-source-path", src.toString(),
 81                 "--module", "m,m2");
 82         checkExit(Exit.OK);
 83 
 84         checkOutput("overview-summary.html", false,
 85                 "Prev",
 86                 "Next",
 87                 "All&nbsp;Classes",
 88                 "<script type=\"text/javascript\"><!--\n"
 89                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
 90                 "<script type=\"text/javascript\"><!--\n"
 91                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
 92 
 93         checkOutput("m/module-summary.html", false,
 94                 "Prev&nbsp;Module",
 95                 "Next&nbsp;Module",
 96                 "All&nbsp;Classes",
 97                 "<script type=\"text/javascript\"><!--\n"
 98                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
 99                 "<script type=\"text/javascript\"><!--\n"
100                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
101 
102         checkOutput("m2/m2p1/package-summary.html", false,
103                 "Prev&nbsp;Package",
104                 "Next&nbsp;Package",
105                 "All&nbsp;Classes",
106                 "<script type=\"text/javascript\"><!--\n"
107                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
108                 "<script type=\"text/javascript\"><!--\n"
109                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
110 
111         checkOutput("m2/m2p1/Am2.html", false,
112                 "Prev&nbsp;Class",
113                 "Next&nbsp;Class",
114                 "All&nbsp;Classes",
115                 "<script type=\"text/javascript\"><!--\n"
116                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
117                 "<script type=\"text/javascript\"><!--\n"
118                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
119 
120         checkOutput("m2/m2p1/class-use/Am2.html", false,
121                 "Prev",
122                 "Next",
123                 "All&nbsp;Classes",
124                 "<script type=\"text/javascript\"><!--\n"
125                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
126                 "<script type=\"text/javascript\"><!--\n"
127                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
128 
129         checkOutput("m2/m2p1/package-tree.html", false,
130                 "Prev",
131                 "Next",
132                 "All&nbsp;Classes",
133                 "<script type=\"text/javascript\"><!--\n"
134                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
135                 "<script type=\"text/javascript\"><!--\n"
136                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
137 
138         checkOutput("deprecated-list.html", false,
139                 "Prev",
140                 "Next",
141                 "All&nbsp;Classes",
142                 "<script type=\"text/javascript\"><!--\n"
143                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
144                 "<script type=\"text/javascript\"><!--\n"
145                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
146 
147         checkOutput("index-all.html", false,
148                 "Prev",
149                 "Next",
150                 "<script type=\"text/javascript\"><!--\n"
151                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_top\");",
152                 "<script type=\"text/javascript\"><!--\n"
153                 + "  allClassesLink = document.getElementById(\"allclasses_navbar_bottom\");");
154     }
