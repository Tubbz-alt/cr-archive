24 /*
25  * @test
26  * @bug 4640745
27  * @summary This test verifys that the -link option handles absolute paths.
28  * @author jamieh
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main TestAbsLinkPath
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestAbsLinkPath extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestAbsLinkPath tester = new TestAbsLinkPath();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test1() {
46         String out1 = "out1";
47         javadoc("-d", out1, "-sourcepath", testSrc, "pkg2");
48         checkExit(Exit.OK);
49 
50         javadoc("-d", "out2",
51                 "-sourcepath", testSrc,
52                 "-link", "../" + out1,
53                 "pkg1");
54         checkExit(Exit.OK);
55 
56         checkOutput("pkg1/C1.html", true,
57                 "C2.html");
58     }
59 }
