24 /*
25  * @test
26  * @bug      4732864 6280605 7064544 8014636 8016328 8025633 8071982 8182765
27  * @summary  Make sure that you can link from one member to another using
28  *           non-qualified name, furthermore, ensure the right one is linked.
29  * @author   jamieh
30  * @library  ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build    javadoc.tester.*
33  * @run main TestLinkTaglet
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestLinkTaglet extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestLinkTaglet tester = new TestLinkTaglet();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-Xdoclint:none",
48                 "-d", "out",
49                 "-sourcepath", testSrc,
50                 "pkg", testSrc("checkPkg/B.java"));
51         checkExit(Exit.OK);
52 
53         checkOutput("pkg/C.html", true,
54                 "Qualified Link: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
55                 + " Unqualified Link1: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
56                 + " Unqualified Link2: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
57                 + " Qualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(pkg.C.InnerC, pkg.C.InnerC2)</code></a>.<br/>\n"
58                 + " Unqualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(C.InnerC, C.InnerC2)</code></a>.<br/>\n"
59                 + " Unqualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(InnerC, InnerC2)</code></a>.<br/>\n"
60                 + " Package Link: <a href=\"package-summary.html\"><code>pkg</code></a>.<br/>");
61 
62         checkOutput("pkg/C.InnerC.html", true,
63                 "Link to member in outer class: <a href=\"C.html#MEMBER\"><code>C.MEMBER</code></a> <br/>\n"
64                 + " Link to member in inner class: <a href=\"C.InnerC2.html#MEMBER2\"><code>C.InnerC2.MEMBER2</code></a> <br/>\n"
65                 + " Link to another inner class: <a href=\"C.InnerC2.html\" title=\"class in pkg\"><code>C.InnerC2</code></a>");
66 
67         checkOutput("pkg/C.InnerC2.html", true,
68                 "<dl>\n"
69                 + "<dt>Enclosing class:</dt>\n"
70                 + "<dd><a href=\"C.html\" title=\"class in pkg\">C</a></dd>\n"
71                 + "</dl>");
72 
73         checkOutput(Output.OUT, false,
74                 "Tag @see: reference not found: A");
75 
76         checkFiles(false, "checkPkg/A.html");
77     }
78 }
