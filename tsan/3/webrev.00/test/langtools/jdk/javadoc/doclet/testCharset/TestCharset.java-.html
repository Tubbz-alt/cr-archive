24 /*
25  * @test
26  * @bug      7052170 8047745
27  * @summary  Run a test on -charset to make sure the charset gets generated as a
28  *           part of the meta tag.
29  * @author   Bhavesh Patel
30  * @library  ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build    javadoc.tester.*
33  * @run main TestCharset
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestCharset extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestCharset tester = new TestCharset();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 "-charset", "ISO-8859-1",
49                 "-sourcepath", testSrc,
50                 "pkg");
51         checkExit(Exit.OK);
52 
53         checkOutput("index.html", true,
54             "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=ISO-8859-1\">");
55         checkOutput("pkg/Foo.html", true,
56             "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=ISO-8859-1\">");
57 
58         checkOutput("index.html", false,
59             "<meta http-equiv=\"Content-Type\" content=\"text/html\" charset=\"ISO-8859-1\">");
60         checkOutput("pkg/Foo.html", false,
61             "<meta http-equiv=\"Content-Type\" content=\"text/html\" charset=\"ISO-8859-1\">");
62     }
63 
64     @Test
65     public void test1() {
66         javadoc("-d", "out-1",
67                 "-sourcepath", testSrc,
68                 "pkg");
69         checkExit(Exit.OK);
70 
71         checkOutput("index.html", true,
72             "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=utf-8\">");
73         checkOutput("pkg/Foo.html", true,
74             "<meta http-equiv=\"Content-Type\" content=\"text/html; charset=utf-8\">");
75     }
76 }
