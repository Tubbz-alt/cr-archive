24 /**
25  * @test
26  * @bug     6551367
27  * @summary javadoc throws ClassCastException when an link tag tries to reference constructor.

28  * @modules jdk.javadoc/jdk.javadoc.internal.tool
29  * @run main T6551367 T6551367.java
30  */
31 
32 /*
33  * ***NOTE: This is not a tool test!, should be moved to doclets.***
34  */
35 import java.io.File;
36 
37 import jdk.javadoc.doclet.DocletEnvironment;
38 
39 public class T6551367 {
40     public T6551367() {}
41     public boolean run(DocletEnvironment root) {
42         return true;
43     }
44     /** Here, in the javadoc for this method, I try to link to
45      *  {@link #<init> a constructor}.
46      */
47     public static void main(String... args) {
48         File testSrc = new File(System.getProperty("test.src", "."));
49         File destDir = new File(System.getProperty("user.dir", "."));
50         for (String file : args) {
51             File source = new File(testSrc, file);
52             String[] array = {
53                 "-Xdoclint:none",
54                 source.getPath(),
55                 "-d",
56                 destDir.getAbsolutePath()
57             };
58 
59             int rc = jdk.javadoc.internal.tool.Main.execute(array);
60             if (rc != 0)
61                 throw new Error("unexpected exit from javadoc: " + rc);
62         }
63     }
64 }
