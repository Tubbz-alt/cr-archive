 24 /*
 25  * @test
 26  * @bug 8224083 8224082
 27  * @summary javadoc Reporter generates "warning" for Kind.NOTE
 28  *          NPE in javadoc Reporter
 29  * @library /tools/lib ../../lib
 30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 31  *          jdk.compiler/com.sun.tools.javac.api
 32  *          jdk.compiler/com.sun.tools.javac.main
 33  *          jdk.javadoc/jdk.javadoc.internal.api
 34  *          jdk.javadoc/jdk.javadoc.internal.tool
 35  * @build toolbox.ToolBox javadoc.tester.*
 36  * @compile pkg/MyDoclet.java
 37  * @run main ReporterGeneratesWarningsInsteadOfNotes
 38  */
 39 
 40 import java.io.PrintWriter;
 41 import java.util.*;
 42 
 43 import java.nio.file.Files;
 44 import java.nio.file.Path;
 45 import java.nio.file.Paths;
 46 
 47 import javax.lang.model.SourceVersion;
 48 import javax.lang.model.element.TypeElement;
 49 import javax.lang.model.util.ElementFilter;
 50 
 51 import jdk.javadoc.doclet.Doclet;
 52 import jdk.javadoc.doclet.DocletEnvironment;
 53 import jdk.javadoc.doclet.Reporter;
 54 
 55 import javadoc.tester.JavadocTester;
 56 import toolbox.JavacTask;
 57 import toolbox.JavadocTask;
 58 import toolbox.TestRunner;
 59 import toolbox.Task;
 60 import toolbox.ToolBox;
 61 
 62 public class ReporterGeneratesWarningsInsteadOfNotes extends TestRunner {
 63     ToolBox tb = new ToolBox();
 64 
 65     static final String[] outputToCheckFor = new String[]{
 66             ": PACKAGE pkg",
 67             ": CLASS pkg.MyDoclet",
 68             ": CLASS pkg.MyDoclet.MyScanner",
 69             ": CONSTRUCTOR MyScanner()",
 70             ": METHOD scan(javax.lang.model.element.Element,java.lang.Integer)",
 71             ": PARAMETER e",
 72             ": PARAMETER depth",
 73             ": CLASS pkg.MyDoclet.Option",
 74             ": FIELD names",
 75             ": FIELD hasArg",
 76             ": FIELD description",
 77             ": CONSTRUCTOR Option(java.lang.String,boolean,java.lang.String)",
 78             ": PARAMETER names",
 79             ": PARAMETER hasArg",
 80             ": PARAMETER description",
 81             ": METHOD getArgumentCount()",
 82             ": METHOD getDescription()",
 83             ": METHOD getKind()",
 84             ": METHOD getNames()",
 85             ": METHOD getParameters()",
 86             ": CONSTRUCTOR MyDoclet()",
 87             ": FIELD OK",
 88             ": FIELD verbose",
 89             ": FIELD reporter",
 90             ": FIELD options",
 91             ": METHOD init(java.util.Locale,jdk.javadoc.doclet.Reporter)",
 92             ": PARAMETER locale",
 93             ": PARAMETER reporter",
 94             ": METHOD getName()",
 95             ": METHOD getSupportedOptions()",
 96             ": METHOD getSupportedSourceVersion()",
 97             ": METHOD run(jdk.javadoc.doclet.DocletEnvironment)",
 98             ": PARAMETER environment"
 99     };
100 
101     ReporterGeneratesWarningsInsteadOfNotes() throws Exception {
102         super(System.err);
103     }
104 
105     public static void main(String... args) throws Exception {
106         ReporterGeneratesWarningsInsteadOfNotes tester = new ReporterGeneratesWarningsInsteadOfNotes();
107         tester.runTests();
108     }
109 
110     protected void runTests() throws Exception {
111         runTests(m -> new Object[] { Paths.get(m.getName()) });
112     }
113 
114     @Test
115     public void testMain(Path base) throws Exception {
116         String testSrc = System.getProperty("test.src");
117         String testClasses = System.getProperty("test.classes");
118         String log = new JavadocTask(tb, Task.Mode.CMDLINE)
119                 .options("-docletpath", testClasses, "-doclet",  "pkg.MyDoclet", "-sourcepath", testSrc, "pkg")
120                 .run(Task.Expect.SUCCESS)
121                 .writeAll()
122                 .getOutput(Task.OutputKind.DIRECT);
123         for (String output : outputToCheckFor) {
124             if (!log.contains(output)) {
125                 throw new AssertionError("was expecting to find: " + output);
126             }
127         }
128     }
129 }
