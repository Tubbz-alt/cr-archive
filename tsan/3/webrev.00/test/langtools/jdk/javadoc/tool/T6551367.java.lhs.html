24 /**
25  * @test
26  * @bug     6551367
27  * @summary javadoc throws ClassCastException when an link tag tries to reference constructor.
28  * @author  A. Sundararajan
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @run main T6551367 T6551367.java
31  */
32 
33 /*
34  * ***NOTE: This is not a tool test!, should be moved to doclets.***
35  */
36 import java.io.File;
37 
38 import jdk.javadoc.doclet.DocletEnvironment;
39 
40 public class T6551367 {
41     public T6551367() {}
42     public boolean run(DocletEnvironment root) {
43         return true;
44     }
45     /** Here, in the javadoc for this method, I try to link to
46      *  {@link #<init> a constructor}.
47      */
48     public static void main(String... args) {
49         File testSrc = new File(System.getProperty("test.src", "."));
50         File destDir = new File(System.getProperty("user.dir", "."));
51         for (String file : args) {
52             File source = new File(testSrc, file);
53             String[] array = {
54                 "-Xdoclint:none",
55                 source.getPath(),
56                 "-d",
57                 destDir.getAbsolutePath()
58             };
59 
60             int rc = jdk.javadoc.internal.tool.Main.execute(array);
61             if (rc != 0)
62                 throw new Error("unexpected exit from javadoc: " + rc);
63         }
64     }
65 }
