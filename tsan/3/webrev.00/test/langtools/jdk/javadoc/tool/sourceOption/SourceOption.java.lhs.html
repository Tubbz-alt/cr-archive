 24 /*
 25  * @test
 26  * @bug     6507179
 27  * @summary Ensure that "-source" option isn't ignored.
 28  * @author  Scott Seligman
 29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 30  * @run main/fail SourceOption 7
 31  * @run main      SourceOption 9
 32  * @run main      SourceOption
 33  */
 34 
 35 /*
 36  * In order to test whether or not the -source option is working
 37  * correctly, this test tries to parse source code that contains
 38  * a feature that is not available in at least one of the currently
 39  * supported previous versions.
 40  *
 41  * Parsing such code should be expected to fail; if the action
 42  * passes, that means the -source option is (incorrectly) ineffective.
 43  *
 44  * Additional actions are performed to ensure that the source
 45  * provided is valid for the current release of the JDK.
 46  *
 47  * As support for older versions of the platform are dropped, the
 48  * source code (currently p/LambdaConstructTest.java) will need to
 49  * be updated with a more recent feature.
 50  */
 51 
 52 import java.util.ArrayList;
 53 import java.util.Collections;
 54 import java.util.List;
 55 import java.util.Locale;
 56 import java.util.Set;
 57 
 58 import javax.lang.model.SourceVersion;
 59 import javax.lang.model.util.ElementFilter;
 60 import javax.tools.Diagnostic.Kind;
 61 
 62 import jdk.javadoc.doclet.Doclet;
 63 import jdk.javadoc.doclet.Doclet.Option;
 64 import jdk.javadoc.doclet.DocletEnvironment;
 65 import jdk.javadoc.doclet.Reporter;
 66 
 67 public class SourceOption implements Doclet {
 68 
 69     public static void main(String[] args) {
 70         List<String> params = new ArrayList<>();
 71         params.add("-sourcepath");
 72         params.add(System.getProperty("test.src"));
 73         params.add("-docletpath");
 74         params.add(System.getProperty("test.classes"));
 75         params.add("-doclet");
 76         params.add("SourceOption");
 77         if ((args == null) || (args.length==0)) {
 78             System.out.println("NOTE : -source not provided, default taken");
 79         } else {
 80             params.add("-source");
 81             params.add(args[0]);
 82             System.out.println("NOTE : -source will be: " + args[0]);
 83         }
 84         params.add("p");
 85         System.out.println("arguments: " + params);
 86         if (jdk.javadoc.internal.tool.Main.execute(params.toArray(new String[params.size()])) != 0)
 87             throw new Error("Javadoc encountered warnings or errors.");
 88     }
 89 
 90     public boolean run(DocletEnvironment root) {
 91         ElementFilter.typesIn(root.getIncludedElements());
 92         return true;
 93     }
 94 
 95     @Override
 96     public String getName() {
 97         return "Test";
 98     }
 99 
100     @Override
101     public Set<Option> getSupportedOptions() {
102         return Collections.emptySet();
103     }
104 
105     @Override
106     public SourceVersion getSupportedSourceVersion() {
107         return SourceVersion.latest();
108     }
109 
110     @Override
111     public void init(Locale locale, Reporter reporter) {
112         reporter.print(Kind.NOTE, "init");
113     }
114 }
