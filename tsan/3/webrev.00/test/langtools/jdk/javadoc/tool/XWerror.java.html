24 /*
25  * @test
26  * @bug 4099527
27  * @summary javadoc tool: want flag to exit nonzero if there were warnings.
28  * @modules jdk.javadoc/jdk.javadoc.internal.tool
29  * @run main XWerror
30  */
31 
32 import java.io.PrintWriter;
33 import java.io.StringWriter;
34 import java.util.Collections;
35 import java.util.Locale;
36 import java.util.Set;
37 
38 import javax.lang.model.SourceVersion;
39 import javax.tools.Diagnostic;
40 
41 import jdk.javadoc.doclet.Doclet;
42 import jdk.javadoc.doclet.Reporter;
43 import jdk.javadoc.doclet.DocletEnvironment;
44 
45 public class XWerror implements Doclet {
46 
47     private static final String message = "warning message";
48 
49     public static void main(String[] args) {
50         StringWriter sw = new StringWriter();
51         PrintWriter output = new PrintWriter(sw);
52 
53         String[] aargs = {
54             "-docletpath", System.getProperty("test.classes", "."),
55             "-doclet", "XWerror",
56             "-Xwerror",
57             System.getProperty("test.src", ".") + java.io.File.separatorChar
58             + "XWerror.java"
59         };
60         if (jdk.javadoc.internal.tool.Main.execute(aargs, output) == 0) {
61             throw new Error("did not get non-zero exit code");
62         }
63         if (!sw.toString().contains(message)) {
64             throw new Error("expected message not found: " + message);
65         }
66     }
67     Reporter reporter;
68 
69     public boolean run(DocletEnvironment root) {
70         reporter.print(Diagnostic.Kind.WARNING , message);
71         return false;
72     }
73 
74     @Override
75     public String getName() {
76         return "Test";
77     }
78 
79     @Override
80     public Set<Option> getSupportedOptions() {
81         return Collections.emptySet();
82     }
83 
84     @Override
85     public SourceVersion getSupportedSourceVersion() {
86         return SourceVersion.latest();
87     }
88 
89     @Override
90     public void init(Locale locale, Reporter reporter) {
91         this.reporter = reporter;
92         return;
93     }
94 }
