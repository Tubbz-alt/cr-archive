  77 // Check core dump limit and report possible place where core can be found
  78 void os::check_dump_limit(char* buffer, size_t bufferSize) {
  79   if (!FLAG_IS_DEFAULT(CreateCoredumpOnCrash) && !CreateCoredumpOnCrash) {
  80     jio_snprintf(buffer, bufferSize, "CreateCoredumpOnCrash is disabled from command line");
  81     VMError::record_coredump_status(buffer, false);
  82     return;
  83   }
  84 
  85   int n;
  86   struct rlimit rlim;
  87   bool success;
  88 
  89   char core_path[PATH_MAX];
  90   n = get_core_path(core_path, PATH_MAX);
  91 
  92   if (n <= 0) {
  93     jio_snprintf(buffer, bufferSize, "core.%d (may not exist)", current_process_id());
  94     success = true;
  95 #ifdef LINUX
  96   } else if (core_path[0] == '"') { // redirect to user process
  97     jio_snprintf(buffer, bufferSize, "Core dumps may be processed with %s", core_path);
  98     success = true;
  99 #endif
 100   } else if (getrlimit(RLIMIT_CORE, &rlim) != 0) {
 101     jio_snprintf(buffer, bufferSize, "%s (may not exist)", core_path);
 102     success = true;
 103   } else {
 104     switch(rlim.rlim_cur) {
 105       case RLIM_INFINITY:
 106         jio_snprintf(buffer, bufferSize, "%s", core_path);
 107         success = true;
 108         break;
 109       case 0:
 110         jio_snprintf(buffer, bufferSize, "Core dumps have been disabled. To enable core dumping, try \"ulimit -c unlimited\" before starting Java again");
 111         success = false;
 112         break;
 113       default:
 114         jio_snprintf(buffer, bufferSize, "%s (max size " UINT64_FORMAT " kB). To ensure a full core dump, try \"ulimit -c unlimited\" before starting Java again", core_path, uint64_t(rlim.rlim_cur) / 1024);
 115         success = true;
 116         break;
 117     }
 118   }
 119 
 120   VMError::record_coredump_status(buffer, success);
 121 }

 123 int os::get_native_stack(address* stack, int frames, int toSkip) {
 124   int frame_idx = 0;
 125   int num_of_frames;  // number of frames captured
 126   frame fr = os::current_frame();
 127   while (fr.pc() && frame_idx < frames) {
 128     if (toSkip > 0) {
 129       toSkip --;
 130     } else {
 131       stack[frame_idx ++] = fr.pc();
 132     }
 133     if (fr.fp() == NULL || fr.cb() != NULL ||
 134         fr.sender_pc() == NULL || os::is_first_C_frame(&fr)) break;
 135 
 136     if (fr.sender_pc() && !os::is_first_C_frame(&fr)) {
 137       fr = os::get_sender_for_C_frame(&fr);
 138     } else {
 139       break;
 140     }
 141   }
 142   num_of_frames = frame_idx;
 143   for (; frame_idx < frames; frame_idx ++) {
 144     stack[frame_idx] = NULL;
 145   }
 146 
 147   return num_of_frames;
 148 }

 160 size_t os::lasterror(char *buf, size_t len) {
 161   if (errno == 0)  return 0;
 162 
 163   const char *s = os::strerror(errno);
 164   size_t n = ::strlen(s);
 165   if (n >= len) {
 166     n = len - 1;
 167   }
 168   ::strncpy(buf, s, n);
 169   buf[n] = '\0';
 170   return n;
 171 }

 178 int os::create_file_for_heap(const char* dir) {
 179   int fd;
 180 
 181 #if defined(LINUX) && defined(O_TMPFILE)
 182   char* native_dir = os::strdup(dir);
 183   if (native_dir == NULL) {
 184     vm_exit_during_initialization(err_msg("strdup failed during creation of backing file for heap (%s)", os::strerror(errno)));
 185     return -1;
 186   }
 187   os::native_path(native_dir);
 188   fd = os::open(dir, O_TMPFILE | O_RDWR, S_IRUSR | S_IWUSR);
 189   os::free(native_dir);
 190 
 191   if (fd == -1)
 192 #endif
 193   {
 194     const char name_template[] = "/jvmheap.XXXXXX";
 195 
 196     size_t fullname_len = strlen(dir) + strlen(name_template);
 197     char *fullname = (char*)os::malloc(fullname_len + 1, mtInternal);
 198     if (fullname == NULL) {
 199       vm_exit_during_initialization(err_msg("Malloc failed during creation of backing file for heap (%s)", os::strerror(errno)));
 200       return -1;
 201     }
 202     int n = snprintf(fullname, fullname_len + 1, "%s%s", dir, name_template);
 203     assert((size_t)n == fullname_len, "Unexpected number of characters in string");
 204 
 205     os::native_path(fullname);
 206 
 207     // create a new file.
 208     fd = mkstemp(fullname);
 209 
 210     if (fd < 0) {
 211       warning("Could not create file for heap with template %s", fullname);
 212       os::free(fullname);
 213       return -1;
 214     } else {
 215       // delete the name from the filesystem. When 'fd' is closed, the file (and space) will be deleted.
 216       int ret = unlink(fullname);
 217       assert_with_errno(ret == 0, "unlink returned error");
 218     }
 219 
 220     os::free(fullname);
 221   }
 222 
 223   return fd;
 224 }

 226 static char* reserve_mmapped_memory(size_t bytes, char* requested_addr) {
 227   char * addr;
 228   int flags = MAP_PRIVATE NOT_AIX( | MAP_NORESERVE ) | MAP_ANONYMOUS;
 229   if (requested_addr != NULL) {
 230     assert((uintptr_t)requested_addr % os::vm_page_size() == 0, "Requested address should be aligned to OS page size");
 231     flags |= MAP_FIXED;
 232   }
 233 
 234   // Map reserved/uncommitted pages PROT_NONE so we fail early if we
 235   // touch an uncommitted page. Otherwise, the read/write might
 236   // succeed if we have enough swap space to back the physical page.
 237   addr = (char*)::mmap(requested_addr, bytes, PROT_NONE,
 238                        flags, -1, 0);
 239 
 240   if (addr != MAP_FAILED) {
 241     MemTracker::record_virtual_memory_reserve((address)addr, bytes, CALLER_PC);
 242     return addr;
 243   }
 244   return NULL;
 245 }

 247 static int util_posix_fallocate(int fd, off_t offset, off_t len) {
 248 #ifdef __APPLE__
 249   fstore_t store = { F_ALLOCATECONTIG, F_PEOFPOSMODE, 0, len };
 250   // First we try to get a continuous chunk of disk space
 251   int ret = fcntl(fd, F_PREALLOCATE, &store);
 252   if (ret == -1) {
 253     // Maybe we are too fragmented, try to allocate non-continuous range
 254     store.fst_flags = F_ALLOCATEALL;
 255     ret = fcntl(fd, F_PREALLOCATE, &store);
 256   }
 257   if(ret != -1) {
 258     return ftruncate(fd, len);
 259   }
 260   return -1;
 261 #else
 262   return posix_fallocate(fd, offset, len);
 263 #endif
 264 }

 266 // Map the given address range to the provided file descriptor.
 267 char* os::map_memory_to_file(char* base, size_t size, int fd) {
 268   assert(fd != -1, "File descriptor is not valid");
 269 
 270   // allocate space for the file
 271   int ret = util_posix_fallocate(fd, 0, (off_t)size);
 272   if (ret != 0) {
 273     vm_exit_during_initialization(err_msg("Error in mapping Java heap at the given filesystem directory. error(%d)", ret));
 274     return NULL;
 275   }
 276 
 277   int prot = PROT_READ | PROT_WRITE;
 278   int flags = MAP_SHARED;
 279   if (base != NULL) {
 280     flags |= MAP_FIXED;
 281   }
 282   char* addr = (char*)mmap(base, size, prot, flags, fd, 0);
 283 
 284   if (addr == MAP_FAILED) {
 285     warning("Failed mmap to file. (%s)", os::strerror(errno));
 286     return NULL;
 287   }
 288   if (base != NULL && addr != base) {
 289     if (!os::release_memory(addr, size)) {
 290       warning("Could not release memory on unsuccessful file mapping");
 291     }
 292     return NULL;
 293   }
 294   return addr;
 295 }

 304 // Multiple threads can race in this code, and can remap over each other with MAP_FIXED,
 305 // so on posix, unmap the section at the start and at the end of the chunk that we mapped
 306 // rather than unmapping and remapping the whole chunk to get requested alignment.
 307 char* os::reserve_memory_aligned(size_t size, size_t alignment, int file_desc) {
 308   assert((alignment & (os::vm_allocation_granularity() - 1)) == 0,
 309       "Alignment must be a multiple of allocation granularity (page size)");
 310   assert((size & (alignment -1)) == 0, "size must be 'alignment' aligned");
 311 
 312   size_t extra_size = size + alignment;
 313   assert(extra_size >= size, "overflow, size is too large to allow alignment");
 314 
 315   char* extra_base;
 316   if (file_desc != -1) {
 317     // For file mapping, we do not call os:reserve_memory(extra_size, NULL, alignment, file_desc) because
 318     // we need to deal with shrinking of the file space later when we release extra memory after alignment.
 319     // We also cannot called os:reserve_memory() with file_desc set to -1 because on aix we might get SHM memory.
 320     // So here to call a helper function while reserve memory for us. After we have a aligned base,
 321     // we will replace anonymous mapping with file mapping.
 322     extra_base = reserve_mmapped_memory(extra_size, NULL);
 323     if (extra_base != NULL) {
 324       MemTracker::record_virtual_memory_reserve((address)extra_base, extra_size, CALLER_PC);
 325     }
 326   } else {
 327     extra_base = os::reserve_memory(extra_size, NULL, alignment);
 328   }
 329 
 330   if (extra_base == NULL) {
 331     return NULL;
 332   }
 333 
 334   // Do manual alignment
 335   char* aligned_base = align_up(extra_base, alignment);
 336 
 337   // [  |                                       |  ]
 338   // ^ extra_base
 339   //    ^ extra_base + begin_offset == aligned_base
 340   //     extra_base + begin_offset + size       ^
 341   //                       extra_base + extra_size ^
 342   // |<>| == begin_offset
 343   //                              end_offset == |<>|
 344   size_t begin_offset = aligned_base - extra_base;
 345   size_t end_offset = (extra_base + extra_size) - (aligned_base + size);
 346 
 347   if (begin_offset > 0) {
 348       os::release_memory(extra_base, begin_offset);
 349   }
 350 
 351   if (end_offset > 0) {
 352       os::release_memory(extra_base + begin_offset + size, end_offset);
 353   }
 354 
 355   if (file_desc != -1) {
 356     // After we have an aligned address, we can replace anonymous mapping with file mapping
 357     if (replace_existing_mapping_with_file_mapping(aligned_base, size, file_desc) == NULL) {
 358       vm_exit_during_initialization(err_msg("Error in mapping Java heap at the given filesystem directory"));
 359     }
 360     MemTracker::record_virtual_memory_commit((address)aligned_base, size, CALLER_PC);
 361   }
 362   return aligned_base;
 363 }

 365 int os::vsnprintf(char* buf, size_t len, const char* fmt, va_list args) {
 366   // All supported POSIX platforms provide C99 semantics.
 367   int result = ::vsnprintf(buf, len, fmt, args);
 368   // If an encoding error occurred (result < 0) then it's not clear
 369   // whether the buffer is NUL terminated, so ensure it is.
 370   if ((result < 0) && (len > 0)) {
 371     buf[len - 1] = '\0';
 372   }
 373   return result;
 374 }

 528 bool os::has_allocatable_memory_limit(julong* limit) {
 529   struct rlimit rlim;
 530   int getrlimit_res = getrlimit(RLIMIT_AS, &rlim);
 531   // if there was an error when calling getrlimit, assume that there is no limitation
 532   // on virtual memory.
 533   bool result;
 534   if ((getrlimit_res != 0) || (rlim.rlim_cur == RLIM_INFINITY)) {
 535     result = false;
 536   } else {
 537     *limit = (julong)rlim.rlim_cur;
 538     result = true;
 539   }
 540 
 541 #if defined(INCLUDE_TSAN) && defined(AARCH64)
 542   // Current TSAN memory mapping for 48bits aarch64, a large continuous space could be allocated between
 543   // kMidAppMemBeg and kMidAppMemEnd, which is only 20GB size.
 544   // Take 16GB here for safer allocation.
 545   const julong max_avail_vmspace = 16ULL * G; // 16GB
 546   const u8 msb_in_aarch64 = 47; // Only support 48-bits space now.
 547 
 548   // Based on current TSAN aarch64 memory mapping, libjvm.so will be loaded to high end address space.
 549   u8 vm_addr_u8 = reinterpret_cast<u8>(&__FUNCTION__);
 550   // High address in 48bits user space is like 0x0000ffffxxxxxxxx.
 551   assert((vm_addr_u8  >> msb_in_aarch64) == 0x1, "warning: allocation could fail in non 48-bit address space.");
 552 
 553   if (result) {
 554     *limit = MIN2(*limit, max_avail_vmspace);
 555   } else {
 556     *limit = max_avail_vmspace;
 557   }
 558 
 559   result = true;
 560 #endif
 561 
 562 #ifdef _LP64
 563   return result;
 564 #else
 565   // arbitrary virtual space limit for 32 bit Unices found by testing. If
 566   // getrlimit above returned a limit, bound it with this limit. Otherwise
 567   // directly use it.
 568   const julong max_virtual_limit = (julong)3800*M;
 569   if (result) {
 570     *limit = MIN2(*limit, max_virtual_limit);
 571   } else {
 572     *limit = max_virtual_limit;
 573   }
 574 
 575   // bound by actually allocatable memory. The algorithm uses two bounds, an
 576   // upper and a lower limit. The upper limit is the current highest amount of
 577   // memory that could not be allocated, the lower limit is the current highest
 578   // amount of memory that could be allocated.
 579   // The algorithm iteratively refines the result by halving the difference
 580   // between these limits, updating either the upper limit (if that value could
 581   // not be allocated) or the lower limit (if the that value could be allocated)
 582   // until the difference between these limits is "small".
 583 
 584   // the minimum amount of memory we care about allocating.
 585   const julong min_allocation_size = M;
 586 
 587   julong upper_limit = *limit;
 588 
 589   // first check a few trivial cases
 590   if (is_allocatable(upper_limit) || (upper_limit <= min_allocation_size)) {
 591     *limit = upper_limit;
 592   } else if (!is_allocatable(min_allocation_size)) {
 593     // we found that not even min_allocation_size is allocatable. Return it
 594     // anyway. There is no point to search for a better value any more.
 595     *limit = min_allocation_size;
 596   } else {
 597     // perform the binary search.
 598     julong lower_limit = min_allocation_size;

 601       temp_limit = align_down(temp_limit, min_allocation_size);
 602       if (is_allocatable(temp_limit)) {
 603         lower_limit = temp_limit;
 604       } else {
 605         upper_limit = temp_limit;
 606       }
 607     }
 608     *limit = lower_limit;
 609   }
 610   return true;

 649 // Builds a platform dependent Agent_OnLoad_<lib_name> function name
 650 // which is used to find statically linked in agents.
 651 // Parameters:
 652 //            sym_name: Symbol in library we are looking for
 653 //            lib_name: Name of library to look in, NULL for shared libs.
 654 //            is_absolute_path == true if lib_name is absolute path to agent
 655 //                                     such as "/a/b/libL.so"
 656 //            == false if only the base name of the library is passed in
 657 //               such as "L"
 658 char* os::build_agent_function_name(const char *sym_name, const char *lib_name,
 659                                     bool is_absolute_path) {
 660   char *agent_entry_name;
 661   size_t len;
 662   size_t name_len;
 663   size_t prefix_len = strlen(JNI_LIB_PREFIX);
 664   size_t suffix_len = strlen(JNI_LIB_SUFFIX);
 665   const char *start;
 666 
 667   if (lib_name != NULL) {
 668     name_len = strlen(lib_name);
 669     if (is_absolute_path) {
 670       // Need to strip path, prefix and suffix
 671       if ((start = strrchr(lib_name, *os::file_separator())) != NULL) {
 672         lib_name = ++start;
 673       }
 674       if (strlen(lib_name) <= (prefix_len + suffix_len)) {
 675         return NULL;
 676       }
 677       lib_name += prefix_len;
 678       name_len = strlen(lib_name) - suffix_len;
 679     }
 680   }
 681   len = (lib_name != NULL ? name_len : 0) + strlen(sym_name) + 2;
 682   agent_entry_name = NEW_C_HEAP_ARRAY_RETURN_NULL(char, len, mtThread);
 683   if (agent_entry_name == NULL) {
 684     return NULL;
 685   }
 686   strcpy(agent_entry_name, sym_name);
 687   if (lib_name != NULL) {
 688     strcat(agent_entry_name, "_");
 689     strncat(agent_entry_name, lib_name, name_len);
 690   }
 691   return agent_entry_name;
 692 }

 874 // Returned string is a constant. For unknown signals "UNKNOWN" is returned.
 875 const char* os::Posix::get_signal_name(int sig, char* out, size_t outlen) {
 876 
 877   const char* ret = NULL;
 878 
 879 #ifdef SIGRTMIN
 880   if (sig >= SIGRTMIN && sig <= SIGRTMAX) {
 881     if (sig == SIGRTMIN) {
 882       ret = "SIGRTMIN";
 883     } else if (sig == SIGRTMAX) {
 884       ret = "SIGRTMAX";
 885     } else {
 886       jio_snprintf(out, outlen, "SIGRTMIN+%d", sig - SIGRTMIN);
 887       return out;
 888     }
 889   }
 890 #endif
 891 
 892   if (sig > 0) {
 893     for (int idx = 0; g_signal_info[idx].sig != -1; idx ++) {
 894       if (g_signal_info[idx].sig == sig) {
 895         ret = g_signal_info[idx].name;
 896         break;
 897       }
 898     }
 899   }
 900 
 901   if (!ret) {
 902     if (!is_valid_signal(sig)) {
 903       ret = "INVALID";
 904     } else {
 905       ret = "UNKNOWN";
 906     }
 907   }
 908 
 909   if (out && outlen > 0) {
 910     strncpy(out, ret, outlen);
 911     out[outlen - 1] = '\0';
 912   }
 913   return out;
 914 }

 935 // Returns true if signal number is valid.
 936 bool os::Posix::is_valid_signal(int sig) {
 937   // MacOS not really POSIX compliant: sigaddset does not return
 938   // an error for invalid signal numbers. However, MacOS does not
 939   // support real time signals and simply seems to have just 33
 940   // signals with no holes in the signal range.
 941 #ifdef __APPLE__
 942   return sig >= 1 && sig < NSIG;
 943 #else
 944   // Use sigaddset to check for signal validity.
 945   sigset_t set;
 946   sigemptyset(&set);
 947   if (sigaddset(&set, sig) == -1 && errno == EINVAL) {
 948     return false;
 949   }
 950   return true;
 951 #endif
 952 }

 958                                  : CAST_FROM_FN_PTR(void*,  oact.sa_handler);
 959   if (ohlr == CAST_FROM_FN_PTR(void*, SIG_IGN)) {
 960     return true;
 961   } else {
 962     return false;

1347 char* os::Posix::realpath(const char* filename, char* outbuf, size_t outbuflen) {
1348 
1349   if (filename == NULL || outbuf == NULL || outbuflen < 1) {
1350     assert(false, "os::Posix::realpath: invalid arguments.");
1351     errno = EINVAL;
1352     return NULL;
1353   }
1354 
1355   char* result = NULL;
1356 
1357   // This assumes platform realpath() is implemented according to POSIX.1-2008.
1358   // POSIX.1-2008 allows to specify NULL for the output buffer, in which case
1359   // output buffer is dynamically allocated and must be ::free()'d by the caller.
1360   char* p = ::realpath(filename, NULL);
1361   if (p != NULL) {
1362     if (strlen(p) < outbuflen) {
1363       strcpy(outbuf, p);
1364       result = outbuf;
1365     } else {
1366       errno = ENAMETOOLONG;
1367     }
1368     ::free(p); // *not* os::free
1369   } else {
1370     // Fallback for platforms struggling with modern Posix standards (AIX 5.3, 6.1). If realpath
1371     // returns EINVAL, this may indicate that realpath is not POSIX.1-2008 compatible and
1372     // that it complains about the NULL we handed down as user buffer.
1373     // In this case, use the user provided buffer but at least check whether realpath caused
1374     // a memory overwrite.
1375     if (errno == EINVAL) {
1376       outbuf[outbuflen - 1] = '\0';
1377       p = ::realpath(filename, outbuf);
1378       if (p != NULL) {
1379         guarantee(outbuf[outbuflen - 1] == '\0', "realpath buffer overwrite detected.");
1380         result = p;
1381       }
1382     }
1383   }
1384   return result;

1420 // Check minimum allowable stack sizes for thread creation and to initialize
1421 // the java system classes, including StackOverflowError - depends on page
1422 // size.
1423 // The space needed for frames during startup is platform dependent. It
1424 // depends on word size, platform calling conventions, C frame layout and
1425 // interpreter/C1/C2 design decisions. Therefore this is given in a
1426 // platform (os/cpu) dependent constant.
1427 // To this, space for guard mechanisms is added, which depends on the
1428 // page size which again depends on the concrete system the VM is running
1429 // on. Space for libc guard pages is not included in this size.
1430 jint os::Posix::set_minimum_stack_sizes() {
1431   size_t os_min_stack_allowed = SOLARIS_ONLY(thr_min_stack()) NOT_SOLARIS(PTHREAD_STACK_MIN);
1432 
1433   _java_thread_min_stack_allowed = _java_thread_min_stack_allowed +
1434                                    JavaThread::stack_guard_zone_size() +
1435                                    JavaThread::stack_shadow_zone_size();
1436 
1437   _java_thread_min_stack_allowed = align_up(_java_thread_min_stack_allowed, vm_page_size());
1438   _java_thread_min_stack_allowed = MAX2(_java_thread_min_stack_allowed, os_min_stack_allowed);
1439 
1440   size_t stack_size_in_bytes = ThreadStackSize * K;
1441   if (stack_size_in_bytes != 0 &&
1442       stack_size_in_bytes < _java_thread_min_stack_allowed) {
1443     // The '-Xss' and '-XX:ThreadStackSize=N' options both set
1444     // ThreadStackSize so we go with "Java thread stack size" instead
1445     // of "ThreadStackSize" to be more friendly.
1446     tty->print_cr("\nThe Java thread stack size specified is too small. "
1447                   "Specify at least " SIZE_FORMAT "k",
1448                   _java_thread_min_stack_allowed / K);
1449     return JNI_ERR;
1450   }
1451 
1452   // Make the stack size a multiple of the page size so that
1453   // the yellow/red zones can be guarded.
1454   JavaThread::set_stack_size_at_create(align_up(stack_size_in_bytes, vm_page_size()));
1455 
1456   // Reminder: a compiler thread is a Java thread.
1457   _compiler_thread_min_stack_allowed = _compiler_thread_min_stack_allowed +
1458                                        JavaThread::stack_guard_zone_size() +
1459                                        JavaThread::stack_shadow_zone_size();
1460 
1461   _compiler_thread_min_stack_allowed = align_up(_compiler_thread_min_stack_allowed, vm_page_size());
1462   _compiler_thread_min_stack_allowed = MAX2(_compiler_thread_min_stack_allowed, os_min_stack_allowed);
1463 
1464   stack_size_in_bytes = CompilerThreadStackSize * K;
1465   if (stack_size_in_bytes != 0 &&
1466       stack_size_in_bytes < _compiler_thread_min_stack_allowed) {
1467     tty->print_cr("\nThe CompilerThreadStackSize specified is too small. "
1468                   "Specify at least " SIZE_FORMAT "k",
1469                   _compiler_thread_min_stack_allowed / K);
1470     return JNI_ERR;
1471   }
1472

1474   _vm_internal_thread_min_stack_allowed = MAX2(_vm_internal_thread_min_stack_allowed, os_min_stack_allowed);
1475 
1476   stack_size_in_bytes = VMThreadStackSize * K;
1477   if (stack_size_in_bytes != 0 &&
1478       stack_size_in_bytes < _vm_internal_thread_min_stack_allowed) {
1479     tty->print_cr("\nThe VMThreadStackSize specified is too small. "
1480                   "Specify at least " SIZE_FORMAT "k",
1481                   _vm_internal_thread_min_stack_allowed / K);
1482     return JNI_ERR;
1483   }

1487 // Called when creating the thread.  The minimum stack sizes have already been calculated
1488 size_t os::Posix::get_initial_stack_size(ThreadType thr_type, size_t req_stack_size) {
1489   size_t stack_size;
1490   if (req_stack_size == 0) {
1491     stack_size = default_stack_size(thr_type);
1492   } else {
1493     stack_size = req_stack_size;
1494   }
1495 
1496   switch (thr_type) {
1497   case os::java_thread:
1498     // Java threads use ThreadStackSize which default value can be
1499     // changed with the flag -Xss
1500     if (req_stack_size == 0 && JavaThread::stack_size_at_create() > 0) {
1501       // no requested size and we have a more specific default value
1502       stack_size = JavaThread::stack_size_at_create();
1503     }
1504     stack_size = MAX2(stack_size,
1505                       _java_thread_min_stack_allowed);
1506     break;
1507   case os::compiler_thread:
1508     if (req_stack_size == 0 && CompilerThreadStackSize > 0) {
1509       // no requested size and we have a more specific default value
1510       stack_size = (size_t)(CompilerThreadStackSize * K);
1511     }
1512     stack_size = MAX2(stack_size,
1513                       _compiler_thread_min_stack_allowed);
1514     break;
1515   case os::vm_thread:
1516   case os::pgc_thread:
1517   case os::cgc_thread:
1518   case os::watcher_thread:
1519   default:  // presume the unknown thr_type is a VM internal
1520     if (req_stack_size == 0 && VMThreadStackSize > 0) {
1521       // no requested size and we have a more specific default value
1522       stack_size = (size_t)(VMThreadStackSize * K);
1523     }
1524 
1525     stack_size = MAX2(stack_size,
1526                       _vm_internal_thread_min_stack_allowed);
1527     break;
1528   }
1529 
1530   // pthread_attr_setstacksize() may require that the size be rounded up to the OS page size.
1531   // Be careful not to round up to 0. Align down in that case.
1532   if (stack_size <= SIZE_MAX - vm_page_size()) {
1533     stack_size = align_up(stack_size, vm_page_size());
1534   } else {
1535     stack_size = align_down(stack_size, vm_page_size());
1536   }
1537 
1538   return stack_size;
1539 }

1560 /*
1561  * See the caveats for this class in os_posix.hpp
1562  * Protects the callback call so that SIGSEGV / SIGBUS jumps back into this
1563  * method and returns false. If none of the signals are raised, returns true.
1564  * The callback is supposed to provide the method that should be protected.
1565  */
1566 bool os::ThreadCrashProtection::call(os::CrashProtectionCallback& cb) {
1567   sigset_t saved_sig_mask;
1568 
1569   Thread::muxAcquire(&_crash_mux, "CrashProtection");

1580     // installed
1581     _crash_protection = this;

1584     _crash_protection = NULL;
1585     _protected_thread = NULL;
1586     Thread::muxRelease(&_crash_mux);
1587     return true;
1588   }
1589   // this happens when we siglongjmp() back
1590   pthread_sigmask(SIG_SETMASK, &saved_sig_mask, NULL);
1591   _crash_protection = NULL;
1592   _protected_thread = NULL;
1593   Thread::muxRelease(&_crash_mux);

1711     (int(*)(clockid_t, struct timespec*))dlsym(handle, "clock_gettime");

1714     // CLOCK_MONOTONIC then the OS provides true high-res monotonic clock.

1794 // Utility to convert the given timeout to an absolute timespec
1795 // (based on the appropriate clock) to use with pthread_cond_timewait,
1796 // and sem_timedwait().
1797 // The clock queried here must be the clock used to manage the
1798 // timeout of the condition variable or semaphore.
1799 //
1800 // The passed in timeout value is either a relative time in nanoseconds
1801 // or an absolute time in milliseconds. A relative timeout will be
1802 // associated with CLOCK_MONOTONIC if available, unless the real-time clock
1803 // is explicitly requested; otherwise, or if absolute,
1804 // the default time-of-day clock will be used.
1805 
1806 // Given time is a 64-bit value and the time_t used in the timespec is
1807 // sometimes a signed-32-bit value we have to watch for overflow if times
1808 // way in the future are given. Further on Solaris versions
1809 // prior to 10 there is a restriction (see cond_timedwait) that the specified
1810 // number of seconds, in abstime, is less than current_time + 100000000.
1811 // As it will be over 20 years before "now + 100000000" will overflow we can
1812 // ignore overflow and just impose a hard-limit on seconds using the value
1813 // of "now + 100000000". This places a limit on the timeout of about 3.17
1814 // years from "now".
1815 //
1816 #define MAX_SECS 100000000
1817 
1818 // Calculate a new absolute time that is "timeout" nanoseconds from "now".
1819 // "unit" indicates the unit of "now_part_sec" (may be nanos or micros depending
1820 // on which clock API is being used).
1821 static void calc_rel_time(timespec* abstime, jlong timeout, jlong now_sec,
1822                           jlong now_part_sec, jlong unit) {
1823   time_t max_secs = now_sec + MAX_SECS;
1824 
1825   jlong seconds = timeout / NANOUNITS;
1826   timeout %= NANOUNITS; // remaining nanos
1827 
1828   if (seconds >= MAX_SECS) {
1829     // More seconds than we can add, so pin to max_secs.
1830     abstime->tv_sec = max_secs;
1831     abstime->tv_nsec = 0;
1832   } else {
1833     abstime->tv_sec = now_sec  + seconds;
1834     long nanos = (now_part_sec * (NANOUNITS / unit)) + timeout;
1835     if (nanos >= NANOUNITS) { // overflow
1836       abstime->tv_sec += 1;
1837       nanos -= NANOUNITS;
1838     }
1839     abstime->tv_nsec = nanos;
1840   }
1841 }
1842 
1843 // Unpack the given deadline in milliseconds since the epoch, into the given timespec.
1844 // The current time in seconds is also passed in to enforce an upper bound as discussed above.
1845 // This is only used with gettimeofday, when clock_gettime is not available.
1846 static void unpack_abs_time(timespec* abstime, jlong deadline, jlong now_sec) {
1847   time_t max_secs = now_sec + MAX_SECS;
1848 
1849   jlong seconds = deadline / MILLIUNITS;
1850   jlong millis = deadline % MILLIUNITS;
1851 
1852   if (seconds >= max_secs) {
1853     // Absolute seconds exceeds allowed max, so pin to max_secs.
1854     abstime->tv_sec = max_secs;
1855     abstime->tv_nsec = 0;
1856   } else {
1857     abstime->tv_sec = seconds;
1858     abstime->tv_nsec = millis_to_nanos(millis);
1859   }
1860 }
1861 
1862 static jlong millis_to_nanos_bounded(jlong millis) {
1863   // We have to watch for overflow when converting millis to nanos,
1864   // but if millis is that large then we will end up limiting to
1865   // MAX_SECS anyway, so just do that here.
1866   if (millis / MILLIUNITS > MAX_SECS) {
1867     millis = jlong(MAX_SECS) * MILLIUNITS;
1868   }
1869   return millis_to_nanos(millis);
1870 }

1872 static void to_abstime(timespec* abstime, jlong timeout,
1873                        bool isAbsolute, bool isRealtime) {
1874   DEBUG_ONLY(int max_secs = MAX_SECS;)
1875 
1876   if (timeout < 0) {
1877     timeout = 0;
1878   }
1879 
1880 #ifdef SUPPORTS_CLOCK_MONOTONIC
1881 
1882   clockid_t clock = CLOCK_MONOTONIC;
1883   // need to ensure we have a runtime check for clock_gettime support
1884   if (!isAbsolute && os::Posix::supports_monotonic_clock()) {
1885     if (!_use_clock_monotonic_condattr || isRealtime) {
1886       clock = CLOCK_REALTIME;
1887     }
1888     struct timespec now;
1889     int status = os::Posix::clock_gettime(clock, &now);
1890     assert_status(status == 0, status, "clock_gettime");
1891     calc_rel_time(abstime, timeout, now.tv_sec, now.tv_nsec, NANOUNITS);
1892     DEBUG_ONLY(max_secs += now.tv_sec;)
1893   } else {
1894 
1895 #else
1896 
1897   { // Match the block scope.
1898 
1899 #endif // SUPPORTS_CLOCK_MONOTONIC
1900 
1901     // Time-of-day clock is all we can reliably use.
1902     struct timeval now;
1903     int status = gettimeofday(&now, NULL);
1904     assert_status(status == 0, errno, "gettimeofday");
1905     if (isAbsolute) {
1906       unpack_abs_time(abstime, timeout, now.tv_sec);
1907     } else {
1908       calc_rel_time(abstime, timeout, now.tv_sec, now.tv_usec, MICROUNITS);
1909     }
1910     DEBUG_ONLY(max_secs += now.tv_sec;)
1911   }
1912 
1913   assert(abstime->tv_sec >= 0, "tv_sec < 0");
1914   assert(abstime->tv_sec <= max_secs, "tv_sec > max_secs");
1915   assert(abstime->tv_nsec >= 0, "tv_nsec < 0");
1916   assert(abstime->tv_nsec < NANOUNITS, "tv_nsec >= NANOUNITS");

1933 //
1934 // Assumption:
1935 //    Only one parker can exist on an event, which is why we allocate
1936 //    them per-thread. Multiple unparkers can coexist.

1956 void os::PlatformEvent::park() {       // AKA "down()"
1957   // Transitions for _event:
1958   //   -1 => -1 : illegal
1959   //    1 =>  0 : pass - return immediately
1960   //    0 => -1 : block; then set _event to 0 before returning
1961 
1962   // Invariant: Only the thread associated with the PlatformEvent
1963   // may call park().
1964   assert(_nParked == 0, "invariant");
1965 
1966   int v;
1967 
1968   // atomically decrement _event
1969   for (;;) {
1970     v = _event;
1971     if (Atomic::cmpxchg(&_event, v, v - 1) == v) break;
1972   }
1973   guarantee(v >= 0, "invariant");
1974 
1975   if (v == 0) { // Do this the hard way by blocking ...
1976     int status = pthread_mutex_lock(_mutex);
1977     assert_status(status == 0, status, "mutex_lock");
1978     guarantee(_nParked == 0, "invariant");
1979     ++_nParked;
1980     while (_event < 0) {
1981       // OS-level "spurious wakeups" are ignored
1982       status = pthread_cond_wait(_cond, _mutex);
1983       assert_status(status == 0, status, "cond_wait");
1984     }
1985     --_nParked;
1986 
1987     _event = 0;
1988     status = pthread_mutex_unlock(_mutex);
1989     assert_status(status == 0, status, "mutex_unlock");
1990     // Paranoia to ensure our locked and lock-free paths interact
1991     // correctly with each other.
1992     OrderAccess::fence();
1993   }
1994   guarantee(_event >= 0, "invariant");
1995 }

2051 void os::PlatformEvent::unpark() {
2052   // Transitions for _event:
2053   //    0 => 1 : just return
2054   //    1 => 1 : just return
2055   //   -1 => either 0 or 1; must signal target thread
2056   //         That is, we can safely transition _event from -1 to either
2057   //         0 or 1.
2058   // See also: "Semaphores in Plan 9" by Mullender & Cox
2059   //
2060   // Note: Forcing a transition from "-1" to "1" on an unpark() means
2061   // that it will take two back-to-back park() calls for the owning
2062   // thread to block. This has the benefit of forcing a spurious return
2063   // from the first park() call after an unpark() call which will help
2064   // shake out uses of park() and unpark() without checking state conditions
2065   // properly. This spurious return doesn't manifest itself in any user code
2066   // but only in the correctly written condition checking loops of ObjectMonitor,
2067   // Mutex/Monitor, Thread::muxAcquire and JavaThread::sleep
2068 
2069   if (Atomic::xchg(&_event, 1) >= 0) return;
2070 
2071   int status = pthread_mutex_lock(_mutex);
2072   assert_status(status == 0, status, "mutex_lock");
2073   int anyWaiters = _nParked;
2074   assert(anyWaiters == 0 || anyWaiters == 1, "invariant");
2075   status = pthread_mutex_unlock(_mutex);
2076   assert_status(status == 0, status, "mutex_unlock");
2077 
2078   // Note that we signal() *after* dropping the lock for "immortal" Events.
2079   // This is safe and avoids a common class of futile wakeups.  In rare
2080   // circumstances this can cause a thread to return prematurely from
2081   // cond_{timed}wait() but the spurious wakeup is benign and the victim
2082   // will simply re-test the condition and re-park itself.
2083   // This provides particular benefit if the underlying platform does not
2084   // provide wait morphing.
2085 
2086   if (anyWaiters != 0) {
2087     status = pthread_cond_signal(_cond);
2088     assert_status(status == 0, status, "cond_signal");
2089   }
2090 }

2105 // Parker::park decrements count if > 0, else does a condvar wait.  Unpark
2106 // sets count to 1 and signals condvar.  Only one thread ever waits
2107 // on the condvar. Contention seen when trying to park implies that someone
2108 // is unparking you, so don't wait. And spurious returns are fine, so there
2109 // is no need to track notifications.
2110 
2111 void Parker::park(bool isAbsolute, jlong time) {
2112 
2113   // Optional fast-path check:
2114   // Return immediately if a permit is available.
2115   // We depend on Atomic::xchg() having full barrier semantics
2116   // since we are doing a lock-free update to _counter.
2117   if (Atomic::xchg(&_counter, 0) > 0) return;
2118 
2119   Thread* thread = Thread::current();
2120   assert(thread->is_Java_thread(), "Must be JavaThread");
2121   JavaThread *jt = (JavaThread *)thread;
2122 
2123   // Optional optimization -- avoid state transitions if there's
2124   // an interrupt pending.
2125   if (jt->is_interrupted(false)) {
2126     return;
2127   }
2128 
2129   // Next, demultiplex/decode time arguments
2130   struct timespec absTime;
2131   if (time < 0 || (isAbsolute && time == 0)) { // don't wait at all
2132     return;
2133   }
2134   if (time > 0) {
2135     to_abstime(&absTime, time, isAbsolute, false);
2136   }
2137 
2138   // Enter safepoint region
2139   // Beware of deadlocks such as 6317397.
2140   // The per-thread Parker:: mutex is a classic leaf-lock.
2141   // In particular a thread must never block on the Threads_lock while
2142   // holding the Parker:: mutex.  If safepoints are pending both the
2143   // the ThreadBlockInVM() CTOR and DTOR may grab Threads_lock.
2144   ThreadBlockInVM tbivm(jt);
2145 
2146   // Can't access interrupt state now that we are _thread_blocked. If we've
2147   // been interrupted since we checked above then _counter will be > 0.
2148 
2149   // Don't wait if cannot get lock since interference arises from
2150   // unparking.
2151   if (pthread_mutex_trylock(_mutex) != 0) {
2152     return;
2153   }
2154 
2155   int status;
2156   if (_counter > 0)  { // no wait needed
2157     _counter = 0;
2158     status = pthread_mutex_unlock(_mutex);
2159     assert_status(status == 0, status, "invariant");
2160     // Paranoia to ensure our locked and lock-free paths interact
2161     // correctly with each other and Java-level accesses.
2162     OrderAccess::fence();
2163     return;
2164   }
2165 
2166   OSThreadWaitState osts(thread->osthread(), false /* not Object.wait() */);
2167   jt->set_suspend_equivalent();
2168   // cleared by handle_special_suspend_equivalent_condition() or java_suspend_self()
2169 
2170   assert(_cur_index == -1, "invariant");
2171   if (time == 0) {
2172     _cur_index = REL_INDEX; // arbitrary choice when not timed
2173     status = pthread_cond_wait(&_cond[_cur_index], _mutex);
2174     assert_status(status == 0, status, "cond_timedwait");
2175   }
2176   else {
2177     _cur_index = isAbsolute ? ABS_INDEX : REL_INDEX;
2178     status = pthread_cond_timedwait(&_cond[_cur_index], _mutex, &absTime);
2179     assert_status(status == 0 || status == ETIMEDOUT,
2180                   status, "cond_timedwait");
2181   }
2182   _cur_index = -1;
2183 
2184   _counter = 0;
2185   status = pthread_mutex_unlock(_mutex);
2186   assert_status(status == 0, status, "invariant");
2187   // Paranoia to ensure our locked and lock-free paths interact
2188   // correctly with each other and Java-level accesses.

2194   }

2341       millis = jlong(MAX_SECS) * MILLIUNITS;
2342     }

2344 
2345     int ret = OS_TIMEOUT;

2347     assert_status(status == 0 || status == ETIMEDOUT,
2348                   status, "cond_timedwait");

2350       ret = OS_OK;

2353   } else {
2354     int status = pthread_cond_wait(cond(), mutex());
