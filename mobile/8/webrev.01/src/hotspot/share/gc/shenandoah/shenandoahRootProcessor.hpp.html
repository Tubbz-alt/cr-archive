230 template <bool CONCURRENT, bool SINGLE_THREADED>
231 class ShenandoahClassLoaderDataRoots {
232 private:
233   ShenandoahSharedSemaphore     _semaphore;
234   ShenandoahPhaseTimings::Phase _phase;
235 
236   static uint worker_count(uint n_workers) {
237     // Limit concurrency a bit, otherwise it wastes resources when workers are tripping
238     // over each other. This also leaves free workers to process other parts of the root
239     // set, while admitted workers are busy with doing the CLDG walk.
240     return MAX2(1u, MIN2(ShenandoahSharedSemaphore::max_tokens(), n_workers / 2));
241   }
242 
243 public:
244   ShenandoahClassLoaderDataRoots(ShenandoahPhaseTimings::Phase phase, uint n_workers);
245   ~ShenandoahClassLoaderDataRoots();
246 
247   void always_strong_cld_do(CLDClosure* clds, uint worker_id);
248   void cld_do(CLDClosure* clds, uint worker_id);
249 };

262 class ShenandoahRootScanner : public ShenandoahRootProcessor {
263 private:
264   ShenandoahSerialRoots                                     _serial_roots;
265   ShenandoahThreadRoots                                     _thread_roots;
266 
267 public:
268   ShenandoahRootScanner(uint n_workers, ShenandoahPhaseTimings::Phase phase);
269   ~ShenandoahRootScanner();
270 
271   // Apply oops, clds and blobs to all strongly reachable roots in the system,
272   // during class unloading cycle
273   void strong_roots_do(uint worker_id, OopClosure* cl);
274   void strong_roots_do(uint worker_id, OopClosure* oops, CLDClosure* clds, CodeBlobClosure* code, ThreadClosure* tc = NULL);
275 
276   // Apply oops, clds and blobs to all strongly reachable roots and weakly reachable
277   // roots when class unloading is disabled during this cycle
278   void roots_do(uint worker_id, OopClosure* cl);
279   void roots_do(uint worker_id, OopClosure* oops, CLDClosure* clds, CodeBlobClosure* code, ThreadClosure* tc = NULL);
280 };
