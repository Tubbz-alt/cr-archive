25 import java.lang.Package;
26 
27 public class PackageSealingTest {
28     public static void main(String args[]) {



29         try {
30             Class c1 = PackageSealingTest.class.forName("sealed.pkg.C1");
31             Class c2 = PackageSealingTest.class.forName("pkg.C2");
32             Package p1 = c1.getPackage();
33             System.out.println("Package 1: " + p1.toString());
34             Package p2 = c2.getPackage();
35             System.out.println("Package 2: " + p2.toString());
36 
37             if (!p1.isSealed()) {
38                 System.out.println("Failed: sealed.pkg is not sealed.");
39                 System.exit(0);
40             }
41 
42             if (p2.isSealed()) {
43                 System.out.println("Failed: pkg is sealed.");
44                 System.exit(0);
45             }
46 
47             System.out.println("OK");
48         } catch (Exception e) {
49             System.out.println(e.getMessage());
50         }
51     }
52 }
