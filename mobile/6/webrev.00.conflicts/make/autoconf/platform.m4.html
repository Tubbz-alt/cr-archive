238 # Expects $host_os $host_cpu $build_os and $build_cpu
239 # and $with_target_bits to have been setup!
240 #
241 # Translate the standard triplet(quadruplet) definition
242 # of the target/build system into OPENJDK_TARGET_OS, OPENJDK_TARGET_CPU,
243 # OPENJDK_BUILD_OS, etc.
244 AC_DEFUN([PLATFORM_EXTRACT_TARGET_AND_BUILD],
245 [
246   # Copy the autoconf trip/quadruplet verbatim to OPENJDK_TARGET_AUTOCONF_NAME
247   # (from the autoconf "host") and OPENJDK_BUILD_AUTOCONF_NAME
248   # Note that we might later on rewrite e.g. OPENJDK_TARGET_CPU due to reduced build,
249   # but this will not change the value of OPENJDK_TARGET_AUTOCONF_NAME.
250   OPENJDK_TARGET_AUTOCONF_NAME="$host"
251   OPENJDK_BUILD_AUTOCONF_NAME="$build"
252   AC_SUBST(OPENJDK_TARGET_AUTOCONF_NAME)
253   AC_SUBST(OPENJDK_BUILD_AUTOCONF_NAME)
254 
255   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
256   PLATFORM_EXTRACT_VARS_FROM_OS($build_os)
257   PLATFORM_EXTRACT_VARS_FROM_CPU($build_cpu)
258   # ..and setup our own variables. (Do this explicitly to facilitate searching)
259   OPENJDK_BUILD_OS="$VAR_OS"
260   if test "x$VAR_OS_TYPE" != x; then
261     OPENJDK_BUILD_OS_TYPE="$VAR_OS_TYPE"
262   else
263     OPENJDK_BUILD_OS_TYPE="$VAR_OS"
264   fi
265   if test "x$VAR_OS_ENV" != x; then
266     OPENJDK_BUILD_OS_ENV="$VAR_OS_ENV"
267   else
268     OPENJDK_BUILD_OS_ENV="$VAR_OS"
269   fi
270   OPENJDK_BUILD_CPU="$VAR_CPU"
271   OPENJDK_BUILD_CPU_ARCH="$VAR_CPU_ARCH"
272   OPENJDK_BUILD_CPU_BITS="$VAR_CPU_BITS"
273   OPENJDK_BUILD_CPU_ENDIAN="$VAR_CPU_ENDIAN"
274   AC_SUBST(OPENJDK_BUILD_OS)
275   AC_SUBST(OPENJDK_BUILD_OS_TYPE)
276   AC_SUBST(OPENJDK_BUILD_OS_ENV)
277   AC_SUBST(OPENJDK_BUILD_CPU)
278   AC_SUBST(OPENJDK_BUILD_CPU_ARCH)
279   AC_SUBST(OPENJDK_BUILD_CPU_BITS)
280   AC_SUBST(OPENJDK_BUILD_CPU_ENDIAN)
281 
282   AC_MSG_CHECKING([openjdk-build os-cpu])
283   AC_MSG_RESULT([$OPENJDK_BUILD_OS-$OPENJDK_BUILD_CPU])
284 
285   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
286   PLATFORM_EXTRACT_VARS_FROM_OS($host_os)
287   PLATFORM_EXTRACT_VARS_FROM_CPU($host_cpu)
288   # ... and setup our own variables. (Do this explicitly to facilitate searching)
289   OPENJDK_TARGET_OS="$VAR_OS"
290   if test "x$VAR_OS_TYPE" != x; then
291     OPENJDK_TARGET_OS_TYPE="$VAR_OS_TYPE"
292   else
293     OPENJDK_TARGET_OS_TYPE="$VAR_OS"
294   fi
295   if test "x$VAR_OS_ENV" != x; then
296     OPENJDK_TARGET_OS_ENV="$VAR_OS_ENV"
297   else
298     OPENJDK_TARGET_OS_ENV="$VAR_OS"
299   fi
300   OPENJDK_TARGET_CPU="$VAR_CPU"
301   OPENJDK_TARGET_CPU_ARCH="$VAR_CPU_ARCH"
302   OPENJDK_TARGET_CPU_BITS="$VAR_CPU_BITS"
303   OPENJDK_TARGET_CPU_ENDIAN="$VAR_CPU_ENDIAN"
304   OPENJDK_TARGET_OS_UPPERCASE=`$ECHO $OPENJDK_TARGET_OS | $TR 'abcdefghijklmnopqrstuvwxyz' 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'`
305 
306   AC_SUBST(OPENJDK_TARGET_OS)
307   AC_SUBST(OPENJDK_TARGET_OS_TYPE)
308   AC_SUBST(OPENJDK_TARGET_OS_ENV)
309   AC_SUBST(OPENJDK_TARGET_OS_UPPERCASE)
310   AC_SUBST(OPENJDK_TARGET_CPU)
311   AC_SUBST(OPENJDK_TARGET_CPU_ARCH)
312   AC_SUBST(OPENJDK_TARGET_CPU_BITS)

315   AC_MSG_CHECKING([openjdk-target os-cpu])
316   AC_MSG_RESULT([$OPENJDK_TARGET_OS-$OPENJDK_TARGET_CPU])

319 # Check if a reduced build (32-bit on 64-bit platforms) is requested, and modify behaviour
320 # accordingly. Must be done after setting up build and target system, but before
321 # doing anything else with these values.
322 AC_DEFUN([PLATFORM_SETUP_TARGET_CPU_BITS],
323 [
324   AC_ARG_WITH(target-bits, [AS_HELP_STRING([--with-target-bits],
325        [build 32-bit or 64-bit binaries (for platforms that support it), e.g. --with-target-bits=32 @<:@guessed@:>@])])
326 
327   # We have three types of compiles:
328   # native  == normal compilation, target system == build system
329   # cross   == traditional cross compilation, target system != build system; special toolchain needed
330   # reduced == using native compilers, but with special flags (e.g. -m32) to produce 32-bit builds on 64-bit machines
331   #
332   if test "x$OPENJDK_BUILD_AUTOCONF_NAME" != "x$OPENJDK_TARGET_AUTOCONF_NAME"; then
333     # We're doing a proper cross-compilation
334     COMPILE_TYPE="cross"
335   else
336     COMPILE_TYPE="native"
337   fi
338 
339   if test "x$with_target_bits" != x; then
340     if test "x$COMPILE_TYPE" = "xcross"; then
341       AC_MSG_ERROR([It is not possible to combine --with-target-bits=X and proper cross-compilation. Choose either.])
342     fi
343 
344     if test "x$with_target_bits" = x32 && test "x$OPENJDK_TARGET_CPU_BITS" = x64; then
345       # A reduced build is requested
346       COMPILE_TYPE="reduced"
347       OPENJDK_TARGET_CPU_BITS=32
348       if test "x$OPENJDK_TARGET_CPU_ARCH" = "xx86"; then
349         OPENJDK_TARGET_CPU=x86
350       elif test "x$OPENJDK_TARGET_CPU_ARCH" = "xsparc"; then
351         OPENJDK_TARGET_CPU=sparc
352       else
353         AC_MSG_ERROR([Reduced build (--with-target-bits=32) is only supported on x86_64 and sparcv9])
354       fi
355     elif test "x$with_target_bits" = x64 && test "x$OPENJDK_TARGET_CPU_BITS" = x32; then
356       AC_MSG_ERROR([It is not possible to use --with-target-bits=64 on a 32 bit system. Use proper cross-compilation instead.])
357     elif test "x$with_target_bits" = "x$OPENJDK_TARGET_CPU_BITS"; then
358       AC_MSG_NOTICE([--with-target-bits are set to build platform address size; argument has no meaning])
359     else
360       AC_MSG_ERROR([--with-target-bits can only be 32 or 64, you specified $with_target_bits!])
361     fi
362   fi
363   AC_SUBST(COMPILE_TYPE)
364 
365   AC_MSG_CHECKING([compilation type])
366   AC_MSG_RESULT([$COMPILE_TYPE])
367 ])

377 # $1 - Either TARGET or BUILD to setup the variables for.
378 AC_DEFUN([PLATFORM_SETUP_LEGACY_VARS_HELPER],
379 [
380   # Also store the legacy naming of the cpu.
381   # Ie i586 and amd64 instead of x86 and x86_64
382   OPENJDK_$1_CPU_LEGACY="$OPENJDK_$1_CPU"
383   if test "x$OPENJDK_$1_CPU" = xx86; then
384     OPENJDK_$1_CPU_LEGACY="i586"
385   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
386     # On all platforms except MacOSX replace x86_64 with amd64.
387     OPENJDK_$1_CPU_LEGACY="amd64"
388   elif test "x$OPENJDK_$1_CPU" = xalpha; then
389     # Avoid name collisions with variables named alpha
390     OPENJDK_$1_CPU_LEGACY="_alpha_"
391   elif test "x$OPENJDK_$1_CPU" = xsh; then
392     # Avoid name collisions with variables named sh
393     OPENJDK_$1_CPU_LEGACY="_sh_"
394   fi
395   AC_SUBST(OPENJDK_$1_CPU_LEGACY)
396 
397   # And the second legacy naming of the cpu.
398   # Ie i386 and amd64 instead of x86 and x86_64.
399   OPENJDK_$1_CPU_LEGACY_LIB="$OPENJDK_$1_CPU"
400   if test "x$OPENJDK_$1_CPU" = xx86; then
401     OPENJDK_$1_CPU_LEGACY_LIB="i386"
402   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
403     OPENJDK_$1_CPU_LEGACY_LIB="amd64"
404   fi
405   AC_SUBST(OPENJDK_$1_CPU_LEGACY_LIB)
406 
407   # Setup OPENJDK_$1_CPU_OSARCH, which is used to set the os.arch Java system property
408   OPENJDK_$1_CPU_OSARCH="$OPENJDK_$1_CPU"
409   if test "x$OPENJDK_$1_OS" = xlinux && test "x$OPENJDK_$1_CPU" = xx86; then
410     # On linux only, we replace x86 with i386.
411     OPENJDK_$1_CPU_OSARCH="i386"
412   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
413     # On all platforms except macosx, we replace x86_64 with amd64.
414     OPENJDK_$1_CPU_OSARCH="amd64"
415   fi
416   AC_SUBST(OPENJDK_$1_CPU_OSARCH)
417 
418   OPENJDK_$1_CPU_JLI="$OPENJDK_$1_CPU"
419   if test "x$OPENJDK_$1_CPU" = xx86; then
420     OPENJDK_$1_CPU_JLI="i386"
421   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
422     # On all platforms except macosx, we replace x86_64 with amd64.
423     OPENJDK_$1_CPU_JLI="amd64"
424   fi
425 
426   # The new version string in JDK 9 also defined new naming of OS and ARCH for bundles
427   # Macosx is osx and x86_64 is x64
428   if test "x$OPENJDK_$1_OS" = xmacosx; then
429     OPENJDK_$1_OS_BUNDLE="osx"
430   else
431     OPENJDK_$1_OS_BUNDLE="$OPENJDK_TARGET_OS"
432   fi
433   if test "x$OPENJDK_$1_CPU" = xx86_64; then
434     OPENJDK_$1_CPU_BUNDLE="x64"
435   else
436     OPENJDK_$1_CPU_BUNDLE="$OPENJDK_$1_CPU"
437   fi
438   OPENJDK_$1_BUNDLE_PLATFORM="${OPENJDK_$1_OS_BUNDLE}-${OPENJDK_$1_CPU_BUNDLE}"
439   AC_SUBST(OPENJDK_$1_BUNDLE_PLATFORM)
440 
441   if test "x$COMPILE_TYPE" = "xcross"; then
442     # FIXME: ... or should this include reduced builds..?
443     DEFINE_CROSS_COMPILE_ARCH="CROSS_COMPILE_ARCH:=$OPENJDK_$1_CPU_LEGACY"
444   else
445     DEFINE_CROSS_COMPILE_ARCH=""
446   fi
447   AC_SUBST(DEFINE_CROSS_COMPILE_ARCH)
448 
449   # Convert openjdk platform names to hotspot names
450 
451   HOTSPOT_$1_OS=${OPENJDK_$1_OS}
452   if test "x$OPENJDK_$1_OS" = xmacosx; then
453     HOTSPOT_$1_OS=bsd
454   fi
455   AC_SUBST(HOTSPOT_$1_OS)
456 
457   HOTSPOT_$1_OS_TYPE=${OPENJDK_$1_OS_TYPE}
458   if test "x$OPENJDK_$1_OS_TYPE" = xunix; then
459     HOTSPOT_$1_OS_TYPE=posix
460   fi
461   AC_SUBST(HOTSPOT_$1_OS_TYPE)
462 
463   HOTSPOT_$1_CPU=${OPENJDK_$1_CPU}
464   if test "x$OPENJDK_$1_CPU" = xx86; then
465     HOTSPOT_$1_CPU=x86_32
466   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
467     HOTSPOT_$1_CPU=sparc
468   elif test "x$OPENJDK_$1_CPU" = xppc64; then
469     HOTSPOT_$1_CPU=ppc_64
470   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
471     HOTSPOT_$1_CPU=ppc_64
472   fi
473   AC_SUBST(HOTSPOT_$1_CPU)
474 
475   # This is identical with OPENJDK_*, but define anyway for consistency.
476   HOTSPOT_$1_CPU_ARCH=${OPENJDK_$1_CPU_ARCH}
477   AC_SUBST(HOTSPOT_$1_CPU_ARCH)
478 
479   # Setup HOTSPOT_$1_CPU_DEFINE
480   if test "x$OPENJDK_$1_CPU" = xx86; then
481     HOTSPOT_$1_CPU_DEFINE=IA32
482   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
483     HOTSPOT_$1_CPU_DEFINE=AMD64
484   elif test "x$OPENJDK_$1_CPU" = xx32; then
485     HOTSPOT_$1_CPU_DEFINE=X32
486   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
487     HOTSPOT_$1_CPU_DEFINE=SPARC
488   elif test "x$OPENJDK_$1_CPU" = xaarch64; then
489     HOTSPOT_$1_CPU_DEFINE=AARCH64
490   elif test "x$OPENJDK_$1_CPU" = xppc64; then
491     HOTSPOT_$1_CPU_DEFINE=PPC64
492   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
493     HOTSPOT_$1_CPU_DEFINE=PPC64
494 
495   # The cpu defines below are for zero, we don't support them directly.
496   elif test "x$OPENJDK_$1_CPU" = xsparc; then
497     HOTSPOT_$1_CPU_DEFINE=SPARC
498   elif test "x$OPENJDK_$1_CPU" = xppc; then
499     HOTSPOT_$1_CPU_DEFINE=PPC32
500   elif test "x$OPENJDK_$1_CPU" = xs390; then

505     HOTSPOT_$1_CPU_DEFINE=RISCV

509   AC_SUBST(HOTSPOT_$1_CPU_DEFINE)
510 
511   # For historical reasons, the OS include directories have odd names.
512   OPENJDK_$1_OS_INCLUDE_SUBDIR="$OPENJDK_TARGET_OS"
513   if test "x$OPENJDK_TARGET_OS" = "xwindows"; then
514     OPENJDK_$1_OS_INCLUDE_SUBDIR="win32"
515   elif test "x$OPENJDK_TARGET_OS" = "xmacosx"; then
516     OPENJDK_$1_OS_INCLUDE_SUBDIR="darwin"
517   fi
518   AC_SUBST(OPENJDK_$1_OS_INCLUDE_SUBDIR)

596 AC_DEFUN_ONCE([PLATFORM_SETUP_OPENJDK_TARGET_BITS],
597 [
598   ###############################################################################
599   #
600   # Now we check if libjvm.so will use 32 or 64 bit pointers for the C/C++ code.
601   # (The JVM can use 32 or 64 bit Java pointers but that decision
602   # is made at runtime.)
603   #
604 
605   # Make compilation sanity check
606   AC_CHECK_HEADERS([stdio.h], , [
607     AC_MSG_NOTICE([Failed to compile stdio.h. This likely implies missing compile dependencies.])
608     if test "x$COMPILE_TYPE" = xreduced; then
609       HELP_MSG_MISSING_DEPENDENCY([reduced])
610       AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
611     elif test "x$COMPILE_TYPE" = xcross; then
612       AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
613     fi
614     AC_MSG_ERROR([Cannot continue.])
615   ])
616 
617   AC_CHECK_SIZEOF([int *], [1111])
618 
619   # AC_CHECK_SIZEOF defines 'ac_cv_sizeof_int_p' to hold the number of bytes used by an 'int*'
620   if test "x$ac_cv_sizeof_int_p" = x; then
621     # The test failed, lets stick to the assumed value.
622     AC_MSG_WARN([The number of bits in the target could not be determined, using $OPENJDK_TARGET_CPU_BITS.])
623   else
624     TESTED_TARGET_CPU_BITS=`expr 8 \* $ac_cv_sizeof_int_p`
625 
626     if test "x$TESTED_TARGET_CPU_BITS" != "x$OPENJDK_TARGET_CPU_BITS"; then
627       AC_MSG_NOTICE([The tested number of bits in the target ($TESTED_TARGET_CPU_BITS) differs from the number of bits expected to be found in the target ($OPENJDK_TARGET_CPU_BITS)])
628       if test "x$COMPILE_TYPE" = xreduced; then
629         HELP_MSG_MISSING_DEPENDENCY([reduced])
630         AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
631       elif test "x$COMPILE_TYPE" = xcross; then
632         AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
633       fi
634       AC_MSG_ERROR([Cannot continue.])
635     fi
636   fi
637 
638   AC_MSG_CHECKING([for target address size])
639   AC_MSG_RESULT([$OPENJDK_TARGET_CPU_BITS bits])
640 ])
