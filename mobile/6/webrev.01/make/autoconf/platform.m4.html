231 # Expects $host_os $host_cpu $build_os and $build_cpu
232 # and $with_target_bits to have been setup!
233 #
234 # Translate the standard triplet(quadruplet) definition
235 # of the target/build system into OPENJDK_TARGET_OS, OPENJDK_TARGET_CPU,
236 # OPENJDK_BUILD_OS, etc.
237 AC_DEFUN([PLATFORM_EXTRACT_TARGET_AND_BUILD],
238 [
239   # Copy the autoconf trip/quadruplet verbatim to OPENJDK_TARGET_AUTOCONF_NAME
240   # (from the autoconf "host") and OPENJDK_BUILD_AUTOCONF_NAME
241   # Note that we might later on rewrite e.g. OPENJDK_TARGET_CPU due to reduced build,
242   # but this will not change the value of OPENJDK_TARGET_AUTOCONF_NAME.
243   OPENJDK_TARGET_AUTOCONF_NAME="$host"
244   OPENJDK_BUILD_AUTOCONF_NAME="$build"
245   AC_SUBST(OPENJDK_TARGET_AUTOCONF_NAME)
246   AC_SUBST(OPENJDK_BUILD_AUTOCONF_NAME)
247 
248   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
249   PLATFORM_EXTRACT_VARS_FROM_OS($build_os)
250   PLATFORM_EXTRACT_VARS_FROM_CPU($build_cpu)
251   # ..and setup our own variables. (Do this explicitly to facilitate searching)
252   OPENJDK_BUILD_OS="$VAR_OS"
253   if test "x$VAR_OS_TYPE" != x; then
254     OPENJDK_BUILD_OS_TYPE="$VAR_OS_TYPE"
255   else
256     OPENJDK_BUILD_OS_TYPE="$VAR_OS"
257   fi
258   if test "x$VAR_OS_ENV" != x; then
259     OPENJDK_BUILD_OS_ENV="$VAR_OS_ENV"
260   else
261     OPENJDK_BUILD_OS_ENV="$VAR_OS"
262   fi
263   OPENJDK_BUILD_CPU="$VAR_CPU"
264   OPENJDK_BUILD_CPU_ARCH="$VAR_CPU_ARCH"
265   OPENJDK_BUILD_CPU_BITS="$VAR_CPU_BITS"
266   OPENJDK_BUILD_CPU_ENDIAN="$VAR_CPU_ENDIAN"
267   AC_SUBST(OPENJDK_BUILD_OS)
268   AC_SUBST(OPENJDK_BUILD_OS_TYPE)
269   AC_SUBST(OPENJDK_BUILD_OS_ENV)
270   AC_SUBST(OPENJDK_BUILD_CPU)
271   AC_SUBST(OPENJDK_BUILD_CPU_ARCH)
272   AC_SUBST(OPENJDK_BUILD_CPU_BITS)
273   AC_SUBST(OPENJDK_BUILD_CPU_ENDIAN)
274 
275   AC_MSG_CHECKING([openjdk-build os-cpu])
276   AC_MSG_RESULT([$OPENJDK_BUILD_OS-$OPENJDK_BUILD_CPU])
277 
278   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
279   PLATFORM_EXTRACT_VARS_FROM_OS($host_os)
280   PLATFORM_EXTRACT_VARS_FROM_CPU($host_cpu)
281   # ... and setup our own variables. (Do this explicitly to facilitate searching)
282   OPENJDK_TARGET_OS="$VAR_OS"
283   if test "x$VAR_OS_TYPE" != x; then
284     OPENJDK_TARGET_OS_TYPE="$VAR_OS_TYPE"
285   else
286     OPENJDK_TARGET_OS_TYPE="$VAR_OS"
287   fi
288   if test "x$VAR_OS_ENV" != x; then
289     OPENJDK_TARGET_OS_ENV="$VAR_OS_ENV"
290   else
291     OPENJDK_TARGET_OS_ENV="$VAR_OS"
292   fi
293   OPENJDK_TARGET_CPU="$VAR_CPU"
294   OPENJDK_TARGET_CPU_ARCH="$VAR_CPU_ARCH"
295   OPENJDK_TARGET_CPU_BITS="$VAR_CPU_BITS"
296   OPENJDK_TARGET_CPU_ENDIAN="$VAR_CPU_ENDIAN"
297   OPENJDK_TARGET_OS_UPPERCASE=`$ECHO $OPENJDK_TARGET_OS | $TR 'abcdefghijklmnopqrstuvwxyz' 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'`
298 
299   AC_SUBST(OPENJDK_TARGET_OS)
300   AC_SUBST(OPENJDK_TARGET_OS_TYPE)
301   AC_SUBST(OPENJDK_TARGET_OS_ENV)
302   AC_SUBST(OPENJDK_TARGET_OS_UPPERCASE)
303   AC_SUBST(OPENJDK_TARGET_CPU)
304   AC_SUBST(OPENJDK_TARGET_CPU_ARCH)
305   AC_SUBST(OPENJDK_TARGET_CPU_BITS)
306   AC_SUBST(OPENJDK_TARGET_CPU_ENDIAN)
307 
308   AC_MSG_CHECKING([openjdk-target os-cpu])
309   AC_MSG_RESULT([$OPENJDK_TARGET_OS-$OPENJDK_TARGET_CPU])
310 ])

312 # Check if a reduced build (32-bit on 64-bit platforms) is requested, and modify behaviour
313 # accordingly. Must be done after setting up build and target system, but before
314 # doing anything else with these values.
315 AC_DEFUN([PLATFORM_SETUP_TARGET_CPU_BITS],
316 [
317   AC_ARG_WITH(target-bits, [AS_HELP_STRING([--with-target-bits],
318        [build 32-bit or 64-bit binaries (for platforms that support it), e.g. --with-target-bits=32 @<:@guessed@:>@])])
319 
320   # We have three types of compiles:
321   # native  == normal compilation, target system == build system
322   # cross   == traditional cross compilation, target system != build system; special toolchain needed
323   # reduced == using native compilers, but with special flags (e.g. -m32) to produce 32-bit builds on 64-bit machines
324   #
325   if test "x$OPENJDK_BUILD_AUTOCONF_NAME" != "x$OPENJDK_TARGET_AUTOCONF_NAME"; then
326     # We're doing a proper cross-compilation
327     COMPILE_TYPE="cross"
328   else
329     COMPILE_TYPE="native"
330   fi
331 
332   if test "x$with_target_bits" != x; then
333     if test "x$COMPILE_TYPE" = "xcross"; then
334       AC_MSG_ERROR([It is not possible to combine --with-target-bits=X and proper cross-compilation. Choose either.])
335     fi
336 
337     if test "x$with_target_bits" = x32 && test "x$OPENJDK_TARGET_CPU_BITS" = x64; then
338       # A reduced build is requested
339       COMPILE_TYPE="reduced"
340       OPENJDK_TARGET_CPU_BITS=32
341       if test "x$OPENJDK_TARGET_CPU_ARCH" = "xx86"; then
342         OPENJDK_TARGET_CPU=x86
343       elif test "x$OPENJDK_TARGET_CPU_ARCH" = "xsparc"; then
344         OPENJDK_TARGET_CPU=sparc
345       else
346         AC_MSG_ERROR([Reduced build (--with-target-bits=32) is only supported on x86_64 and sparcv9])
347       fi
348     elif test "x$with_target_bits" = x64 && test "x$OPENJDK_TARGET_CPU_BITS" = x32; then
349       AC_MSG_ERROR([It is not possible to use --with-target-bits=64 on a 32 bit system. Use proper cross-compilation instead.])
350     elif test "x$with_target_bits" = "x$OPENJDK_TARGET_CPU_BITS"; then
351       AC_MSG_NOTICE([--with-target-bits are set to build platform address size; argument has no meaning])
352     else
353       AC_MSG_ERROR([--with-target-bits can only be 32 or 64, you specified $with_target_bits!])
354     fi
355   fi
356   AC_SUBST(COMPILE_TYPE)
357 
358   AC_MSG_CHECKING([compilation type])
359   AC_MSG_RESULT([$COMPILE_TYPE])
360 ])

370 # $1 - Either TARGET or BUILD to setup the variables for.
371 AC_DEFUN([PLATFORM_SETUP_LEGACY_VARS_HELPER],
372 [
373   # Also store the legacy naming of the cpu.
374   # Ie i586 and amd64 instead of x86 and x86_64
375   OPENJDK_$1_CPU_LEGACY="$OPENJDK_$1_CPU"
376   if test "x$OPENJDK_$1_CPU" = xx86; then
377     OPENJDK_$1_CPU_LEGACY="i586"
378   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
379     # On all platforms except MacOSX replace x86_64 with amd64.
380     OPENJDK_$1_CPU_LEGACY="amd64"
381   elif test "x$OPENJDK_$1_CPU" = xalpha; then
382     # Avoid name collisions with variables named alpha
383     OPENJDK_$1_CPU_LEGACY="_alpha_"
384   elif test "x$OPENJDK_$1_CPU" = xsh; then
385     # Avoid name collisions with variables named sh
386     OPENJDK_$1_CPU_LEGACY="_sh_"
387   fi
388   AC_SUBST(OPENJDK_$1_CPU_LEGACY)
389 
390   # And the second legacy naming of the cpu.
391   # Ie i386 and amd64 instead of x86 and x86_64.
392   OPENJDK_$1_CPU_LEGACY_LIB="$OPENJDK_$1_CPU"
393   if test "x$OPENJDK_$1_CPU" = xx86; then
394     OPENJDK_$1_CPU_LEGACY_LIB="i386"
395   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
396     OPENJDK_$1_CPU_LEGACY_LIB="amd64"
397   fi
398   AC_SUBST(OPENJDK_$1_CPU_LEGACY_LIB)
399 
400   # Setup OPENJDK_$1_CPU_OSARCH, which is used to set the os.arch Java system property
401   OPENJDK_$1_CPU_OSARCH="$OPENJDK_$1_CPU"
402   if test "x$OPENJDK_$1_OS" = xlinux && test "x$OPENJDK_$1_CPU" = xx86; then
403     # On linux only, we replace x86 with i386.
404     OPENJDK_$1_CPU_OSARCH="i386"
405   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
406     # On all platforms except macosx, we replace x86_64 with amd64.
407     OPENJDK_$1_CPU_OSARCH="amd64"
408   fi
409   AC_SUBST(OPENJDK_$1_CPU_OSARCH)
410 
411   OPENJDK_$1_CPU_JLI="$OPENJDK_$1_CPU"
412   if test "x$OPENJDK_$1_CPU" = xx86; then
413     OPENJDK_$1_CPU_JLI="i386"
414   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
415     # On all platforms except macosx, we replace x86_64 with amd64.
416     OPENJDK_$1_CPU_JLI="amd64"
417   fi
418 
419   # The new version string in JDK 9 also defined new naming of OS and ARCH for bundles
420   # Macosx is osx and x86_64 is x64
421   if test "x$OPENJDK_$1_OS" = xmacosx; then
422     OPENJDK_$1_OS_BUNDLE="osx"
423   else
424     OPENJDK_$1_OS_BUNDLE="$OPENJDK_TARGET_OS"
425   fi
426   if test "x$OPENJDK_$1_CPU" = xx86_64; then
427     OPENJDK_$1_CPU_BUNDLE="x64"
428   else
429     OPENJDK_$1_CPU_BUNDLE="$OPENJDK_$1_CPU"
430   fi
431   OPENJDK_$1_BUNDLE_PLATFORM="${OPENJDK_$1_OS_BUNDLE}-${OPENJDK_$1_CPU_BUNDLE}"
432   AC_SUBST(OPENJDK_$1_BUNDLE_PLATFORM)
433 
434   if test "x$COMPILE_TYPE" = "xcross"; then
435     # FIXME: ... or should this include reduced builds..?
436     DEFINE_CROSS_COMPILE_ARCH="CROSS_COMPILE_ARCH:=$OPENJDK_$1_CPU_LEGACY"
437   else
438     DEFINE_CROSS_COMPILE_ARCH=""
439   fi
440   AC_SUBST(DEFINE_CROSS_COMPILE_ARCH)
441 
442   # Convert openjdk platform names to hotspot names
443 
444   HOTSPOT_$1_OS=${OPENJDK_$1_OS}
445   if test "x$OPENJDK_$1_OS" = xmacosx; then
446     HOTSPOT_$1_OS=bsd
447   fi
448   AC_SUBST(HOTSPOT_$1_OS)
449 
450   HOTSPOT_$1_OS_TYPE=${OPENJDK_$1_OS_TYPE}
451   if test "x$OPENJDK_$1_OS_TYPE" = xunix; then
452     HOTSPOT_$1_OS_TYPE=posix
453   fi
454   AC_SUBST(HOTSPOT_$1_OS_TYPE)
455 
456   HOTSPOT_$1_CPU=${OPENJDK_$1_CPU}
457   if test "x$OPENJDK_$1_CPU" = xx86; then
458     HOTSPOT_$1_CPU=x86_32
459   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
460     HOTSPOT_$1_CPU=sparc
461   elif test "x$OPENJDK_$1_CPU" = xppc64; then
462     HOTSPOT_$1_CPU=ppc_64
463   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
464     HOTSPOT_$1_CPU=ppc_64
465   fi
466   AC_SUBST(HOTSPOT_$1_CPU)
467 
468   # This is identical with OPENJDK_*, but define anyway for consistency.
469   HOTSPOT_$1_CPU_ARCH=${OPENJDK_$1_CPU_ARCH}
470   AC_SUBST(HOTSPOT_$1_CPU_ARCH)
471 
472   # Setup HOTSPOT_$1_CPU_DEFINE
473   if test "x$OPENJDK_$1_CPU" = xx86; then
474     HOTSPOT_$1_CPU_DEFINE=IA32
475   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
476     HOTSPOT_$1_CPU_DEFINE=AMD64
477   elif test "x$OPENJDK_$1_CPU" = xx32; then
478     HOTSPOT_$1_CPU_DEFINE=X32
479   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
480     HOTSPOT_$1_CPU_DEFINE=SPARC
481   elif test "x$OPENJDK_$1_CPU" = xaarch64; then
482     HOTSPOT_$1_CPU_DEFINE=AARCH64
483   elif test "x$OPENJDK_$1_CPU" = xppc64; then
484     HOTSPOT_$1_CPU_DEFINE=PPC64
485   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
486     HOTSPOT_$1_CPU_DEFINE=PPC64
487 
488   # The cpu defines below are for zero, we don't support them directly.
489   elif test "x$OPENJDK_$1_CPU" = xsparc; then
490     HOTSPOT_$1_CPU_DEFINE=SPARC
491   elif test "x$OPENJDK_$1_CPU" = xppc; then
492     HOTSPOT_$1_CPU_DEFINE=PPC32
493   elif test "x$OPENJDK_$1_CPU" = xs390; then
494     HOTSPOT_$1_CPU_DEFINE=S390
495   elif test "x$OPENJDK_$1_CPU" = xs390x; then
496     HOTSPOT_$1_CPU_DEFINE=S390
497   elif test "x$OPENJDK_$1_CPU" = xriscv64; then
498     HOTSPOT_$1_CPU_DEFINE=RISCV
499   elif test "x$OPENJDK_$1_CPU" != x; then
500     HOTSPOT_$1_CPU_DEFINE=$(echo $OPENJDK_$1_CPU | tr a-z A-Z)
501   fi
502   AC_SUBST(HOTSPOT_$1_CPU_DEFINE)
503 
504   # For historical reasons, the OS include directories have odd names.
505   OPENJDK_$1_OS_INCLUDE_SUBDIR="$OPENJDK_TARGET_OS"
506   if test "x$OPENJDK_TARGET_OS" = "xwindows"; then
507     OPENJDK_$1_OS_INCLUDE_SUBDIR="win32"
508   elif test "x$OPENJDK_TARGET_OS" = "xmacosx"; then
509     OPENJDK_$1_OS_INCLUDE_SUBDIR="darwin"
510   fi
511   AC_SUBST(OPENJDK_$1_OS_INCLUDE_SUBDIR)
512 ])

552 #%%% Build and target systems %%%
553 AC_DEFUN_ONCE([PLATFORM_SETUP_OPENJDK_BUILD_AND_TARGET],
554 [
555   # Figure out the build and target systems. # Note that in autoconf terminology, "build" is obvious, but "target"
556   # is confusing; it assumes you are cross-compiling a cross-compiler (!)  and "target" is thus the target of the
557   # product you're building. The target of this build is called "host". Since this is confusing to most people, we
558   # have not adopted that system, but use "target" as the platform we are building for. In some places though we need
559   # to use the configure naming style.

589 AC_DEFUN_ONCE([PLATFORM_SETUP_OPENJDK_TARGET_BITS],
590 [
591   ###############################################################################
592   #
593   # Now we check if libjvm.so will use 32 or 64 bit pointers for the C/C++ code.
594   # (The JVM can use 32 or 64 bit Java pointers but that decision
595   # is made at runtime.)
596   #
597 
598   # Make compilation sanity check
599   AC_CHECK_HEADERS([stdio.h], , [
600     AC_MSG_NOTICE([Failed to compile stdio.h. This likely implies missing compile dependencies.])
601     if test "x$COMPILE_TYPE" = xreduced; then
602       HELP_MSG_MISSING_DEPENDENCY([reduced])
603       AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
604     elif test "x$COMPILE_TYPE" = xcross; then
605       AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
606     fi
607     AC_MSG_ERROR([Cannot continue.])
608   ])
609 
610   AC_CHECK_SIZEOF([int *], [1111])
611 
612   # AC_CHECK_SIZEOF defines 'ac_cv_sizeof_int_p' to hold the number of bytes used by an 'int*'
613   if test "x$ac_cv_sizeof_int_p" = x; then
614     # The test failed, lets stick to the assumed value.
615     AC_MSG_WARN([The number of bits in the target could not be determined, using $OPENJDK_TARGET_CPU_BITS.])
616   else
617     TESTED_TARGET_CPU_BITS=`expr 8 \* $ac_cv_sizeof_int_p`
618 
619     if test "x$TESTED_TARGET_CPU_BITS" != "x$OPENJDK_TARGET_CPU_BITS"; then
620       AC_MSG_NOTICE([The tested number of bits in the target ($TESTED_TARGET_CPU_BITS) differs from the number of bits expected to be found in the target ($OPENJDK_TARGET_CPU_BITS)])
621       if test "x$COMPILE_TYPE" = xreduced; then
622         HELP_MSG_MISSING_DEPENDENCY([reduced])
623         AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
624       elif test "x$COMPILE_TYPE" = xcross; then
625         AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
626       fi
627       AC_MSG_ERROR([Cannot continue.])
628     fi
629   fi
630 
631   AC_MSG_CHECKING([for target address size])
632   AC_MSG_RESULT([$OPENJDK_TARGET_CPU_BITS bits])
633 ])
