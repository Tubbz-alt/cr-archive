 51 int
 52 dbgsysConnect(int fd, struct sockaddr *name, socklen_t namelen) {
 53     int rv = connect(fd, name, namelen);
 54     if (rv < 0 && (errno == EINPROGRESS || errno == EINTR)) {
 55         return DBG_EINPROGRESS;
 56     } else {
 57         return rv;
