30 void VMRegImpl::set_regName() {
31   Register reg = ::as_Register(0);
32   int i;
33   for (i = 0; i < ConcreteRegisterImpl::max_gpr ; ) {
34     regName[i++  ] = reg->name();
35     regName[i++  ] = reg->name();
36     reg = reg->successor();
37   }
38 
39   FloatRegister freg = ::as_FloatRegister(0);
40   for ( ; i < ConcreteRegisterImpl::max_fpr ; ) {
41     regName[i++] = freg->name();
42     if (freg->encoding() > 31) {
43       regName[i++] = freg->name();
44     }
45     freg = freg->successor();
46   }
47 
48   for ( ; i < ConcreteRegisterImpl::number_of_registers ; i ++ ) {
49     regName[i] = "NON-GPR-FPR";
50   }
51 }
