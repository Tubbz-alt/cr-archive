25 #ifndef CPU_SPARC_C1_LINEARSCAN_SPARC_HPP
26 #define CPU_SPARC_C1_LINEARSCAN_SPARC_HPP
27 
28 inline bool LinearScan::is_processed_reg_num(int reg_num) {
29   return reg_num < 26 || reg_num > 31;
30 }
31 
32 inline int LinearScan::num_physical_regs(BasicType type) {
33   // Sparc requires two cpu registers for long
34   // and two cpu registers for double
35   if (type == T_DOUBLE) {
36     return 2;
37   }
38   return 1;
39 }
40 
41 
42 inline bool LinearScan::requires_adjacent_regs(BasicType type) {
43   return type == T_DOUBLE;
44 }
45 
46 inline bool LinearScan::is_caller_save(int assigned_reg) {
47   return assigned_reg > pd_last_callee_saved_reg && assigned_reg <= pd_last_fpu_reg;
48 }
49 
50 
51 inline void LinearScan::pd_add_temps(LIR_Op* op) {
52   // No special case behaviours yet
53 }
54 
55 
56 inline bool LinearScanWalker::pd_init_regs_for_alloc(Interval* cur) {
57   if (allocator()->gen()->is_vreg_flag_set(cur->reg_num(), LIRGenerator::callee_saved)) {
58     assert(cur->type() != T_FLOAT && cur->type() != T_DOUBLE, "cpu regs only");
59     _first_reg = pd_first_callee_saved_reg;
60     _last_reg = pd_last_callee_saved_reg;
61     return true;
62   } else if (cur->type() == T_INT || cur->type() == T_LONG || cur->type() == T_OBJECT || cur->type() == T_ADDRESS || cur->type() == T_METADATA) {
63     _first_reg = pd_first_cpu_reg;
64     _last_reg = pd_last_allocatable_cpu_reg;
65     return true;
66   }
67   return false;
68 }
