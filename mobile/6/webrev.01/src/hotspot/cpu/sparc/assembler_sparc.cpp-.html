35 #ifdef VALIDATE_PIPELINE
36 /* Walk over the current code section and verify that there are no obvious
37  * pipeline hazards exposed in the code generated.
38  */
39 void Assembler::validate_no_pipeline_hazards() {
40   const CodeSection* csect = code_section();
41 
42   address addr0 = csect->start();
43   address addrN = csect->end();
44   uint32_t prev = 0;
45 
46   assert((addrN - addr0) % BytesPerInstWord == 0, "must be");
47 
48   for (address pc = addr0; pc != addrN; pc += BytesPerInstWord) {
49     uint32_t insn = *reinterpret_cast<uint32_t*>(pc);
50 
51     // 1. General case: No CTI immediately after other CTI
52     assert(!(is_cti(prev) && is_cti(insn)), "CTI-CTI not allowed.");
53 
54     // 2. Special case: No CTI immediately after/before RDPC
55     assert(!(is_cti(prev) && is_rdpc(insn)), "CTI-RDPC not allowed.");
56     assert(!(is_rdpc(prev) && is_cti(insn)), "RDPC-CTI not allowed.");
57 
58     prev = insn;
59   }
60 }
61 #endif
