25 #ifndef CPU_SPARC_VMREG_SPARC_HPP
26 #define CPU_SPARC_VMREG_SPARC_HPP
27 
28 inline bool is_Register() { return value() >= 0 && value() < ConcreteRegisterImpl::max_gpr; }
29 inline bool is_FloatRegister() { return value() >= ConcreteRegisterImpl::max_gpr &&
30                                         value()  < ConcreteRegisterImpl::max_fpr; }
31 inline Register as_Register() {
32   assert( is_Register() && is_even(value()), "even-aligned GPR name" );
33   // Yuk
34   return ::as_Register(value()>>1);
35 }
36 
37 inline FloatRegister as_FloatRegister() {
38   assert( is_FloatRegister(), "must be" );
39   // Yuk
40   return ::as_FloatRegister( value() - ConcreteRegisterImpl::max_gpr );
41 }
42 
43 inline   bool is_concrete() {
44   assert(is_reg(), "must be");
45   int v = value();
46   if ( v  <  ConcreteRegisterImpl::max_gpr ) {
47     return is_even(v);
48   }
49   // F0..F31
50   if ( v <= ConcreteRegisterImpl::max_gpr + 31) return true;
51   if ( v <  ConcreteRegisterImpl::max_fpr) {
52     return is_even(v);
53   }
54   assert(false, "what register?");
55   return false;
56 }
57 
58 #endif // CPU_SPARC_VMREG_SPARC_HPP
