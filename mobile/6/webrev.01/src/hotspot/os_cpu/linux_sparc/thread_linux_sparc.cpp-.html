 30 frame JavaThread::pd_last_frame() {
 31   assert(has_last_Java_frame(), "must have last_Java_sp() when suspended");
 32   assert(_anchor.walkable(), "thread has not dumped its register windows yet");
 33 
 34   assert(_anchor.last_Java_pc() != NULL, "Ack no pc!");
 35   return frame(last_Java_sp(), frame::unpatchable, _anchor.last_Java_pc());
 36 }

 38 bool JavaThread::pd_get_top_frame_for_profiling(frame* fr_addr, void* ucontext, bool isInJava) {
 39   ucontext_t* uc = (ucontext_t*) ucontext;
 40   *fr_addr = frame((intptr_t*)uc->uc_mcontext.mc_i7, frame::unpatchable,
 41                    (address)uc->uc_mcontext.mc_gregs[MC_PC]);
 42   return true;
 43 }

 45 // For Forte Analyzer AsyncGetCallTrace profiling support - thread is
 46 // currently interrupted by SIGPROF
 47 bool JavaThread::pd_get_top_frame_for_signal_handler(frame* fr_addr,
 48                                                      void* ucontext,
 49                                                      bool isInJava) {
 50   assert(Thread::current() == this, "caller must be current thread");
 51   assert(this->is_Java_thread(), "must be JavaThread");
 52 
 53   JavaThread* jt = (JavaThread *)this;
 54 
 55   if (!isInJava) {
 56     // make_walkable flushes register windows and grabs last_Java_pc
 57     // which can not be done if the ucontext sp matches last_Java_sp
 58     // stack walking utilities assume last_Java_pc set if marked flushed
 59     jt->frame_anchor()->make_walkable(jt);
 60   }
 61 
 62   // If we have a walkable last_Java_frame, then we should use it
 63   // even if isInJava == true. It should be more reliable than
 64   // ucontext info.
 65   if (jt->has_last_Java_frame() && jt->frame_anchor()->walkable()) {
 66     *fr_addr = jt->pd_last_frame();
 67     return true;
 68   }
 69 
 70   ucontext_t* uc = (ucontext_t*) ucontext;
 71 
 72   // At this point, we don't have a walkable last_Java_frame, so
 73   // we try to glean some information out of the ucontext.
 74   intptr_t* ret_sp;
 75   ExtendedPC addr =
 76     os::fetch_frame_from_context(uc, &ret_sp,
 77                                  NULL /* ret_fp only used on X86 */);
 78   if (addr.pc() == NULL || ret_sp == NULL) {
 79     // ucontext wasn't useful
 80     return false;
 81   }
 82 
 83   if (MetaspaceShared::is_in_trampoline_frame(addr.pc())) {
 84     // In the middle of a trampoline call. Bail out for safety.
 85     // This happens rarely so shouldn't affect profiling.
 86     return false;
 87   }
 88 
 89   // we were running Java code when SIGPROF came in
 90   if (isInJava) {
 91     // If we have a last_Java_sp, then the SIGPROF signal caught us
 92     // right when we were transitioning from _thread_in_Java to a new
 93     // JavaThreadState. We use last_Java_sp instead of the sp from
 94     // the ucontext since it should be more reliable.
 95     if (jt->has_last_Java_frame()) {
 96       ret_sp = jt->last_Java_sp();
 97     }
 98     // Implied else: we don't have a last_Java_sp so we use what we
 99     // got from the ucontext.
100 
101     frame ret_frame(ret_sp, frame::unpatchable, addr.pc());
102     if (!ret_frame.safe_for_sender(jt)) {
103       // nothing else to try if the frame isn't good
104       return false;
105     }
106     *fr_addr = ret_frame;
107     return true;
108   }
109 
110   // At this point, we know we weren't running Java code. We might
111   // have a last_Java_sp, but we don't have a walkable frame.
112   // However, we might still be able to construct something useful
113   // if the thread was running native code.
114   if (jt->has_last_Java_frame()) {
115     assert(!jt->frame_anchor()->walkable(), "case covered above");
116 
117     if (jt->thread_state() == _thread_in_native) {
118       frame ret_frame(jt->last_Java_sp(), frame::unpatchable, addr.pc());
119       if (!ret_frame.safe_for_sender(jt)) {
120         // nothing else to try if the frame isn't good
121         return false;
122       }
123       *fr_addr = ret_frame;
124       return true;
125     }
126   }
127 
128   // nothing else to try
129   return false;
130 }
