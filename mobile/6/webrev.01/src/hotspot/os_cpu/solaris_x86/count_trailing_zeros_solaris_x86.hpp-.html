25 #ifndef OS_CPU_SOLARIS_X86_COUNT_TRAILING_ZEROS_SOLARIS_X86_HPP
26 #define OS_CPU_SOLARIS_X86_COUNT_TRAILING_ZEROS_SOLARIS_X86_HPP
27 
28 #include "utilities/globalDefinitions.hpp"
29 
30 inline unsigned count_trailing_zeros(uintx x) {
31   assert(x != 0, "precondition");
32   uintx result;
33   __asm__(" rep bsfq %1, %0" : "=r" (result) : "rm" (x));
34   return result;
35 }
