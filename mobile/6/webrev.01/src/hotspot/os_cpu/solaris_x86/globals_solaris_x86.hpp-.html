25 #ifndef OS_CPU_SOLARIS_X86_GLOBALS_SOLARIS_X86_HPP
26 #define OS_CPU_SOLARIS_X86_GLOBALS_SOLARIS_X86_HPP
27 
28 // Sets the default values for platform dependent flags used by the runtime system.
29 // (see globals.hpp)
30 
31 define_pd_global(bool, DontYieldALot,            true); // Determined in the design center
32 #ifdef AMD64
33 define_pd_global(intx, CompilerThreadStackSize,  1024);
34 define_pd_global(intx, ThreadStackSize,          1024); // 0 => use system default
35 define_pd_global(intx, VMThreadStackSize,        1024);
36 define_pd_global(size_t, JVMInvokeMethodSlack,   8*K);
37 #else
38 define_pd_global(intx, CompilerThreadStackSize,  512);
39 // ThreadStackSize 320 allows a couple of test cases to run while
40 // keeping the number of threads that can be created high.
41 define_pd_global(intx, ThreadStackSize,          320);
42 define_pd_global(intx, VMThreadStackSize,        512);
43 define_pd_global(size_t, JVMInvokeMethodSlack,   10*K);
44 #endif // AMD64
45 
46 
47 // Used on 64 bit platforms for UseCompressedOops base address
48 define_pd_global(size_t, HeapBaseMinAddress,     2*G);
49 
50 #endif // OS_CPU_SOLARIS_X86_GLOBALS_SOLARIS_X86_HPP
