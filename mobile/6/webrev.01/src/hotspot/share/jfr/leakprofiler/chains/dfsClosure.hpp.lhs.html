31 class BitSet;
32 class Edge;
33 class EdgeStore;
34 class EdgeQueue;

36 // Class responsible for iterating the heap depth-first
37 class DFSClosure : public BasicOopIterateClosure {
38  private:
39   static EdgeStore* _edge_store;
40   static BitSet*    _mark_bits;
41   static const Edge*_start_edge;
42   static size_t _max_depth;
43   static bool _ignore_root_set;
44   DFSClosure* _parent;
45   UnifiedOopRef _reference;
46   size_t _depth;





47 
48   void add_chain();
49   void closure_impl(UnifiedOopRef reference, const oop pointee);
50 
51   DFSClosure* parent() const { return _parent; }
52   UnifiedOopRef reference() const { return _reference; }
53 
54   DFSClosure(DFSClosure* parent, size_t depth);
55   DFSClosure();
56 
57  public:
58   virtual ReferenceIterationMode reference_iteration_mode() { return DO_FIELDS_EXCEPT_REFERENT; }
59   virtual bool should_verify_oops() { return false; }
60 
61   static void find_leaks_from_edge(EdgeStore* edge_store, BitSet* mark_bits, const Edge* start_edge);
62   static void find_leaks_from_root_set(EdgeStore* edge_store, BitSet* mark_bits);
63   void do_root(UnifiedOopRef ref);
64 
65   virtual void do_oop(oop* ref);
66   virtual void do_oop(narrowOop* ref);
67 };
