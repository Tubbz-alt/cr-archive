102 void DFSClosure::closure_impl(UnifiedOopRef reference, const oop pointee) {
103   assert(pointee != NULL, "invariant");
104   assert(!reference.is_null(), "invariant");
105 
106   if (GranularTimer::is_finished()) {
107      return;
108   }
109   if (_depth == 0 && _ignore_root_set) {
110     // Root set is already marked, but we want
111     // to continue, so skip is_marked check.
112     assert(_mark_bits->is_marked(pointee), "invariant");
113   } else {
114     if (_mark_bits->is_marked(pointee)) {
115       return;
116     }
117   }
118 
119   _reference = reference;
120   _mark_bits->mark_obj(pointee);
121   assert(_mark_bits->is_marked(pointee), "invariant");
122 
123   // is the pointee a sample object?
124   if (pointee->mark().is_marked()) {
125     add_chain();
126   }
127 
128   assert(_max_depth >= 1, "invariant");
129   if (_depth < _max_depth - 1) {
130     DFSClosure next_level(this, _depth + 1);
131     pointee->oop_iterate(&next_level);


132   }
133 }

135 void DFSClosure::add_chain() {
136   const size_t array_length = _depth + 2;
137 
138   ResourceMark rm;
139   Edge* const chain = NEW_RESOURCE_ARRAY(Edge, array_length);
140   size_t idx = 0;
141 
142   // aggregate from depth-first search
143   const DFSClosure* c = this;
144   while (c != NULL) {
145     const size_t next = idx + 1;
146     chain[idx++] = Edge(&chain[next], c->reference());
147     c = c->parent();
148   }
149   assert(_depth + 1 == idx, "invariant");
150   assert(array_length == idx + 1, "invariant");
151 
152   // aggregate from breadth-first search
153   if (_start_edge != NULL) {
154     chain[idx++] = *_start_edge;
155   } else {
156     chain[idx - 1] = Edge(NULL, chain[idx - 1].reference());
157   }
158   _edge_store->put_chain(chain, idx + (_start_edge != NULL ? _start_edge->distance_to_root() : 0));
159 }

162   assert(ref != NULL, "invariant");
163   assert(is_aligned(ref, HeapWordSize), "invariant");
164   const oop pointee = HeapAccess<AS_NO_KEEPALIVE>::oop_load(ref);
165   if (pointee != NULL) {
166     closure_impl(UnifiedOopRef::encode_in_heap(ref), pointee);
167   }

 82 void DFSClosure::closure_impl(UnifiedOopRef reference, const oop pointee) {
 83   assert(pointee != NULL, "invariant");
 84   assert(!reference.is_null(), "invariant");
 85 
 86   if (GranularTimer::is_finished()) {
 87     return;
 88   }
 89   if (_depth == 0 && _ignore_root_set) {
 90     // Root set is already marked, but we want
 91     // to continue, so skip is_marked check.
 92     assert(_mark_bits->is_marked(pointee), "invariant");
 93   }  else {
 94     if (_mark_bits->is_marked(pointee)) {
 95       return;
 96     }
 97   }
 98   _reference_stack[_depth] = reference;

 99   _mark_bits->mark_obj(pointee);
100   assert(_mark_bits->is_marked(pointee), "invariant");
101 
102   // is the pointee a sample object?
103   if (pointee->mark().is_marked()) {
104     add_chain();
105   }
106 
107   assert(_max_depth >= 1, "invariant");
108   if (_depth < _max_depth - 1) {
109     _depth++;
110     pointee->oop_iterate(this);
111     assert(_depth > 0, "invariant");
112     _depth--;
113   }
114 }

116 void DFSClosure::add_chain() {
117   const size_t array_length = _depth + 2;
118 
119   ResourceMark rm;
120   Edge* const chain = NEW_RESOURCE_ARRAY(Edge, array_length);
121   size_t idx = 0;
122 
123   // aggregate from depth-first search
124   for (size_t i = 0; i <= _depth; i++) {

125     const size_t next = idx + 1;
126     const size_t depth = _depth - i;
127     chain[idx++] = Edge(&chain[next], _reference_stack[depth]);
128   }
129   assert(_depth + 1 == idx, "invariant");
130   assert(array_length == idx + 1, "invariant");
131 
132   // aggregate from breadth-first search
133   if (_start_edge != NULL) {
134     chain[idx++] = *_start_edge;
135   } else {
136     chain[idx - 1] = Edge(NULL, chain[idx - 1].reference());
137   }
138   _edge_store->put_chain(chain, idx + (_start_edge != NULL ? _start_edge->distance_to_root() : 0));
139 }
