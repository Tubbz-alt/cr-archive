25 #ifndef SHARE_GC_SHARED_MEMSET_WITH_CONCURRENT_READERS_HPP
26 #define SHARE_GC_SHARED_MEMSET_WITH_CONCURRENT_READERS_HPP
27 
28 #include "utilities/macros.hpp"
29 
30 #include <stddef.h>
31 #include <string.h>
32 
33 // Fill a block of memory with value, like memset, but with the
34 // understanding that there may be concurrent readers of that memory.
35 inline void memset_with_concurrent_readers(void* to, int value, size_t size) {
36   ::memset(to, value, size);
37 }
38 
39 #endif // SHARE_GC_SHARED_MEMSET_WITH_CONCURRENT_READERS_HPP
