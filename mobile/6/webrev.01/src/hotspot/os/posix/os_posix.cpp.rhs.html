  77 // Check core dump limit and report possible place where core can be found
  78 void os::check_dump_limit(char* buffer, size_t bufferSize) {
  79   if (!FLAG_IS_DEFAULT(CreateCoredumpOnCrash) && !CreateCoredumpOnCrash) {
  80     jio_snprintf(buffer, bufferSize, "CreateCoredumpOnCrash is disabled from command line");
  81     VMError::record_coredump_status(buffer, false);
  82     return;
  83   }
  84 
  85   int n;
  86   struct rlimit rlim;
  87   bool success;
  88 
  89   char core_path[PATH_MAX];
  90   n = get_core_path(core_path, PATH_MAX);
  91 
  92   if (n <= 0) {
  93     jio_snprintf(buffer, bufferSize, "core.%d (may not exist)", current_process_id());
  94     success = true;
  95 #ifdef LINUX
  96   } else if (core_path[0] == '"') { // redirect to user process
  97     jio_snprintf(buffer, bufferSize, "Core dumps may be processed with %s", core_path);
  98     success = true;
  99 #endif
 100   } else if (getrlimit(RLIMIT_CORE, &rlim) != 0) {
 101     jio_snprintf(buffer, bufferSize, "%s (may not exist)", core_path);
 102     success = true;
 103   } else {
 104     switch(rlim.rlim_cur) {
 105       case RLIM_INFINITY:
 106         jio_snprintf(buffer, bufferSize, "%s", core_path);
 107         success = true;
 108         break;
 109       case 0:
 110         jio_snprintf(buffer, bufferSize, "Core dumps have been disabled. To enable core dumping, try \"ulimit -c unlimited\" before starting Java again");
 111         success = false;
 112         break;
 113       default:
 114         jio_snprintf(buffer, bufferSize, "%s (max size " UINT64_FORMAT " kB). To ensure a full core dump, try \"ulimit -c unlimited\" before starting Java again", core_path, uint64_t(rlim.rlim_cur) / 1024);
 115         success = true;
 116         break;
 117     }
 118   }
 119 
 120   VMError::record_coredump_status(buffer, success);
 121 }

 123 int os::get_native_stack(address* stack, int frames, int toSkip) {
 124   int frame_idx = 0;
 125   int num_of_frames;  // number of frames captured
 126   frame fr = os::current_frame();
 127   while (fr.pc() && frame_idx < frames) {
 128     if (toSkip > 0) {
 129       toSkip --;
 130     } else {
 131       stack[frame_idx ++] = fr.pc();
 132     }
 133     if (fr.fp() == NULL || fr.cb() != NULL ||
 134         fr.sender_pc() == NULL || os::is_first_C_frame(&fr)) break;
 135 
 136     if (fr.sender_pc() && !os::is_first_C_frame(&fr)) {
 137       fr = os::get_sender_for_C_frame(&fr);
 138     } else {
 139       break;
 140     }
 141   }
 142   num_of_frames = frame_idx;
 143   for (; frame_idx < frames; frame_idx ++) {
 144     stack[frame_idx] = NULL;
 145   }
 146 
 147   return num_of_frames;
 148 }

 160 size_t os::lasterror(char *buf, size_t len) {
 161   if (errno == 0)  return 0;
 162 
 163   const char *s = os::strerror(errno);
 164   size_t n = ::strlen(s);
 165   if (n >= len) {
 166     n = len - 1;
 167   }
 168   ::strncpy(buf, s, n);
 169   buf[n] = '\0';
 170   return n;
 171 }

 178 int os::create_file_for_heap(const char* dir) {
 179   int fd;
 180 
 181 #if defined(LINUX) && defined(O_TMPFILE)
 182   char* native_dir = os::strdup(dir);
 183   if (native_dir == NULL) {
 184     vm_exit_during_initialization(err_msg("strdup failed during creation of backing file for heap (%s)", os::strerror(errno)));
 185     return -1;
 186   }
 187   os::native_path(native_dir);
 188   fd = os::open(dir, O_TMPFILE | O_RDWR, S_IRUSR | S_IWUSR);
 189   os::free(native_dir);
 190 
 191   if (fd == -1)
 192 #endif
 193   {
 194     const char name_template[] = "/jvmheap.XXXXXX";
 195 
 196     size_t fullname_len = strlen(dir) + strlen(name_template);
 197     char *fullname = (char*)os::malloc(fullname_len + 1, mtInternal);
 198     if (fullname == NULL) {
 199       vm_exit_during_initialization(err_msg("Malloc failed during creation of backing file for heap (%s)", os::strerror(errno)));
 200       return -1;
 201     }
 202     int n = snprintf(fullname, fullname_len + 1, "%s%s", dir, name_template);
 203     assert((size_t)n == fullname_len, "Unexpected number of characters in string");
 204 
 205     os::native_path(fullname);
 206 
 207     // create a new file.
 208     fd = mkstemp(fullname);
 209 
 210     if (fd < 0) {
 211       warning("Could not create file for heap with template %s", fullname);
 212       os::free(fullname);
 213       return -1;
 214     } else {
 215       // delete the name from the filesystem. When 'fd' is closed, the file (and space) will be deleted.
 216       int ret = unlink(fullname);
 217       assert_with_errno(ret == 0, "unlink returned error");
 218     }
 219 
 220     os::free(fullname);
 221   }
 222 
 223   return fd;
 224 }

 226 static char* reserve_mmapped_memory(size_t bytes, char* requested_addr) {
 227   char * addr;
 228   int flags = MAP_PRIVATE NOT_AIX( | MAP_NORESERVE ) | MAP_ANONYMOUS;
 229   if (requested_addr != NULL) {
 230     assert((uintptr_t)requested_addr % os::vm_page_size() == 0, "Requested address should be aligned to OS page size");
 231     flags |= MAP_FIXED;
 232   }
 233 
 234   // Map reserved/uncommitted pages PROT_NONE so we fail early if we
 235   // touch an uncommitted page. Otherwise, the read/write might
 236   // succeed if we have enough swap space to back the physical page.
 237   addr = (char*)::mmap(requested_addr, bytes, PROT_NONE,
 238                        flags, -1, 0);
 239 
 240   if (addr != MAP_FAILED) {
 241     MemTracker::record_virtual_memory_reserve((address)addr, bytes, CALLER_PC);
 242     return addr;
 243   }
 244   return NULL;
 245 }

 247 static int util_posix_fallocate(int fd, off_t offset, off_t len) {
 248 #ifdef __APPLE__
 249   fstore_t store = { F_ALLOCATECONTIG, F_PEOFPOSMODE, 0, len };
 250   // First we try to get a continuous chunk of disk space
 251   int ret = fcntl(fd, F_PREALLOCATE, &store);
 252   if (ret == -1) {
 253     // Maybe we are too fragmented, try to allocate non-continuous range
 254     store.fst_flags = F_ALLOCATEALL;
 255     ret = fcntl(fd, F_PREALLOCATE, &store);
 256   }
 257   if(ret != -1) {
 258     return ftruncate(fd, len);
 259   }
 260   return -1;
 261 #else
 262   return posix_fallocate(fd, offset, len);
 263 #endif
 264 }

 266 // Map the given address range to the provided file descriptor.
 267 char* os::map_memory_to_file(char* base, size_t size, int fd) {
 268   assert(fd != -1, "File descriptor is not valid");
 269 
 270   // allocate space for the file
 271   int ret = util_posix_fallocate(fd, 0, (off_t)size);
 272   if (ret != 0) {
 273     vm_exit_during_initialization(err_msg("Error in mapping Java heap at the given filesystem directory. error(%d)", ret));
 274     return NULL;
 275   }
 276 
 277   int prot = PROT_READ | PROT_WRITE;
 278   int flags = MAP_SHARED;
 279   if (base != NULL) {
 280     flags |= MAP_FIXED;
 281   }
 282   char* addr = (char*)mmap(base, size, prot, flags, fd, 0);
 283 
 284   if (addr == MAP_FAILED) {
 285     warning("Failed mmap to file. (%s)", os::strerror(errno));
 286     return NULL;
 287   }
 288   if (base != NULL && addr != base) {
 289     if (!os::release_memory(addr, size)) {
 290       warning("Could not release memory on unsuccessful file mapping");
 291     }
 292     return NULL;
 293   }
 294   return addr;
 295 }

 304 // Multiple threads can race in this code, and can remap over each other with MAP_FIXED,
 305 // so on posix, unmap the section at the start and at the end of the chunk that we mapped
 306 // rather than unmapping and remapping the whole chunk to get requested alignment.
 307 char* os::reserve_memory_aligned(size_t size, size_t alignment, int file_desc) {
 308   assert((alignment & (os::vm_allocation_granularity() - 1)) == 0,
 309       "Alignment must be a multiple of allocation granularity (page size)");
 310   assert((size & (alignment -1)) == 0, "size must be 'alignment' aligned");
 311 
 312   size_t extra_size = size + alignment;
 313   assert(extra_size >= size, "overflow, size is too large to allow alignment");
 314 
 315   char* extra_base;
 316   if (file_desc != -1) {
 317     // For file mapping, we do not call os:reserve_memory(extra_size, NULL, alignment, file_desc) because
 318     // we need to deal with shrinking of the file space later when we release extra memory after alignment.
 319     // We also cannot called os:reserve_memory() with file_desc set to -1 because on aix we might get SHM memory.
 320     // So here to call a helper function while reserve memory for us. After we have a aligned base,
 321     // we will replace anonymous mapping with file mapping.
 322     extra_base = reserve_mmapped_memory(extra_size, NULL);
 323     if (extra_base != NULL) {
 324       MemTracker::record_virtual_memory_reserve((address)extra_base, extra_size, CALLER_PC);
 325     }
 326   } else {
 327     extra_base = os::reserve_memory(extra_size, NULL, alignment);
 328   }
 329 
 330   if (extra_base == NULL) {
 331     return NULL;
 332   }
 333 
 334   // Do manual alignment
 335   char* aligned_base = align_up(extra_base, alignment);
 336 
 337   // [  |                                       |  ]
 338   // ^ extra_base
 339   //    ^ extra_base + begin_offset == aligned_base
 340   //     extra_base + begin_offset + size       ^
 341   //                       extra_base + extra_size ^
 342   // |<>| == begin_offset
 343   //                              end_offset == |<>|
 344   size_t begin_offset = aligned_base - extra_base;
 345   size_t end_offset = (extra_base + extra_size) - (aligned_base + size);
 346 
 347   if (begin_offset > 0) {
 348       os::release_memory(extra_base, begin_offset);
 349   }
 350 
 351   if (end_offset > 0) {
 352       os::release_memory(extra_base + begin_offset + size, end_offset);
 353   }
 354 
 355   if (file_desc != -1) {
 356     // After we have an aligned address, we can replace anonymous mapping with file mapping
 357     if (replace_existing_mapping_with_file_mapping(aligned_base, size, file_desc) == NULL) {
 358       vm_exit_during_initialization(err_msg("Error in mapping Java heap at the given filesystem directory"));
 359     }
 360     MemTracker::record_virtual_memory_commit((address)aligned_base, size, CALLER_PC);
 361   }
 362   return aligned_base;
 363 }

 365 // On Posix platforms, reservations are done using mmap which can be released in parts. So splitting is a no-op.
 366 void os::split_reserved_memory(char *base, size_t size, size_t split) {
 367   char* const split_address = base + split;
 368   assert(size > 0, "Sanity");
 369   assert(size > split, "Sanity");
 370   assert(split > 0, "Sanity");
 371   assert(is_aligned(base, os::vm_allocation_granularity()), "Sanity");
 372   assert(is_aligned(split_address, os::vm_allocation_granularity()), "Sanity");
 373 }

 375 int os::vsnprintf(char* buf, size_t len, const char* fmt, va_list args) {
 376   // All supported POSIX platforms provide C99 semantics.
 377   int result = ::vsnprintf(buf, len, fmt, args);
 378   // If an encoding error occurred (result < 0) then it's not clear
 379   // whether the buffer is NUL terminated, so ensure it is.
 380   if ((result < 0) && (len > 0)) {
 381     buf[len - 1] = '\0';
 382   }
 383   return result;
 384 }

 426 static void print_rlimit(outputStream* st, const char* msg,
 427                          int resource, bool output_k = false) {
 428   struct rlimit rlim;
 429 
 430   st->print(" %s ", msg);
 431   int res = getrlimit(resource, &rlim);
 432   if (res == -1) {
 433     st->print("could not obtain value");
 434   } else {
 435     // soft limit
 436     if (rlim.rlim_cur == RLIM_INFINITY) { st->print("infinity"); }
 437     else {
 438       if (output_k) { st->print(UINT64_FORMAT "k", uint64_t(rlim.rlim_cur) / 1024); }
 439       else { st->print(UINT64_FORMAT, uint64_t(rlim.rlim_cur)); }
 440     }
 441     // hard limit
 442     st->print("/");
 443     if (rlim.rlim_max == RLIM_INFINITY) { st->print("infinity"); }
 444     else {
 445       if (output_k) { st->print(UINT64_FORMAT "k", uint64_t(rlim.rlim_max) / 1024); }
 446       else { st->print(UINT64_FORMAT, uint64_t(rlim.rlim_max)); }
 447     }
 448   }
 449 }

 541 bool os::has_allocatable_memory_limit(julong* limit) {
 542   struct rlimit rlim;
 543   int getrlimit_res = getrlimit(RLIMIT_AS, &rlim);
 544   // if there was an error when calling getrlimit, assume that there is no limitation
 545   // on virtual memory.
 546   bool result;
 547   if ((getrlimit_res != 0) || (rlim.rlim_cur == RLIM_INFINITY)) {
 548     result = false;
 549   } else {
 550     *limit = (julong)rlim.rlim_cur;
 551     result = true;
 552   }
 553 #ifdef _LP64
 554   return result;
 555 #else
 556   // arbitrary virtual space limit for 32 bit Unices found by testing. If
 557   // getrlimit above returned a limit, bound it with this limit. Otherwise
 558   // directly use it.
 559   const julong max_virtual_limit = (julong)3800*M;
 560   if (result) {
 561     *limit = MIN2(*limit, max_virtual_limit);
 562   } else {
 563     *limit = max_virtual_limit;
 564   }
 565 
 566   // bound by actually allocatable memory. The algorithm uses two bounds, an
 567   // upper and a lower limit. The upper limit is the current highest amount of
 568   // memory that could not be allocated, the lower limit is the current highest
 569   // amount of memory that could be allocated.
 570   // The algorithm iteratively refines the result by halving the difference
 571   // between these limits, updating either the upper limit (if that value could
 572   // not be allocated) or the lower limit (if the that value could be allocated)
 573   // until the difference between these limits is "small".
 574 
 575   // the minimum amount of memory we care about allocating.
 576   const julong min_allocation_size = M;
 577 
 578   julong upper_limit = *limit;
 579 
 580   // first check a few trivial cases
 581   if (is_allocatable(upper_limit) || (upper_limit <= min_allocation_size)) {
 582     *limit = upper_limit;
 583   } else if (!is_allocatable(min_allocation_size)) {
 584     // we found that not even min_allocation_size is allocatable. Return it
 585     // anyway. There is no point to search for a better value any more.
 586     *limit = min_allocation_size;
 587   } else {
 588     // perform the binary search.
 589     julong lower_limit = min_allocation_size;
 590     while ((upper_limit - lower_limit) > min_allocation_size) {
 591       julong temp_limit = ((upper_limit - lower_limit) / 2) + lower_limit;
 592       temp_limit = align_down(temp_limit, min_allocation_size);
 593       if (is_allocatable(temp_limit)) {
 594         lower_limit = temp_limit;
 595       } else {
 596         upper_limit = temp_limit;
 597       }
 598     }
 599     *limit = lower_limit;
 600   }
 601   return true;
 602 #endif
 603 }

 640 // Builds a platform dependent Agent_OnLoad_<lib_name> function name
 641 // which is used to find statically linked in agents.
 642 // Parameters:
 643 //            sym_name: Symbol in library we are looking for
 644 //            lib_name: Name of library to look in, NULL for shared libs.
 645 //            is_absolute_path == true if lib_name is absolute path to agent
 646 //                                     such as "/a/b/libL.so"
 647 //            == false if only the base name of the library is passed in
 648 //               such as "L"
 649 char* os::build_agent_function_name(const char *sym_name, const char *lib_name,
 650                                     bool is_absolute_path) {
 651   char *agent_entry_name;
 652   size_t len;
 653   size_t name_len;
 654   size_t prefix_len = strlen(JNI_LIB_PREFIX);
 655   size_t suffix_len = strlen(JNI_LIB_SUFFIX);
 656   const char *start;
 657 
 658   if (lib_name != NULL) {
 659     name_len = strlen(lib_name);
 660     if (is_absolute_path) {
 661       // Need to strip path, prefix and suffix
 662       if ((start = strrchr(lib_name, *os::file_separator())) != NULL) {
 663         lib_name = ++start;
 664       }
 665       if (strlen(lib_name) <= (prefix_len + suffix_len)) {
 666         return NULL;
 667       }
 668       lib_name += prefix_len;
 669       name_len = strlen(lib_name) - suffix_len;
 670     }
 671   }
 672   len = (lib_name != NULL ? name_len : 0) + strlen(sym_name) + 2;
 673   agent_entry_name = NEW_C_HEAP_ARRAY_RETURN_NULL(char, len, mtThread);
 674   if (agent_entry_name == NULL) {
 675     return NULL;
 676   }
 677   strcpy(agent_entry_name, sym_name);
 678   if (lib_name != NULL) {
 679     strcat(agent_entry_name, "_");
 680     strncat(agent_entry_name, lib_name, name_len);
 681   }
 682   return agent_entry_name;
 683 }

 865 // Returned string is a constant. For unknown signals "UNKNOWN" is returned.
 866 const char* os::Posix::get_signal_name(int sig, char* out, size_t outlen) {
 867 
 868   const char* ret = NULL;
 869 
 870 #ifdef SIGRTMIN
 871   if (sig >= SIGRTMIN && sig <= SIGRTMAX) {
 872     if (sig == SIGRTMIN) {
 873       ret = "SIGRTMIN";
 874     } else if (sig == SIGRTMAX) {
 875       ret = "SIGRTMAX";
 876     } else {
 877       jio_snprintf(out, outlen, "SIGRTMIN+%d", sig - SIGRTMIN);
 878       return out;
 879     }
 880   }
 881 #endif
 882 
 883   if (sig > 0) {
 884     for (int idx = 0; g_signal_info[idx].sig != -1; idx ++) {
 885       if (g_signal_info[idx].sig == sig) {
 886         ret = g_signal_info[idx].name;
 887         break;
 888       }
 889     }
 890   }
 891 
 892   if (!ret) {
 893     if (!is_valid_signal(sig)) {
 894       ret = "INVALID";
 895     } else {
 896       ret = "UNKNOWN";
 897     }
 898   }
 899 
 900   if (out && outlen > 0) {
 901     strncpy(out, ret, outlen);
 902     out[outlen - 1] = '\0';
 903   }
 904   return out;
 905 }

 926 // Returns true if signal number is valid.
 927 bool os::Posix::is_valid_signal(int sig) {
 928   // MacOS not really POSIX compliant: sigaddset does not return
 929   // an error for invalid signal numbers. However, MacOS does not
 930   // support real time signals and simply seems to have just 33
 931   // signals with no holes in the signal range.
 932 #ifdef __APPLE__
 933   return sig >= 1 && sig < NSIG;
 934 #else
 935   // Use sigaddset to check for signal validity.
 936   sigset_t set;
 937   sigemptyset(&set);
 938   if (sigaddset(&set, sig) == -1 && errno == EINVAL) {
 939     return false;
 940   }
 941   return true;
 942 #endif
 943 }

1313 char* os::Posix::realpath(const char* filename, char* outbuf, size_t outbuflen) {
1314 
1315   if (filename == NULL || outbuf == NULL || outbuflen < 1) {
1316     assert(false, "os::Posix::realpath: invalid arguments.");
1317     errno = EINVAL;
1318     return NULL;
1319   }
1320 
1321   char* result = NULL;
1322 
1323   // This assumes platform realpath() is implemented according to POSIX.1-2008.
1324   // POSIX.1-2008 allows to specify NULL for the output buffer, in which case
1325   // output buffer is dynamically allocated and must be ::free()'d by the caller.
1326   char* p = ::realpath(filename, NULL);
1327   if (p != NULL) {
1328     if (strlen(p) < outbuflen) {
1329       strcpy(outbuf, p);
1330       result = outbuf;
1331     } else {
1332       errno = ENAMETOOLONG;
1333     }
1334     ::free(p); // *not* os::free
1335   } else {
1336     // Fallback for platforms struggling with modern Posix standards (AIX 5.3, 6.1). If realpath
1337     // returns EINVAL, this may indicate that realpath is not POSIX.1-2008 compatible and
1338     // that it complains about the NULL we handed down as user buffer.
1339     // In this case, use the user provided buffer but at least check whether realpath caused
1340     // a memory overwrite.
1341     if (errno == EINVAL) {
1342       outbuf[outbuflen - 1] = '\0';
1343       p = ::realpath(filename, outbuf);
1344       if (p != NULL) {
1345         guarantee(outbuf[outbuflen - 1] == '\0', "realpath buffer overwrite detected.");
1346         result = p;
1347       }
1348     }
1349   }
1350   return result;
1351

1386 // Check minimum allowable stack sizes for thread creation and to initialize
1387 // the java system classes, including StackOverflowError - depends on page
1388 // size.
1389 // The space needed for frames during startup is platform dependent. It
1390 // depends on word size, platform calling conventions, C frame layout and
1391 // interpreter/C1/C2 design decisions. Therefore this is given in a
1392 // platform (os/cpu) dependent constant.
1393 // To this, space for guard mechanisms is added, which depends on the
1394 // page size which again depends on the concrete system the VM is running
1395 // on. Space for libc guard pages is not included in this size.
1396 jint os::Posix::set_minimum_stack_sizes() {
1397   size_t os_min_stack_allowed = PTHREAD_STACK_MIN;
1398 
1399   _java_thread_min_stack_allowed = _java_thread_min_stack_allowed +
1400                                    JavaThread::stack_guard_zone_size() +
1401                                    JavaThread::stack_shadow_zone_size();
1402 
1403   _java_thread_min_stack_allowed = align_up(_java_thread_min_stack_allowed, vm_page_size());
1404   _java_thread_min_stack_allowed = MAX2(_java_thread_min_stack_allowed, os_min_stack_allowed);
1405 
1406   size_t stack_size_in_bytes = ThreadStackSize * K;
1407   if (stack_size_in_bytes != 0 &&
1408       stack_size_in_bytes < _java_thread_min_stack_allowed) {
1409     // The '-Xss' and '-XX:ThreadStackSize=N' options both set
1410     // ThreadStackSize so we go with "Java thread stack size" instead
1411     // of "ThreadStackSize" to be more friendly.
1412     tty->print_cr("\nThe Java thread stack size specified is too small. "
1413                   "Specify at least " SIZE_FORMAT "k",
1414                   _java_thread_min_stack_allowed / K);
1415     return JNI_ERR;
1416   }
1417 
1418   // Make the stack size a multiple of the page size so that
1419   // the yellow/red zones can be guarded.
1420   JavaThread::set_stack_size_at_create(align_up(stack_size_in_bytes, vm_page_size()));
1421 
1422   // Reminder: a compiler thread is a Java thread.
1423   _compiler_thread_min_stack_allowed = _compiler_thread_min_stack_allowed +
1424                                        JavaThread::stack_guard_zone_size() +
1425                                        JavaThread::stack_shadow_zone_size();
1426 
1427   _compiler_thread_min_stack_allowed = align_up(_compiler_thread_min_stack_allowed, vm_page_size());
1428   _compiler_thread_min_stack_allowed = MAX2(_compiler_thread_min_stack_allowed, os_min_stack_allowed);
1429 
1430   stack_size_in_bytes = CompilerThreadStackSize * K;
1431   if (stack_size_in_bytes != 0 &&
1432       stack_size_in_bytes < _compiler_thread_min_stack_allowed) {
1433     tty->print_cr("\nThe CompilerThreadStackSize specified is too small. "
1434                   "Specify at least " SIZE_FORMAT "k",
1435                   _compiler_thread_min_stack_allowed / K);
1436     return JNI_ERR;
1437   }
1438 
1439   _vm_internal_thread_min_stack_allowed = align_up(_vm_internal_thread_min_stack_allowed, vm_page_size());
1440   _vm_internal_thread_min_stack_allowed = MAX2(_vm_internal_thread_min_stack_allowed, os_min_stack_allowed);
1441 
1442   stack_size_in_bytes = VMThreadStackSize * K;
1443   if (stack_size_in_bytes != 0 &&
1444       stack_size_in_bytes < _vm_internal_thread_min_stack_allowed) {
1445     tty->print_cr("\nThe VMThreadStackSize specified is too small. "
1446                   "Specify at least " SIZE_FORMAT "k",
1447                   _vm_internal_thread_min_stack_allowed / K);
1448     return JNI_ERR;
1449   }
1450   return JNI_OK;
1451 }

1453 // Called when creating the thread.  The minimum stack sizes have already been calculated
1454 size_t os::Posix::get_initial_stack_size(ThreadType thr_type, size_t req_stack_size) {
1455   size_t stack_size;
1456   if (req_stack_size == 0) {
1457     stack_size = default_stack_size(thr_type);
1458   } else {
1459     stack_size = req_stack_size;
1460   }
1461 
1462   switch (thr_type) {
1463   case os::java_thread:
1464     // Java threads use ThreadStackSize which default value can be
1465     // changed with the flag -Xss
1466     if (req_stack_size == 0 && JavaThread::stack_size_at_create() > 0) {
1467       // no requested size and we have a more specific default value
1468       stack_size = JavaThread::stack_size_at_create();
1469     }
1470     stack_size = MAX2(stack_size,
1471                       _java_thread_min_stack_allowed);
1472     break;
1473   case os::compiler_thread:
1474     if (req_stack_size == 0 && CompilerThreadStackSize > 0) {
1475       // no requested size and we have a more specific default value
1476       stack_size = (size_t)(CompilerThreadStackSize * K);
1477     }
1478     stack_size = MAX2(stack_size,
1479                       _compiler_thread_min_stack_allowed);
1480     break;
1481   case os::vm_thread:
1482   case os::pgc_thread:
1483   case os::cgc_thread:
1484   case os::watcher_thread:
1485   default:  // presume the unknown thr_type is a VM internal
1486     if (req_stack_size == 0 && VMThreadStackSize > 0) {
1487       // no requested size and we have a more specific default value
1488       stack_size = (size_t)(VMThreadStackSize * K);
1489     }
1490 
1491     stack_size = MAX2(stack_size,
1492                       _vm_internal_thread_min_stack_allowed);
1493     break;
1494   }
1495 
1496   // pthread_attr_setstacksize() may require that the size be rounded up to the OS page size.
1497   // Be careful not to round up to 0. Align down in that case.
1498   if (stack_size <= SIZE_MAX - vm_page_size()) {
1499     stack_size = align_up(stack_size, vm_page_size());
1500   } else {
1501     stack_size = align_down(stack_size, vm_page_size());
1502   }
1503 
1504   return stack_size;
1505 }

1523 os::ThreadCrashProtection::ThreadCrashProtection() {
1524 }
1525 
1526 /*
1527  * See the caveats for this class in os_posix.hpp
1528  * Protects the callback call so that SIGSEGV / SIGBUS jumps back into this
1529  * method and returns false. If none of the signals are raised, returns true.
1530  * The callback is supposed to provide the method that should be protected.
1531  */
1532 bool os::ThreadCrashProtection::call(os::CrashProtectionCallback& cb) {
1533   sigset_t saved_sig_mask;
1534 
1535   Thread::muxAcquire(&_crash_mux, "CrashProtection");
1536 
1537   _protected_thread = Thread::current_or_null();
1538   assert(_protected_thread != NULL, "Cannot crash protect a NULL thread");
1539 
1540   // we cannot rely on sigsetjmp/siglongjmp to save/restore the signal mask
1541   // since on at least some systems (OS X) siglongjmp will restore the mask
1542   // for the process, not the thread
1543   pthread_sigmask(0, NULL, &saved_sig_mask);
1544   if (sigsetjmp(_jmpbuf, 0) == 0) {
1545     // make sure we can see in the signal handler that we have crash protection
1546     // installed
1547     _crash_protection = this;
1548     cb.call();
1549     // and clear the crash protection
1550     _crash_protection = NULL;
1551     _protected_thread = NULL;
1552     Thread::muxRelease(&_crash_mux);
1553     return true;
1554   }
1555   // this happens when we siglongjmp() back
1556   pthread_sigmask(SIG_SETMASK, &saved_sig_mask, NULL);
1557   _crash_protection = NULL;
1558   _protected_thread = NULL;
1559   Thread::muxRelease(&_crash_mux);
1560   return false;
1561 }

1750 // Utility to convert the given timeout to an absolute timespec
1751 // (based on the appropriate clock) to use with pthread_cond_timewait,
1752 // and sem_timedwait().
1753 // The clock queried here must be the clock used to manage the
1754 // timeout of the condition variable or semaphore.
1755 //
1756 // The passed in timeout value is either a relative time in nanoseconds
1757 // or an absolute time in milliseconds. A relative timeout will be
1758 // associated with CLOCK_MONOTONIC if available, unless the real-time clock
1759 // is explicitly requested; otherwise, or if absolute,
1760 // the default time-of-day clock will be used.
1761 
1762 // Given time is a 64-bit value and the time_t used in the timespec is
1763 // sometimes a signed-32-bit value we have to watch for overflow if times
1764 // way in the future are given. Further on Solaris versions
1765 // prior to 10 there is a restriction (see cond_timedwait) that the specified
1766 // number of seconds, in abstime, is less than current_time + 100000000.
1767 // As it will be over 20 years before "now + 100000000" will overflow we can
1768 // ignore overflow and just impose a hard-limit on seconds using the value
1769 // of "now + 100000000". This places a limit on the timeout of about 3.17
1770 // years from "now".
1771 //
1772 #define MAX_SECS 100000000
1773 
1774 // Calculate a new absolute time that is "timeout" nanoseconds from "now".
1775 // "unit" indicates the unit of "now_part_sec" (may be nanos or micros depending
1776 // on which clock API is being used).
1777 static void calc_rel_time(timespec* abstime, jlong timeout, jlong now_sec,
1778                           jlong now_part_sec, jlong unit) {
1779   time_t max_secs = now_sec + MAX_SECS;
1780 
1781   jlong seconds = timeout / NANOUNITS;
1782   timeout %= NANOUNITS; // remaining nanos
1783 
1784   if (seconds >= MAX_SECS) {
1785     // More seconds than we can add, so pin to max_secs.
1786     abstime->tv_sec = max_secs;
1787     abstime->tv_nsec = 0;
1788   } else {
1789     abstime->tv_sec = now_sec  + seconds;
1790     long nanos = (now_part_sec * (NANOUNITS / unit)) + timeout;
1791     if (nanos >= NANOUNITS) { // overflow
1792       abstime->tv_sec += 1;
1793       nanos -= NANOUNITS;
1794     }
1795     abstime->tv_nsec = nanos;
1796   }
1797 }

1799 // Unpack the given deadline in milliseconds since the epoch, into the given timespec.
1800 // The current time in seconds is also passed in to enforce an upper bound as discussed above.
1801 // This is only used with gettimeofday, when clock_gettime is not available.
1802 static void unpack_abs_time(timespec* abstime, jlong deadline, jlong now_sec) {
1803   time_t max_secs = now_sec + MAX_SECS;
1804 
1805   jlong seconds = deadline / MILLIUNITS;
1806   jlong millis = deadline % MILLIUNITS;
1807 
1808   if (seconds >= max_secs) {
1809     // Absolute seconds exceeds allowed max, so pin to max_secs.
1810     abstime->tv_sec = max_secs;
1811     abstime->tv_nsec = 0;
1812   } else {
1813     abstime->tv_sec = seconds;
1814     abstime->tv_nsec = millis_to_nanos(millis);
1815   }
1816 }

1818 static jlong millis_to_nanos_bounded(jlong millis) {
1819   // We have to watch for overflow when converting millis to nanos,
1820   // but if millis is that large then we will end up limiting to
1821   // MAX_SECS anyway, so just do that here.
1822   if (millis / MILLIUNITS > MAX_SECS) {
1823     millis = jlong(MAX_SECS) * MILLIUNITS;
1824   }
1825   return millis_to_nanos(millis);
1826 }

1830   DEBUG_ONLY(int max_secs = MAX_SECS;)

1911 void os::PlatformEvent::park() {       // AKA "down()"
1912   // Transitions for _event:
1913   //   -1 => -1 : illegal
1914   //    1 =>  0 : pass - return immediately
1915   //    0 => -1 : block; then set _event to 0 before returning
1916 
1917   // Invariant: Only the thread associated with the PlatformEvent
1918   // may call park().
1919   assert(_nParked == 0, "invariant");
1920 
1921   int v;
1922 
1923   // atomically decrement _event
1924   for (;;) {
1925     v = _event;
1926     if (Atomic::cmpxchg(&_event, v, v - 1) == v) break;
1927   }
1928   guarantee(v >= 0, "invariant");
1929 
1930   if (v == 0) { // Do this the hard way by blocking ...
1931     int status = pthread_mutex_lock(_mutex);
1932     assert_status(status == 0, status, "mutex_lock");
1933     guarantee(_nParked == 0, "invariant");
1934     ++_nParked;
1935     while (_event < 0) {
1936       // OS-level "spurious wakeups" are ignored
1937       status = pthread_cond_wait(_cond, _mutex);
1938       assert_status(status == 0 MACOS_ONLY(|| status == ETIMEDOUT),
1939                     status, "cond_wait");
1940     }
1941     --_nParked;
1942 
1943     _event = 0;
1944     status = pthread_mutex_unlock(_mutex);
1945     assert_status(status == 0, status, "mutex_unlock");
1946     // Paranoia to ensure our locked and lock-free paths interact
1947     // correctly with each other.
1948     OrderAccess::fence();
1949   }
1950   guarantee(_event >= 0, "invariant");
1951 }

2007 void os::PlatformEvent::unpark() {
2008   // Transitions for _event:
2009   //    0 => 1 : just return
2010   //    1 => 1 : just return
2011   //   -1 => either 0 or 1; must signal target thread
2012   //         That is, we can safely transition _event from -1 to either
2013   //         0 or 1.
2014   // See also: "Semaphores in Plan 9" by Mullender & Cox
2015   //
2016   // Note: Forcing a transition from "-1" to "1" on an unpark() means
2017   // that it will take two back-to-back park() calls for the owning
2018   // thread to block. This has the benefit of forcing a spurious return
2019   // from the first park() call after an unpark() call which will help
2020   // shake out uses of park() and unpark() without checking state conditions
2021   // properly. This spurious return doesn't manifest itself in any user code
2022   // but only in the correctly written condition checking loops of ObjectMonitor,
2023   // Mutex/Monitor, Thread::muxAcquire and JavaThread::sleep
2024 
2025   if (Atomic::xchg(&_event, 1) >= 0) return;
2026 
2027   int status = pthread_mutex_lock(_mutex);
2028   assert_status(status == 0, status, "mutex_lock");
2029   int anyWaiters = _nParked;
2030   assert(anyWaiters == 0 || anyWaiters == 1, "invariant");
2031   status = pthread_mutex_unlock(_mutex);
2032   assert_status(status == 0, status, "mutex_unlock");
2033 
2034   // Note that we signal() *after* dropping the lock for "immortal" Events.
2035   // This is safe and avoids a common class of futile wakeups.  In rare
2036   // circumstances this can cause a thread to return prematurely from
2037   // cond_{timed}wait() but the spurious wakeup is benign and the victim
2038   // will simply re-test the condition and re-park itself.
2039   // This provides particular benefit if the underlying platform does not
2040   // provide wait morphing.
2041 
2042   if (anyWaiters != 0) {
2043     status = pthread_cond_signal(_cond);
2044     assert_status(status == 0, status, "cond_signal");
2045   }
2046 }

2061 // Parker::park decrements count if > 0, else does a condvar wait.  Unpark
2062 // sets count to 1 and signals condvar.  Only one thread ever waits
2063 // on the condvar. Contention seen when trying to park implies that someone
2064 // is unparking you, so don't wait. And spurious returns are fine, so there
2065 // is no need to track notifications.
2066 
2067 void Parker::park(bool isAbsolute, jlong time) {
2068 
2069   // Optional fast-path check:
2070   // Return immediately if a permit is available.
2071   // We depend on Atomic::xchg() having full barrier semantics
2072   // since we are doing a lock-free update to _counter.
2073   if (Atomic::xchg(&_counter, 0) > 0) return;
2074 
2075   Thread* thread = Thread::current();
2076   assert(thread->is_Java_thread(), "Must be JavaThread");
2077   JavaThread *jt = (JavaThread *)thread;
2078 
2079   // Optional optimization -- avoid state transitions if there's
2080   // an interrupt pending.
2081   if (jt->is_interrupted(false)) {
2082     return;
2083   }
2084 
2085   // Next, demultiplex/decode time arguments
2086   struct timespec absTime;
2087   if (time < 0 || (isAbsolute && time == 0)) { // don't wait at all
2088     return;
2089   }
2090   if (time > 0) {
2091     to_abstime(&absTime, time, isAbsolute, false);
2092   }
2093 
2094   // Enter safepoint region
2095   // Beware of deadlocks such as 6317397.
2096   // The per-thread Parker:: mutex is a classic leaf-lock.
2097   // In particular a thread must never block on the Threads_lock while
2098   // holding the Parker:: mutex.  If safepoints are pending both the
2099   // the ThreadBlockInVM() CTOR and DTOR may grab Threads_lock.
2100   ThreadBlockInVM tbivm(jt);
2101 
2102   // Can't access interrupt state now that we are _thread_blocked. If we've
2103   // been interrupted since we checked above then _counter will be > 0.
2104 
2105   // Don't wait if cannot get lock since interference arises from
2106   // unparking.
2107   if (pthread_mutex_trylock(_mutex) != 0) {
2108     return;
2109   }
2110 
2111   int status;
2112   if (_counter > 0)  { // no wait needed
2113     _counter = 0;
2114     status = pthread_mutex_unlock(_mutex);
2115     assert_status(status == 0, status, "invariant");
2116     // Paranoia to ensure our locked and lock-free paths interact
2117     // correctly with each other and Java-level accesses.
2118     OrderAccess::fence();
2119     return;
2120   }
2121 
2122   OSThreadWaitState osts(thread->osthread(), false /* not Object.wait() */);
2123   jt->set_suspend_equivalent();
2124   // cleared by handle_special_suspend_equivalent_condition() or java_suspend_self()
2125 
2126   assert(_cur_index == -1, "invariant");
2127   if (time == 0) {
2128     _cur_index = REL_INDEX; // arbitrary choice when not timed
2129     status = pthread_cond_wait(&_cond[_cur_index], _mutex);
2130     assert_status(status == 0 MACOS_ONLY(|| status == ETIMEDOUT),
2131                   status, "cond_wait");
2132   }
2133   else {
2134     _cur_index = isAbsolute ? ABS_INDEX : REL_INDEX;
2135     status = pthread_cond_timedwait(&_cond[_cur_index], _mutex, &absTime);
2136     assert_status(status == 0 || status == ETIMEDOUT,
2137                   status, "cond_timedwait");
2138   }
2139   _cur_index = -1;
2140 
2141   _counter = 0;
2142   status = pthread_mutex_unlock(_mutex);
2143   assert_status(status == 0, status, "invariant");
2144   // Paranoia to ensure our locked and lock-free paths interact
2145   // correctly with each other and Java-level accesses.
2146   OrderAccess::fence();
2147 
2148   // If externally suspended while waiting, re-suspend
2149   if (jt->handle_special_suspend_equivalent_condition()) {
2150     jt->java_suspend_self();
2151   }
2152 }
