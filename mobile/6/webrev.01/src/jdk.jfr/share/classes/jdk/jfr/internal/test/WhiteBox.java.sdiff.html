26 package jdk.jfr.internal.test;
27 
28 public final class WhiteBox {
29 
30     private static boolean writeAllObjectSamples;
31     private static boolean skipBFS;
32 
33     /**
34      * If OldObjectSample event is enabled, calling this method
35      * ensures that all object samples are written, including short-lived objects.
36      * Purpose of this method is to increase determinism in tests.
37      *
38      * @param writeAllObjectSamples if all samples should be written or not
39      *
40      */
41     public static void setWriteAllObjectSamples(boolean writeAllSamples) {
42         writeAllObjectSamples = writeAllSamples;
43     }
44 
45     public static boolean getWriteAllObjectSamples() {
46         return writeAllObjectSamples;
47     }
48 
49     /**
50      * If OldObjectSample event is enabled, calling this method
51      * ensures that BFS is not used when searching for path to GC root.
52      * Purpose of this method is to trigger code paths that are
53      * hard to provoke reliably in testing.
54      *
55      * @param skipBFS if only DFS should be used
56      */
57     public static void setSkipBFS(boolean skip) {
58         skipBFS = skip;
59     }
60 
61     public static boolean getSkipBFS() {
62         return skipBFS;
63     }
64 }
