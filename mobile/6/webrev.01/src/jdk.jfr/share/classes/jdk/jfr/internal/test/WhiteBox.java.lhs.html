26 package jdk.jfr.internal.test;
27 
28 public final class WhiteBox {
29 
30     private static boolean writeAllObjectSamples;

31 
32     /**
33      * If OldObjectSample event is enabled, calling this method
34      * ensures that all object samples are written, including short-lived objects.
35      * Purpose of this method is to increase determinism in tests.
36      *
37      * @param writeAllObjectSamples if all samples should be written or not
38      *
39      */
40     public static void setWriteAllObjectSamples(boolean writeAllSamples) {
41         writeAllObjectSamples = writeAllSamples;
42     }
43 
44     public static boolean getWriteAllObjectSamples() {
45         return writeAllObjectSamples;
46     }
47 















48 }
