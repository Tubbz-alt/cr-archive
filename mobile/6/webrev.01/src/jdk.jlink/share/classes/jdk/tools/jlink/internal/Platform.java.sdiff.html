25 package jdk.tools.jlink.internal;
26 
27 import jdk.tools.jlink.plugin.ResourcePoolModule;
28 
29 import java.util.Locale;
30 
31 /**
32  * Supported platforms
33  */
34 public enum Platform {
35     WINDOWS,
36     LINUX,
37     SOLARIS,
38     MACOS,
39     AIX,
40     UNKNOWN;
41 
42     /**
43      * Returns the {@code Platform} derived from the target platform
44      * in the {@code ModuleTarget} attribute.
45      */
46     public static Platform toPlatform(String targetPlatform) {
47         String osName;
48         int index = targetPlatform.indexOf("-");
49         if (index < 0) {
50             osName = targetPlatform;
51         } else {
52             osName = targetPlatform.substring(0, index);
53         }
54         try {
55             return Platform.valueOf(osName.toUpperCase(Locale.ENGLISH));
56         } catch (IllegalArgumentException e) {
57             return Platform.UNKNOWN;
