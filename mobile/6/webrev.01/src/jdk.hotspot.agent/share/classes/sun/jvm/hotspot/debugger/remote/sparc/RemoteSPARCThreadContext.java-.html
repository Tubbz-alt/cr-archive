25 package sun.jvm.hotspot.debugger.remote.sparc;
26 
27 import sun.jvm.hotspot.debugger.*;
28 import sun.jvm.hotspot.debugger.sparc.*;
29 import sun.jvm.hotspot.debugger.remote.*;
30 
31 public class RemoteSPARCThreadContext extends SPARCThreadContext {
32   private RemoteDebuggerClient debugger;
33 
34   public RemoteSPARCThreadContext(RemoteDebuggerClient debugger) {
35     super();
36     this.debugger = debugger;
37   }
38 
39   /** This can't be implemented in this class since we would have to
40       tie the implementation to, for example, the debugging system */
41   public void setRegisterAsAddress(int index, Address value) {
42     setRegister(index, debugger.getAddressValue(value));
43   }
44 
45   /** This can't be implemented in this class since we would have to
46       tie the implementation to, for example, the debugging system */
47   public Address getRegisterAsAddress(int index) {
48     return debugger.newAddress(getRegister(index));
49   }
50 }
