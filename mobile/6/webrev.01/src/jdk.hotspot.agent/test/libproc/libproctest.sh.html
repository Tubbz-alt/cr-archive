27 # This script is used to run consistency check of Serviceabilty Agent
28 # after making any libproc.so changes. Prints "PASSED" or "FAILED" in
29 # standard output.
30 
31 usage() {
32     echo "usage: $0"
33     echo "   set SA_JAVA to be java executable from JDK 1.5"
34     exit 1 
35 }
36 
37 STARTDIR=`dirname $0`
38 
39 if [ "$1" == "-help" ]; then
40     usage
41 fi
42 
43 if [ "x$SA_JAVA" = "x" ]; then
44    SA_JAVA=java
45 fi
46 
47 # create java process with test case
48 tmp=/tmp/libproctest
49 rm -f $tmp
50 $SA_JAVA -classpath $STARTDIR LibprocTest > $tmp &
51 pid=$!
52 while [ ! -s $tmp ] ; do
53   # Kludge alert!
54   sleep 2
55 done
56 
57 # dump core
58 gcore $pid
59 kill -9 $pid
60 
61 
62 # run libproc client
63 $SA_JAVA -showversion -cp $STARTDIR/../../build/classes::$STARTDIR/../sa.jar:$STARTDIR LibprocClient x core.$pid
64 
65 # delete core
66 rm -f core.$pid
