25 package org.graalvm.compiler.lir.sparc;
26 
27 import static jdk.vm.ci.code.ValueUtil.asRegister;
28 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.REG;
29 
30 import org.graalvm.compiler.asm.sparc.SPARCAssembler;
31 import org.graalvm.compiler.asm.sparc.SPARCAssembler.CC;
32 import org.graalvm.compiler.asm.sparc.SPARCAssembler.Opfs;
33 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
34 import org.graalvm.compiler.lir.LIRInstructionClass;
35 import org.graalvm.compiler.lir.Opcode;
36 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
37 
38 import jdk.vm.ci.meta.AllocatableValue;
39 
40 public class SPARCFloatCompareOp extends SPARCLIRInstruction implements SPARCTailDelayedLIRInstruction {
41     public static final LIRInstructionClass<SPARCFloatCompareOp> TYPE = LIRInstructionClass.create(SPARCFloatCompareOp.class);
42     public static final SizeEstimate SIZE = SizeEstimate.create(1);
43 
44     private final CC cc;
45     @Opcode protected final Opfs opf;
46     @Use({REG}) protected AllocatableValue a;
47     @Use({REG}) protected AllocatableValue b;
48 
49     public SPARCFloatCompareOp(Opfs opf, CC cc, AllocatableValue a, AllocatableValue b) {
50         super(TYPE, SIZE);
51         this.cc = cc;
52         this.opf = opf;
53         this.a = a;
54         this.b = b;
55     }
56 
57     @Override
58     protected void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
59         getDelayedControlTransfer().emitControlTransfer(crb, masm);
60         SPARCAssembler.OpfOp.emitFcmp(masm, opf, cc, asRegister(a), asRegister(b));
61     }
62 
63     @Override
64     public void verify() {
65         assert a.getPlatformKind().equals(b.getPlatformKind()) : "a: " + a + " b: " + b;
66         assert opf.equals(Opfs.Fcmpd) || opf.equals(Opfs.Fcmps) : opf;
67     }
68 }
