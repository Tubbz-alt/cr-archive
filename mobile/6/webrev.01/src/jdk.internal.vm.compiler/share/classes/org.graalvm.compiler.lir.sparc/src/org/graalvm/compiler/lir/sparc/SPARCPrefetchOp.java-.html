26 package org.graalvm.compiler.lir.sparc;
27 
28 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.COMPOSITE;
29 
30 import org.graalvm.compiler.asm.sparc.SPARCAssembler;
31 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
32 import org.graalvm.compiler.lir.LIRInstructionClass;
33 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
34 
35 public final class SPARCPrefetchOp extends SPARCLIRInstruction {
36     public static final LIRInstructionClass<SPARCPrefetchOp> TYPE = LIRInstructionClass.create(SPARCPrefetchOp.class);
37     public static final SizeEstimate SIZE = SizeEstimate.create(1);
38 
39     private final SPARCAssembler.Fcn fcn;
40     @Alive({COMPOSITE}) protected SPARCAddressValue address;
41 
42     public SPARCPrefetchOp(SPARCAddressValue address, SPARCAssembler.Fcn fcn) {
43         super(TYPE, SIZE);
44         this.address = address;
45         this.fcn = fcn;
46     }
47 
48     @Override
49     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
50         masm.prefetch(address.toAddress(), fcn);
51     }
52 }
