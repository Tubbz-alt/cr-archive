25 package org.graalvm.compiler.lir.sparc;
26 
27 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.REG;
28 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.STACK;
29 
30 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
31 import org.graalvm.compiler.lir.LIRInstructionClass;
32 import org.graalvm.compiler.lir.Opcode;
33 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
34 
35 import jdk.vm.ci.meta.Value;
36 
37 /**
38  * Emits a breakpoint.
39  */
40 @Opcode("BREAKPOINT")
41 public final class SPARCBreakpointOp extends SPARCLIRInstruction {
42     public static final LIRInstructionClass<SPARCBreakpointOp> TYPE = LIRInstructionClass.create(SPARCBreakpointOp.class);
43     public static final SizeEstimate SIZE = SizeEstimate.create(1);
44 
45     // historical - from hotspot src/cpu/sparc/vm
46     // <sys/trap.h> promises that the system will not use traps 16-31
47     // We want to use ST_BREAKPOINT here, but the debugger is confused by it.
48     public static final int ST_RESERVED_FOR_USER_0 = 0x10;
49 
50     /**
51      * A set of values loaded into the Java ABI parameter locations (for inspection by a debugger).
52      */
53     @Use({REG, STACK}) protected Value[] parameters;
54 
55     public SPARCBreakpointOp(Value[] parameters) {
56         super(TYPE, SIZE);
57         this.parameters = parameters;
58     }
59 
60     @Override
61     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
62         masm.ta(ST_RESERVED_FOR_USER_0);
63     }
64 }
