25 package org.graalvm.compiler.lir.sparc;
26 
27 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
28 import org.graalvm.compiler.lir.LIRInstruction;
29 
30 /**
31  * Implementors of this interface are able to place its last instruction into the delay slot of a
32  * {@link SPARCDelayedControlTransfer} instruction.
33  *
34  * The implementor has to do following steps to emit code into the delay slot for the
35  * DelayedControlTransfer instruction:
36  * <ol>
37  * <li>Emit everything up to the second last instruction.</li>
38  * <li>Call
39  * {@link SPARCDelayedControlTransfer#emitControlTransfer(org.graalvm.compiler.lir.asm.CompilationResultBuilder, SPARCMacroAssembler)}
40  * to let the DelayedControlTransfer instruction emit its own code (But must not stuff the delay
41  * slot with Nop)</li>
42  * <li>emit the last instruction for this {@link LIRInstruction}</li>
43  * </ol>
44  *
45  * Note: If this instruction decides not to use the delay slot, it can skip the call of
46  * {@link SPARCDelayedControlTransfer#emitControlTransfer(org.graalvm.compiler.lir.asm.CompilationResultBuilder, SPARCMacroAssembler)}
47  * . The DelayedControlTransfer instruction will emit the code just with Nop in the delay slot.
48  */
49 public interface SPARCTailDelayedLIRInstruction extends SPARCLIRInstructionMixin {
50     default void setDelayedControlTransfer(SPARCDelayedControlTransfer holder) {
51         getSPARCLIRInstructionStore().delayedControlTransfer = holder;
52     }
53 
54     default SPARCDelayedControlTransfer getDelayedControlTransfer() {
55         return getSPARCLIRInstructionStore().delayedControlTransfer;
56     }
57 }
