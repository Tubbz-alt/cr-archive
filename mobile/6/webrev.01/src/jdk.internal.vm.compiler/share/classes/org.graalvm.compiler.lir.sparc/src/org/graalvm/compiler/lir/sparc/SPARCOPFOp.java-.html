25 package org.graalvm.compiler.lir.sparc;
26 
27 import static jdk.vm.ci.code.ValueUtil.asRegister;
28 import static org.graalvm.compiler.lir.LIRInstruction.OperandFlag.REG;
29 
30 import org.graalvm.compiler.asm.sparc.SPARCAssembler;
31 import org.graalvm.compiler.asm.sparc.SPARCAssembler.Opfs;
32 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
33 import org.graalvm.compiler.core.common.LIRKind;
34 import org.graalvm.compiler.lir.LIRFrameState;
35 import org.graalvm.compiler.lir.LIRInstructionClass;
36 import org.graalvm.compiler.lir.Opcode;
37 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
38 
39 import jdk.vm.ci.meta.AllocatableValue;
40 import jdk.vm.ci.sparc.SPARC;
41 import jdk.vm.ci.sparc.SPARCKind;
42 
43 public final class SPARCOPFOp extends SPARCLIRInstruction implements SPARCTailDelayedLIRInstruction {
44     public static final LIRInstructionClass<SPARCOPFOp> TYPE = LIRInstructionClass.create(SPARCOPFOp.class);
45     public static final SizeEstimate SIZE = SizeEstimate.create(1);
46 
47     @Opcode protected final Opfs opf;
48     @Use({REG}) protected AllocatableValue rs1;
49     @Use({REG}) protected AllocatableValue rs2;
50     @Def({REG}) protected AllocatableValue rd;
51     @State protected LIRFrameState state;
52 
53     public SPARCOPFOp(Opfs opf, AllocatableValue rs2, AllocatableValue rd) {
54         this(opf, SPARC.g0.asValue(LIRKind.value(SPARCKind.SINGLE)), rs2, rd);
55     }
56 
57     public SPARCOPFOp(Opfs opf, AllocatableValue rs1, AllocatableValue rs2, AllocatableValue rd) {
58         this(opf, rs1, rs2, rd, null);
59     }
60 
61     public SPARCOPFOp(Opfs opf, AllocatableValue rs1, AllocatableValue rs2, AllocatableValue rd, LIRFrameState state) {
62         super(TYPE, SIZE);
63         this.opf = opf;
64         this.rs1 = rs1;
65         this.rs2 = rs2;
66         this.rd = rd;
67         this.state = state;
68     }
69 
70     @Override
71     protected void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
72         getDelayedControlTransfer().emitControlTransfer(crb, masm);
73         if (state != null) {
74             crb.recordImplicitException(masm.position(), state);
75         }
76         SPARCAssembler.OpfOp.emit(masm, opf, asRegister(rs1), asRegister(rs2), asRegister(rd));
77     }
78 }
