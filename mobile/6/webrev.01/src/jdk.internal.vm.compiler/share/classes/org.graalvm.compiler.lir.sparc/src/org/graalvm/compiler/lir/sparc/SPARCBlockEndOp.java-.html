25 package org.graalvm.compiler.lir.sparc;
26 
27 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
28 import org.graalvm.compiler.lir.LIRInstruction;
29 import org.graalvm.compiler.lir.LIRInstructionClass;
30 import org.graalvm.compiler.lir.StandardOp.BlockEndOp;
31 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
32 
33 public abstract class SPARCBlockEndOp extends LIRInstruction implements BlockEndOp, SPARCLIRInstructionMixin {
34     public static final LIRInstructionClass<SPARCBlockEndOp> TYPE = LIRInstructionClass.create(SPARCBlockEndOp.class);
35     private final SPARCLIRInstructionMixinStore store;
36 
37     protected SPARCBlockEndOp(LIRInstructionClass<? extends SPARCBlockEndOp> c) {
38         this(c, null);
39     }
40 
41     protected SPARCBlockEndOp(LIRInstructionClass<? extends SPARCBlockEndOp> c, SizeEstimate sizeEstimate) {
42         super(c);
43         store = new SPARCLIRInstructionMixinStore(sizeEstimate);
44     }
45 
46     @Override
47     public SPARCLIRInstructionMixinStore getSPARCLIRInstructionStore() {
48         return store;
49     }
50 
51     @Override
52     public void emitCode(CompilationResultBuilder crb) {
53         emitCode(crb, (SPARCMacroAssembler) crb.asm);
54     }
55 
56     protected abstract void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm);
57 }
