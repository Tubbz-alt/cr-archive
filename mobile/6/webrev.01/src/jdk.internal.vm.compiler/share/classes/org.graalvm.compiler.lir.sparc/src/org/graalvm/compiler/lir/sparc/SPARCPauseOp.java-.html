25 package org.graalvm.compiler.lir.sparc;
26 
27 import org.graalvm.compiler.asm.sparc.SPARCMacroAssembler;
28 import org.graalvm.compiler.lir.LIRInstructionClass;
29 import org.graalvm.compiler.lir.Opcode;
30 import org.graalvm.compiler.lir.asm.CompilationResultBuilder;
31 
32 /**
33  * Emits a pause.
34  */
35 @Opcode("PAUSE")
36 public final class SPARCPauseOp extends SPARCLIRInstruction {
37     public static final LIRInstructionClass<SPARCPauseOp> TYPE = LIRInstructionClass.create(SPARCPauseOp.class);
38     public static final SizeEstimate SIZE = SizeEstimate.create(1);
39 
40     public SPARCPauseOp() {
41         super(TYPE, SIZE);
42     }
43 
44     @Override
45     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
46         masm.pause();
47     }
48 }
