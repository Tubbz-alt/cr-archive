47 @Opcode("CRUNTIME_CALL_PROLOGUE")
48 final class SPARCHotSpotCRuntimeCallPrologueOp extends SPARCLIRInstruction implements SPARCTailDelayedLIRInstruction {
49     public static final LIRInstructionClass<SPARCHotSpotCRuntimeCallPrologueOp> TYPE = LIRInstructionClass.create(SPARCHotSpotCRuntimeCallPrologueOp.class);
50     public static final SizeEstimate SIZE = SizeEstimate.create(10);
51 
52     private final int threadLastJavaSpOffset;
53     private final Register thread;
54     private final Register stackPointer;
55     @Def({REG, STACK}) protected Value threadTemp;
56     @Temp({REG}) protected AllocatableValue spScratch;
57 
58     SPARCHotSpotCRuntimeCallPrologueOp(int threadLastJavaSpOffset, Register thread, Register stackPointer, Value threadTemp, AllocatableValue spScratch) {
59         super(TYPE, SIZE);
60         this.threadLastJavaSpOffset = threadLastJavaSpOffset;
61         this.thread = thread;
62         this.stackPointer = stackPointer;
63         this.threadTemp = threadTemp;
64         this.spScratch = spScratch;
65     }
66 
67     @Override
68     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
69         // Save last Java frame.
70         Register scratchRegister = asRegister(spScratch);
71         masm.add(stackPointer, STACK_BIAS, scratchRegister);
72         masm.stx(scratchRegister, new SPARCAddress(thread, threadLastJavaSpOffset));
73 
74         // Save the thread register when calling out to the runtime.
75         SPARCMove.move(crb, masm, threadTemp, thread.asValue(LIRKind.value(XWORD)), getDelayedControlTransfer());
76     }
77 }
