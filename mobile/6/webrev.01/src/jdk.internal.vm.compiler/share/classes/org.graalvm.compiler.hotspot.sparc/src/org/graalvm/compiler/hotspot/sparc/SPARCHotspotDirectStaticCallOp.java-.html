34 import org.graalvm.compiler.nodes.CallTargetNode.InvokeKind;
35 
36 import jdk.vm.ci.meta.ResolvedJavaMethod;
37 import jdk.vm.ci.meta.Value;
38 
39 /**
40  * A direct call that complies with the conventions for such calls in HotSpot. It doesn't use an
41  * inline cache so it's just a patchable call site.
42  */
43 @Opcode("CALL_DIRECT")
44 final class SPARCHotspotDirectStaticCallOp extends DirectCallOp {
45     public static final LIRInstructionClass<SPARCHotspotDirectStaticCallOp> TYPE = LIRInstructionClass.create(SPARCHotspotDirectStaticCallOp.class);
46     public static final SizeEstimate SIZE = SizeEstimate.create(8);
47 
48     private final InvokeKind invokeKind;
49     private final GraalHotSpotVMConfig config;
50 
51     SPARCHotspotDirectStaticCallOp(ResolvedJavaMethod target, Value result, Value[] parameters, Value[] temps, LIRFrameState state, InvokeKind invokeKind, GraalHotSpotVMConfig config) {
52         super(TYPE, SIZE, target, result, parameters, temps, state);
53         assert invokeKind.isDirect();
54         this.invokeKind = invokeKind;
55         this.config = config;
56     }
57 
58     @Override
59     public void emitCallPrefixCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
60         crb.recordMark(invokeKind == InvokeKind.Static ? config.MARKID_INVOKESTATIC : config.MARKID_INVOKESPECIAL);
61     }
62 }
