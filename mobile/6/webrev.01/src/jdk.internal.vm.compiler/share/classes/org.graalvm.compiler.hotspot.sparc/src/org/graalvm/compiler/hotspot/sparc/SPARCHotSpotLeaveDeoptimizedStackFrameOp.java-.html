44 /**
45  * Pops the current frame off the stack including the return address.
46  */
47 @Opcode("LEAVE_DEOPTIMIZED_STACK_FRAME")
48 final class SPARCHotSpotLeaveDeoptimizedStackFrameOp extends SPARCLIRInstruction {
49 
50     public static final LIRInstructionClass<SPARCHotSpotLeaveDeoptimizedStackFrameOp> TYPE = LIRInstructionClass.create(SPARCHotSpotLeaveDeoptimizedStackFrameOp.class);
51 
52     protected SPARCHotSpotLeaveDeoptimizedStackFrameOp() {
53         super(TYPE);
54     }
55 
56     @Override
57     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
58         // Save O registers over restore.
59         masm.mov(o0, i0);
60         masm.mov(o1, i1);
61         masm.mov(o2, i2);
62         masm.mov(o3, i3);
63         masm.mov(o4, i4);
64 
65         masm.restoreWindow();
66     }
67 
68     @Override
69     public boolean leavesRegisterWindow() {
70         return true;
71     }
72 }
