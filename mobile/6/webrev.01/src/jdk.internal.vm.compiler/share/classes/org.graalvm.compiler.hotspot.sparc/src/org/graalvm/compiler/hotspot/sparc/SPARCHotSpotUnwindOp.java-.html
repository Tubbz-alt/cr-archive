48 /**
49  * Removes the current frame and jumps to the {@link UnwindExceptionToCallerStub}.
50  */
51 @Opcode("UNWIND")
52 final class SPARCHotSpotUnwindOp extends SPARCHotSpotEpilogueOp {
53     public static final LIRInstructionClass<SPARCHotSpotUnwindOp> TYPE = LIRInstructionClass.create(SPARCHotSpotUnwindOp.class);
54     public static final SizeEstimate SIZE = SizeEstimate.create(32);
55 
56     @Use({REG}) protected RegisterValue exception;
57 
58     SPARCHotSpotUnwindOp(RegisterValue exception) {
59         super(TYPE, SIZE);
60         this.exception = exception;
61     }
62 
63     @Override
64     public void emitCode(CompilationResultBuilder crb, SPARCMacroAssembler masm) {
65         // This Frame is left but the called unwind (which is sibling) method needs the exception as
66         // input in i0
67         masm.mov(o0, i0);
68         leaveFrame(crb);
69 
70         ForeignCallLinkage linkage = crb.foreignCalls.lookupForeignCall(UNWIND_EXCEPTION_TO_CALLER);
71         CallingConvention cc = linkage.getOutgoingCallingConvention();
72         assert cc.getArgumentCount() == 2;
73         assert exception.equals(cc.getArgument(0));
74 
75         // Get return address (is in o7 after leave).
76         Register returnAddress = asRegister(cc.getArgument(1));
77         masm.add(o7, SPARCAssembler.PC_RETURN_OFFSET, returnAddress);
78         Register scratch = g5;
79         SPARCCall.indirectJmp(crb, masm, scratch, linkage);
80     }
81 }
