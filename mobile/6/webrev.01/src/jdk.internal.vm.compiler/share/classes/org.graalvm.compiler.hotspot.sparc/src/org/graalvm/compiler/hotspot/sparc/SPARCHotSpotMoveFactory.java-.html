25 package org.graalvm.compiler.hotspot.sparc;
26 
27 import static jdk.vm.ci.hotspot.HotSpotCompressedNullConstant.COMPRESSED_NULL;
28 import static jdk.vm.ci.meta.JavaConstant.INT_0;
29 import static jdk.vm.ci.meta.JavaConstant.LONG_0;
30 
31 import org.graalvm.compiler.core.sparc.SPARCLIRGenerator.ConstantTableBaseProvider;
32 import org.graalvm.compiler.core.sparc.SPARCMoveFactory;
33 import org.graalvm.compiler.lir.LIRInstruction;
34 
35 import jdk.vm.ci.hotspot.HotSpotCompressedNullConstant;
36 import jdk.vm.ci.hotspot.HotSpotConstant;
37 import jdk.vm.ci.hotspot.HotSpotMetaspaceConstant;
38 import jdk.vm.ci.hotspot.HotSpotObjectConstant;
39 import jdk.vm.ci.meta.AllocatableValue;
40 import jdk.vm.ci.meta.Constant;
41 
42 public class SPARCHotSpotMoveFactory extends SPARCMoveFactory {
43 
44     public SPARCHotSpotMoveFactory(ConstantTableBaseProvider constantTableBaseProvider) {
45         super(constantTableBaseProvider);
46     }
47 
48     @Override
49     public boolean canInlineConstant(Constant c) {
50         if (HotSpotCompressedNullConstant.COMPRESSED_NULL.equals(c)) {
51             return true;
52         } else if (c instanceof HotSpotObjectConstant || c instanceof HotSpotMetaspaceConstant) {
53             return false;
54         } else {
55             return super.canInlineConstant(c);
56         }
57     }
58 
59     @Override
60     public LIRInstruction createLoad(AllocatableValue dst, Constant src) {
61         Constant usedSource;
62         if (COMPRESSED_NULL.equals(src)) {
63             usedSource = INT_0;
64         } else if (src instanceof HotSpotObjectConstant && ((HotSpotObjectConstant) src).isNull()) {
65             usedSource = LONG_0;
66         } else {
67             usedSource = src;
68         }
69         if (usedSource instanceof HotSpotConstant) {
70             HotSpotConstant constant = (HotSpotConstant) usedSource;
71             if (constant.isCompressed()) {
72                 return new SPARCHotSpotMove.LoadHotSpotObjectConstantInline(constant, dst);
73             } else {
74                 return new SPARCHotSpotMove.LoadHotSpotObjectConstantFromTable(constant, dst, constantTableBaseProvider.getConstantTableBase());
75             }
76         } else {
77             return super.createLoad(dst, usedSource);
78         }
79     }
80 }
