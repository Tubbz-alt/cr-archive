25 package org.graalvm.compiler.core.sparc;
26 
27 import java.util.ListIterator;
28 
29 import org.graalvm.compiler.java.DefaultSuitesCreator;
30 import org.graalvm.compiler.nodes.graphbuilderconf.GraphBuilderConfiguration.Plugins;
31 import org.graalvm.compiler.options.OptionValues;
32 import org.graalvm.compiler.phases.BasePhase;
33 import org.graalvm.compiler.phases.PhaseSuite;
34 import org.graalvm.compiler.phases.common.ExpandLogicPhase;
35 import org.graalvm.compiler.phases.tiers.CompilerConfiguration;
36 import org.graalvm.compiler.phases.tiers.LowTierContext;
37 import org.graalvm.compiler.phases.tiers.Suites;
38 
39 public class SPARCSuitesCreator extends DefaultSuitesCreator {
40 
41     public SPARCSuitesCreator(CompilerConfiguration compilerConfiguration, Plugins plugins) {
42         super(compilerConfiguration, plugins);
43     }
44 
45     @Override
46     public Suites createSuites(OptionValues options) {
47         Suites s = super.createSuites(options);
48         ListIterator<BasePhase<? super LowTierContext>> l = s.getLowTier().findPhase(ExpandLogicPhase.class);
49         while (PhaseSuite.findNextPhase(l, ExpandLogicPhase.class)) {
50             // Search for last occurrence of ExpandLogicPhase
51         }
52         l.previous();
53         l.add(new SPARCIntegerCompareCanonicalizationPhase());
54         return s;
55     }
56 }
