36 import org.graalvm.compiler.nodes.spi.LIRLowerable;
37 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
38 
39 import jdk.vm.ci.meta.AllocatableValue;
40 
41 /**
42  * Represents an address of the form [base + simm13].
43  */
44 @NodeInfo
45 public class SPARCImmediateAddressNode extends AddressNode implements LIRLowerable {
46 
47     public static final NodeClass<SPARCImmediateAddressNode> TYPE = NodeClass.create(SPARCImmediateAddressNode.class);
48 
49     @Input private ValueNode base;
50     private int displacement;
51 
52     public SPARCImmediateAddressNode(ValueNode base, int displacement) {
53         super(TYPE);
54         assert SPARCAssembler.isSimm13(displacement);
55         this.base = base;
56         this.displacement = displacement;
57     }
58 
59     @Override
60     public void generate(NodeLIRBuilderTool gen) {
61         SPARCLIRGenerator tool = (SPARCLIRGenerator) gen.getLIRGeneratorTool();
62 
63         AllocatableValue baseValue = tool.asAllocatable(gen.operand(base));
64 
65         LIRKind kind = tool.getLIRKind(stamp(NodeView.DEFAULT));
66         AllocatableValue baseReference = LIRKind.derivedBaseFromValue(baseValue);
67         if (baseReference != null) {
68             kind = kind.makeDerivedReference(baseReference);
69         }
70 
71         gen.setResult(this, new SPARCImmediateAddressValue(kind, baseValue, displacement));
72     }
73 
74     @Override
75     public ValueNode getBase() {
76         return base;
77     }
78 
79     @Override
80     public long getMaxConstantDisplacement() {
81         return displacement;
82     }
83 
84     @Override
85     public ValueNode getIndex() {
86         return null;
87     }
88 }
