26 package org.graalvm.compiler.core.sparc;
27 
28 import org.graalvm.compiler.core.common.LIRKind;
29 import org.graalvm.compiler.graph.NodeClass;
30 import org.graalvm.compiler.lir.sparc.SPARCIndexedAddressValue;
31 import org.graalvm.compiler.nodeinfo.NodeInfo;
32 import org.graalvm.compiler.nodes.NodeView;
33 import org.graalvm.compiler.nodes.ValueNode;
34 import org.graalvm.compiler.nodes.memory.address.AddressNode;
35 import org.graalvm.compiler.nodes.spi.LIRLowerable;
36 import org.graalvm.compiler.nodes.spi.NodeLIRBuilderTool;
37 
38 import jdk.vm.ci.meta.AllocatableValue;
39 
40 /**
41  * Represents an address of the form [base + index].
42  */
43 @NodeInfo
44 public class SPARCIndexedAddressNode extends AddressNode implements LIRLowerable {
45 
46     public static final NodeClass<SPARCIndexedAddressNode> TYPE = NodeClass.create(SPARCIndexedAddressNode.class);
47 
48     @Input private ValueNode base;
49     @Input private ValueNode index;
50 
51     public SPARCIndexedAddressNode(ValueNode base, ValueNode index) {
52         super(TYPE);
53         this.base = base;
54         this.index = index;
55     }
56 
57     @Override
58     public void generate(NodeLIRBuilderTool gen) {
59         SPARCLIRGenerator tool = (SPARCLIRGenerator) gen.getLIRGeneratorTool();
60 
61         AllocatableValue baseValue = tool.asAllocatable(gen.operand(base));
62         AllocatableValue indexValue = tool.asAllocatable(gen.operand(index));
63 
64         AllocatableValue baseReference = LIRKind.derivedBaseFromValue(baseValue);
65         AllocatableValue indexReference = LIRKind.derivedBaseFromValue(indexValue);
66 
67         LIRKind kind = LIRKind.combineDerived(tool.getLIRKind(stamp(NodeView.DEFAULT)), baseReference, indexReference);
68         gen.setResult(this, new SPARCIndexedAddressValue(kind, baseValue, indexValue));
69     }
70 
71     @Override
72     public ValueNode getBase() {
73         return base;
74     }
75 
76     public void setBase(ValueNode base) {
77         updateUsages(this.base, base);
78         this.base = base;
79     }
80 
81     @Override
82     public ValueNode getIndex() {
83         return index;
84     }
85 
86     public void setIndex(ValueNode index) {
87         updateUsages(this.index, index);
88         this.index = index;
89     }
90 
91     @Override
92     public long getMaxConstantDisplacement() {
93         return Long.MAX_VALUE;
94     }
95 }
