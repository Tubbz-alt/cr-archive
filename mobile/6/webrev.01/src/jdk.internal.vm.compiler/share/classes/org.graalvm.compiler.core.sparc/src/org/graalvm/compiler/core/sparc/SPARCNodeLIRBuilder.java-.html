26 package org.graalvm.compiler.core.sparc;
27 
28 import org.graalvm.compiler.core.gen.NodeLIRBuilder;
29 import org.graalvm.compiler.lir.LabelRef;
30 import org.graalvm.compiler.lir.StandardOp.JumpOp;
31 import org.graalvm.compiler.lir.gen.LIRGeneratorTool;
32 import org.graalvm.compiler.lir.sparc.SPARCJumpOp;
33 import org.graalvm.compiler.nodes.StructuredGraph;
34 import org.graalvm.compiler.nodes.ValueNode;
35 
36 /**
37  * This class implements the SPARC specific portion of the LIR generator.
38  */
39 public abstract class SPARCNodeLIRBuilder extends NodeLIRBuilder {
40 
41     public SPARCNodeLIRBuilder(StructuredGraph graph, LIRGeneratorTool lirGen, SPARCNodeMatchRules nodeMatchRules) {
42         super(graph, lirGen, nodeMatchRules);
43     }
44 
45     @Override
46     protected boolean peephole(ValueNode valueNode) {
47         // No peephole optimizations for now
48         return false;
49     }
50 
51     @Override
52     protected JumpOp newJumpOp(LabelRef ref) {
53         return new SPARCJumpOp(ref);
54     }
55 
56     @Override
57     public SPARCLIRGenerator getLIRGeneratorTool() {
58         return (SPARCLIRGenerator) super.getLIRGeneratorTool();
59     }
60 
61     @Override
62     protected void emitPrologue(StructuredGraph graph) {
63         getLIRGeneratorTool().emitLoadConstantTableBase();
64         super.emitPrologue(graph);
65     }
66 }
