25 package org.graalvm.compiler.core.sparc;
26 
27 import org.graalvm.compiler.core.common.LIRKind;
28 import org.graalvm.compiler.core.common.spi.LIRKindTool;
29 import org.graalvm.compiler.debug.GraalError;
30 
31 import jdk.vm.ci.sparc.SPARCKind;
32 
33 public class SPARCLIRKindTool implements LIRKindTool {
34 
35     @Override
36     public LIRKind getIntegerKind(int bits) {
37         if (bits <= 8) {
38             return LIRKind.value(SPARCKind.BYTE);
39         } else if (bits <= 16) {
40             return LIRKind.value(SPARCKind.HWORD);
41         } else if (bits <= 32) {
42             return LIRKind.value(SPARCKind.WORD);
43         } else {
44             assert bits <= 64;
45             return LIRKind.value(SPARCKind.XWORD);
46         }
47     }
48 
49     @Override
50     public LIRKind getFloatingKind(int bits) {
51         switch (bits) {
52             case 32:
53                 return LIRKind.value(SPARCKind.SINGLE);
54             case 64:
55                 return LIRKind.value(SPARCKind.DOUBLE);
56             default:
57                 throw GraalError.shouldNotReachHere();
58         }
59     }
60 
61     @Override
62     public LIRKind getObjectKind() {
63         return LIRKind.reference(SPARCKind.XWORD);
64     }
65 
66     @Override
67     public LIRKind getWordKind() {
68         return LIRKind.value(SPARCKind.XWORD);
69     }
70 
71     @Override
72     public LIRKind getNarrowOopKind() {
73         return LIRKind.compressedReference(SPARCKind.WORD);
74     }
75 
76     @Override
77     public LIRKind getNarrowPointerKind() {
78         return LIRKind.value(SPARCKind.WORD);
79     }
80 }
