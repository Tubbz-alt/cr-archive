26 package sun.nio.fs;
27 
28 import java.nio.file.*;
29 import java.nio.file.attribute.*;
30 import java.nio.file.spi.FileTypeDetector;
31 import java.io.IOException;
32 import jdk.internal.util.StaticProperty;
33 import sun.security.action.GetPropertyAction;
34 
35 /**
36  * Solaris implementation of FileSystemProvider
37  */
38 
39 class SolarisFileSystemProvider extends UnixFileSystemProvider {
40     public SolarisFileSystemProvider() {
41         super();
42     }
43 
44     @Override
45     SolarisFileSystem newFileSystem(String dir) {
46         return new SolarisFileSystem(this, dir);
47     }
48 
49     @Override
50     SolarisFileStore getFileStore(UnixPath path) throws IOException {
51         return new SolarisFileStore(path);
52     }
53 
54 
55     @Override
56     @SuppressWarnings("unchecked")
57     public <V extends FileAttributeView> V getFileAttributeView(Path obj,
58                                                                 Class<V> type,
59                                                                 LinkOption... options)
60     {
61         if (type == AclFileAttributeView.class) {
62             return (V) new SolarisAclFileAttributeView(UnixPath.toUnixPath(obj),
63                                                        Util.followLinks(options));
64         }
65         if (type == UserDefinedFileAttributeView.class) {
66             return(V) new SolarisUserDefinedFileAttributeView(UnixPath.toUnixPath(obj),
67                                                               Util.followLinks(options));
68         }
69         return super.getFileAttributeView(obj, type, options);
70     }
71 
72     @Override
73     public DynamicFileAttributeView getFileAttributeView(Path obj,
74                                                          String name,
75                                                          LinkOption... options)
76     {
77         if (name.equals("acl"))
78             return new SolarisAclFileAttributeView(UnixPath.toUnixPath(obj),
79                                                    Util.followLinks(options));
80         if (name.equals("user"))
81             return new SolarisUserDefinedFileAttributeView(UnixPath.toUnixPath(obj),
82                                                            Util.followLinks(options));
83         return super.getFileAttributeView(obj, name, options);
84     }
85 
86     @Override
87     FileTypeDetector getFileTypeDetector() {
88         Path userMimeTypes = Path.of(StaticProperty.userHome(), ".mime.types");
89         Path etcMimeTypes = Path.of("/etc/mime.types");
90 
91         return chain(new MimeTypesFileTypeDetector(userMimeTypes),
92                      new MimeTypesFileTypeDetector(etcMimeTypes));
93     }
94 }
