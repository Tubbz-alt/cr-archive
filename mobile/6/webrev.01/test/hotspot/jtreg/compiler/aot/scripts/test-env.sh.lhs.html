22 # set platform-dependent variables
23 OS=`uname -s`
24 case "$OS" in
25   Linux | SunOS )
26     SO_TYPE=so
27     ;;
28   Darwin )
29     SO_TYPE=dylib
30     ;;
31   Windows_* )
32     SO_TYPE=dll
33     ;;
34   CYGWIN_* )
35     SO_TYPE=dll
36     ;;
37   * )
38     echo "Unrecognized system!"
39     exit 1;
40     ;;
41 esac
