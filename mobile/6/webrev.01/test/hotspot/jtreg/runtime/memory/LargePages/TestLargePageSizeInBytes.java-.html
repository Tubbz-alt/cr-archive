24 /* @test TestLargePageSizeInBytes
25  * @summary Tests that the flag -XX:LargePageSizeInBytes does not cause warnings on Solaris
26  * @bug 8049536
27  * @requires os.family == "solaris"
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @run driver TestLargePageSizeInBytes
32  */
33 
34 import jdk.test.lib.process.OutputAnalyzer;
35 import jdk.test.lib.process.ProcessTools;
36 
37 public class TestLargePageSizeInBytes {
38     private static long M = 1024L * 1024L;
39     private static long G = 1024L * M;
40 
41     public static void main(String[] args) throws Exception {
42         testLargePageSizeInBytes(4 * M);
43         testLargePageSizeInBytes(256 * M);
44         testLargePageSizeInBytes(512 * M);
45         testLargePageSizeInBytes(2 * G);
46     }
47 
48     private static void testLargePageSizeInBytes(long size) throws Exception {
49         ProcessBuilder pb =
50             ProcessTools.createJavaProcessBuilder("-XX:+UseLargePages",
51                                                   "-XX:LargePageSizeInBytes=" + size,
52                                                   "-version");
53 
54         OutputAnalyzer out = new OutputAnalyzer(pb.start());
55         out.shouldNotContain("Attempt to use MPSS failed.");
56         out.shouldHaveExitValue(0);
57     }
58 }
