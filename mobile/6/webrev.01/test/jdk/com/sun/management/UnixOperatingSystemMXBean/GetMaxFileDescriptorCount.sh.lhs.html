33 if [ "${TESTJAVA}" = "" ] ; then
34      echo "--Error: TESTJAVA must be defined as the pathname of a jdk to test."
35      exit 1
36 fi
37 if [ "${COMPILEJAVA}" = "" ]; then
38     COMPILEJAVA="${TESTJAVA}"
39 fi
40 
41 runOne()
42 {
43    echo "runOne $@"
44    $COMPILEJAVA/bin/javac ${TESTJAVACOPTS} ${TESTTOOLVMOPTS} -d $TESTCLASSES \
45        $TESTSRC/$@.java || exit 2
46    $TESTJAVA/bin/java ${TESTVMOPTS} -classpath $TESTCLASSES $@        || exit 3
47 }
48 
49 # Test GetMaxFileDescriptorCount if we are running on Unix
50 case `uname -s` in
51     SunOS | Linux )
52        runOne GetMaxFileDescriptorCount
53        ;;
54     * )
55        echo "Ignore test when not run on Solaris or Linux"
56        exit 0
57        ;;
58 esac
59 
60 exit 0
