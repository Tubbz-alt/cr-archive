36         String arch = System.getProperty("os.arch");
37         System.err.println("os.arch:" + arch);
38         if (((arch.equals("i386") && (bo != ByteOrder.LITTLE_ENDIAN))) ||
39             ((arch.equals("amd64") && (bo != ByteOrder.LITTLE_ENDIAN))) ||
