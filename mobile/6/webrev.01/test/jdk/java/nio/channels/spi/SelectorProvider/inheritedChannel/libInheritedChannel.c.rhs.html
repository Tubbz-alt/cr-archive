 46 /*
 47  * Throws the exception of the given class name and detail message
 48  */
 49 static void ThrowException(JNIEnv *env, const char *name, const char *msg) {
 50     jclass cls = (*env)->FindClass(env, name);
 51     if (cls != NULL) {
 52         (*env)->ThrowNew(env, cls, msg);
 53     }
 54 }

 56 /*
 57  * Convert a jstring to an ISO 8859_1 encoded C string
 58  */
 59 static char* getString8859_1Chars(JNIEnv *env, jstring jstr) {
 60     int i;
 61     char *result;
 62     jint len = (*env)->GetStringLength(env, jstr);
 63     const jchar *str = (*env)->GetStringCritical(env, jstr, 0);
 64     if (str == 0) {
 65         return NULL;
 66     }
 67 
 68     result = (char*)malloc(len+1);
 69     if (result == 0) {
 70         (*env)->ReleaseStringCritical(env, jstr, str);
 71         ThrowException(env, "java/lang/OutOfMemoryError", NULL);
 72         return NULL;
 73     }
 74 
 75     for (i=0; i<len; i++) {
 76         jchar unicode = str[i];
 77         if (unicode <= 0x00ff)
 78             result[i] = unicode;
 79         else
 80             result[i] = '?';
 81     }
 82 
 83     result[len] = 0;
 84     (*env)->ReleaseStringCritical(env, jstr, str);
 85     return result;
 86 }

 89 /*
 90  * Class:     Launcher
 91  * Method:    launch0
 92  * Signature: ([Ljava/lang/String;I)V
 93  */
 94 JNIEXPORT void JNICALL Java_Launcher_launch0
 95   (JNIEnv *env, jclass cls, jobjectArray cmdarray, jint serviceFd)
 96 {
 97     pid_t pid;
 98     DIR* dp;
 99     struct dirent* dirp;
100     int thisFd;
101     char** cmdv;
102     int i, cmdlen;
103 
104     /*
105      * Argument 0 of the command array is the program name.
106      * Here we just extract the program name and any arguments into
107      * a command array suitable for use with execvp.
108      */
109     cmdlen = (*env)->GetArrayLength(env, cmdarray);
110     if (cmdlen == 0) {
111         ThrowException(env, "java/lang/IllegalArgumentException",
112             "command array must at least include the program name");
113         return;
114     }
115     cmdv = (char **)malloc((cmdlen + 1) * sizeof(char *));
116     if (cmdv == NULL) {
117         ThrowException(env, "java/lang/OutOfMemoryError", NULL);
118         return;
119     }
120 
121     for (i=0; i<cmdlen; i++) {
122         jstring str = (*env)->GetObjectArrayElement(env, cmdarray, i);
123         cmdv[i] = (char *) getString8859_1Chars(env, str);
124         if (cmdv[i] == NULL) {
125             return;
126         }
127     }
128 
129     /*
130      * Command array must have NULL as the last entry
131      */
132     cmdv[cmdlen] = NULL;
133 
134     /*
135      * Launch the program. As this isn't a complete inetd or Runtime.exec
136      * implementation we don't have a reaper to pick up child exit status.
137      */



138     pid = fork();

139     if (pid != 0) {
140         if (pid < 0) {
141             ThrowException(env, "java/io/IOException", "fork failed");
142         }
143         return;
144     }
145 
146     /*
147      * We need to close all file descriptors except for serviceFd. To
148      * get the list of open file descriptos we read through /proc/self/fd (/dev/fd)
149      * but to open this requires a file descriptor. We could use a specific
150      * file descriptor and fdopendir but Linux doesn't seem to support
151      * fdopendir. Instead we use opendir and make an assumption on the
152      * file descriptor that is used (by opening & closing a file).
153      */
154     thisFd = open("/dev/fd", O_RDONLY);
155     if (thisFd < 0) {
156         _exit(-1);
157     }
158 
159     if ((dp = fdopendir(thisFd)) == NULL) {
160         _exit(-1);
161     }
162 
163     while ((dirp = readdir(dp)) != NULL) {
164         if (isdigit(dirp->d_name[0])) {
165             int fd = strtol(dirp->d_name, NULL, 10);
166             if (fd != serviceFd && fd != thisFd) {
167                 close(fd);
168             }
169         }
170     }
171     closedir(dp);
172 
173     /*
174      * At this point all file descriptors are closed except for
175      * serviceFd. We not dup 0,1,2 to this file descriptor and
176      * close serviceFd. This should leave us with only 0,1,2
177      * open and all connected to the same socket.
178      */
179     dup2(serviceFd, STDIN_FILENO);
180     dup2(serviceFd, STDOUT_FILENO);
181     dup2(serviceFd, STDERR_FILENO);
182     close(serviceFd);
183 
184     execvp(cmdv[0], cmdv);
185     _exit(-1);
186 }

188 JNIEXPORT void JNICALL Java_UnixDomainSocket_init(JNIEnv *env, jclass cls) {
189     CHECK(unixSocketClass = (*env)->FindClass(env, "UnixDomainSocket"));
190     CHECK(unixSocketClass = (*env)->NewGlobalRef(env, unixSocketClass));
191     CHECK(unixSocketCtor = (*env)->GetMethodID(env, unixSocketClass, "<init>", "(I)V"));
192 }

194 /*
195  * Class:     UnixDomainSocket
196  * Method:    socketpair
197  * Signature: ()[LUnixDomainSocket
198  */
199 JNIEXPORT jobjectArray JNICALL Java_UnixDomainSocket_socketpair
200   (JNIEnv *env, jclass cls)
201 {
202     int fds[2];
203     jobject socket;
204     jobjectArray result = (*env)->NewObjectArray(env, 2, unixSocketClass, 0);
205     if (socketpair(AF_UNIX, SOCK_STREAM, 0, fds) < 0) {
206         perror("socketpair");
207         return result;
208     }
209     socket = (*env)->NewObject(env, unixSocketClass, unixSocketCtor, fds[0]);
210     (*env)->SetObjectArrayElement(env, result, 0, socket);
211     socket = (*env)->NewObject(env, unixSocketClass, unixSocketCtor, fds[1]);
212     (*env)->SetObjectArrayElement(env, result, 1, socket);
213     return result;
214 }

216 JNIEXPORT jint JNICALL Java_UnixDomainSocket_create
217   (JNIEnv *env, jclass cls)
218 {
219     int sock = socket(AF_UNIX, SOCK_STREAM, 0);
220     if (sock == -1) {
221         ThrowException(env, "java/io/IOException", "socket create error");
222     }
223     return sock;
224 }

226 JNIEXPORT void JNICALL Java_UnixDomainSocket_bind0
227   (JNIEnv *env, jclass cls, jint sock, jstring name)
228 {
229     struct sockaddr_un addr;
230     const char *nameUtf = (*env)->GetStringUTFChars(env, name, NULL);
231     int ret = -1;
232     int length = sizeof(addr.sun_path);
233     unlink(nameUtf);
234     memset(&addr, 0, sizeof(addr));
235     addr.sun_family = AF_UNIX;
236     strncpy(addr.sun_path, nameUtf, length);
237     addr.sun_path[length - 1] = '\0';
238     ret = bind(sock, (const struct sockaddr*)&addr, sizeof(addr));
239     if (ret == -1) {
240         ThrowException(env, "java/io/IOException", "socket bind error");
241     }
242     ret = listen(sock, 5);
243     if (ret == -1) {
244         ThrowException(env, "java/io/IOException", "socket bind error");
245     }
246     (*env)->ReleaseStringUTFChars(env, name, nameUtf);
247 }

249 JNIEXPORT jint JNICALL Java_UnixDomainSocket_accept0
250   (JNIEnv *env, jclass cls, jint sock)
251 {
252     struct sockaddr_storage addr;
253     socklen_t len = sizeof(addr);
254     int ret = accept(sock, (struct sockaddr *)&addr, &len);
255     if (ret == -1)
256         ThrowException(env, "java/io/IOException", "socket accept error");
257     return ret;
258 }

279 JNIEXPORT jint JNICALL Java_UnixDomainSocket_read0
280   (JNIEnv *env, jclass cls, jint fd)
281 {
282     int ret;
283     unsigned char res;
284     ret = read(fd, &res, 1);
285     if (ret == 0)
286         return -1; /* EOF */
287     else if (ret < 0) {
288         ThrowException(env, "java/io/IOException", "read error");
289         return -1;
290     }
291     return res;
292 }

305 JNIEXPORT void JNICALL Java_UnixDomainSocket_close0
306   (JNIEnv *env, jclass cls, jint fd, jstring name)
307 {
308     close(fd);
309     if (name != NULL) {
310         const char *nameUtf = (*env)->GetStringUTFChars(env, name, NULL);
311         unlink(nameUtf);
312         (*env)->ReleaseStringUTFChars(env, name, nameUtf);
313     }
