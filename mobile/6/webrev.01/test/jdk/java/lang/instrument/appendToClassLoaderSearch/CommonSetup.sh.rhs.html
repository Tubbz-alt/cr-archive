28 # Common setup for unit tests. Setups up the following variables:
29 #
30 # PS - path sep.
31 # FS - file sep.
32 # JAVA - java cmd.
33 # JAVAC - javac cmd.
34 # JAR - jar cmd.
35 
36 OS=`uname -s`
37 case "$OS" in




38   Linux )
39     PS=":"
40     FS="/"
41     ;;
42   Darwin )
43     PS=":"
44     FS="/"
45     ;;
46   AIX )
47     PS=":"
48     FS="/"
49     ;;
50   Windows*)
51     PS=";"
52     OS="Windows"
53     FS="\\"
54     ;;
55   CYGWIN*)
56     PS=";"
57     OS="Windows"
58     FS="\\"
59     isCygwin=true
60     ;;
61   * )
62     echo "Unrecognized system!"
63     exit 1;
64     ;;
65 esac
66 
67 if [ "${TESTJAVA}" = "" ]
68 then
69   echo "TESTJAVA not set.  Test cannot execute.  Failed."
70   exit 1
71 fi
72 
73 if [ "${COMPILEJAVA}" = "" ]
74 then
75   COMPILEJAVA="${TESTJAVA}"
76 fi
77 echo "COMPILEJAVA=${COMPILEJAVA}"
78 
79 if [ "${TESTSRC}" = "" ]
80 then
81   echo "TESTSRC not set.  Test cannot execute.  Failed."
82   exit 1
83 fi
84 
85 if [ "${TESTCLASSES}" = "" ]
86 then
87   echo "TESTCLASSES not set.  Test cannot execute.  Failed."
88   exit 1
89 fi
90 
91 JAVA="${TESTJAVA}/bin/java"
92 JAVAC="${COMPILEJAVA}/bin/javac"
93 JAR="${COMPILEJAVA}/bin/jar"
