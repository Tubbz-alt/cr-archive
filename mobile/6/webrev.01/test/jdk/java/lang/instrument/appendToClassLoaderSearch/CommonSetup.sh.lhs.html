28 # Common setup for unit tests. Setups up the following variables:
29 #
30 # PS - path sep.
31 # FS - file sep.
32 # JAVA - java cmd.
33 # JAVAC - javac cmd.
34 # JAR - jar cmd.
35 
36 OS=`uname -s`
37 case "$OS" in
38   SunOS )
39     PS=":"
40     FS="/"
41     ;;
42   Linux )
43     PS=":"
44     FS="/"
45     ;;
46   Darwin )
47     PS=":"
48     FS="/"
49     ;;
50   AIX )
51     PS=":"
52     FS="/"
53     ;;
54   Windows*)
55     PS=";"
56     OS="Windows"
57     FS="\\"
58     ;;
59   CYGWIN*)
60     PS=";"
61     OS="Windows"
62     FS="\\"
63     isCygwin=true
64     ;;
65   * )
66     echo "Unrecognized system!"
67     exit 1;
68     ;;
69 esac
70 
71 if [ "${TESTJAVA}" = "" ]
72 then
73   echo "TESTJAVA not set.  Test cannot execute.  Failed."
74   exit 1
75 fi
76 
77 if [ "${COMPILEJAVA}" = "" ]
78 then
79   COMPILEJAVA="${TESTJAVA}"
80 fi
81 echo "COMPILEJAVA=${COMPILEJAVA}"
82 
83 if [ "${TESTSRC}" = "" ]
84 then
85   echo "TESTSRC not set.  Test cannot execute.  Failed."
86   exit 1
87 fi
88 
89 if [ "${TESTCLASSES}" = "" ]
90 then
91   echo "TESTCLASSES not set.  Test cannot execute.  Failed."
92   exit 1
93 fi
94 
95 JAVA="${TESTJAVA}/bin/java"
96 JAVAC="${COMPILEJAVA}/bin/javac"
97 JAR="${COMPILEJAVA}/bin/jar"
98
