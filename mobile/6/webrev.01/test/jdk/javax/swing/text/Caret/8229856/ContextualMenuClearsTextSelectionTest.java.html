 24 /*
 25  * @test
 26  * @key headful
 27  * @bug 8229856
 28  * @requires (os.family == "mac")
 29  * @summary [macos] Opening a menu on a JTextField can clear the text selection
 30  * @run main ContextualMenuClearsTextSelectionTest
 31  */
 32 
 33 import javax.swing.JFrame;
 34 import javax.swing.JMenuItem;
 35 import javax.swing.JPopupMenu;
 36 import javax.swing.JTextField;
 37 import javax.swing.SwingUtilities;
 38 import javax.swing.WindowConstants;
 39 import java.awt.BorderLayout;
 40 import java.awt.Component;
 41 import java.awt.event.InputEvent;
 42 import java.awt.event.KeyEvent;
 43 import java.awt.event.MouseAdapter;
 44 import java.awt.event.MouseEvent;
 45 import java.awt.Point;
 46 import java.awt.Robot;
 47 
 48 public class ContextualMenuClearsTextSelectionTest {
 49     private static Robot robot;
 50     private static Point p;
 51     private static JTextField textField;
 52     private static JPopupMenu popupMenu;
 53     private static JFrame frame;
 54     private static boolean isSelectionCleared;
 55     public static void main(String[] args) throws Exception{
 56         if (!System.getProperty("os.name").startsWith("Mac")) {
 57             System.out.println("This test is meant for Mac platform only");
 58             return;
 59         }
 60         try {
 61             robot = new Robot();
 62             robot.setAutoDelay(50);
 63             SwingUtilities.invokeAndWait(() -> {
 64                 createAndShowGUI();
 65             });
 66             robot.waitForIdle();
 67             SwingUtilities.invokeAndWait(() -> {
 68                 p = textField.getLocationOnScreen();
 69                 textField.requestFocusInWindow();
 70             });
 71             robot.delay(200);
 72             robot.waitForIdle();
 73 
 74             robot.mouseMove(p.x + 10, p.y + 10);
 75             robot.waitForIdle();
 76 
 77             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
 78             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
 79             robot.waitForIdle();
 80 
 81             robot.delay(2000);
 82 
 83             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
 84             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
 85             robot.waitForIdle();
 86 
 87             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
 88             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
 89             robot.waitForIdle();
 90 
 91             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
 92             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
 93             robot.waitForIdle();
 94 
 95             robot.keyPress(KeyEvent.VK_CONTROL);
 96             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
 97             robot.waitForIdle();
 98 
 99             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
100             robot.keyRelease(KeyEvent.VK_CONTROL);
101             robot.waitForIdle();
102 
103             SwingUtilities.invokeAndWait(() -> {
104                 isSelectionCleared = textField.getSelectedText() == null
105                                         ? true : false ;
106             });
107 
108             if (isSelectionCleared) {
109                 throw new RuntimeException("Text selection is cleared");
110             }
111         } finally {
112             if (frame != null) SwingUtilities.invokeAndWait(() ->
113                 frame.dispose());
114         }
115     }
116 
117     private static void createAndShowGUI() {
118         frame =  new JFrame();
119         frame.setLayout(new BorderLayout());
120         frame.setSize(200,200);
121         textField = new JTextField("word");
122         textField.setSize(100,20);
123         popupMenu = new JPopupMenu();
124         popupMenu.add(new JMenuItem("Apple"));
125         popupMenu.add(new JMenuItem("Pear"));
126         popupMenu.add(new JMenuItem("Grape"));
127         createPopup(textField);
128 
129         frame.add(textField, BorderLayout.NORTH);
130         frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
131         frame.pack();
132         frame.setVisible(true);
133     }
134 
135     private static void createPopup(JTextField f) {
136         MouseAdapter a = new MouseAdapter() {
137             public void mousePressed(MouseEvent e) {
138                 if (e.isPopupTrigger()) {
139                     showPopupMenu(e);
140                 }
141             }
142             public void mouseReleased(MouseEvent e) {
143                 if (e.isPopupTrigger()) {
144                     showPopupMenu(e);
145                 }
146             }
147             private void showPopupMenu(MouseEvent e) {
148                 popupMenu.show((Component) e.getSource(), e.getX(), e.getY());
149             }
150         };
151         f.addMouseListener(a);
152     }
153 }
