29 # Common setup for the script API unit tests. Setups up the following variables:
30 #
31 # PS - path sep.
32 # FS - file sep.
33 # JAVA - java cmd.
34 # JAVAC - javac cmd.
35 # JAR - jar cmd.
36 
37 OS=`uname -s`
38 case "$OS" in
39   SunOS | Linux | Darwin | AIX )
40     PS=":"
41     FS="/"
42     ;;
43   Windows* )
44     PS=";"
45     OS="Windows"
46     FS="\\"
47     ;;
48   CYGWIN* )
49     PS=";"
50     OS="Windows"
51     FS="\\"
52     isCygwin=true
53     ;;
54   * )
55     echo "Unrecognized system!"
56     exit 1;
57     ;;
58 esac
