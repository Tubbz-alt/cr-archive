 24 # @test
 25 # @bug 5070773
 26 # @summary SunPKCS11 provider does not support spaces config's provider name
 27 # @run shell ConfigQuotedString.sh
 28 
 29 # set a few environment variables so that the shell-script can run stand-alone
 30 # in the source directory
 31 
 32 # if running by hand on windows, change TESTSRC and TESTCLASSES to "."
 33 if [ "${TESTSRC}" = "" ] ; then
 34     TESTSRC=`pwd`
 35 fi
 36 if [ "${TESTCLASSES}" = "" ] ; then
 37     TESTCLASSES=`pwd`
 38 fi
 39 
 40 # if running by hand on windows, change this to appropriate value
 41 if [ "${TESTJAVA}" = "" ] ; then
 42     TESTJAVA="/net/radiant/export1/charlie/mustang/build/solaris-sparc"
 43 fi
 44 if [ "${COMPILEJAVA}" = "" ]; then
 45     COMPILEJAVA="${TESTJAVA}"
 46 fi
 47 echo TESTSRC=${TESTSRC}
 48 echo TESTCLASSES=${TESTCLASSES}
 49 echo TESTJAVA=${TESTJAVA}
 50 echo COMPILEJAVA=${COMPILEJAVA}
 51 echo CPAPPEND=${CPAPPEND}
 52 echo ""
 53 
 54 # let java test exit if platform unsupported
 55 
 56 OS=`uname -s`
 57 case "$OS" in
 58   SunOS )
 59     FS="/"
 60     PS=":"
 61     CP="${FS}bin${FS}cp"
 62     CHMOD="${FS}bin${FS}chmod"
 63     ;;
 64   Linux )
 65     FS="/"
 66     PS=":"
 67     CP="${FS}bin${FS}cp"
 68     CHMOD="${FS}bin${FS}chmod"
 69     ;;
 70   Darwin )
 71     FS="/"
 72     PS=":"
 73     CP="${FS}bin${FS}cp"
 74     CHMOD="${FS}bin${FS}chmod"
 75     ;;
 76   AIX )
 77     FS="/"
 78     PS=":"
 79     CP="${FS}bin${FS}cp"
 80     CHMOD="${FS}bin${FS}chmod"
 81     ;;
 82   Windows* )
 83     FS="\\"
 84     PS=";"
 85     CP="cp"
 86     CHMOD="chmod"
 87     ;;
 88   CYGWIN* )
 89     FS="/"
 90     PS=";"
 91     CP="cp"
 92     CHMOD="chmod"
 93     #
 94     # javac does not like /cygdrive produced by `pwd`
 95     #
 96     TESTSRC=`cygpath -d ${TESTSRC}`
 97     ;;
 98   * )
 99     echo "Unrecognized system!"
100     exit 1;
101     ;;
102 esac
103 
104 # compile test
105 
106 ${COMPILEJAVA}${FS}bin${FS}javac ${TESTJAVACOPTS} ${TESTTOOLVMOPTS} \
107         -classpath ${TESTSRC}${FS}.. \
108         -d ${TESTCLASSES} \
109         ${TESTSRC}${FS}..${FS}..${FS}..${FS}..${FS}..${FS}lib${FS}jdk${FS}test${FS}lib${FS}artifacts${FS}*.java \
110         ${TESTSRC}${FS}ConfigQuotedString.java \
111         ${TESTSRC}${FS}..${FS}PKCS11Test.java
112 
113 # run test
114 
115 ${TESTJAVA}${FS}bin${FS}java ${TESTVMOPTS} \
116         -classpath ${TESTCLASSES}${PS}${CPAPPEND} \
117         -DCUSTOM_P11_CONFIG=${TESTSRC}${FS}ConfigQuotedString-nss.txt \
118         -Dtest.src=${TESTSRC} \
119         -Dtest.classes=${TESTCLASSES} \
120         ConfigQuotedString
121 
122 # save error status
123 status=$?
124 
125 # return
126 exit $status
