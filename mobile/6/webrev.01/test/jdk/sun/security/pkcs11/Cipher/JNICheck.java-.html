25  * @test
26  * @bug 8063700
27  * @library /test/lib
28  * @run main/othervm -Xcheck:jni JNICheck
29  */
30 
31 import javax.crypto.Cipher;
32 import javax.crypto.SealedObject;
33 import javax.crypto.SecretKey;
34 import javax.crypto.KeyGenerator;
35 import java.security.NoSuchProviderException;
36 
37 public class JNICheck {
38 
39     /* This test is similar to the JCK test that found 8063700. */
40     static class SealedObjectTest {
41         Cipher c;
42 
43         SealedObjectTest() throws Exception {
44             try {
45                 c = Cipher.getInstance("AES", "SunPKCS11-Solaris");
46             } catch (NoSuchProviderException nspe) {
47                 System.out.println("No SunPKCS11-Solaris provider.  Test skipped");
48                 return;
49             }
50 
51             String s = "Test string";
52             SealedObject so;
53             SecretKey key = KeyGenerator.getInstance("AES").generateKey();
54 
55             c.init(Cipher.ENCRYPT_MODE, key);
56             so = new SealedObject(s, c);
57 
58             so.getObject(key, "SunPKCS11-Solaris");
59         }
60     }
61 
62     public static void main(String args[]) throws Exception {
63         new SealedObjectTest();
64     }
65 }
