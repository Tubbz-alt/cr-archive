28 # set a few environment variables so that the shell-script can run stand-alone
29 # in the source directory
30 if [ "${TESTSRC}" = "" ] ; then
31    TESTSRC="."
32 fi
33 
34 if [ "${TESTCLASSES}" = "" ] ; then
35    TESTCLASSES="."
36 fi
37 
38 if [ "${TESTJAVA}" = "" ] ; then
39    echo "TESTJAVA not set.  Test cannot execute."
40    echo "FAILED!!!"
41    exit 1
42 fi
43 
44 if [ "${COMPILEJAVA}" = "" ]; then
45    COMPILEJAVA="${TESTJAVA}"
46 fi
47 
48 # set platform-dependent variables
49 OS=`uname -s`
50 case "$OS" in




51   Linux )
52     PATHSEP=":"
53     FILESEP="/"
54     ;;
55   Windows* )
56     PATHSEP=";"
57     FILESEP="\\"
58     ;;
59   * )
60     echo "Unrecognized system!"
61     exit 1;
62     ;;
63 esac
64 
65 # compile the test program
66 cd ${TESTSRC}${FILESEP}
67 rm GrantAllPermToExtWhenNoPolicy.class
68 ${COMPILEJAVA}${FILESEP}bin${FILESEP}javac ${TESTJAVACOPTS} ${TESTTOOLVMOPTS} \
69     -d ${TESTSRC}${FILESEP} ${TESTSRC}${FILESEP}SomeExtensionClass.java
70 ${COMPILEJAVA}${FILESEP}bin${FILESEP}javac ${TESTJAVACOPTS} ${TESTTOOLVMOPTS} \
71     -d ${TESTSRC}${FILESEP} ${TESTSRC}${FILESEP}GrantAllPermToExtWhenNoPolicy.java
72 
73 # create the extension JAR file
74 cd ${TESTCLASSES}
75 ${COMPILEJAVA}${FILESEP}bin${FILESEP}jar cvf SomeExt.jar SomeExtensionClass*.class
76 rm SomeExtensionClass.class
77 
78 # move the extension JAR file to the extension directory
79 mv SomeExt.jar ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}ext
80 
81 # remove the system policy file
82 mv \
83  ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}security${FILESEP}java.policy \
84  ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}security${FILESEP}tmp_pol
85 
86 # run the test program
87 ${TESTJAVA}${FILESEP}bin${FILESEP}java ${TESTVMOPTS} -Djava.security.manager \
88  GrantAllPermToExtWhenNoPolicy
89 
90 # save error status
91 status=$?
92 
93 # restore system policy and remove extension JAR file
94 mv ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}security${FILESEP}tmp_pol \
95  ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}security${FILESEP}java.policy
96 rm ${TESTJAVA}${FILESEP}jre${FILESEP}lib${FILESEP}ext${FILESEP}SomeExt.jar
97 
98 exit $status
--- EOF ---
