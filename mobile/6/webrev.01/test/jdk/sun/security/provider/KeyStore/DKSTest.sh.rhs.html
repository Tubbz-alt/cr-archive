31 # in the source directory

33    TESTSRC="."
34 fi
35 
36 if [ "${TESTCLASSES}" = "" ] ; then
37    TESTCLASSES="."
38 fi
39 
40 if [ "${TESTJAVA}" = "" ] ; then
41    echo "TESTJAVA not set.  Test cannot execute."

43    exit 1

45 
46 if [ "${COMPILEJAVA}" = "" ]; then
47     COMPILEJAVA="${TESTJAVA}"
48 fi
49 
50 # set platform-dependent variables
51 OS=`uname -s`
52 case "$OS" in
53   Linux | Darwin | AIX)
54     PS=":"
55     FS="/"
56     ;;
57   CYGWIN* )
58     PS=";"
59     FS="/"
60     ;;
61   Windows* )
62     PS=";"
63     FS="\\"
64     ;;
65   * )
66     echo "Unrecognized system!"
67     exit 1;
68     ;;
69 esac
70
