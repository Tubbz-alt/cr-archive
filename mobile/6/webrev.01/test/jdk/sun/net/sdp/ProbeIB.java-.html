24 import java.io.File;
25 import java.io.IOException;
26 import java.net.NetworkInterface;
27 import java.net.InetAddress;
28 import java.util.Scanner;
29 import java.util.Enumeration;
30 
31 /**
32  * Probes for InfiniBand devices plumbed with IP addresses.
33  */
34 
35 public class ProbeIB {
36     public static void main(String[] args) throws IOException {
37         Scanner s = new Scanner(new File(args[0]));
38         try {
39             while (s.hasNextLine()) {
40                 String link = s.nextLine();
41                 NetworkInterface ni = NetworkInterface.getByName(link);
42                 if (ni != null) {
43                     Enumeration<InetAddress> addrs = ni.getInetAddresses();
44                     while (addrs.hasMoreElements()) {
45                         InetAddress addr = addrs.nextElement();
46                         System.out.println(addr.getHostAddress());
47                     }
48                 }
49             }
50         } finally {
51             s.close();
52         }
53     }
54 }
