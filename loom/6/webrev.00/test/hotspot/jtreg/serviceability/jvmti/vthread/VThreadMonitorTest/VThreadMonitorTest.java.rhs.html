 25 /**
 26  * @test
 27  * @summary Test JVMTI Monitor functions for virtual threads
 28  * @compile VThreadMonitorTest.java
 29  * @run main/othervm/native -agentlib:VThreadMonitorTest VThreadMonitorTest
 30  */
 31 
 32 import java.io.PrintStream;
 33 
 34 public class VThreadMonitorTest {
 35 
 36     static {
 37         try {
 38             System.loadLibrary("VThreadMonitorTest");
 39         } catch (UnsatisfiedLinkError ule) {
 40             System.err.println("Could not load VThreadMonitorTest library");
 41             System.err.println("java.library.path: "
 42                                + System.getProperty("java.library.path"));
 43             throw ule;
 44         }
 45     }
 46     private static native boolean hasEventPosted();
 47     private static native void checkContendedMonitor(Thread thread, Object monitor);
 48     private static native int check();
 49 
 50     private static void log(String str) { System.out.println(str); }
 51     private static String thrName() { return Thread.currentThread().getName(); }
 52 
 53     private static final VThreadMonitorTest lock0 = new VThreadMonitorTest();
 54     private static final Object lock1 = new Object();
 55     private static final Object lock2 = new Object();
 56 
 57     static void m0() {
 58         synchronized (lock0) {
 59             log("Thread entered sync section with lock0: " + thrName());
 60         }
 61     }
 62     static void m1() {
 63         synchronized (lock1) {
 64             log("Thread entered sync section with lock1: " + thrName());
 65             m0();
 66         }
 67     }
 68     static void m2() {
 69         synchronized (lock2) {
 70             log("Thread entered sync section with lock2: " + thrName());
 71             m1();
 72         }
 73     }
 74 
 75     static final Runnable VT1 = () -> {
 76         m2();
 77     };
 78 
 79     public static void main(String[] args) throws Exception {
 80         Thread vt1 = Thread.newThread("VirtualThread-1", Thread.VIRTUAL, VT1);
 81 
 82         // Make sure VT1 is blocked on monitor lock0
 83         synchronized (lock0) {
 84             log("Main starting VT2 virtual thread.");
 85             vt1.start();
 86 
 87             // Wait for the MonitorContendedEnter event
 88             while (!hasEventPosted()) {
 89                 log("Main thread is waiting for event.");
 90                 Thread.sleep(100);
 91             }
 92             checkContendedMonitor(vt1, lock0);
 93         }
 94 
 95         vt1.join();
 96 
 97         if (check() != 0) {
 98             throw new RuntimeException("FAILED status returned from the agent");
 99         }
100     }
101 }
