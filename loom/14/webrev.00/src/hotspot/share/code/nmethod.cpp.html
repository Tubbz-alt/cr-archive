 199 struct pc_nmethod_stats_struct {
 200   int pc_desc_resets;   // number of resets (= number of caches)
 201   int pc_desc_queries;  // queries to nmethod::find_pc_desc
 202   int pc_desc_approx;   // number of those which have approximate true
 203   int pc_desc_repeats;  // number of _pc_descs[0] hits
 204   int pc_desc_hits;     // number of LRU cache hits
 205   int pc_desc_tests;    // total number of PcDesc examinations
 206   int pc_desc_searches; // total number of quasi-binary search steps
 207   int pc_desc_adds;     // number of LUR cache insertions
 208 
 209   void print_pc_stats() {
 210     tty->print_cr("PcDesc Statistics:  %d queries, %.2f comparisons per query",
 211                   pc_desc_queries,
 212                   (double)(pc_desc_tests + pc_desc_searches)
 213                   / pc_desc_queries);
 214     tty->print_cr("  caches=%d queries=%d/%d, hits=%d+%d, tests=%d+%d, adds=%d",
 215                   pc_desc_resets,
 216                   pc_desc_queries, pc_desc_approx,
 217                   pc_desc_repeats, pc_desc_hits,
 218                   pc_desc_tests, pc_desc_searches, pc_desc_adds);
 219   }
 220 };

 275 address ExceptionCache::match(Handle exception, address pc) {
 276   assert(pc != NULL,"Must be non null");
 277   assert(exception.not_null(),"Must be non null");
 278   if (exception->klass() == exception_type()) {
 279     return (test_address(pc));
 280   }
 281 
 282   return NULL;
 283 }

 286 bool ExceptionCache::match_exception_with_space(Handle exception) {
 287   assert(exception.not_null(),"Must be non null");
 288   if (exception->klass() == exception_type() && count() < cache_size) {
 289     return true;
 290   }
 291   return false;
 292 }

 295 address ExceptionCache::test_address(address addr) {
 296   int limit = count();
 297   for (int i = 0; i < limit; i++) {
 298     if (pc_at(i) == addr) {
 299       return handler_at(i);
 300     }
 301   }
 302   return NULL;
 303 }

 306 bool ExceptionCache::add_address_and_handler(address addr, address handler) {
 307   if (test_address(addr) == handler) return true;
 308 
 309   int index = count();
 310   if (index < cache_size) {
 311     set_pc_at(index, addr);
 312     set_handler_at(index, handler);
 313     increment_count();
 314     return true;
 315   }
 316   return false;
 317 }

 330 // Helper used by both find_pc_desc methods.
 331 static inline bool match_desc(PcDesc* pc, int pc_offset, bool approximate) {
 332   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_tests);
 333   if (!approximate)
 334     return pc->pc_offset() == pc_offset;
 335   else
 336     return (pc-1)->pc_offset() < pc_offset && pc_offset <= pc->pc_offset();
 337 }
 338 
 339 void PcDescCache::reset_to(PcDesc* initial_pc_desc) {
 340   if (initial_pc_desc == NULL) {
 341     _pc_descs[0] = NULL; // native method; no PcDescs at all
 342     return;
 343   }
 344   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_resets);
 345   // reset the cache by filling it with benign (non-null) values
 346   assert(initial_pc_desc->pc_offset() < 0, "must be sentinel");
 347   for (int i = 0; i < cache_size; i++)
 348     _pc_descs[i] = initial_pc_desc;
 349 }
 350 
 351 PcDesc* PcDescCache::find_pc_desc(int pc_offset, bool approximate) {
 352   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_queries);
 353   NOT_PRODUCT(if (approximate) ++pc_nmethod_stats.pc_desc_approx);
 354 
 355   // Note: one might think that caching the most recently
 356   // read value separately would be a win, but one would be
 357   // wrong.  When many threads are updating it, the cache
 358   // line it's in would bounce between caches, negating
 359   // any benefit.
 360 
 361   // In order to prevent race conditions do not load cache elements
 362   // repeatedly, but use a local copy:
 363   PcDesc* res;
 364 
 365   // Step one:  Check the most recently added value.
 366   res = _pc_descs[0];
 367   if (res == NULL) return NULL;  // native method; no PcDescs at all
 368   if (match_desc(res, pc_offset, approximate)) {
 369     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_repeats);
 370     return res;
 371   }
 372 
 373   // Step two:  Check the rest of the LRU cache.
 374   for (int i = 1; i < cache_size; ++i) {
 375     res = _pc_descs[i];
 376     if (res->pc_offset() < 0) break;  // optimization: skip empty cache
 377     if (match_desc(res, pc_offset, approximate)) {
 378       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_hits);
 379       return res;
 380     }
 381   }
 382 
 383   // Report failure.
 384   return NULL;
 385 }
 386 
 387 void PcDescCache::add_pc_desc(PcDesc* pc_desc) {
 388   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_adds);
 389   // Update the LRU cache by shifting pc_desc forward.
 390   for (int i = 0; i < cache_size; i++)  {
 391     PcDesc* next = _pc_descs[i];
 392     _pc_descs[i] = pc_desc;
 393     pc_desc = next;
 394   }
 395 }
 396 
 397 // adjust pcs_size so that it is a multiple of both oopSize and
 398 // sizeof(PcDesc) (assumes that if sizeof(PcDesc) is not a multiple
 399 // of oopSize, then 2*sizeof(PcDesc) is)
 400 static int adjust_pcs_size(int pcs_size) {
 401   int nsize = align_up(pcs_size,   oopSize);
 402   if ((nsize % sizeof(PcDesc)) != 0) {
 403     nsize = pcs_size + sizeof(PcDesc);
 404   }
 405   assert((nsize % oopSize) == 0, "correct alignment");
 406   return nsize;
 407 }

 453 nmethod* nmethod::new_native_nmethod(const methodHandle& method,
 454   int compile_id,
 455   CodeBuffer *code_buffer,
 456   int vep_offset,
 457   int frame_complete,
 458   int frame_size,
 459   ByteSize basic_lock_owner_sp_offset,
 460   ByteSize basic_lock_sp_offset,
 461   OopMapSet* oop_maps,
 462   int exception_handler) {
 463   code_buffer->finalize_oop_references(method);
 464   // create nmethod
 465   nmethod* nm = NULL;
 466   {
 467     MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 468     int native_nmethod_size = CodeBlob::allocation_size(code_buffer, sizeof(nmethod));
 469 
 470     CodeOffsets offsets;
 471     offsets.set_value(CodeOffsets::Verified_Entry, vep_offset);
 472     offsets.set_value(CodeOffsets::Frame_Complete, frame_complete);
 473     if (exception_handler != -1) {
 474       offsets.set_value(CodeOffsets::Exceptions, exception_handler);
 475     }
 476     nm = new (native_nmethod_size, CompLevel_none)
 477     nmethod(method(), compiler_none, native_nmethod_size,
 478             compile_id, &offsets,
 479             code_buffer, frame_size,
 480             basic_lock_owner_sp_offset,
 481             basic_lock_sp_offset,
 482             oop_maps);
 483     NOT_PRODUCT(if (nm != NULL)  native_nmethod_stats.note_native_nmethod(nm));
 484   }
 485 
 486   if (nm != NULL) {
 487     // verify nmethod
 488     debug_only(nm->verify();) // might block
 489 
 490     nm->log_new_nmethod();
 491   }
 492   return nm;
 493 }

 495 nmethod* nmethod::new_nmethod(const methodHandle& method,
 496   int compile_id,
 497   int entry_bci,
 498   CodeOffsets* offsets,
 499   int orig_pc_offset,
 500   DebugInformationRecorder* debug_info,
 501   Dependencies* dependencies,
 502   CodeBuffer* code_buffer, int frame_size,
 503   OopMapSet* oop_maps,
 504   ExceptionHandlerTable* handler_table,
 505   ImplicitExceptionTable* nul_chk_table,
 506   AbstractCompiler* compiler,
 507   int comp_level
 508 #if INCLUDE_JVMCI
 509   , char* speculations,
 510   int speculations_len,
 511   int nmethod_mirror_index,
 512   const char* nmethod_mirror_name,
 513   FailedSpeculation** failed_speculations
 514 #endif
 515 )
 516 {
 517   assert(debug_info->oop_recorder() == code_buffer->oop_recorder(), "shared OR");
 518   code_buffer->finalize_oop_references(method);
 519   // create nmethod
 520   nmethod* nm = NULL;
 521   { MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
 522 #if INCLUDE_JVMCI
 523     int jvmci_data_size = !compiler->is_jvmci() ? 0 : JVMCINMethodData::compute_size(nmethod_mirror_name);
 524 #endif
 525     int nmethod_size =
 526       CodeBlob::allocation_size(code_buffer, sizeof(nmethod))
 527       + adjust_pcs_size(debug_info->pcs_size())
 528       + align_up((int)dependencies->size_in_bytes(), oopSize)
 529       + align_up(handler_table->size_in_bytes()    , oopSize)
 530       + align_up(nul_chk_table->size_in_bytes()    , oopSize)
 531 #if INCLUDE_JVMCI
 532       + align_up(speculations_len                  , oopSize)
 533       + align_up(jvmci_data_size                   , oopSize)
 534 #endif
 535       + align_up(debug_info->data_size()           , oopSize);
 536 
 537     nm = new (nmethod_size, comp_level)
 538     nmethod(method(), compiler->type(), nmethod_size, compile_id, entry_bci, offsets,
 539             orig_pc_offset, debug_info, dependencies, code_buffer, frame_size,
 540             oop_maps,
 541             handler_table,
 542             nul_chk_table,
 543             compiler,
 544             comp_level
 545 #if INCLUDE_JVMCI
 546             , speculations,
 547             speculations_len,
 548             jvmci_data_size
 549 #endif
 550             );
 551 
 552     if (nm != NULL) {
 553 #if INCLUDE_JVMCI
 554       if (compiler->is_jvmci()) {
 555         // Initialize the JVMCINMethodData object inlined into nm
 556         nm->jvmci_nmethod_data()->initialize(nmethod_mirror_index, nmethod_mirror_name, failed_speculations);
 557       }
 558 #endif
 559       // To make dependency checking during class loading fast, record
 560       // the nmethod dependencies in the classes it is dependent on.
 561       // This allows the dependency checking code to simply walk the
 562       // class hierarchy above the loaded class, checking only nmethods
 563       // which are dependent on those classes.  The slow way is to
 564       // check every nmethod for dependencies which makes it linear in
 565       // the number of methods compiled.  For applications with a lot
 566       // classes the slow way is too slow.
 567       for (Dependencies::DepStream deps(nm); deps.next(); ) {
 568         if (deps.type() == Dependencies::call_site_target_value) {
 569           // CallSite dependencies are managed on per-CallSite instance basis.
 570           oop call_site = deps.argument_oop(0);
 571           MethodHandles::add_dependent_nmethod(call_site, nm);
 572         } else {
 573           Klass* klass = deps.context_type();
 574           if (klass == NULL) {
 575             continue;  // ignore things like evol_method
 576           }
 577           // record this nmethod as dependent on this klass
 578           InstanceKlass::cast(klass)->add_dependent_nmethod(nm);
 579         }
 580       }
 581       NOT_PRODUCT(if (nm != NULL)  note_java_nmethod(nm));
 582     }
 583   }
 584   // Do verification and logging outside CodeCache_lock.
 585   if (nm != NULL) {
 586     // Safepoints in nmethod::verify aren't allowed because nm hasn't been installed yet.
 587     DEBUG_ONLY(nm->verify();)
 588     nm->log_new_nmethod();
 589   }
 590   return nm;
 591 }

1052 // Have to have the same name because it's called by a template
1053 void nmethod::copy_values(GrowableArray<jobject>* array) {
1054   int length = array->length();
1055   assert((address)(oops_begin() + length) <= (address)oops_end(), "oops big enough");
1056   oop* dest = oops_begin();
1057   for (int index = 0 ; index < length; index++) {
1058     initialize_immediate_oop(&dest[index], array->at(index));
1059   }
1060 
1061   // Now we can fix up all the oops in the code.  We need to do this
1062   // in the code because the assembler uses jobjects as placeholders.
1063   // The code and relocations have already been initialized by the
1064   // CodeBlob constructor, so it is valid even at this early point to
1065   // iterate over relocations and patch the code.
1066   fix_oop_relocations(NULL, NULL, /*initialize_immediates=*/ true);

1078 void nmethod::fix_oop_relocations(address begin, address end, bool initialize_immediates) {
1079   // re-patch all oop-bearing instructions, just in case some oops moved
1080   RelocIterator iter(this, begin, end);
1081   while (iter.next()) {
1082     if (iter.type() == relocInfo::oop_type) {
1083       oop_Relocation* reloc = iter.oop_reloc();
1084       if (initialize_immediates && reloc->oop_is_immediate()) {
1085         oop* dest = reloc->oop_addr();
1086         initialize_immediate_oop(dest, cast_from_oop<jobject>(*dest));
1087       }
1088       // Refresh the oop-related bits of this instruction.
1089       reloc->fix_oop_relocation();
1090     } else if (iter.type() == relocInfo::metadata_type) {
1091       metadata_Relocation* reloc = iter.metadata_reloc();
1092       reloc->fix_metadata_relocation();
1093     }
1094   }
1095 }

1098 void nmethod::verify_clean_inline_caches() {
1099   assert(CompiledICLocker::is_safe(this), "mt unsafe call");
1100 
1101   ResourceMark rm;
1102   RelocIterator iter(this, oops_reloc_begin());
1103   while(iter.next()) {
1104     switch(iter.type()) {
1105       case relocInfo::virtual_call_type:
1106       case relocInfo::opt_virtual_call_type: {
1107         CompiledIC *ic = CompiledIC_at(&iter);
1108         // Ok, to lookup references to zombies here
1109         CodeBlob *cb = CodeCache::find_blob_unsafe(ic->ic_destination());
1110         assert(cb != NULL, "destination not in CodeBlob?");
1111         nmethod* nm = cb->as_nmethod_or_null();
1112         if( nm != NULL ) {
1113           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1114           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1115             assert(ic->is_clean(), "IC should be clean");
1116           }
1117         }
1118         break;
1119       }
1120       case relocInfo::static_call_type: {
1121         CompiledStaticCall *csc = compiledStaticCall_at(iter.reloc());
1122         CodeBlob *cb = CodeCache::find_blob_unsafe(csc->destination());
1123         assert(cb != NULL, "destination not in CodeBlob?");
1124         nmethod* nm = cb->as_nmethod_or_null();
1125         if( nm != NULL ) {
1126           // Verify that inline caches pointing to both zombie and not_entrant methods are clean
1127           if (!nm->is_in_use() || (nm->method()->code() != nm)) {
1128             assert(csc->is_clean(), "IC should be clean");
1129           }
1130         }
1131         break;
1132       }
1133       default:
1134         break;
1135     }
1136   }
1137 }

1152 bool nmethod::is_not_on_continuation_stack() {
1153   // Odd marking cycles are found during concurrent marking. Even numbers are found
1154   // in nmethods that are marked when GC is inactive (e.g. nmethod entry barriers during
1155   // normal execution). Therefore we align up by 2 so that nmethods encountered during
1156   // concurrent marking are treated as if they were encountered in the inactive phase
1157   // after that concurrent GC. Each GC increments the marking cycle twice - once when
1158   // it starts and once when it ends. So we can only be sure there are no new continuations
1159   // when they have not been encountered from before a GC to after a GC.
1160   bool not_on_new_vthread_stack = CodeCache::marking_cycle() >= align_up(_marking_cycle, 2) + 2;
1161 
1162   // As for old vthread stacks, they are kept alive by a WeakHandle.
1163   bool not_on_old_vthread_stack = false;
1164   if (_keepalive != NULL) {
1165     WeakHandle wh = WeakHandle::from_raw(_keepalive);
1166     not_on_old_vthread_stack = wh.resolve() == NULL;
1167   }
1168   return not_on_new_vthread_stack && not_on_old_vthread_stack;
1169 }

1171 // Tell if a non-entrant method can be converted to a zombie (i.e.,
1172 // there are no activations on the stack, not in use by the VM,
1173 // and not in use by the ServiceThread)
1174 bool nmethod::can_convert_to_zombie() {
1175   // Note that this is called when the sweeper has observed the nmethod to be
1176   // not_entrant. However, with concurrent code cache unloading, the state
1177   // might have moved on to unloaded if it is_unloading(), due to racing
1178   // concurrent GC threads.
1179   assert(is_not_entrant() || is_unloading() ||
1180          !Thread::current()->is_Code_cache_sweeper_thread(),
1181          "must be a non-entrant method if called from sweeper");
1182 
1183   // Since the nmethod sweeper only does partial sweep the sweeper's traversal
1184   // count can be greater than the stack traversal count before it hits the
1185   // nmethod for the second time.
1186   // If an is_unloading() nmethod is still not_entrant, then it is not safe to
1187   // convert it to zombie due to GC unloading interactions. However, if it
1188   // has become unloaded, then it is okay to convert such nmethods to zombie.
1189   return stack_traversal_mark()+1 < NMethodSweeper::traversal_count() && is_not_on_continuation_stack() &&
1190           !is_locked_by_vm() && (!is_unloading() || is_unloaded());
1191 }

1204 bool nmethod::try_transition(int new_state_int) {
1205   signed char new_state = new_state_int;
1206 #ifdef DEBUG
1207   if (new_state != unloaded) {
1208     assert_lock_strong(CompiledMethod_lock);
1209   }
1210 #endif
1211   for (;;) {
1212     signed char old_state = Atomic::load(&_state);
1213     if (old_state >= new_state) {
1214       // Ensure monotonicity of transitions.
1215       return false;
1216     }
1217     if (Atomic::cmpxchg(&_state, old_state, new_state) == old_state) {
1218       return true;
1219     }
1220   }

1223 void nmethod::make_unloaded() {
1224   post_compiled_method_unload();
1225 
1226   // This nmethod is being unloaded, make sure that dependencies
1227   // recorded in instanceKlasses get flushed.
1228   // Since this work is being done during a GC, defer deleting dependencies from the
1229   // InstanceKlass.
1230   assert(Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread(),
1231          "should only be called during gc");
1232   flush_dependencies(/*delete_immediately*/false);
1233 
1234   // Break cycle between nmethod & method
1235   LogTarget(Trace, class, unload, nmethod) lt;
1236   if (lt.is_enabled()) {
1237     LogStream ls(lt);
1238     ls.print("making nmethod " INTPTR_FORMAT
1239              " unloadable, Method*(" INTPTR_FORMAT
1240              ") ",
1241              p2i(this), p2i(_method));
1242      ls.cr();
1243   }
1244   // Unlink the osr method, so we do not look this up again
1245   if (is_osr_method()) {
1246     // Invalidate the osr nmethod only once. Note that with concurrent
1247     // code cache unloading, OSR nmethods are invalidated before they
1248     // are made unloaded. Therefore, this becomes a no-op then.
1249     if (is_in_use()) {
1250       invalidate_osr_method();
1251     }
1252 #ifdef ASSERT
1253     if (method() != NULL) {
1254       // Make sure osr nmethod is invalidated, i.e. not on the list
1255       bool found = method()->method_holder()->remove_osr_nmethod(this);
1256       assert(!found, "osr nmethod should have been invalidated");
1257     }
1258 #endif
1259   }
1260 
1261   // If _method is already NULL the Method* is about to be unloaded,
1262   // so we don't have to break the cycle. Note that it is possible to
1263   // have the Method* live here, in case we unload the nmethod because
1264   // it is pointing to some oop (other than the Method*) being unloaded.
1265   if (_method != NULL) {
1266     _method->unlink_code(this);
1267   }
1268 
1269   // Make the class unloaded - i.e., change state and notify sweeper
1270   assert(SafepointSynchronize::is_at_safepoint() || Thread::current()->is_ConcurrentGC_thread(),
1271          "must be at safepoint");
1272 
1273   {
1274     // Clear ICStubs and release any CompiledICHolders.
1275     CompiledICLocker ml(this);
1276     clear_ic_callsites();
1277   }
1278 
1279   // Unregister must be done before the state change
1280   {
1281     MutexLocker ml(SafepointSynchronize::is_at_safepoint() ? NULL : CodeCache_lock,
1282                      Mutex::_no_safepoint_check_flag);
1283     Universe::heap()->unregister_nmethod(this);
1284   }
1285 
1286   // Clear the method of this dead nmethod
1287   set_method(NULL);
1288 
1289   // Log the unloading.
1290   log_state_change();
1291 
1292   // The Method* is gone at this point
1293   assert(_method == NULL, "Tautology");
1294 
1295   set_osr_link(NULL);
1296   NMethodSweeper::report_state_change(this);
1297 
1298   bool transition_success = try_transition(unloaded);
1299 
1300   // It is an important invariant that there exists no race between
1301   // the sweeper and GC thread competing for making the same nmethod
1302   // zombie and unloaded respectively. This is ensured by
1303   // can_convert_to_zombie() returning false for any is_unloading()
1304   // nmethod, informing the sweeper not to step on any GC toes.
1305   assert(transition_success, "Invalid nmethod transition to unloaded");
1306 
1307 #if INCLUDE_JVMCI
1308   // Clear the link between this nmethod and a HotSpotNmethod mirror
1309   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1310   if (nmethod_data != NULL) {
1311     nmethod_data->invalidate_nmethod_mirror(this);
1312     nmethod_data->clear_nmethod_mirror(this);
1313   }
1314 #endif
1315 }

1356 /**
1357  * Common functionality for both make_not_entrant and make_zombie
1358  */
1359 bool nmethod::make_not_entrant_or_zombie(int state) {
1360   assert(state == zombie || state == not_entrant, "must be zombie or not_entrant");
1361 
1362   if (Atomic::load(&_state) >= state) {
1363     // Avoid taking the lock if already in required state.
1364     // This is safe from races because the state is an end-state,
1365     // which the nmethod cannot back out of once entered.
1366     // No need for fencing either.
1367     return false;
1368   }
1369 
1370   // Make sure the nmethod is not flushed.
1371   nmethodLocker nml(this);
1372   // This can be called while the system is already at a safepoint which is ok
1373   NoSafepointVerifier nsv;
1374 
1375   // during patching, depending on the nmethod state we must notify the GC that
1376   // code has been unloaded, unregistering it. We cannot do this right while
1377   // holding the CompiledMethod_lock because we need to use the CodeCache_lock. This
1378   // would be prone to deadlocks.
1379   // This flag is used to remember whether we need to later lock and unregister.
1380   bool nmethod_needs_unregister = false;
1381 
1382   {
1383     // Enter critical section.  Does not block for safepoint.
1384     MutexLocker ml(CompiledMethod_lock->owned_by_self() ? NULL : CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1385 
1386     // This logic is equivalent to the logic below for patching the
1387     // verified entry point of regular methods. We check that the
1388     // nmethod is in use to ensure that it is invalidated only once.
1389     if (is_osr_method() && is_in_use()) {
1390       // this effectively makes the osr nmethod not entrant
1391       invalidate_osr_method();
1392     }
1393 
1394     if (Atomic::load(&_state) >= state) {
1395       // another thread already performed this transition so nothing
1396       // to do, but return false to indicate this.
1397       return false;
1398     }
1399 
1400     // The caller can be calling the method statically or through an inline
1401     // cache call.
1402     if (!is_osr_method() && !is_not_entrant()) {
1403       NativeJump::patch_verified_entry(entry_point(), verified_entry_point(),
1404                   SharedRuntime::get_handle_wrong_method_stub());
1405     }
1406 
1407     if (is_in_use() && update_recompile_counts()) {
1408       // It's a true state change, so mark the method as decompiled.
1409       // Do it only for transition from alive.
1410       inc_decompile_count();
1411     }
1412 
1413     // If the state is becoming a zombie, signal to unregister the nmethod with
1414     // the heap.
1415     // This nmethod may have already been unloaded during a full GC.
1416     if ((state == zombie) && !is_unloaded()) {
1417       nmethod_needs_unregister = true;
1418     }
1419 
1420     // Must happen before state change. Otherwise we have a race condition in
1421     // nmethod::can_convert_to_zombie(). I.e., a method can immediately
1422     // transition its state from 'not_entrant' to 'zombie' without having to wait
1423     // for stack scanning.
1424     if (state == not_entrant) {
1425       mark_as_seen_on_stack();
1426       OrderAccess::storestore(); // _stack_traversal_mark and _state
1427     }
1428 
1429     // Change state
1430     if (!try_transition(state)) {
1431       // If the transition fails, it is due to another thread making the nmethod more
1432       // dead. In particular, one thread might be making the nmethod unloaded concurrently.
1433       // If so, having patched in the jump in the verified entry unnecessarily is fine.
1434       // The nmethod is no longer possible to call by Java threads.
1435       // Incrementing the decompile count is also fine as the caller of make_not_entrant()
1436       // had a valid reason to deoptimize the nmethod.
1437       // Marking the nmethod as seen on stack also has no effect, as the nmethod is now
1438       // !is_alive(), and the seen on stack value is only used to convert not_entrant
1439       // nmethods to zombie in can_convert_to_zombie().
1440       return false;
1441     }
1442 
1443     // Log the transition once
1444     log_state_change();
1445 
1446     // Remove nmethod from method.
1447     unlink_from_method();
1448 
1449   } // leave critical region under CompiledMethod_lock
1450 
1451 #if INCLUDE_JVMCI
1452   // Invalidate can't occur while holding the Patching lock
1453   JVMCINMethodData* nmethod_data = jvmci_nmethod_data();
1454   if (nmethod_data != NULL) {
1455     nmethod_data->invalidate_nmethod_mirror(this);
1456   }
1457 #endif
1458 
1459 #ifdef ASSERT
1460   if (is_osr_method() && method() != NULL) {
1461     // Make sure osr nmethod is invalidated, i.e. not on the list
1462     bool found = method()->method_holder()->remove_osr_nmethod(this);
1463     assert(!found, "osr nmethod should have been invalidated");
1464   }
1465 #endif
1466 
1467   // When the nmethod becomes zombie it is no longer alive so the
1468   // dependencies must be flushed.  nmethods in the not_entrant
1469   // state will be flushed later when the transition to zombie
1470   // happens or they get unloaded.
1471   if (state == zombie) {
1472     {
1473       // Flushing dependencies must be done before any possible
1474       // safepoint can sneak in, otherwise the oops used by the
1475       // dependency logic could have become stale.
1476       MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1477       if (nmethod_needs_unregister) {
1478         Universe::heap()->unregister_nmethod(this);
1479       }
1480       flush_dependencies(/*delete_immediately*/true);
1481     }
1482 
1483 #if INCLUDE_JVMCI
1484     // Now that the nmethod has been unregistered, it's
1485     // safe to clear the HotSpotNmethod mirror oop.
1486     if (nmethod_data != NULL) {
1487       nmethod_data->clear_nmethod_mirror(this);
1488     }
1489 #endif
1490 
1491     // Clear ICStubs to prevent back patching stubs of zombie or flushed
1492     // nmethods during the next safepoint (see ICStub::finalize), as well
1493     // as to free up CompiledICHolder resources.
1494     {
1495       CompiledICLocker ml(this);
1496       clear_ic_callsites();
1497     }
1498 
1499     // zombie only - if a JVMTI agent has enabled the CompiledMethodUnload
1500     // event and it hasn't already been reported for this nmethod then
1501     // report it now. The event may have been reported earlier if the GC
1502     // marked it for unloading). JvmtiDeferredEventQueue support means
1503     // we no longer go to a safepoint here.
1504     post_compiled_method_unload();
1505 
1506 #ifdef ASSERT
1507     // It's no longer safe to access the oops section since zombie
1508     // nmethods aren't scanned for GC.
1509     _oops_are_stale = true;
1510 #endif
1511      // the Method may be reclaimed by class unloading now that the
1512      // nmethod is in zombie state
1513     set_method(NULL);
1514   } else {
1515     assert(state == not_entrant, "other cases may need to be handled differently");
1516   }
1517 
1518   if (TraceCreateZombies && state == zombie) {
1519     ResourceMark m;
1520     tty->print_cr("nmethod <" INTPTR_FORMAT "> %s code made %s", p2i(this), this->method() ? this->method()->name_and_sig_as_C_string() : "null", (state == not_entrant) ? "not entrant" : "zombie");
1521   }
1522 
1523   NMethodSweeper::report_state_change(this);
1524   return true;
1525 }

1527 void nmethod::flush() {
1528   MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
1529   // Note that there are no valid oops in the nmethod anymore.
1530   assert(!is_osr_method() || is_unloaded() || is_zombie(),
1531          "osr nmethod must be unloaded or zombie before flushing");
1532   assert(is_zombie() || is_osr_method(), "must be a zombie method");
1533   assert (!is_locked_by_vm(), "locked methods shouldn't be flushed");
1534   assert_locked_or_safepoint(CodeCache_lock);
1535 
1536   // completely deallocate this method
1537   Events::log(JavaThread::current(), "flushing nmethod " INTPTR_FORMAT, p2i(this));
1538   if (PrintMethodFlushing) {
1539     tty->print_cr("*flushing %s nmethod %3d/" INTPTR_FORMAT ". Live blobs:" UINT32_FORMAT
1540                   "/Free CodeCache:" SIZE_FORMAT "Kb",
1541                   is_osr_method() ? "osr" : "",_compile_id, p2i(this), CodeCache::blob_count(),
1542                   CodeCache::unallocated_capacity(CodeCache::get_code_blob_type(this))/1024);
1543   }
1544 
1545   // We need to deallocate any ExceptionCache data.
1546   // Note that we do not need to grab the nmethod lock for this, it
1547   // better be thread safe if we're disposing of it!
1548   ExceptionCache* ec = exception_cache();
1549   set_exception_cache(NULL);
1550   while(ec != NULL) {
1551     ExceptionCache* next = ec->next();
1552     delete ec;
1553     ec = next;
1554   }
1555 
1556   Universe::heap()->flush_nmethod(this);
1557   CodeCache::unregister_old_nmethod(this);
1558 
1559   CodeBlob::flush();
1560   CodeCache::free(this);
1561 }

1577 //
1578 // Notify all classes this nmethod is dependent on that it is no
1579 // longer dependent. This should only be called in two situations.
1580 // First, when a nmethod transitions to a zombie all dependents need
1581 // to be clear.  Since zombification happens at a safepoint there's no
1582 // synchronization issues.  The second place is a little more tricky.
1583 // During phase 1 of mark sweep class unloading may happen and as a
1584 // result some nmethods may get unloaded.  In this case the flushing
1585 // of dependencies must happen during phase 1 since after GC any
1586 // dependencies in the unloaded nmethod won't be updated, so
1587 // traversing the dependency information in unsafe.  In that case this
1588 // function is called with a boolean argument and this function only
1589 // notifies instanceKlasses that are reachable
1590 
1591 void nmethod::flush_dependencies(bool delete_immediately) {
1592   DEBUG_ONLY(bool called_by_gc = Universe::heap()->is_gc_active() || Thread::current()->is_ConcurrentGC_thread();)
1593   assert(called_by_gc != delete_immediately,
1594   "delete_immediately is false if and only if we are called during GC");
1595   if (!has_flushed_dependencies()) {
1596     set_has_flushed_dependencies();
1597     for (Dependencies::DepStream deps(this); deps.next(); ) {
1598       if (deps.type() == Dependencies::call_site_target_value) {
1599         // CallSite dependencies are managed on per-CallSite instance basis.
1600         oop call_site = deps.argument_oop(0);
1601         if (delete_immediately) {
1602           assert_locked_or_safepoint(CodeCache_lock);
1603           MethodHandles::remove_dependent_nmethod(call_site, this);
1604         } else {
1605           MethodHandles::clean_dependency_context(call_site);
1606         }
1607       } else {
1608         Klass* klass = deps.context_type();
1609         if (klass == NULL) {
1610           continue;  // ignore things like evol_method
1611         }
1612         // During GC delete_immediately is false, and liveness
1613         // of dependee determines class that needs to be updated.
1614         if (delete_immediately) {
1615           assert_locked_or_safepoint(CodeCache_lock);
1616           InstanceKlass::cast(klass)->remove_dependent_nmethod(this);
1617         } else if (klass->is_loader_alive()) {
1618           // The GC may clean dependency contexts concurrently and in parallel.
1619           InstanceKlass::cast(klass)->clean_dependency_context();
1620         }
1621       }
1622     }
1623   }
1624 }

1626 // ------------------------------------------------------------------
1627 // post_compiled_method_load_event
1628 // new method for install_code() path
1629 // Transfer information from compilation to jvmti
1630 void nmethod::post_compiled_method_load_event(JvmtiThreadState* state) {
1631 
1632   // Don't post this nmethod load event if it is already dying
1633   // because the sweeper might already be deleting this nmethod.
1634   if (is_not_entrant() && can_convert_to_zombie()) {
1635     return;
1636   }
1637 
1638   // This is a bad time for a safepoint.  We don't want
1639   // this nmethod to get unloaded while we're queueing the event.
1640   NoSafepointVerifier nsv;
1641 
1642   Method* m = method();
1643   HOTSPOT_COMPILED_METHOD_LOAD(
1644       (char *) m->klass_name()->bytes(),
1645       m->klass_name()->utf8_length(),
1646       (char *) m->name()->bytes(),
1647       m->name()->utf8_length(),
1648       (char *) m->signature()->bytes(),
1649       m->signature()->utf8_length(),
1650       insts_begin(), insts_size());
1651 
1652 
1653   if (JvmtiExport::should_post_compiled_method_load()) {
1654     // Only post unload events if load events are found.
1655     set_load_reported();
1656     // If a JavaThread hasn't been passed in, let the Service thread
1657     // (which is a real Java thread) post the event
1658     JvmtiDeferredEvent event = JvmtiDeferredEvent::compiled_method_load_event(this);
1659     if (state == NULL) {
1660       // Execute any barrier code for this nmethod as if it's called, since
1661       // keeping it alive looks like stack walking.
1662       run_nmethod_entry_barrier();
1663       ServiceThread::enqueue_deferred_event(&event);
1664     } else {
1665       // This enters the nmethod barrier outside in the caller.
1666       state->enqueue_event(&event);
1667     }
1668   }
1669 }

1671 void nmethod::post_compiled_method_unload() {
1672   if (unload_reported()) {
1673     // During unloading we transition to unloaded and then to zombie
1674     // and the unloading is reported during the first transition.
1675     return;
1676   }

1700   // attempt to report the event in the unlikely scenario where the
1701   // event is enabled at the time the nmethod is made a zombie.
1702   set_unload_reported();

1752 // The _is_unloading_state encodes a tuple comprising the unloading cycle
1753 // and the result of IsUnloadingBehaviour::is_unloading() fpr that cycle.
1754 // This is the bit layout of the _is_unloading_state byte: 00000CCU
1755 // CC refers to the cycle, which has 2 bits, and U refers to the result of
1756 // IsUnloadingBehaviour::is_unloading() for that unloading cycle.
1757 
1758 class IsUnloadingState: public AllStatic {
1759   static const uint8_t _is_unloading_mask = 1;
1760   static const uint8_t _is_unloading_shift = 0;
1761   static const uint8_t _unloading_cycle_mask = 6;
1762   static const uint8_t _unloading_cycle_shift = 1;
1763 
1764   static uint8_t set_is_unloading(uint8_t state, bool value) {
1765     state &= ~_is_unloading_mask;
1766     if (value) {
1767       state |= 1 << _is_unloading_shift;
1768     }
1769     assert(is_unloading(state) == value, "unexpected unloading cycle overflow");
1770     return state;
1771   }
1772 
1773   static uint8_t set_unloading_cycle(uint8_t state, uint8_t value) {
1774     state &= ~_unloading_cycle_mask;
1775     state |= value << _unloading_cycle_shift;
1776     assert(unloading_cycle(state) == value, "unexpected unloading cycle overflow");
1777     return state;
1778   }
1779 
1780 public:
1781   static bool is_unloading(uint8_t state) { return (state & _is_unloading_mask) >> _is_unloading_shift == 1; }
1782   static uint8_t unloading_cycle(uint8_t state) { return (state & _unloading_cycle_mask) >> _unloading_cycle_shift; }
1783 
1784   static uint8_t create(bool is_unloading, uint8_t unloading_cycle) {
1785     uint8_t state = 0;
1786     state = set_is_unloading(state, is_unloading);
1787     state = set_unloading_cycle(state, unloading_cycle);
1788     return state;
1789   }
1790 };
1791 
1792 bool nmethod::is_unloading() {
1793   uint8_t state = RawAccess<MO_RELAXED>::load(&_is_unloading_state);
1794   bool state_is_unloading = IsUnloadingState::is_unloading(state);
1795   uint8_t state_unloading_cycle = IsUnloadingState::unloading_cycle(state);
1796   if (state_is_unloading) {
1797     return true;
1798   }
1799   uint8_t current_cycle = CodeCache::unloading_cycle();
1800   if (state_unloading_cycle == current_cycle) {
1801     return false;
1802   }
1803 
1804   // The IsUnloadingBehaviour is responsible for checking if there are any dead
1805   // oops in the CompiledMethod, by calling oops_do on it.
1806   state_unloading_cycle = current_cycle;
1807 
1808   if (is_zombie()) {
1809     // Zombies without calculated unloading epoch are never unloading due to GC.
1810 
1811     // There are no races where a previously observed is_unloading() nmethod
1812     // suddenly becomes not is_unloading() due to here being observed as zombie.
1813 
1814     // With STW unloading, all is_alive() && is_unloading() nmethods are unlinked
1815     // and unloaded in the safepoint. That makes races where an nmethod is first
1816     // observed as is_alive() && is_unloading() and subsequently observed as
1817     // is_zombie() impossible.
1818 
1819     // With concurrent unloading, all references to is_unloading() nmethods are
1820     // first unlinked (e.g. IC caches and dependency contexts). Then a global
1821     // handshake operation is performed with all JavaThreads before finally
1822     // unloading the nmethods. The sweeper never converts is_alive() && is_unloading()
1823     // nmethods to zombies; it waits for them to become is_unloaded(). So before
1824     // the global handshake, it is impossible for is_unloading() nmethods to
1825     // racingly become is_zombie(). And is_unloading() is calculated for all is_alive()
1826     // nmethods before taking that global handshake, meaning that it will never
1827     // be recalculated after the handshake.
1828 
1829     // After that global handshake, is_unloading() nmethods are only observable
1830     // to the iterators, and they will never trigger recomputation of the cached
1831     // is_unloading_state, and hence may not suffer from such races.
1832 
1833     state_is_unloading = false;
1834   } else {
1835     state_is_unloading = IsUnloadingBehaviour::current()->is_unloading(this);
1836   }
1837 
1838   state = IsUnloadingState::create(state_is_unloading, state_unloading_cycle);
1839 
1840   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1841 
1842   return state_is_unloading;
1843 }
1844 
1845 void nmethod::clear_unloading_state() {
1846   uint8_t state = IsUnloadingState::create(false, CodeCache::unloading_cycle());
1847   RawAccess<MO_RELAXED>::store(&_is_unloading_state, state);
1848 }

2014 void nmethod::oops_do_process_weak(OopsDoProcessor* p) {
2015   if (!oops_do_try_claim_weak_request()) {
2016     // Failed to claim for weak processing.
2017     oops_do_log_change("oops_do, mark weak request fail");
2018     return;
2019   }
2020 
2021   p->do_regular_processing(this);
2022 
2023   nmethod* old_head = oops_do_try_add_to_list_as_weak_done();
2024   if (old_head == NULL) {
2025     return;
2026   }
2027   oops_do_log_change("oops_do, mark weak done fail");
2028   // Adding to global list failed, another thread added a strong request.
2029   assert(extract_state(_oops_do_mark_link) == claim_strong_request_tag,
2030          "must be but is %u", extract_state(_oops_do_mark_link));
2031 
2032   oops_do_log_change("oops_do, mark weak request -> mark strong done");
2033 
2034   oops_do_set_strong_done(old_head);
2035   // Do missing strong processing.
2036   p->do_remaining_strong_processing(this);
2037 }

2039 void nmethod::oops_do_process_strong(OopsDoProcessor* p) {
2040   oops_do_mark_link* next_raw = oops_do_try_claim_strong_done();
2041   if (next_raw == NULL) {
2042     p->do_regular_processing(this);
2043     oops_do_add_to_list_as_strong_done();
2044     return;
2045   }
2046   // Claim failed. Figure out why and handle it.
2047   if (oops_do_has_weak_request(next_raw)) {
2048     oops_do_mark_link* old = next_raw;
2049     // Claim failed because being weak processed (state == "weak request").
2050     // Try to request deferred strong processing.
2051     next_raw = oops_do_try_add_strong_request(old);
2052     if (next_raw == old) {
2053       // Successfully requested deferred strong processing.
2054       return;
2055     }
2056     // Failed because of a concurrent transition. No longer in "weak request" state.
2057   }
2058   if (oops_do_has_any_strong_state(next_raw)) {
2059     // Already claimed for strong processing or requested for such.
2060     return;
2061   }
2062   if (oops_do_try_claim_weak_done_as_strong_done(next_raw)) {
2063     // Successfully claimed "weak done" as "strong done". Do the missing marking.
2064     p->do_remaining_strong_processing(this);
2065     return;
2066   }
2067   // Claim failed, some other thread got it.
2068 }

2177 PcDesc* PcDescContainer::find_pc_desc_internal(address pc, bool approximate, const PcDescSearch& search) {
2178   address base_address = search.code_begin();
2179   if ((pc < base_address) ||
2180       (pc - base_address) >= (ptrdiff_t) PcDesc::upper_offset_limit) {
2181     return NULL;  // PC is wildly out of range
2182   }

2184

2188   if (res != NULL) {
2189     assert(res == linear_search(search, pc_offset, approximate), "cache ok");
2190     return res;
2191   }
2192

2194   // Find the last pc_offset less than the given offset.
2195   // The successor must be the required match, if there is a match at all.
2196   // (Use a fixed radix to avoid expensive affine pointer arithmetic.)

2203   /* invariant on lower..upper during the following search: */ \
2204   assert(lower->pc_offset() <  pc_offset, "sanity"); \
2205   assert(upper->pc_offset() >= pc_offset, "sanity")
2206   assert_LU_OK;
2207 
2208   // Use the last successful return as a split point.
2209   PcDesc* mid = _pc_desc_cache.last_pc_desc();
2210   NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2211   if (mid->pc_offset() < pc_offset) {
2212     lower = mid;
2213   } else {
2214     upper = mid;
2215   }
2216 
2217   // Take giant steps at first (4096, then 256, then 16, then 1)
2218   const int LOG2_RADIX = 4 /*smaller steps in debug mode:*/ debug_only(-1);
2219   const int RADIX = (1 << LOG2_RADIX);
2220   for (int step = (1 << (LOG2_RADIX*3)); step > 1; step >>= LOG2_RADIX) {
2221     while ((mid = lower + step) < upper) {
2222       assert_LU_OK;
2223       NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2224       if (mid->pc_offset() < pc_offset) {
2225         lower = mid;
2226       } else {
2227         upper = mid;
2228         break;
2229       }
2230     }
2231     assert_LU_OK;
2232   }
2233 
2234   // Sneak up on the value with a linear search of length ~16.
2235   while (true) {
2236     assert_LU_OK;
2237     mid = lower + 1;
2238     NOT_PRODUCT(++pc_nmethod_stats.pc_desc_searches);
2239     if (mid->pc_offset() < pc_offset) {
2240       lower = mid;
2241     } else {
2242       upper = mid;
2243       break;
2244     }
2245   }
2246 #undef assert_LU_OK
2247 
2248   if (match_desc(upper, pc_offset, approximate)) {
2249     assert(upper == linear_search(search, pc_offset, approximate), "search ok");
2250     _pc_desc_cache.add_pc_desc(upper);
2251     return upper;
2252   } else {
2253     assert(NULL == linear_search(search, pc_offset, approximate), "search ok");
2254     return NULL;
2255   }
2256 }
