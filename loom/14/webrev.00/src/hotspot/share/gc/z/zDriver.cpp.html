 54 class VM_ZOperation : public VM_Operation {
 55 private:
 56   const uint _gc_id;
 57   bool       _gc_locked;
 58   bool       _success;
 59 
 60 public:
 61   VM_ZOperation() :
 62       _gc_id(GCId::current()),
 63       _gc_locked(false),
 64       _success(false) {}
 65 
 66   virtual bool needs_inactive_gc_locker() const {
 67     // An inactive GC locker is needed in operations where we change the bad
 68     // mask or move objects. Changing the bad mask will invalidate all oops,
 69     // which makes it conceptually the same thing as moving all objects.
 70     return false;
 71   }
 72 
 73   virtual bool do_operation() = 0;
 74 
 75   virtual bool doit_prologue() {
 76     Heap_lock->lock();
 77     return true;
 78   }
 79 
 80   virtual void doit() {
 81     // Abort if GC locker state is incompatible
 82     if (needs_inactive_gc_locker() && GCLocker::check_active_before_gc()) {
 83       _gc_locked = true;
 84       return;
 85     }
 86 
 87     // Setup GC id and active marker
 88     GCIdMark gc_id_mark(_gc_id);
 89     IsGCActiveMark gc_active_mark;
 90 
 91     // Verify before operation
 92     ZVerify::before_zoperation();
 93 
 94     // Execute operation
 95     _success = do_operation();
 96 
 97     // Update statistics
 98     ZStatSample(ZSamplerJavaThreads, Threads::number_of_threads());
 99   }
100 
101   virtual void doit_epilogue() {
102     Heap_lock->unlock();
103   }
104 
105   bool gc_locked() const {
106     return _gc_locked;
107   }
108 
109   bool success() const {
110     return _success;
111   }
112 };
113 
114 static bool should_clear_soft_references() {
115   // Clear if one or more allocations have stalled
116   const bool stalled = ZHeap::heap()->is_alloc_stalled();
117   if (stalled) {
118     // Clear
119     return true;
120   }
121 
122   // Clear if implied by the GC cause
123   const GCCause::Cause cause = ZCollectedHeap::heap()->gc_cause();
124   if (cause == GCCause::_wb_full_gc ||
125       cause == GCCause::_metadata_GC_clear_soft_refs) {
126     // Clear
127     return true;
128   }
129 
130   // Don't clear
131   return false;
132 }
133 
134 static bool should_boost_worker_threads() {
135   // Boost worker threads if one or more allocations have stalled
136   const bool stalled = ZHeap::heap()->is_alloc_stalled();
137   if (stalled) {
138     // Boost
139     return true;
140   }
141 
142   // Boost worker threads if implied by the GC cause
143   const GCCause::Cause cause = ZCollectedHeap::heap()->gc_cause();
144   if (cause == GCCause::_wb_full_gc ||
145       cause == GCCause::_java_lang_system_gc ||
146       cause == GCCause::_metadata_GC_clear_soft_refs) {
147     // Boost
148     return true;
149   }
150 
151   // Don't boost
152   return false;
153 }

155 class VM_ZMarkStart : public VM_ZOperation {
156 public:
157   virtual VMOp_Type type() const {
158     return VMOp_ZMarkStart;
159   }
160 
161   virtual bool needs_inactive_gc_locker() const {
162     return true;
163   }
164 
165   virtual bool do_operation() {
166     ZStatTimer timer(ZPhasePauseMarkStart);
167     ZServiceabilityPauseTracer tracer;
168 
169     // Set up soft reference policy
170     const bool clear = should_clear_soft_references();
171     ZHeap::heap()->set_soft_reference_policy(clear);
172 
173     // Set up boost mode
174     const bool boost = should_boost_worker_threads();
175     ZHeap::heap()->set_boost_worker_threads(boost);
176 
177     ZCollectedHeap::heap()->increment_total_collections(true /* full */);
178 
179     ZHeap::heap()->mark_start();
180     return true;
181   }
182 };

233 void ZDriver::collect(GCCause::Cause cause) {
234   switch (cause) {
235   case GCCause::_wb_young_gc:
236   case GCCause::_wb_conc_mark:
237   case GCCause::_wb_full_gc:
238   case GCCause::_dcmd_gc_run:
239   case GCCause::_java_lang_system_gc:
240   case GCCause::_full_gc_alot:
241   case GCCause::_scavenge_alot:
242   case GCCause::_jvmti_force_gc:
243   case GCCause::_metadata_GC_clear_soft_refs:
244   case GCCause::_codecache_GC_threshold:
245     // Start synchronous GC
246     _gc_cycle_port.send_sync(cause);
247     break;
248 
249   case GCCause::_z_timer:
250   case GCCause::_z_warmup:
251   case GCCause::_z_allocation_rate:
252   case GCCause::_z_allocation_stall:
253   case GCCause::_z_proactive:
254   case GCCause::_z_high_usage:
255   case GCCause::_metadata_GC_threshold:
256     // Start asynchronous GC
257     _gc_cycle_port.send_async(cause);
258     break;
259 
260   case GCCause::_gc_locker:
261     // Restart VM operation previously blocked by the GC locker
262     _gc_locker_port.signal();
263     break;
264 
265   case GCCause::_wb_breakpoint:
266     ZBreakpoint::start_gc();
267     _gc_cycle_port.send_async(cause);
268     break;
269 
270   default:
271     // Other causes not supported
272     fatal("Unsupported GC cause (%s)", GCCause::to_string(cause));
273     break;
274   }
275 }

277 template <typename T>
278 bool ZDriver::pause() {
279   for (;;) {
280     T op;
281     VMThread::execute(&op);
282     if (op.gc_locked()) {
283       // Wait for GC to become unlocked and restart the VM operation
284       ZStatTimer timer(ZCriticalPhaseGCLockerStall);
285       _gc_locker_port.wait();
286       continue;
287     }
288 
289     // Notify VM operation completed
290     _gc_locker_port.ack();
291 
292     return op.success();
293   }
294 }

326 void ZDriver::pause_verify() {
327   if (VerifyBeforeGC || VerifyDuringGC || VerifyAfterGC) {
328     // Full verification
329     VM_Verify op;
330     VMThread::execute(&op);
331   } else if (ZVerifyRoots || ZVerifyObjects) {
332     // Limited verification
333     VM_ZVerify op;
334     VMThread::execute(&op);
335   }
336 }

356 class ZDriverGCScope : public StackObj {
357 private:
358   GCIdMark                   _gc_id;
359   GCCause::Cause             _gc_cause;
360   GCCauseSetter              _gc_cause_setter;
361   ZStatTimer                 _timer;
362   ZServiceabilityCycleTracer _tracer;
363 
364 public:
365   ZDriverGCScope(GCCause::Cause cause) :
366       _gc_id(),
367       _gc_cause(cause),
368       _gc_cause_setter(ZCollectedHeap::heap(), cause),
369       _timer(ZPhaseCycle),
370       _tracer() {
371     // Update statistics
372     ZStatCycle::at_start();
373   }
374 
375   ~ZDriverGCScope() {
376     // Calculate boost factor
377     const double boost_factor = (double)ZHeap::heap()->nconcurrent_worker_threads() /
378                                 (double)ZHeap::heap()->nconcurrent_no_boost_worker_threads();
379 
380     // Update statistics
381     ZStatCycle::at_end(_gc_cause, boost_factor);
382 
383     // Update data used by soft reference policy
384     Universe::update_heap_info_at_gc();
385   }
386 };

388 void ZDriver::gc(GCCause::Cause cause) {
389   ZDriverGCScope scope(cause);
390 
391   // Phase 1: Pause Mark Start
392   pause_mark_start();
393 
394   // Phase 2: Concurrent Mark
395   concurrent_mark();
396 
397   // Phase 3: Pause Mark End
398   while (!pause_mark_end()) {
399     // Phase 3.5: Concurrent Mark Continue
400     concurrent_mark_continue();
401   }
402 
403   // Phase 4: Concurrent Process Non-Strong References
404   concurrent_process_non_strong_references();
405 
406   // Phase 5: Concurrent Reset Relocation Set
407   concurrent_reset_relocation_set();
408 
409   // Phase 6: Pause Verify
410   pause_verify();
411 
412   // Phase 7: Concurrent Select Relocation Set
413   concurrent_select_relocation_set();
414 
415   // Phase 8: Pause Relocate Start
416   pause_relocate_start();
417 
418   // Phase 9: Concurrent Relocate
419   concurrent_relocate();
420 }

422 void ZDriver::run_service() {
423   // Main loop
424   while (!should_terminate()) {
425     // Wait for GC request
426     const GCCause::Cause cause = _gc_cycle_port.receive();
427     if (cause == GCCause::_no_gc) {
428       continue;
429     }
430 
431     ZBreakpoint::at_before_gc();
432 
433     // Run GC
434     gc(cause);
435 
436     // Notify GC completed
437     _gc_cycle_port.ack();
438 
439     // Check for out of memory condition
440     check_out_of_memory();
441 
442     ZBreakpoint::at_after_gc();
443   }
444 }
445 
446 void ZDriver::stop_service() {
447   _gc_cycle_port.send_async(GCCause::_no_gc);
