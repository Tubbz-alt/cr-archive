 82 template <typename T, void (T::*F)(ZRootsIteratorClosure*)>
 83 ZSerialOopsDo<T, F>::ZSerialOopsDo(T* iter) :
 84     _iter(iter),
 85     _claimed(false) {}
 86 
 87 template <typename T, void (T::*F)(ZRootsIteratorClosure*)>
 88 void ZSerialOopsDo<T, F>::oops_do(ZRootsIteratorClosure* cl) {
 89   if (!_claimed && Atomic::cmpxchg(&_claimed, false, true) == false) {
 90     (_iter->*F)(cl);
 91   }
 92 }

136 class ZRootsIteratorCodeBlobClosure : public CodeBlobClosure {
137 private:
138   ZRootsIteratorClosure* const _cl;
139   const bool                   _should_disarm_nmethods;
140 
141 public:
142   ZRootsIteratorCodeBlobClosure(ZRootsIteratorClosure* cl) :
143       _cl(cl),
144       _should_disarm_nmethods(cl->should_disarm_nmethods()) {}
145 
146   virtual void do_code_blob(CodeBlob* cb) {
147     nmethod* const nm = cb->as_nmethod_or_null();
148     if (nm != NULL && nm->oops_do_try_claim()) {
149       ZNMethod::nmethod_oops_do(nm, _cl, _should_disarm_nmethods);
150       nm->mark_as_maybe_on_continuation();
151       assert(!ZNMethod::supports_entry_barrier(nm) ||
152              ZNMethod::is_armed(nm) == _should_disarm_nmethods, "Invalid state");
153       if (_should_disarm_nmethods) {
154         ZNMethod::disarm(nm);
155       }
156     }
157   }
158 };

160 class ZRootsIteratorThreadClosure : public ThreadClosure {
161 private:
162   ZRootsIteratorClosure* const _cl;
163   ResourceMark                 _rm;
164 
165 public:
166   ZRootsIteratorThreadClosure(ZRootsIteratorClosure* cl) :
167       _cl(cl) {}
168 
169   virtual void do_thread(Thread* thread) {
170     ZRootsIteratorCodeBlobClosure code_cl(_cl);
171     thread->oops_do(_cl, ClassUnloading ? &code_cl : NULL);
172     _cl->do_thread(thread);
173   }
174 };

176 ZJavaThreadsIterator::ZJavaThreadsIterator() :
177     _threads(),
178     _claimed(0) {}
179 
180 uint ZJavaThreadsIterator::claim() {
181   return Atomic::fetch_and_add(&_claimed, 1u);
182 }
183 
184 void ZJavaThreadsIterator::threads_do(ThreadClosure* cl) {
185   for (uint i = claim(); i < _threads.length(); i = claim()) {
186     cl->do_thread(_threads.thread_at(i));
187   }
188 }

190 ZRootsIterator::ZRootsIterator(bool visit_jvmti_weak_export) :
191     _visit_jvmti_weak_export(visit_jvmti_weak_export),
192     _java_threads_iter(),
193     _universe(this),
194     _object_synchronizer(this),
195     _jvmti_weak_export(this),
196     _vm_thread(this),
197     _java_threads(this),
198     _code_cache(this) {
199   assert(SafepointSynchronize::is_at_safepoint(), "Should be at safepoint");
200   ZStatTimer timer(ZSubPhasePauseRootsSetup);
201   COMPILER2_OR_JVMCI_PRESENT(DerivedPointerTable::clear());
202   if (ClassUnloading) {
203     nmethod::oops_do_marking_prologue();
204   } else {
205     ZNMethod::oops_do_begin();
206   }
207 }
208 
209 ZRootsIterator::~ZRootsIterator() {
210   ZStatTimer timer(ZSubPhasePauseRootsTeardown);
211   ResourceMark rm;
212   if (ClassUnloading) {
213     nmethod::oops_do_marking_epilogue();
214   } else {
215     ZNMethod::oops_do_end();
216   }
217 
218   COMPILER2_OR_JVMCI_PRESENT(DerivedPointerTable::update_pointers());
219 }

254 void ZRootsIterator::oops_do(ZRootsIteratorClosure* cl) {
255   ZStatTimer timer(ZSubPhasePauseRoots);
256   _universe.oops_do(cl);
257   _object_synchronizer.oops_do(cl);
258   _vm_thread.oops_do(cl);
259   _java_threads.oops_do(cl);
260   if (!ClassUnloading) {
261     _code_cache.oops_do(cl);
262   }
263   if (_visit_jvmti_weak_export) {
264     _jvmti_weak_export.oops_do(cl);
265   }
266 }
