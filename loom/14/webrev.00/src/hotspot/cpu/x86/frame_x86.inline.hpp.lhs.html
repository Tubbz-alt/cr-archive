 37 class ContinuationCodeBlobLookup {
 38 public:
 39   enum { has_oopmap_lookup = true };

 41   static CodeBlob* find_blob(address pc) {
 42     CodeBlob* cb = CodeCache::find_blob_fast(pc);
 43     /*Prefetch::read(cb, PrefetchScanIntervalInBytes);
 44     Prefetch::read((void*)cb->is_compiled_addr(), PrefetchScanIntervalInBytes);

 50     return CodeCache::find_blob_and_oopmap(pc, slot);
 51   }
 52 };

141 inline void frame::setup(address pc) {
142   adjust_unextended_sp();
143 
144   address original_pc = CompiledMethod::get_deopt_original_pc(this);
145   if (original_pc != NULL) {
146     _pc = original_pc;
147     _deopt_state = is_deoptimized;
148     assert(_cb == NULL || _cb->as_compiled_method()->insts_contains_inclusive(_pc),
149            "original PC must be in the main code section of the the compiled method (or must be immediately following it)");
150   } else {
151     if (_cb == SharedRuntime::deopt_blob()) {
152       _deopt_state = is_deoptimized;
153     } else {
154       _deopt_state = not_deoptimized;
155     }
156   }
157 }

159 inline frame::frame(intptr_t* sp, intptr_t* fp) {
160   _sp = sp;
161   _unextended_sp = sp;
162   _fp = fp;
163   _pc = (address)(sp[-1]);
164 
165   // Here's a sticky one. This constructor can be called via AsyncGetCallTrace
166   // when last_Java_sp is non-null but the pc fetched is junk. If we are truly
167   // unlucky the junk value could be to a zombied method and we'll die on the
168   // find_blob call. This is also why we can have no asserts on the validity
169   // of the pc we find here. AsyncGetCallTrace -> pd_get_top_frame_for_signal_handler
170   // -> pd_last_frame should use a specialized version of pd_last_frame which could
171   // call a specialized frame constructor instead of this one.
172   // Then we could use the assert below. However this assert is of somewhat dubious
173   // value.
174   // UPDATE: this constructor is only used by trace_method_handle_stub() now.
175   // assert(_pc != NULL, "no pc?");
176 
177   _cb = CodeCache::find_blob(_pc);
178   adjust_unextended_sp();
179 
180   address original_pc = CompiledMethod::get_deopt_original_pc(this);
181   if (original_pc != NULL) {
182     _pc = original_pc;
183     _deopt_state = is_deoptimized;
184   } else {
185     _deopt_state = not_deoptimized;
186   }
187   _oop_map = NULL;
188 }

190 inline frame::frame(int sp, int ref_sp, intptr_t fp, address pc, CodeBlob* cb, bool deopt) {
191   _cont_sp._sp = sp;
192   _cont_sp._ref_sp = ref_sp;
193   _unextended_sp = NULL;
194   _fp = (intptr_t*)fp;
195   _pc = pc;
196   assert(pc != NULL, "no pc?");
197   _cb = cb;
198   _deopt_state = deopt ? is_deoptimized : not_deoptimized;
199   _oop_map = NULL;
200 }

204 inline bool frame::equal(frame other) const {
205   bool ret =  sp() == other.sp()
206               && unextended_sp() == other.unextended_sp()
207               && fp() == other.fp()
208               && pc() == other.pc();
209   assert(!ret || ret && cb() == other.cb() && _deopt_state == other._deopt_state, "inconsistent construction");
210   return ret;
211 }

228 inline intptr_t* frame::real_fp() const {
229   if (_cb != NULL) {
230     // use the frame size if valid
231     int size = _cb->frame_size();
232     if (size > 0) {
233       return unextended_sp() + size;
234     }
235   }
236   // else rely on fp()
237   assert(! is_compiled_frame(), "unknown compiled frame size");
238   return fp();
239 }

241 // helper to update a map with callee-saved RBP
242 
243 template <typename RegisterMapT>
244 void frame::update_map_with_saved_link(RegisterMapT* map, intptr_t** link_addr) {
245   // The interpreter and compiler(s) always save EBP/RBP in a known
246   // location on entry. We must record where that location is
247   // so this if EBP/RBP was live on callout from c2 we can find
248   // the saved copy no matter what it called.
249 
250   // Since the interpreter always saves EBP/RBP if we record where it is then
251   // we don't have to always save EBP/RBP on entry and exit to c2 compiled
252   // code, on entry will be enough.
253   map->set_location(rbp->as_VMReg(), (address) link_addr);
254 #ifdef AMD64
255   // this is weird "H" ought to be at a higher address however the
256   // oopMaps seems to have the "H" regs at the same address and the
257   // vanilla register.
258   // XXXX make this go away
259   if (true) {
260     map->set_location(rbp->as_VMReg()->next(), (address) link_addr);
261   }
262 #endif // AMD64
263 }

267   return (intptr_t**)map->location(rbp->as_VMReg());

314 // top of expression stack
315 inline intptr_t* frame::interpreter_frame_tos_address() const {
316   intptr_t* last_sp = interpreter_frame_last_sp();
317   if (last_sp == NULL) {
318     return sp();
319   } else {
320     // sp() may have been extended or shrunk by an adapter.  At least
321     // check that we don't fall behind the legal region.
322     // For top deoptimized frame last_sp == interpreter_frame_monitor_end.
323     assert(last_sp <= (intptr_t*) interpreter_frame_monitor_end(), "bad tos");
324     return last_sp;
325   }
326 }

371 template <typename LOOKUP>
372 frame frame::frame_sender(RegisterMap* map) const {
373   // Default is we done have to follow them. The sender_for_xxx will
374   // update it accordingly
375   map->set_include_argument_oops(false);
376 
377   if (is_entry_frame())       return sender_for_entry_frame(map);
378   if (is_interpreted_frame()) return sender_for_interpreter_frame(map);
379 
380   assert(_cb == CodeCache::find_blob(pc()), "Must be the same");
381 
382   if (_cb != NULL) {
383     return _cb->is_compiled() ? sender_for_compiled_frame<LOOKUP, false>(map) : sender_for_compiled_frame<LOOKUP, true>(map);
384   }
385   // Must be native-compiled frame, i.e. the marshaling code for native
386   // methods that exists in the core system.
387   return frame(sender_sp(), link(), sender_pc());
388 }

390 //------------------------------------------------------------------------------
391 // frame::sender_for_compiled_frame
392 template <typename LOOKUP, bool stub>
393 frame frame::sender_for_compiled_frame(RegisterMap* map) const {
394   assert(map != NULL, "map must be set");
395 
396   if (map->in_cont()) { // already in an h-stack
397     return Continuation::sender_for_compiled_frame(*this, map);
398   }
399 
400   // frame owned by optimizing compiler
401   assert(_cb->frame_size() >= 0, "must have non-zero frame size");
402   intptr_t* sender_sp = unextended_sp() + _cb->frame_size();
403 
404   assert (sender_sp == real_fp(), "sender_sp: " INTPTR_FORMAT " real_fp: " INTPTR_FORMAT, p2i(sender_sp), p2i(real_fp()));
405 
406   // On Intel the return_address is always the word on the stack
407   address sender_pc = (address) *(sender_sp-1);
408 
409   // This is the saved value of EBP which may or may not really be an FP.
410   // It is only an FP if the sender is an interpreter frame (or C1?).
411   // saved_fp_addr should be correct even for a bottom thawed frame (with a return barrier)
412   intptr_t** saved_fp_addr = (intptr_t**) (sender_sp - frame::sender_sp_offset);
413   intptr_t* sender_fp = *saved_fp_addr;
414   if (map->update_map()) {
415     // Tell GC to use argument oopmaps for some runtime stubs that need it.
416     // For C1, the runtime stub might not have oop maps, so set this flag
417     // outside of update_register_map.
418     if (stub) { // compiled frames do not use callee-saved registers
419       map->set_include_argument_oops(_cb->caller_must_gc_arguments(map->thread()));
420       if (oop_map() != NULL) { 
421         _oop_map->update_register_map(this, map);
422       }
423     } else {
424       assert (!_cb->caller_must_gc_arguments(map->thread()), "");
425       assert (!map->include_argument_oops(), "");
426       assert (oop_map() == NULL || !oop_map()->has_any(OopMapValue::callee_saved_value), "callee-saved value in compiled frame");
427     }
428 
429     // Since the prolog does the save and restore of EBP there is no oopmap
430     // for it so we must fill in its location as if there was an oopmap entry
431     // since if our caller was compiled code there could be live jvm state in it.
432     update_map_with_saved_link(map, saved_fp_addr);
433   }
434 
435   assert(sender_sp != sp(), "must have changed");
436 
437   if (Continuation::is_return_barrier_entry(sender_pc)) {	
438     if (map->walk_cont()) { // about to walk into an h-stack 	
439       return Continuation::top_frame(*this, map);	
440     } else {
441       Continuation::fix_continuation_bottom_sender(map->thread(), *this, &sender_pc, &sender_sp);	
442     }
443   }
444 
445   intptr_t* unextended_sp = sender_sp;
446   CodeBlob* sender_cb = LOOKUP::find_blob(sender_pc);
447   if (sender_cb != NULL) {
448     return frame(sender_sp, unextended_sp, sender_fp, sender_pc, sender_cb);
449   }
450   // tty->print_cr(">>>> NO CB sender_pc: %p", sender_pc); os::print_location(tty, (intptr_t)sender_pc); print_on(tty);
451   return frame(sender_sp, unextended_sp, sender_fp, sender_pc);
452 }

454 inline const ImmutableOopMap* frame::get_oop_map() const {
455   if (_cb == NULL) return NULL;
456   if (_cb->oop_maps() != NULL) {
457     NativePostCallNop* nop = nativePostCallNop_at(_pc);
458     if (nop != NULL &&
459 #ifdef CONT_DOUBLE_NOP
460       !nop->is_mode2() &&
461 #endif
462       nop->displacement() != 0
463     ) {
464       int slot = ((nop->displacement() >> 24) & 0xff);
465       return _cb->oop_map_for_slot(slot, _pc);
466     }
467     const ImmutableOopMap* oop_map = OopMapSet::find_map(this);
468     return oop_map;
469   }
470   return NULL;
471 }
